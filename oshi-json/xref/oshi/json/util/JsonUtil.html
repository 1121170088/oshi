19  package oshi.json.util;
20  
21  import java.io.StringWriter;
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import javax.json.Json;
26  import javax.json.JsonObject;
27  import javax.json.JsonWriter;
28  import javax.json.JsonWriterFactory;
29  import javax.json.stream.JsonGenerator;
30  
31  /**
32   * Formatting utility for json strings
33   *
34   * @author widdis[at]gmail[dot]com
35   */
36  public class JsonUtil {
37      private JsonUtil() {
38      }
39  
40      /**
41       * Pretty print a JSON string.
42       *
43       * @param json
44       *            A JSON object
45       * @return String representing the object with added whitespace, new lines,
46       *         indentation
47       */
48      public static String jsonPrettyPrint(JsonObject json) {
49          // Pretty printing using JsonWriterFactory
50          // Output stream
51          StringWriter stringWriter = new StringWriter();
52          // Config
53          Map<String, Boolean> config = new HashMap<>();
54          config.put(JsonGenerator.PRETTY_PRINTING, true);
55          JsonWriterFactory writerFactory = Json.createWriterFactory(config);
56          // Writer
57          try (JsonWriter jsonWriter = writerFactory.createWriter(stringWriter)) {
58              jsonWriter.write(json);
59          }
60          // Return
61          return stringWriter.toString();
62      }
63  }

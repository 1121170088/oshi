38      /**
39       * Test operating system
40       */
41      @Test
42      public void testOperatingSystem() {
43          SystemInfo si = new SystemInfo();
44          OperatingSystem os = si.getOperatingSystem();
45          assertNotNull(os.getFamily());
46          assertNotNull(os.getManufacturer());
47          OperatingSystemVersion version = os.getVersion();
48          assertNotNull(version);
49          assertNotNull(version.getVersion());
50          assertNotNull(version.getCodeName());
51          assertNotNull(version.getBuildNumber());
52  
53          assertTrue(os.getProcessCount() >= 1);
54          assertTrue(os.getThreadCount() >= 1);
55          assertTrue(os.getProcessId() > 0);
56  
57          assertTrue(os.getProcesses(0, null).length > 0);
58          OSProcess proc = os.getProcess(os.getProcessId());
59          assertTrue(proc.getName().length() > 0);
60          assertTrue(proc.getPath().length() > 0);
61          assertTrue(proc.getCommandLine().length() > 0);
62          assertNotNull(proc.getCurrentWorkingDirectory());
63          assertNotNull(proc.getUser());
64          assertNotNull(proc.getUserID());
65          assertNotNull(proc.getGroup());
66          assertNotNull(proc.getGroupID());
67          assertNotNull(proc.getState());
68          assertEquals(proc.getProcessID(), os.getProcessId());
69          assertTrue(proc.getParentProcessID() > 0);
70          assertTrue(proc.getThreadCount() > 0);
71          assertTrue(proc.getPriority() >= -20 && proc.getPriority() <= 128);
72          assertTrue(proc.getVirtualSize() >= proc.getResidentSetSize());
73          assertTrue(proc.getResidentSetSize() >= 0);
74          assertTrue(proc.getKernelTime() >= 0);
75          assertTrue(proc.getUserTime() >= 0);
76          assertTrue(proc.getUpTime() >= 0);
77          assertTrue(proc.getStartTime() >= 0);
78          assertTrue(proc.getBytesRead() >= 0);
79          assertTrue(proc.getBytesWritten() >= 0);
80          assertTrue(proc.getOpenFiles() >= -1);
81      }

83      /**
84       * tests the
85       */
86      @Test
87      public void testProcessQueryByList() {
88          SystemInfo si = new SystemInfo();
89          OperatingSystem os = si.getOperatingSystem();
90          assertNotNull(os.getFamily());
91          assertNotNull(os.getManufacturer());
92          OperatingSystemVersion version = os.getVersion();
93          assertNotNull(version);
94          assertNotNull(version.getVersion());
95          assertNotNull(version.getCodeName());
96          assertNotNull(version.getBuildNumber());
97  
98          assertTrue(os.getProcessCount() >= 1);
99          assertTrue(os.getThreadCount() >= 1);
100         assertTrue(os.getProcessId() > 0);
101 
102         OSProcess[] processes = os.getProcesses(5, null);
103         assertNotNull(processes);
104         // every OS should have at least one process running on it
105         assertTrue(processes.length > 0);
106         // the list of pids we want info on
107         List<Integer> pids = new ArrayList<>();
108         for (OSProcess p : processes) {
109             pids.add(p.getProcessID());
110         }
111         // query for just those processes
112         Collection<OSProcess> processes1 = os.getProcesses(pids);
113         // theres a potential for a race condition here, if a process we queried
114         // for initially wasn't running during the second query. In this case,
115         // try again with the shorter list
116         if (processes1.size() < processes.length) {
117             pids.clear();
118             for (OSProcess p : processes1) {
119                 pids.add(p.getProcessID());
120             }
121             // query for just those processes
122             processes1 = os.getProcesses(pids);
123         }
124         assertEquals(processes.length, processes1.size());
125 
126     }

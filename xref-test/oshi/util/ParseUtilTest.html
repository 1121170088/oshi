331      * Test parse string to array
332      */
333     @Test
334     public void testParseStringToLongArray() {
335         int[] indices = { 1, 3 };
336         long now = System.currentTimeMillis();
337 
338         String foo = String.format("The numbers are %d %d %d %d", 123, 456, 789, now);
339         int count = ParseUtil.countStringToLongArray(foo, ' ');
340         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
341         long[] result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
342         assertEquals("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", 456L, result[0]);
343         assertEquals("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", now, result[1]);
344 
345         foo = String.format("The numbers are %d %d %d %d %s", 123, 456, 789, now,
346                 "709af748-5f8e-41b3-b73a-b440ef4406c8");
347         count = ParseUtil.countStringToLongArray(foo, ' ');
348         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
349         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
350         assertEquals("result[0] should be 456 using parseStringToLongArray on \"" + foo + "\"", 456L, result[0]);
351         assertEquals("result[1] should be " + now + " using parseStringToLongArray on \"" + foo + "\"", now, result[1]);
352 
353         foo = String.format("The numbers are %d -%d %d +%d", 123, 456, 789, now);
354         count = ParseUtil.countStringToLongArray(foo, ' ');
355         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
356         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
357         assertEquals("result[0] should be -4456 using parseStringToLongArray on \"" + foo + "\"", -456L, result[0]);
358         assertEquals("result[1] index should be 456 using parseStringToLongArray on \"" + foo + "\"", now, result[1]);
359 
360         foo = String.format("Invalid character %d %s %d %d", 123, "4v6", 789, now);
361         count = ParseUtil.countStringToLongArray(foo, ' ');
362         assertEquals("countStringToLongArray should return 2 for \"" + foo + "\"", 2, count);
363         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
364         assertEquals("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", 0, result[1]);
365 
366         foo = String.format("Exceeds max long %d %d %d 1%d", 123, 456, 789, Long.MAX_VALUE);
367         result = ParseUtil.parseStringToLongArray(foo, indices, 4, ' ');
368         assertEquals("result[1] index should be " + Long.MAX_VALUE
369                 + " (Long.MAX_VALUE) using parseStringToLongArray on \"" + foo + "\"", Long.MAX_VALUE, result[1]);
370 
371         foo = String.format("String too short %d %d %d %d", 123, 456, 789, now);
372         result = ParseUtil.parseStringToLongArray(foo, indices, 9, ' ');
373         assertEquals("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", 0, result[1]);
374 
375         foo = String.format("Array too short %d %d %d %d", 123, 456, 789, now);
376         result = ParseUtil.parseStringToLongArray(foo, indices, 2, ' ');
377         assertEquals("result[1] index should be 0 using parseStringToLongArray on \"" + foo + "\"", 0, result[1]);
378 
379         foo = String.format("%d %d %d %d", 123, 456, 789, now);
380         count = ParseUtil.countStringToLongArray(foo, ' ');
381         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
382 
383         foo = String.format("%d %d %d %d nonNumeric", 123, 456, 789, now);
384         count = ParseUtil.countStringToLongArray(foo, ' ');
385         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
386 
387         foo = String.format("%d %d %d %d 123-456", 123, 456, 789, now);
388         count = ParseUtil.countStringToLongArray(foo, ' ');
389         assertEquals("countStringToLongArray should return 4 for \"" + foo + "\"", 4, count);
390     }

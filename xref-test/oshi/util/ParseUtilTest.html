17  package oshi.util;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertTrue;
21  
22  import java.util.Arrays;
23  
24  import org.junit.Test;
25  
26  /**
27   * The Class ParseUtilTest.
28   */
29  public class ParseUtilTest
30  {
31  
32      /**
33       * Test parse hertz.
34       */
35      @Test
36      public void testParseHertz()
37      {
38          assertEquals( 1L, ParseUtil.parseHertz( "1Hz" ) );
39          assertEquals( 500L, ParseUtil.parseHertz( "500 Hz" ) );
40          assertEquals( 1000L, ParseUtil.parseHertz( "1kHz" ) );
41          assertEquals( 1000000L, ParseUtil.parseHertz( "1MHz" ) );
42          assertEquals( 1000000000L, ParseUtil.parseHertz( "1GHz" ) );
43          assertEquals( 1500000000L, ParseUtil.parseHertz( "1.5GHz" ) );
44          assertEquals( 1000000000000L, ParseUtil.parseHertz( "1THz" ) );
45      }
46  
47      /**
48       * Test parse string.
49       */
50      @Test
51      public void testParseString()
52      {
53          assertEquals( 1, ParseUtil.parseString( "foo : 1", 0 ) );
54          assertEquals( 2, ParseUtil.parseString( "foo", 2 ) );
55      }
56  
57      /**
58       * Test parse string.
59       */
60      @Test
61      public void testHexStringToByteArray()
62      {
63          byte[] temp = { (byte) 0x12, (byte) 0xaf };
64          assertTrue( Arrays.equals( temp, ParseUtil.hexStringToByteArray( "12af" ) ) );
65      }
66  }

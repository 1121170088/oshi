19  package oshi.util;
20  
21  import static org.junit.Assert.assertEquals;
22  
23  import java.util.List;
24  
25  import org.junit.Test;
26  
27  /**
28   * Tests FileUtil
29   */
30  public class FileUtilTest {
31  
32      /*
33       * File sources
34       */
35      private static String THISCLASS = "src/test/java/oshi/util/FileUtilTest.java";
36      private static String INT_FILE = "src/test/resources/test.integer.txt";
37      private static String STRING_FILE = "src/test/resources/test.string.txt";
38      private static String NO_FILE = "does/not/exist";
39  
40      /**
41       * Test read file.
42       */
43      @Test
44      public void testReadFile() {
45          List<String> thisFile = null;
46          // Try file not found
47          thisFile = FileUtil.readFile(NO_FILE);
48          assertEquals(0, thisFile.size());
49          // Try this file
50          thisFile = FileUtil.readFile(THISCLASS);
51          // Comment ONE line
52          int lineOne = 0;
53          // Comment TWO line
54          int lineTwo = 0;
55          for (int i = 0; i < thisFile.size(); i++) {
56              String line = thisFile.get(i);
57              if (lineOne == 0 && line.contains("Comment ONE line")) {
58                  lineOne = i;
59                  continue;
60              }
61              if (lineTwo == 0 && line.contains("Comment TWO line")) {
62                  lineTwo = i;
63                  break;
64              }
65          }
66          assertEquals(2, lineTwo - lineOne);
67      }
68  
69      /**
70       * Test get*FromFile
71       */
72      @Test
73      public void testGetFromFile() {
74          assertEquals(123L, FileUtil.getLongFromFile(INT_FILE));
75          assertEquals(0L, FileUtil.getLongFromFile(STRING_FILE));
76          assertEquals(0L, FileUtil.getLongFromFile(NO_FILE));
77  
78          assertEquals(123, FileUtil.getIntFromFile(INT_FILE));
79          assertEquals(0, FileUtil.getIntFromFile(STRING_FILE));
80          assertEquals(0, FileUtil.getIntFromFile(NO_FILE));
81  
82          assertEquals("123", FileUtil.getStringFromFile(INT_FILE));
83          assertEquals("", FileUtil.getStringFromFile(NO_FILE));
84  
85          assertEquals(5, FileUtil.getSplitFromFile(STRING_FILE).length);
86          assertEquals(0, FileUtil.getSplitFromFile(NO_FILE).length);
87      }
88  }

19  package oshi.software.os;
20  
21  import static org.junit.Assert.assertEquals;
22  import static org.junit.Assert.assertNotNull;
23  import static org.junit.Assert.assertTrue;
24  
25  import org.junit.Test;
26  
27  import oshi.SystemInfo;
28  
29  /**
30   * Test OS
31   */
32  public class OperatingSystemTest {
33  
34      /**
35       * Test operating system
36       */
37      @Test
38      public void testOperatingSystem() {
39          SystemInfo si = new SystemInfo();
40          OperatingSystem os = si.getOperatingSystem();
41          assertNotNull(os.getFamily());
42          assertNotNull(os.getManufacturer());
43          OperatingSystemVersion version = os.getVersion();
44          assertNotNull(version);
45          assertNotNull(version.getVersion());
46          assertNotNull(version.getCodeName());
47          assertNotNull(version.getBuildNumber());
48  
49          assertTrue(os.getProcessCount() >= 1);
50          assertTrue(os.getThreadCount() >= 1);
51          assertTrue(os.getProcessId() > 0);
52  
53          assertTrue(os.getProcesses(0, null).length > 0);
54          OSProcess proc = os.getProcess(os.getProcessId());
55          assertTrue(proc.getName().length() > 0);
56          assertTrue(proc.getPath().length() > 0);
57          assertNotNull(proc.getState());
58          assertEquals(proc.getProcessID(), os.getProcessId());
59          assertTrue(proc.getParentProcessID() > 0);
60          assertTrue(proc.getThreadCount() > 0);
61          assertTrue(proc.getPriority() >= -20 && proc.getPriority() <= 128);
62          assertTrue(proc.getVirtualSize() >= proc.getResidentSetSize());
63          assertTrue(proc.getResidentSetSize() >= 0);
64          assertTrue(proc.getKernelTime() >= 0);
65          assertTrue(proc.getUserTime() >= 0);
66          assertTrue(proc.getUpTime() >= 0);
67          assertTrue(proc.getStartTime() >= 0);
68      }
69  }

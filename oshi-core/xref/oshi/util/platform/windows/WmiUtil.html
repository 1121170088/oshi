87      /**
88       * Enum for the value type of WMI queries, used to properly parse/cast the
89       * returned VARIANT. Names are intended to match the WMI documentation and
90       * do not actually represent types (e.g., unsigned). Options are
91       * {@link #STRING}, {@link #UINT64}, {@link #UINT32}, {@link #UINT16},
92       * {@link #FLOAT}, {@link #DATETIME}, {@link #BOOLEAN}
93       */
94      public enum ValueType {
95          /**
96           * String type. Returns from WMI as a BSTR. Parsed to a String.
97           */
98          STRING,
99          /**
100          * Unsigned 64-bit Integer. UINT64s return from WMI as a BSTR, parsed to
101          * a String. With the UINT64 type mapping, OSHI will parse the object to
102          * a (signed) long which may incorrectly represent large values. If true
103          * unsigned longs are required, users may prefer to map UINT64 variables
104          * to the String type and then parse the returned String to a
105          * BigInteger.
106          */
107         UINT64,
108         /**
109          * Unsigned 32-bit Integer. Parsed to a long.
110          */
111         UINT32,
112         /**
113          * Unsigned 16-bit Integer. Parsed to an int.
114          */
115         UINT16,
116         /**
117          * Single precision Float. Parsed to a float.
118          */
119         FLOAT,
120         /**
121          * Date/time. Returns from WMI as a String in the CIM_DATETIME format.
122          * Parsed to a long, number of milliseconds since the 1970 epoch.
123          */
124         DATETIME,
125         /**
126          * Boolean. Parsed to a boolean.
127          */
128         BOOLEAN
129         // If more values are added here, add the appropriate parsing as a case
130         // in the enumerateProperties method.
131     }

235     /**
236      * Helper class wrapping an EnumMap containing the results of a query.
237      */
238     public class WmiResult<T extends Enum<T> & WmiProperty> {
239         private Map<T, List<Object>> propertyMap;
240         private int resultCount = 0;
241 
242         /**
243          * @param propertyEnum
244          *            The enum associated with this map
245          */
246         public WmiResult(Class<T> propertyEnum) {
247             propertyMap = new EnumMap<>(propertyEnum);
248             for (T type : propertyEnum.getEnumConstants()) {
249                 propertyMap.put(type, new ArrayList<>());
250             }
251         }
252 
253         /**
254          * @param item
255          *            An element of the Enum
256          * @return the list associated with that enum
257          */
258         public List<Object> get(final T item) {
259             return this.propertyMap.get(item);
260         }
261 
262         /**
263          * @return The number of results in each mapped list
264          */
265         public int getResultCount() {
266             return this.resultCount;
267         }
268 
269         /**
270          * Increment the result count by one.
271          */
272         public void incrementResultCount() {
273             this.resultCount++;
274         }
275     }

277     /**
278      * Helper class for WMI connection caching
279      */
280     private class WmiConnection {
281         // Connection invalid after this long
282         public static final long STALE_CONNECTION = 300_000L;
283 
284         private long staleAfter;
285         private WbemServices svc;
286 
287         WmiConnection(PointerByReference pSvc) {
288             this.svc = new WbemServices(pSvc.getValue());
289             refresh();
290         }
291 
292         public WbemServices getService() {
293             return this.svc;
294         }
295 
296         public boolean isStale() {
297             return System.currentTimeMillis() > staleAfter;
298         }
299 
300         public void refresh() {
301             this.staleAfter = STALE_CONNECTION + System.currentTimeMillis();
302         }
303 
304         public void close() {
305             this.svc.Release();
306         }
307     }

365     /**
366      * Determine if WMI has the requested namespace. Some namespaces only exist
367      * on newer versions of Windows.
368      *
369      * @param namespace
370      *            The namespace to test
371      * @return true if the namespace exists, false otherwise
372      */
373     public static boolean hasNamespace(String namespace) {
374         if (hasNamespaceCache.contains(namespace)) {
375             return true;
376         } else if (hasNotNamespaceCache.contains(namespace)) {
377             return false;
378         }
379         WmiResult<NamespaceProperty> namespaces = queryWMI(NAMESPACE_QUERY);
380         for (int i = 0; i < namespaces.getResultCount(); i++) {
381             if (namespace.equals(namespaces.get(NamespaceProperty.NAME).get(i))) {
382                 return true;
383             }
384         }
385         return false;
386     }

389      * Query WMI for values, with no timeout.
390      * 
391      * @param <T>
392      *            an enum implementing WmiProperty
393      * @param query
394      *            A WmiQuery object encapsulating the namespace, class,
395      *            properties, and timeout
396      * @return a WmiResult object containing the query results, wrapping an
397      *         EnumMap
398      */
399     public static <T extends Enum<T> & WmiProperty> WmiResult<T> queryWMI(WmiQuery<T> query) {
400         try {
401             return queryWMI(query, EnumWbemClassObject.WBEM_INFINITE);
402         } catch (TimeoutException e) {
403             LOG.error("Got a WMI timeout when infinite wait specified. This should never happen.", e);
404             return INSTANCE.new WmiResult<>(query.getPropertyEnum());
405         }

408     /**
409      * Query WMI for values, with a specified timeout.
410      * 
411      * @param <T>
412      *            an enum implementing WmiProperty
413      * @param query
414      *            A WmiQuery object encapsulating the namespace, class,
415      *            properties, and timeout
416      * @param timeout
417      *            Number of milliseconds to wait for results before timing out.
418      *            If {@link EnumWbemClassObject#WBEM_INFINITE} (-1), will always
419      *            wait for results. If a timeout occurs, throws a
420      *            {@link TimeoutException}.
421      * @return a WmiResult object containing the query results, wrapping an
422      *         EnumMap
423      * @throws TimeoutException
424      *             if the query times out before completion
425      */
426     public static <T extends Enum<T> & WmiProperty> WmiResult<T> queryWMI(WmiQuery<T> query, int timeout)
427             throws TimeoutException {
428         // Idiot check
429         if (query.getPropertyEnum().getEnumConstants().length < 1) {
430             throw new IllegalArgumentException("The query's property enum has no values.");
431         }
432 
433         // Set up empty map
434         WmiResult<T> values = INSTANCE.new WmiResult<>(query.getPropertyEnum());
435 
436         // Initialize COM if not already done
437         if (!comInitialized && !initCOM()) {
438             unInitCOM();
439             return values;
440         }
441 
442         // Connect to the server
443         WmiConnection conn = connectToNamespace(query.getNameSpace());
444         if (conn == null) {
445             return values;
446         }
447 
448         // Send query
449         PointerByReference pEnumerator = new PointerByReference();
450         if (!selectProperties(conn.getService(), pEnumerator, query)) {
451             conn.close();
452             return values;
453         }
454         EnumWbemClassObject enumerator = new EnumWbemClassObject(pEnumerator.getValue());
455         try {
456             enumerateProperties(values, enumerator, query.getPropertyEnum(), timeout);
457         } catch (TimeoutException e) {
458             throw new TimeoutException(e.getMessage());
459         } finally {
460             // Cleanup
461             enumerator.Release();
462         }
463         return values;
464     }

466     /**
467      * Below methods ported from: Getting WMI Data from Local Computer
468      * https://docs.microsoft.com/en-us/windows/desktop/WmiSdk/example--getting-
469      * wmi-data-from-the-local-computer
470      *
471      * Steps 1 - 7 in the comments correspond to the above link. Steps 1 through
472      * 5 contain all the steps required to set up and connect to WMI, and steps
473      * 6 and 7 are where data is queried and received.
474      */
475     /**
476      * Initializes COM library and sets security to impersonate the local user
477      *
478      * @return true if COM successfully initialized
479      */
480     private static boolean initCOM() {
481         // Step 1: --------------------------------------------------
482         // Initialize COM. ------------------------------------------
483         HRESULT hres = Ole32.INSTANCE.CoInitializeEx(null, Ole32.COINIT_MULTITHREADED);
484         if (COMUtils.FAILED(hres)) {
485             if (hres.intValue() == WinError.RPC_E_CHANGED_MODE) {
486                 // Com already initialized, ignore error
487                 LOG.debug("COM already initialized.");
488                 securityInitialized = true;
489                 return true;
490             }
491             if (LOG.isErrorEnabled()) {
492                 LOG.error(String.format("Failed to initialize COM library. Error code = 0x%08x", hres.intValue()));
493             }
494             return false;
495         }
496         comInitialized = true;
497         if (securityInitialized) {
498             // Only run CoInitializeSecuirty once
499             return true;
500         }
501         // Step 2: --------------------------------------------------
502         // Set general COM security levels --------------------------
503         hres = Ole32.INSTANCE.CoInitializeSecurity(null, new NativeLong(-1), null, null,
504                 Ole32.RPC_C_AUTHN_LEVEL_DEFAULT, Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE, null);
505         if (COMUtils.FAILED(hres) && hres.intValue() != WinError.RPC_E_TOO_LATE) {
506             if (LOG.isErrorEnabled()) {
507                 LOG.error(String.format("Failed to initialize security. Error code = 0x%08x", hres.intValue()));
508             }
509             Ole32.INSTANCE.CoUninitialize();
510             return false;
511         }
512         securityInitialized = true;
513         return true;
514     }

516     /**
517      * Find an existing open connection to a namespace if one exists, otherwise
518      * set up a new one
519      * 
520      * @param namespace
521      *            The namespace to connect to
522      * @return The new or cached connection if successful; null if connection
523      *         failed
524      */
525     private static WmiConnection connectToNamespace(String namespace) {
526         // Every once in a while clear any other stale connections
527         if (System.currentTimeMillis() > nextCacheClear) {
528             closeStaleConnections();
529             nextCacheClear = System.currentTimeMillis() + WmiConnection.STALE_CONNECTION;
530         }
531         // Check if connection already open
532         if (connectionCache.containsKey(namespace)) {
533             WmiConnection conn = connectionCache.get(namespace);
534             if (conn.isStale()) {
535                 // Connection expired. Close it.
536                 conn.close();
537                 connectionCache.remove(namespace);
538             } else {
539                 return conn;
540             }
541         }
542         // Connect to the server
543         PointerByReference pSvc = new PointerByReference();
544         if (!connectServer(namespace, pSvc)) {
545             return null;
546         }
547         WmiConnection conn = INSTANCE.new WmiConnection(pSvc);
548         // Add to cache
549         connectionCache.put(namespace, conn);
550         return conn;
551     }

553     /**
554      * Obtains a locator to the WMI server and connects to the specified
555      * namespace
556      *
557      * @param namespace
558      *            The namespace to connect to
559      * @param pSvc
560      *            A pointer to receive an indirect to the WMI service
561      * @return true if successful; pSvc will contain an indirect pointer to the
562      *         WMI service for future IWbemServices calls
563      */
564     private static boolean connectServer(String namespace, PointerByReference pSvc) {
565         // Step 3: ---------------------------------------------------
566         // Obtain the initial locator to WMI -------------------------
567         WbemLocator loc = WbemLocator.create();
568         if (loc == null) {
569             return false;
570         }
571         // Step 4: -----------------------------------------------------
572         // Connect to WMI through the IWbemLocator::ConnectServer method
573         // Connect to the namespace with the current user and obtain pointer
574         // pSvc to make IWbemServices calls.
575         HRESULT hres = loc.ConnectServer(new BSTR(namespace), null, null, null, null, null, null, pSvc);
576         if (COMUtils.FAILED(hres)) {
577             // Don't error on OpenHardwareMonitor
578             if (!OHM_NAMESPACE.equals(namespace) && LOG.isErrorEnabled()) {
579                 LOG.error(String.format("Could not connect to namespace %s. Error code = 0x%08x", namespace,
580                         hres.intValue()));
581             }
582             loc.Release();
583             unInitCOM();
584             return false;
585         }
586         LOG.debug("Connected to {} WMI namespace", namespace);
587         loc.Release();
588 
589         // Step 5: --------------------------------------------------
590         // Set security levels on the proxy -------------------------
591         hres = Ole32.INSTANCE.CoSetProxyBlanket(pSvc.getValue(), Ole32.RPC_C_AUTHN_WINNT, Ole32.RPC_C_AUTHZ_NONE, null,
592                 Ole32.RPC_C_AUTHN_LEVEL_CALL, Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE);
593         if (COMUtils.FAILED(hres)) {
594             if (LOG.isErrorEnabled()) {
595                 LOG.error(String.format("Could not set proxy blanket. Error code = 0x%08x", hres.intValue()));
596             }
597             new WbemServices(pSvc.getValue()).Release();
598             unInitCOM();
599             return false;
600         }
601         LOG.debug("Proxy blanket set.");
602         return true;
603     }

605     /**
606      * Selects properties from WMI. Returns immediately, even while results are
607      * being retrieved; results may begun to be enumerated in the forward
608      * direction only.
609      *
610      * @param svc
611      *            A WbemServices object to make the calls
612      * @param pEnumerator
613      *            An enumerator to receive the results of the query
614      * @param query
615      *            A WmiQuery object encapsulating the details of the query
616      * @return True if successful. The enumerator will allow enumeration of
617      *         results of the query
618      */
619     private static <T extends Enum<T> & WmiProperty> boolean selectProperties(WbemServices svc,
620             PointerByReference pEnumerator, WmiQuery<T> query) {
621         // Step 6: --------------------------------------------------
622         // Use the IWbemServices pointer to make requests of WMI ----
623         T[] props = query.getPropertyEnum().getEnumConstants();
624         StringBuilder sb = new StringBuilder("SELECT ");
625         // We earlier checked for at least one enum constant
626         sb.append(props[0].name());
627         for (int i = 1; i < props.length; i++) {
628             sb.append(',').append(props[i].name());
629         }
630         sb.append(" FROM ").append(query.getWmiClassName());
631         LOG.debug("Query: {}", sb);
632         // Send the query. The flags allow us to return immediately and begin
633         // enumerating in the forward direction as results come in.
634         HRESULT hres = svc.ExecQuery(new BSTR("WQL"), new BSTR(sb.toString().replaceAll("\\\\", "\\\\\\\\")),
635                 new NativeLong(
636                         EnumWbemClassObject.WBEM_FLAG_FORWARD_ONLY | EnumWbemClassObject.WBEM_FLAG_RETURN_IMMEDIATELY),
637                 null, pEnumerator);
638         if (COMUtils.FAILED(hres)) {
639             if (LOG.isErrorEnabled()) {
640                 LOG.error(String.format("Query '%s' failed. Error code = 0x%08x", sb, hres.intValue()));
641             }
642             svc.Release();
643             unInitCOM();
644             return false;
645         }
646         LOG.debug("Query succeeded.");
647         return true;
648     }

650     /**
651      * Enumerate the results of a WMI query. This method is called while results
652      * are still being retrieved and may iterate in the forward direction only.
653      * 
654      * @param values
655      *            A WmiResult object encapsulating an EnumMap which will hold
656      *            the results.
657      * @param enumerator
658      *            The enumerator with the results
659      * @param propertyEnum
660      *            The enum containing the properties to enumerate, which are the
661      *            keys to the WmiResult map
662      * @param timeout
663      *            Number of milliseconds to wait for results before timing out.
664      *            If {@link EnumWbemClassObject#WBEM_INFINITE} (-1), will always
665      *            wait for results.
666      * @throws TimeoutException
667      *             if the query times out before completion
668      */
669     private static <T extends Enum<T> & WmiProperty> void enumerateProperties(WmiResult<T> values,
670             EnumWbemClassObject enumerator, Class<T> propertyEnum, int timeout) throws TimeoutException {
671         // Step 7: -------------------------------------------------
672         // Get the data from the query in step 6 -------------------
673         PointerByReference pclsObj = new PointerByReference();
674         LongByReference uReturn = new LongByReference(0L);
675         while (enumerator.getPointer() != Pointer.NULL) {
676             // Enumerator will be released by calling method so no need to
677             // release it here.
678             HRESULT hres = enumerator.Next(new NativeLong(timeout), new NativeLong(1), pclsObj, uReturn);
679             // Warn user of timeout and abort. This should never happen if
680             // timeout is set to infinite.
681             if (hres.intValue() == EnumWbemClassObject.WBEM_S_TIMEDOUT) {
682                 throw new TimeoutException("No results after " + timeout + " ms.");
683             }
684             // Requested 1; if 0 objects returned, we're done
685             if (COMUtils.FAILED(hres) || 0L == uReturn.getValue()
686                     || hres.intValue() == EnumWbemClassObject.WBEM_S_NO_MORE_DATA) {
687                 LOG.debug("Returned {} results.", values.getResultCount());
688                 break;
689             }
690 
691             VARIANT.ByReference vtProp = new VARIANT.ByReference();
692 
693             // Get the value of the properties
694             WbemClassObject clsObj = new WbemClassObject(pclsObj.getValue());
695             for (T property : propertyEnum.getEnumConstants()) {
696                 clsObj.Get(new BSTR(property.name()), new NativeLong(0L), vtProp, null, null);
697 
698                 switch (property.getType()) {
699                 case STRING:
700                     values.get(property).add(vtProp.getValue() == null ? "unknown" : vtProp.stringValue());
701                     break;
702                 // uint16 == VT_I4, a 32-bit number
703                 case UINT16:
704                     values.get(property).add(vtProp.getValue() == null ? 0L : vtProp.intValue());
705                     break;
706                 // WMI Uint32s will return as longs
707                 case UINT32:
708                     values.get(property).add(vtProp.getValue() == null ? 0L : vtProp.longValue());
709                     break;
710                 // WMI Longs will return as strings so we have the option of
711                 // calling a string and parsing later, or calling UINT64 and
712                 // letting this method do the parsing
713                 case UINT64:
714                     values.get(property).add(
715                             vtProp.getValue() == null ? 0L : ParseUtil.parseLongOrDefault(vtProp.stringValue(), 0L));
716                     break;
717                 case FLOAT:
718                     values.get(property).add(vtProp.getValue() == null ? 0f : vtProp.floatValue());
719                     break;
720                 case DATETIME:
721                     // Read a string in format 20160513072950.782000-420 and
722                     // parse to a long representing ms since eopch
723                     values.get(property)
724                             .add(vtProp.getValue() == null ? 0L : ParseUtil.cimDateTimeToMillis(vtProp.stringValue()));
725                     break;
726                 case BOOLEAN:
727                     values.get(property).add(vtProp.getValue() == null ? 0L : vtProp.booleanValue());
728                     break;
729                 default:
730                     // Should never get here! If you get this exception you've
731                     // added something to the ValueType enum without adding a
732                     // case for it here. Tsk.
733                     throw new IllegalArgumentException("Unimplemented enum type: " + property.getType().toString());
734                 }
735                 OleAuto.INSTANCE.VariantClear(vtProp);
736             }
737             clsObj.Release();
738 
739             values.incrementResultCount();
740         }
741     }

743     /**
744      * UnInitializes COM library if it was initialized by the {@link #initCOM()}
745      * method. Otherwise, does nothing.
746      */
747     private static void unInitCOM() {
748         for (Entry<String, WmiConnection> entry : connectionCache.entrySet()) {
749             entry.getValue().close();
750         }
751         connectionCache.clear();
752         if (comInitialized) {
753             Ole32.INSTANCE.CoUninitialize();
754             comInitialized = false;
755         }
756     }
757 
758     /**
759      * Closes WMI connections that haven't been used recently, freeing up
760      * resources.
761      */
762     private static void closeStaleConnections() {
763         for (Iterator<Map.Entry<String, WmiConnection>> iter = connectionCache.entrySet().iterator(); iter.hasNext();) {
764             Map.Entry<String, WmiConnection> entry = iter.next();
765             if (entry.getValue().isStale()) {
766                 entry.getValue().close();
767                 iter.remove();
768             }
769         }
770     }

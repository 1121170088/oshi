37  public interface IPHlpAPI extends Library {
38      IPHlpAPI INSTANCE = (IPHlpAPI) Native.loadLibrary("IPHlpAPI", IPHlpAPI.class);

40      int IF_MAX_STRING_SIZE = 256;
41      int IF_MAX_PHYS_ADDRESS_LENGTH = 32;
42      int MAX_INTERFACE_NAME_LEN = 256;
43      int MAXLEN_IFDESCR = 256;
44      int MAXLEN_PHYSADDR = 8;
45  
46      class MIB_IFROW extends Structure {
47          public char[] wszName = new char[MAX_INTERFACE_NAME_LEN];
48          public int dwIndex;
49          public int dwType;
50          public int dwMtu;
51          public int dwSpeed;
52          public int dwPhysAddrLen;
53          public byte[] bPhysAddr = new byte[MAXLEN_PHYSADDR];
54          public int dwAdminStatus;
55          public int dwOperStatus;
56          public int dwLastChange;
57          public int dwInOctets;
58          public int dwInUcastPkts;
59          public int dwInNUcastPkts;
60          public int dwInDiscards;
61          public int dwInErrors;
62          public int dwInUnknownProtos;
63          public int dwOutOctets;
64          public int dwOutUcastPkts;
65          public int dwOutNUcastPkts;
66          public int dwOutDiscards;
67          public int dwOutErrors;
68          public int dwOutQLen;
69          public int dwDescrLen;
70          public byte[] bDescr = new byte[MAXLEN_IFDESCR];
71  
72          @Override
73          protected List<String> getFieldOrder() {
74              return Arrays.asList(new String[] { "wszName", "dwIndex", "dwType", "dwMtu", "dwSpeed", "dwPhysAddrLen",
75                      "bPhysAddr", "dwAdminStatus", "dwOperStatus", "dwLastChange", "dwInOctets", "dwInUcastPkts",
76                      "dwInNUcastPkts", "dwInDiscards", "dwInErrors", "dwInUnknownProtos", "dwOutOctets",
77                      "dwOutUcastPkts", "dwOutNUcastPkts", "dwOutDiscards", "dwOutErrors", "dwOutQLen", "dwDescrLen",
78                      "bDescr" });
79          }
80      }

141     /**
142      * The GetIfEntry function retrieves information for the specified interface
143      * on the local computer
144      * 
145      * @param pIfRow
146      *            A pointer to a MIB_IFROW structure that, on successful return,
147      *            receives information for an interface on the local computer.
148      *            On input, set the dwIndex member of MIB_IFROW to the index of
149      *            the interface for which to retrieve information.
150      * @return If the function succeeds, the return value is NO_ERROR.
151      */
152     int GetIfEntry(MIB_IFROW pIfRow);
153 
154     /**

162      *            the interface for which to retrieve information.

164      */
165     int GetIfEntry2(MIB_IFROW2 pIfRow2);

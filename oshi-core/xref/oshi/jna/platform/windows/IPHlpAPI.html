161      *
162      * @see <A HREF=
163      *      "https://docs.microsoft.com/en-us/windows/desktop/api/iptypes/ns-iptypes-ip_address_string">IP_ADDRESS_STRING</A>

166         // Null terminated string
167         // up to 3 chars (decimal 0-255) and dot
168         // ending with null
169         public byte[] String = new byte[16];
170

172         protected List<String> getFieldOrder() {
173             return Arrays.asList(new String[] { "String" });
174         }
175     }

199     /**
200      * The FIXED_INFO structure contains information that is the same across all
201      * the interfaces on a computer.
202      *
203      * @see <A HREF=
204      *      "https://docs.microsoft.com/en-us/windows/desktop/api/iptypes/ns-iptypes-fixed_info_w2ksp1">FIXED_INFO</A>
205      */
206     class FIXED_INFO extends Structure {
207         public byte[] HostName = new byte[MAX_HOSTNAME_LEN + 4];
208         public byte[] DomainName = new byte[MAX_DOMAIN_NAME_LEN + 4];
209         public IP_ADDR_STRING.ByReference CurrentDnsServer;
210         public IP_ADDR_STRING DnsServerList;
211         public int NodeType;
212         public byte[] ScopeId = new byte[MAX_SCOPE_ID_LEN + 4];
213         public int EnableRouting;
214         public int EnableProxy;
215         public int EnableDns;
216 
217         public FIXED_INFO(Pointer p) {
218             super(p);
219             read();
220         }
221 
222         public FIXED_INFO() {
223             super();
224         }
225 
226         @Override
227         protected List<String> getFieldOrder() {
228             return Arrays.asList(new String[] { "HostName", "DomainName", "CurrentDnsServer", "DnsServerList",
229                     "NodeType", "ScopeId", "EnableRouting", "EnableProxy", "EnableDns" });
230         }
231     }

233     /**
234      * The GetIfEntry function retrieves information for the specified interface
235      * on the local computer.
236      *
237      * The dwIndex member in the MIB_IFROW structure pointed to by the pIfRow
238      * parameter must be initialized to a valid network interface index
239      * retrieved by a previous call to the GetIfTable, GetIfTable2, or
240      * GetIfTable2Ex function. The GetIfEntry function will fail if the dwIndex
241      * member of the MIB_IFROW pointed to by the pIfRow parameter does not match
242      * an existing interface index on the local computer.
243      *
244      * @param pIfRow
245      *            A pointer to a MIB_IFROW structure that, on successful return,
246      *            receives information for an interface on the local computer.
247      *            On input, set the dwIndex member of MIB_IFROW to the index of
248      *            the interface for which to retrieve information.
249      * @return If the function succeeds, the return value is NO_ERROR.
250      */
251     int GetIfEntry(MIB_IFROW pIfRow);
252 
253     /**
254      * The GetIfEntry2 function retrieves information for the specified
255      * interface on the local computer.
256      *
257      * On input, at least one of the following members in the MIB_IF_ROW2
258      * structure passed in the Row parameter must be initialized: InterfaceLuid
259      * or InterfaceIndex. The fields are used in the order listed above. So if
260      * the InterfaceLuid is specified, then this member is used to determine the
261      * interface. If no value was set for the InterfaceLuid member (the value of
262      * this member was set to zero), then the InterfaceIndex member is next used
263      * to determine the interface. On output, the remaining fields of the
264      * MIB_IF_ROW2 structure pointed to by the Row parameter are filled in.
265      *
266      * @param pIfRow2
267      *            A pointer to a MIB_IF_ROW2 structure that, on successful
268      *            return, receives information for an interface on the local
269      *            computer. On input, the InterfaceLuid or the InterfaceIndex
270      *            member of the MIB_IF_ROW2 must be set to the interface for
271      *            which to retrieve information.
272      * @return If the function succeeds, the return value is NO_ERROR.
273      */
274     int GetIfEntry2(MIB_IF_ROW2 pIfRow2);
275 
276     /**
277      * The GetNetworkParams function retrieves network parameters for the local
278      * computer.
279      *
280      * @param pFixedInfo
281      *            A pointer to a buffer that contains a FIXED_INFO structure
282      *            that receives the network parameters for the local computer,
283      *            if the function was successful. This buffer must be allocated
284      *            by the caller prior to calling the GetNetworkParams function.
285      * @param pOutBufLen
286      *            A pointer to a ULONG variable that specifies the size of the
287      *            FIXED_INFO structure. If this size is insufficient to hold the
288      *            information, GetNetworkParams fills in this variable with the
289      *            required size, and returns an error code of
290      *            ERROR_BUFFER_OVERFLOW.
291      * @return If the function succeeds, the return value is ERROR_SUCCESS.
292      */
293     int GetNetworkParams(Pointer pFixedInfo, IntByReference pOutBufLen);

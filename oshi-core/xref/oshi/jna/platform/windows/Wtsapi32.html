83      /**
84       * Retrieves information about the active processes on the specified Remote
85       * Desktop Session Host (RD Session Host) server or Remote Desktop
86       * Virtualization Host (RD Virtualization Host) server.
87       * 
88       * @param hServer
89       *            A handle to an RD Session Host server. Specify a handle opened
90       *            by the WTSOpenServer function, or specify
91       *            WTS_CURRENT_SERVER_HANDLE to indicate the server on which your
92       *            application is running.
93       * @param pLevel
94       *            A pointer to a DWORD variable that, on input, specifies the
95       *            type of information to return. To return an array of
96       *            WTS_PROCESS_INFO structures, specify zero. To return an array
97       *            of WTS_PROCESS_INFO_EX structures, specify one. If you do not
98       *            specify a valid value for this parameter, on output,
99       *            WTSEnumerateProcessesEx sets this parameter to one and returns
100      *            an error. Otherwise, on output, WTSEnumerateProcessesEx does
101      *            not change the value of this parameter.
102      * @param SessionID
103      *            The session for which to enumerate processes. To enumerate
104      *            processes for all sessions on the server, specify
105      *            WTS_ANY_SESSION.
106      * @param ppProcessInfo
107      *            A pointer to a variable that receives a pointer to an array of
108      *            WTS_PROCESS_INFO or WTS_PROCESS_INFO_EX structures. The type
109      *            of structure is determined by the value passed to the pLevel
110      *            parameter. Each structure in the array contains information
111      *            about an active process. When you have finished using the
112      *            array, free it by calling the WTSFreeMemoryEx function. You
113      *            should also set the pointer to NULL.
114      * @param pCount
115      *            A pointer to a variable that receives the number of structures
116      *            returned in the buffer referenced by the ppProcessInfo
117      *            parameter.
118      * @return If the function succeeds, the return value is a nonzero value. If
119      *         the function fails, the return value is zero. To get extended
120      *         error information, call the GetLastError function.
121      */
122     boolean WTSEnumerateProcessesEx(HANDLE hServer, IntByReference pLevel, DWORD SessionID,
123             PointerByReference ppProcessInfo, IntByReference pCount);
124 
125     /**
126      * Frees memory that contains WTS_PROCESS_INFO_EX or WTS_SESSION_INFO_1
127      * structures allocated by a Remote Desktop Services function.
128      * 
129      * @param WTSTypeClass
130      *            A value of the WTS_TYPE_CLASS enumeration type that specifies
131      *            the type of structures contained in the buffer referenced by
132      *            the pMemory parameter.
133      * @param pMemory
134      *            A pointer to the buffer to free.
135      * @param NumberOfEntries
136      *            The number of elements in the buffer referenced by the pMemory
137      *            parameter.
138      * @return If the function succeeds, the return value is a nonzero value. If
139      *         the function fails, the return value is zero. To get extended
140      *         error information, call the GetLastError function.
141      */
142     boolean WTSFreeMemoryEx(int WTSTypeClass, Pointer pMemory, long NumberOfEntries);

63      private static void getGroups() {
64          SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX[] processors = Kernel32Util
65                  .getLogicalProcessorInformationEx(WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationGroup);
66          System.out.format("Info for %d group structure (should only be 1):%n", processors.length);
67          for (int i = 0; i < processors.length; i++) {
68              System.out.format("Relationship should be %d, it is %d%n", LOGICAL_PROCESSOR_RELATIONSHIP.RelationGroup,
69                      processors[i].relationship);
70              GROUP_RELATIONSHIP group = (GROUP_RELATIONSHIP) processors[i];
71              System.out.format("Details for %d group(s):%n", group.activeGroupCount);
72              for (int j = 0; j < group.activeGroupCount; j++) {
73                  System.out.format("Group %d had %d active of %d max processors with bitmask %s.%n", j,
74                          group.groupInfo[j].activeProcessorCount, group.groupInfo[j].maximumProcessorCount,
75                          Long.toBinaryString(group.groupInfo[j].activeProcessorMask.longValue()));
76              }
77          }
78      }

145     private static void getItAll() {
146         System.out.println("Grabbing all info using Kernel32Util...");
147         SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX[] procInfo = Kernel32Util
148                 .getLogicalProcessorInformationEx(WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationAll);
149         List<PROCESSOR_GROUP_INFO> groups = new ArrayList<>();
150         List<GROUP_AFFINITY[]> packages = new ArrayList<>();
151         List<NUMA_NODE_RELATIONSHIP> numaNodes = new ArrayList<>();
152         List<CACHE_RELATIONSHIP> caches = new ArrayList<>();
153         List<GROUP_AFFINITY> cores = new ArrayList<>();
154 
155         for (int i = 0; i < procInfo.length; i++) {
156             switch (procInfo[i].relationship) {
157             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationGroup:
158                 groups = Arrays.asList(((GROUP_RELATIONSHIP) procInfo[i]).groupInfo);
159                 break;
160             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage:
161                 packages.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask);
162                 break;
163             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationNumaNode:
164                 numaNodes.add((NUMA_NODE_RELATIONSHIP) procInfo[i]);
165                 break;
166             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationCache:
167                 caches.add((CACHE_RELATIONSHIP) procInfo[i]);
168                 break;
169             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore:
170                 cores.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask[0]);
171                 break;
172             default:
173                 System.out.println("You should never see this message.");
174                 break;
175             }
176         }
177         System.out.format("Retrieved %d structures: %d groups, %d packages, %d numa nodes, %d caches, %d cores%n",
178                 procInfo.length, groups.size(), packages.size(), numaNodes.size(), caches.size(), cores.size());
179         System.out.println("Topology:");
180         System.out.format("%6s %7s %11s %5s %9s%n", "Group", "package", "NUMA node", "core", "logical processor");
181         for (int g = 0; g < groups.size(); g++) {
182             for (int lp = 0; lp < 64; lp++) {
183                 int pkg = getMatchingPackage(packages, g, lp);
184                 if (pkg < 0) {
185                     continue;
186                 }
187                 System.out.format("%4d %8d %8d %8d %8d%n", g, pkg,
188                         getMatchingNumaNode(numaNodes, g, lp), getMatchingCore(cores, g, lp), lp);
189             }
190         }
191     }

194     private static int getMatchingPackage(List<GROUP_AFFINITY[]> packages, int g, int lp) {
195         for (int i = 0; i < packages.size(); i++) {
196             for (int j = 0; j < packages.get(i).length; j++) {
197                 if ((packages.get(i)[j].mask.longValue() & (1L << lp)) > 0 && packages.get(i)[j].group == g) {
198                     return i;
199                 }
200             }
201         }
202         return -1;
203     }
204 
205     private static int getMatchingNumaNode(List<NUMA_NODE_RELATIONSHIP> numaNodes, int g, int lp) {
206         for (int j = 0; j < numaNodes.size(); j++) {
207             if ((numaNodes.get(j).groupMask.mask.longValue() & (1L << lp)) > 0
208                     && numaNodes.get(j).groupMask.group == g) {
209                 return numaNodes.get(j).nodeNumber;
210             }
211         }
212         return -1;
213     }
214 
215     private static int getMatchingCore(List<GROUP_AFFINITY> cores, int g, int lp) {
216         for (int j = 0; j < cores.size(); j++) {
217             if ((cores.get(j).mask.longValue() & (1L << lp)) > 0 && cores.get(j).group == g) {
218                 return j;
219             }
220         }
221         return -1;
222     }

37  /**
38   * The perfstat API uses the perfstat kernel extension to extract various AIX®
39   * performance metrics.
40   *
41   * System component information is also retrieved from the Object Data Manager
42   * (ODM) and returned with the performance metrics.
43   *
44   * The perfstat API is thread–safe, and does not require root authority.
45   */
46  public interface Perfstat extends Library {
47  
48      Perfstat INSTANCE = getInstance();
49  
50      static Perfstat getInstance() {
51          int RTLD_MEMBER = 0x40000; // allows "lib.a(obj.o)" syntax
52          int RTLD_GLOBAL = 0x10000;
53          int RTLD_LAZY = 0x4;
54          Map<String, Object> options = new HashMap<>();
55          options.put(Library.OPTION_OPEN_FLAGS, RTLD_MEMBER | RTLD_GLOBAL | RTLD_LAZY);
56          options = Collections.unmodifiableMap(options);
57          try {
58              return Native.load("/usr/lib/libperfstat.a(shr_64.o)", Perfstat.class, options);
59          } catch (UnsatisfiedLinkError e) {
60              // failed 64 bit, try 32 bit
61          }
62          return Native.load("/usr/lib/libperfstat.a(shr.o)", Perfstat.class, options);
63      }

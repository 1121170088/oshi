41  /**
42   * Core Foundation is a framework that provides fundamental software services
43   * useful to application services, application environments, and to applications
44   * themselves. Core Foundation also provides abstractions for common data types.
45   * <p>
46   * Core Foundation functions have names that indicate when you own a returned
47   * object: Object-creation functions have “Create” embedded in the name, and
48   * Object-duplication functions that have “Copy” embedded in the name. If you
49   * own an object, it is your responsibility to relinquish ownership (using
50   * {@link #CFRelease}) when you have finished with it.
51   * <p>
52   * If you receive an object from any Core Foundation function other than a
53   * creation or copy function—such as a Get function—you do not own it and cannot
54   * be certain of the object’s life span. If you want to ensure that such an
55   * object is not disposed of while you are using it, you must claim ownership
56   * (with the {@link #CFRetain} function).
57   */
58  public interface CoreFoundation extends Library {

480         /**
481          * Convert a reference to a Core Foundations String into its
482          * {@link java.lang.String}
483          *
484          * @return The corresponding {@link java.lang.String}, or null if the conversion
485          *         failed.
486          */
487         public String stringValue() {
488             CFIndex length = INSTANCE.CFStringGetLength(this);
489             CFIndex maxSize = INSTANCE.CFStringGetMaximumSizeForEncoding(length, kCFStringEncodingUTF8);
490             if (maxSize.intValue() == kCFNotFound) {
491                 return null;
492             }
493             Memory buf = new Memory(maxSize.longValue());
494             if (0 != INSTANCE.CFStringGetCString(this, buf, maxSize, kCFStringEncodingUTF8)) {
495                 return buf.getString(0, "UTF8");
496             }
497             return null;
498         }

517     /**
518      * A type for unique, constant integer values that identify particular Core
519      * Foundation opaque types.
520      * <p>
521      * Because the value for a type ID can change from release to release, your code
522      * should not rely on stored or hard-coded type IDs nor should it hard-code any
523      * observed properties of a type ID (such as, for example, it being a small
524      * integer).
525      */
526     class CFTypeID extends NativeLong {
527         private static final long serialVersionUID = 1L;
528 
529         public CFTypeID() {
530             super();
531         }
532 
533         public CFTypeID(long value) {
534             super(value);
535         }
536 
537         @Override
538         public String toString() {
539             if (this.equals(ARRAY_TYPE_ID)) {
540                 return "CFArray";
541             } else if (this.equals(BOOLEAN_TYPE_ID)) {
542                 return "CFBoolean";
543             } else if (this.equals(DATA_TYPE_ID)) {
544                 return "CFData";
545             } else if (this.equals(DATE_TYPE_ID)) {
546                 return "CFDate";
547             } else if (this.equals(DICTIONARY_TYPE_ID)) {
548                 return "CFDictionary";
549             } else if (this.equals(NUMBER_TYPE_ID)) {
550                 return "CFNumber";
551             } else if (this.equals(STRING_TYPE_ID)) {
552                 return "CFString";
553             } else {
554                 return super.toString();
555             }
556         }
557     }

615      * @param values
616      *            A C array of the pointer-sized values to be in the new array. The
617      *            values in the new array are ordered in the same order in which

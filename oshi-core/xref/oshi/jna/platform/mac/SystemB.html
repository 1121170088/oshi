506     /**
507      * The getfsstat() function returns information about all mounted file
508      * systems. The buf argument is a pointer to an array of statfs structures.
509      *
510      * Fields that are undefined for a particular file system are set to -1. The
511      * buffer is filled with an array of statfs structures, one for each mounted
512      * file system up to the size specified by bufsize.
513      *
514      * @param buf
515      *            Array of statfs structures that will be filled with results.
516      *            If buf is given as NULL, getfsstat() returns just the number
517      *            of mounted file systems.
518      * @param bufsize
519      *            Size of the buffer to fill
520      * @param flags
521      *            If flags is set to MNT_NOWAIT, getfsstat() will directly
522      *            return the information retained in the kernel to avoid delays
523      *            caused by waiting for updated information from a file system
524      *            that is perhaps temporarily unable to respond. Some of the
525      *            information returned may be out of date, however; if flags is
526      *            set to MNT_WAIT or MNT_DWAIT instead, getfsstat() will request
527      *            updated information from each mounted filesystem before
528      *            returning.
529      * @return Upon successful completion, the number of statfs structures is
530      *         returned. Otherwise, -1 is returned and the global variable errno
531      *         is set to indicate the error.
532      */
533     int getfsstat64(Statfs[] buf, int bufsize, int flags);

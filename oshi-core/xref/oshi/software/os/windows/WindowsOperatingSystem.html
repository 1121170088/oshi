144     @Override
145     public FamilyVersionInfo queryFamilyVersionInfo() {
146         WmiQuery<OSVersionProperty> osVersionQuery = new WmiQuery<>("Win32_OperatingSystem", OSVersionProperty.class);
147         WmiResult<OSVersionProperty> versionInfo = WmiQueryHandler.createInstance().queryWMI(osVersionQuery);
148         if (versionInfo.getResultCount() < 1) {
149             return new FamilyVersionInfo("Windows", new OSVersionInfo(System.getProperty("os.version"), null, null));
150         }
151         // Guaranteed that versionInfo is not null and lists non-empty
152         // before calling the parse*() methods
153         int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SuiteMask, 0);
154         String buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BuildNumber, 0);
155         String version = parseVersion(versionInfo, suiteMask, buildNumber);
156         String codeName = parseCodeName(suiteMask);
157         return new FamilyVersionInfo("Windows", new OSVersionInfo(version, codeName, buildNumber));
158     }

160     private static String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask, String buildNumber) {
161 
162         // Initialize a default, sane value
163         String version = System.getProperty("os.version");
164 
165         // Version is major.minor.build. Parse the version string for
166         // major/minor and get the build number separately
167         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.Version, 0).split("\\D");
168         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
169         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
170 
171         // see
172         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
173         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.ProductType,
174                 0) == WinNT.VER_NT_WORKSTATION;
175         switch (major) {
176         case 10:
177             if (minor == 0) {
178                 if (ntWorkstation) {
179                     version = "10";
180                 } else {
181                     // Build numbers greater than 17762 is Server 2019 for OS
182                     // Version 10.0
183                     version = (ParseUtil.parseLongOrDefault(buildNumber, 0L) > 17762) ? "Server 2019" : "Server 2016";
184                 }
185             }
186             break;
187         case 6:
188             if (minor == 3) {
189                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
190             } else if (minor == 2) {
191                 version = ntWorkstation ? "8" : "Server 2012";
192             } else if (minor == 1) {
193                 version = ntWorkstation ? "7" : "Server 2008 R2";
194             } else if (minor == 0) {
195                 version = ntWorkstation ? "Vista" : "Server 2008";
196             }
197             break;
198         case 5:
199             if (minor == 2) {
200                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
201                     version = "Home Server";
202                 } else if (ntWorkstation) {
203                     version = "XP"; // 64 bits
204                 } else {
205                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
206                             : "Server 2003 R2";
207                 }
208             } else if (minor == 1) {
209                 version = "XP"; // 32 bits
210             } else if (minor == 0) {
211                 version = "2000";
212             }
213             break;
214         default:
215             break;
216         }
217 
218         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVersion, 0);
219         if (!sp.isEmpty() && !"unknown".equals(sp)) {
220             version = version + " " + sp.replace("Service Pack ", "SP");
221         }
222 
223         return version;
224     }

226     /**
227      * Gets suites available on the system and return as a codename
228      *
229      * @param suiteMask
230      *
231      * @return Suites
232      */
233     private static String parseCodeName(int suiteMask) {
234         List<String> suites = new ArrayList<>();
235         if ((suiteMask & 0x00000002) != 0) {
236             suites.add("Enterprise");
237         }
238         if ((suiteMask & 0x00000004) != 0) {
239             suites.add("BackOffice");
240         }
241         if ((suiteMask & 0x00000008) != 0) {
242             suites.add("Communication Server");
243         }
244         if ((suiteMask & 0x00000080) != 0) {
245             suites.add("Datacenter");
246         }
247         if ((suiteMask & 0x00000200) != 0) {
248             suites.add("Home");
249         }
250         if ((suiteMask & 0x00000400) != 0) {
251             suites.add("Web Server");
252         }
253         if ((suiteMask & 0x00002000) != 0) {
254             suites.add("Storage Server");
255         }
256         if ((suiteMask & 0x00004000) != 0) {
257             suites.add("Compute Cluster");
258         }
259         // 0x8000, Home Server, is included in main version name
260         return String.join(",", suites);
261     }

263     @Override
264     protected int queryBitness(int jvmBitness) {
265         WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
266         if (jvmBitness < 64 && System.getenv("ProgramFiles(x86)") != null && IS_VISTA_OR_GREATER) {
267             WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<>("Win32_Processor", BitnessProperty.class);
268             WmiResult<BitnessProperty> bitnessMap = wmiQueryHandler.queryWMI(bitnessQuery);
269             if (bitnessMap.getResultCount() > 0) {
270                 return WmiUtil.getUint16(bitnessMap, BitnessProperty.AddressWidth, 0);
271             }
272         }
273         return jvmBitness;
274     }

303     @Override
304     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
305         Set<Integer> childPids = new HashSet<>();
306         // Get processes from ToolHelp API for parent PID
307         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
308         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
309         try {
310             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
311                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
312                     childPids.add(processEntry.th32ProcessID.intValue());
313                 }
314             }
315         } finally {
316             Kernel32.INSTANCE.CloseHandle(snapshot);
317         }
318         List<OSProcess> procList = getProcesses(childPids);
319         List<OSProcess> sorted = processSort(procList, limit, sort);
320         return sorted.toArray(new OSProcess[0]);
321     }

329     /**
330      * Private method to do the heavy lifting for all the getProcess functions.
331      *
332      * @param pids
333      *            A collection of pids to query. If null, the entire process list
334      *            will be queried.
335      * @param slowFields
336      *            Whether to include fields that incur processor latency
337      * @return A corresponding list of processes
338      */
339     private List<OSProcess> processMapToList(Collection<Integer> pids, boolean slowFields) {
340         WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
341         // Get data from the registry if possible, otherwise performance counters with
342         // WMI backup
343         Map<Integer, OSProcess> processMap = (HKEY_PERFORMANCE_DATA != null)
344                 ? HKEY_PERFORMANCE_DATA.buildProcessMapFromRegistry(this, pids)
345                 : buildProcessMapFromPerfCounters(pids);
346 
347         // define here to avoid object repeated creation overhead later
348         List<String> groupList = new ArrayList<>();
349         List<String> groupIDList = new ArrayList<>();
350         int myPid = getProcessId();
351 
352         // Structure we'll fill from native memory pointer for Vista+
353         Pointer pProcessInfo = null;
354         WTS_PROCESS_INFO_EX[] processInfo = null;
355         IntByReference pCount = new IntByReference(0);
356 
357         // WMI result we'll use for pre-Vista
358         WmiResult<ProcessXPProperty> processWmiResult = null;
359 
360         // Get processes from WTS (post-XP)
361         if (IS_WINDOWS7_OR_GREATER) {
362             final PointerByReference ppProcessInfo = new PointerByReference();
363             if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
364                     new IntByReference(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1), Wtsapi32.WTS_ANY_SESSION, ppProcessInfo,
365                     pCount)) {
366                 LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
367                 return new ArrayList<>(0);
368             }
369             // extract the pointed-to pointer and create array
370             pProcessInfo = ppProcessInfo.getValue();
371             final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
372             processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
373         } else {
374             // Pre-Vista we can't use WTSEnumerateProcessesEx so we'll grab the
375             // same info from WMI and fake the array
376             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
377             if (pids != null) {
378                 boolean first = true;
379                 for (Integer pid : pids) {
380                     if (first) {
381                         sb.append(" WHERE ProcessID=");
382                         first = false;
383                     } else {
384                         sb.append(" OR ProcessID=");
385                     }
386                     sb.append(pid);
387                 }
388             }
389             WmiQuery<ProcessXPProperty> processQueryXP = new WmiQuery<>(sb.toString(), ProcessXPProperty.class);
390             processWmiResult = wmiQueryHandler.queryWMI(processQueryXP);
391         }
392 
393         // Store a subset of processes in a list to later return.
394         List<OSProcess> processList = new ArrayList<>();
395 
396         int procCount = IS_WINDOWS7_OR_GREATER ? processInfo.length : processWmiResult.getResultCount();
397         for (int i = 0; i < procCount; i++) {
398             int pid = IS_WINDOWS7_OR_GREATER ? processInfo[i].ProcessId
399                     : WmiUtil.getUint32(processWmiResult, ProcessXPProperty.ProcessId, i);
400             OSProcess proc = null;
401             // If the cache is empty, there was a problem with
402             // filling the cache using performance information.
403             if (processMap.isEmpty()) {
404                 if (pids != null && !pids.contains(pid)) {
405                     continue;
406                 }
407                 proc = new OSProcess(this);
408                 proc.setProcessID(pid);
409                 proc.setName(IS_WINDOWS7_OR_GREATER ? processInfo[i].pProcessName
410                         : WmiUtil.getString(processWmiResult, ProcessXPProperty.Name, i));
411             } else {
412                 proc = processMap.get(pid);
413                 if (proc == null || pids != null && !pids.contains(pid)) {
414                     continue;
415                 }
416             }
417             // For my own process, set CWD
418             if (pid == myPid) {
419                 String cwd = new File(".").getAbsolutePath();
420                 // trim off trailing "."
421                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
422             }
423 
424             if (IS_WINDOWS7_OR_GREATER) {
425                 WTS_PROCESS_INFO_EX procInfo = processInfo[i];
426                 proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
427                 proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
428                 proc.setThreadCount(procInfo.NumberOfThreads);
429                 proc.setVirtualSize(procInfo.PagefileUsage & 0xffff_ffffL);
430                 proc.setOpenFiles(procInfo.HandleCount);
431             } else {
432                 proc.setKernelTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KernelModeTime, i) / 10000L);
433                 proc.setUserTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.UserModeTime, i) / 10000L);
434                 proc.setThreadCount(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.ThreadCount, i));
435                 // WMI Pagefile usage is in KB
436                 proc.setVirtualSize(1024
437                         * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PageFileUsage, i) & 0xffff_ffffL));
438                 proc.setOpenFiles(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HandleCount, i));
439             }
440 
441             // Get a handle to the process for various extended info. Only gets
442             // current user unless running as administrator
443             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
444                     proc.getProcessID());
445             if (pHandle != null) {
446                 proc.setBitness(this.getBitness());
447                 // Only test for 32-bit process on 64-bit windows
448                 if (IS_VISTA_OR_GREATER && this.getBitness() == 64) {
449                     IntByReference wow64 = new IntByReference(0);
450                     if (Kernel32.INSTANCE.IsWow64Process(pHandle, wow64)) {
451                         proc.setBitness(wow64.getValue() > 0 ? 32 : 64);
452                     }
453                 }
454                 // Full path
455                 final HANDLEByReference phToken = new HANDLEByReference();
456                 try {// EXECUTABLEPATH
457                     proc.setPath(IS_WINDOWS7_OR_GREATER ? Kernel32Util.QueryFullProcessImageName(pHandle, 0)
458                             : WmiUtil.getString(processWmiResult, ProcessXPProperty.ExecutablePath, i));
459                     if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY,
460                             phToken)) {
461                         Account account = Advapi32Util.getTokenAccount(phToken.getValue());
462                         proc.setUser(account.name);
463                         proc.setUserID(account.sidString);
464                         // Fetching group information incurs ~10ms per process.
465                         if (slowFields) {
466                             Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
467                             // get groups
468                             groupList.clear();
469                             groupIDList.clear();
470                             for (Account a : accounts) {
471                                 groupList.add(a.name);
472                                 groupIDList.add(a.sidString);
473                             }
474                             proc.setGroup(String.join(",", groupList));
475                             proc.setGroupID(String.join(",", groupIDList));
476                         }
477                     } else {
478                         int error = Kernel32.INSTANCE.GetLastError();
479                         // Access denied errors are common. Fail silently.
480                         if (error != WinError.ERROR_ACCESS_DENIED) {
481                             LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
482                                     Kernel32.INSTANCE.GetLastError());
483                         }
484                     }
485                 } catch (Win32Exception e) {
486                     handleWin32ExceptionOnGetProcessInfo(proc, e);
487                 } finally {
488                     final HANDLE token = phToken.getValue();
489                     if (token != null) {
490                         Kernel32.INSTANCE.CloseHandle(token);
491                     }
492                 }
493                 Kernel32.INSTANCE.CloseHandle(pHandle);
494             }
495 
496             // There is no easy way to get ExecutuionState for a process.
497             // The WMI value is null. It's possible to get thread Execution
498             // State and possibly roll up.
499             proc.setState(OSProcess.State.RUNNING);
500 
501             // Initialize default
502             proc.setCommandLine("");
503 
504             processList.add(proc);
505         }
506         // Clean up memory allocated in C (only Vista+ but null pointer
507         // effectively tests)
508         if (pProcessInfo != null && !Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1, pProcessInfo,
509                 pCount.getValue())) {
510             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
511             return new ArrayList<>(0);
512         }
513 
514         // Command Line only accessible via WMI.
515         if (slowFields) {
516             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
517             if (pids != null) {
518                 Set<Integer> pidsToQuery = new HashSet<>();
519                 for (OSProcess process : processList) {
520                     pidsToQuery.add(process.getProcessID());
521                 }
522                 boolean first = true;
523                 for (Integer pid : pidsToQuery) {
524                     if (first) {
525                         sb.append(" WHERE ProcessID=");
526                         first = false;
527                     } else {
528                         sb.append(" OR ProcessID=");
529                     }
530                     sb.append(pid);
531                 }
532             }
533             WmiQuery<ProcessProperty> processQuery = new WmiQuery<>(sb.toString(), ProcessProperty.class);
534             WmiResult<ProcessProperty> commandLineProcs = wmiQueryHandler.queryWMI(processQuery);
535 
536             for (int p = 0; p < commandLineProcs.getResultCount(); p++) {
537                 int pid = WmiUtil.getUint32(commandLineProcs, ProcessProperty.ProcessId, p);
538                 if (processMap.containsKey(pid)) {
539                     OSProcess proc = processMap.get(pid);
540                     proc.setCommandLine(WmiUtil.getString(commandLineProcs, ProcessProperty.CommandLine, p));
541                 }
542             }
543         }
544         return processList;
545     }

552     private Map<Integer, OSProcess> buildProcessMapFromPerfCounters(Collection<Integer> pids) {
553         Map<Integer, OSProcess> processMap = new HashMap<>();
554         Map<ProcessPerformanceProperty, List<Long>> valueMap = this.processPerformancePerfCounters
555                 .queryValuesWildcard();
556         long now = System.currentTimeMillis(); // 1970 epoch
557         List<String> instances = this.processPerformancePerfCounters.getInstancesFromLastQuery();
558         List<Long> pidList = valueMap.get(ProcessPerformanceProperty.ProcessId);
559         List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.ParentProcessId);
560         List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.Priority);
561         List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.ReadTransferCount);
562         List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.WriteTransferCount);
563         List<Long> workingSetSizeList = valueMap.get(ProcessPerformanceProperty.PrivatePageCount);
564         List<Long> creationTimeList = valueMap.get(ProcessPerformanceProperty.CreationDate);
565 
566         for (int inst = 0; inst < instances.size(); inst++) {
567             int pid = pidList.get(inst).intValue();
568             if (pids == null || pids.contains(pid)) {
569                 OSProcessml#OSProcess">OSProcess proc = new OSProcess(this);
570                 processMap.put(pid, proc);
571 
572                 proc.setProcessID(pid);
573                 proc.setName(instances.get(inst));
574                 proc.setParentProcessID(ppidList.get(inst).intValue());
575                 proc.setPriority(priorityList.get(inst).intValue());
576                 // if creation time value is less than current millis, it's in 1970 epoch,
577                 // otherwise it's 1601 epoch and we must convert
578                 long ctime = creationTimeList.get(inst);
579                 if (ctime > now) {
580                     ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
581                 }
582                 proc.setUpTime(now - ctime);
583                 proc.setStartTime(ctime);
584                 proc.setBytesRead(ioReadList.get(inst));
585                 proc.setBytesWritten(ioWriteList.get(inst));
586                 proc.setResidentSetSize(workingSetSizeList.get(inst));
587             }
588         }
589         return processMap;
590     }

651     private static long querySystemBootTime() {
652         String eventLog = systemLog.get();
653         if (eventLog != null) {
654             try {
655                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
656                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
657                 // Startup" is enabled we may not see event 12, so also check for most recent ID
658                 // 6005 (Event log startup) as a reasonably close backup.
659                 long event6005Time = 0L;
660                 while (iter.hasNext()) {
661                     EventLogRecord record = iter.next();
662                     if (record.getStatusCode() == 12) {
663                         // Event 12 is system boot. We want this value unless we find two 6005 events
664                         // first (may occur with Fast Boot)
665                         return record.getRecord().TimeGenerated.longValue();
666                     } else if (record.getStatusCode() == 6005) {
667                         // If we already found one, this means we've found a second one without finding
668                         // an event 12. Return the latest one.
669                         if (event6005Time > 0) {
670                             return event6005Time;
671                         }
672                         // First 6005; tentatively assign
673                         event6005Time = record.getRecord().TimeGenerated.longValue();
674                     }
675                 }
676                 // Only one 6005 found, return
677                 if (event6005Time > 0) {
678                     return event6005Time;
679                 }
680             } catch (Win32Exception e) {
681                 LOG.warn("Can't open event log \"{}\".", eventLog);
682             }
683         }
684         // If we get this far, event log reading has failed, either from no log or no
685         // startup times. Subtract up time from current time as a reasonable proxy.
686         return System.currentTimeMillis() / 1000L - querySystemUptime();
687     }

698     private static void enableDebugPrivilege() {
699         HANDLEByReference hToken = new HANDLEByReference();
700         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
701                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
702         if (!success) {
703             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
704             return;
705         }
706         WinNT.LUID luid = new WinNT.LUID();
707         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
708         if (!success) {
709             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
710             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
711             return;
712         }
713         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
714         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
715         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
716         if (!success) {
717             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
718         }
719         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
720     }

811     private static class HkeyPerformanceData {
812         /*
813          * Grow as needed but persist
814          */
815         private int perfDataBufferSize = 8192;
816         /*
817          * Process counter index in integer and string form
818          */
819         private int processIndex; // 6
820         private String processIndexStr; // "6"
821         /*
822          * Registry counter data byte offsets
823          */
824         private int priorityBaseOffset; // 92
825         private int elapsedTimeOffset; // 96
826         private int idProcessOffset; // 104
827         private int creatingProcessIdOffset; // 108
828         private int ioReadOffset; // 160
829         private int ioWriteOffset; // 168
830         private int workingSetPrivateOffset; // 192

832         private HkeyPerformanceData() throws InstantiationException {
833             // Get the title indices
834             int priorityBaseIndex = 0;
835             int elapsedTimeIndex = 0;
836             int idProcessIndex = 0;
837             int creatingProcessIdIndex = 0;
838             int ioReadIndex = 0;
839             int ioWriteIndex = 0;
840             int workingSetPrivateIndex = 0;
841 
842             try {
843                 final String ENGLISH_COUNTER_KEY = "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Perflib\\009";
844                 final String ENGLISH_COUNTER_VALUE = "Counter";
845 
846                 // Look up list of english names and ids
847                 String[] counters = Advapi32Util.registryGetStringArray(WinReg.HKEY_LOCAL_MACHINE, ENGLISH_COUNTER_KEY,
848                         ENGLISH_COUNTER_VALUE);
849                 // Array contains alternating index/name pairs
850                 // {"1", "1847", "2", "System", "4", "Memory", ... }
851                 // Get position of name in the array (odd index), return parsed value of
852                 // previous even index
853                 for (int i = 1; i < counters.length; i += 2) {
854                     if (counters[i].equals("Process")) {
855                         this.processIndex = Integer.parseInt(counters[i - 1]);
856                     } else if (counters[i].equals("Priority Base")) {
857                         priorityBaseIndex = Integer.parseInt(counters[i - 1]);
858                     } else if (counters[i].equals("Elapsed Time")) {
859                         elapsedTimeIndex = Integer.parseInt(counters[i - 1]);
860                     } else if (counters[i].equals("ID Process")) {
861                         idProcessIndex = Integer.parseInt(counters[i - 1]);
862                     } else if (counters[i].equals("Creating Process ID")) {
863                         creatingProcessIdIndex = Integer.parseInt(counters[i - 1]);
864                     } else if (counters[i].equals("IO Read Bytes/sec")) {
865                         ioReadIndex = Integer.parseInt(counters[i - 1]);
866                     } else if (counters[i].equals("IO Write Bytes/sec")) {
867                         ioWriteIndex = Integer.parseInt(counters[i - 1]);
868                     } else if (counters[i].equals("Working Set - Private")) {
869                         workingSetPrivateIndex = Integer.parseInt(counters[i - 1]);
870                     }
871                 }
872             } catch (NumberFormatException e) {
873                 // Unexpected but handle anyway
874                 throw new InstantiationException("Failed to parse counter index/name array.");
875             } catch (Win32Exception e) {
876                 throw new InstantiationException("Unable to locate English counter names in registry Perflib 009.");
877             }
878             // If any of the indices are 0, we failed
879             if (this.processIndex == 0 || priorityBaseIndex == 0 || elapsedTimeIndex == 0 || idProcessIndex == 0
880                     || creatingProcessIdIndex == 0 || ioReadIndex == 0 || ioWriteIndex == 0
881                     || workingSetPrivateIndex == 0) {
882                 throw new InstantiationException("Failed to parse counter index/name array.");
883             }
884             this.processIndexStr = Integer.toString(this.processIndex);
885 
886             // now load the Process registry to match up the offsets
887             // Sequentially increase the buffer until everything fits.
888             // Save this buffer size for later use
889             IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
890             Pointer pPerfData = new Memory(this.perfDataBufferSize);
891             int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
892                     pPerfData, lpcbData);
893             if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
894                 throw new InstantiationException("Error " + ret + " reading HKEY_PERFORMANCE_DATA from the registry.");
895             }
896             while (ret == WinError.ERROR_MORE_DATA) {
897                 this.perfDataBufferSize += 4096;
898                 lpcbData.setValue(this.perfDataBufferSize);
899                 pPerfData = new Memory(this.perfDataBufferSize);
900                 ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
901                         pPerfData, lpcbData);
902             }
903 
904             PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
905 
906             // See format at
907             // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
908             // [ ] Object Type
909             // [ ][ ][ ] Multiple counter definitions
910             // Then multiple:
911             // [ ] Instance Definition
912             // [ ] Instance name
913             // [ ] Counter Block
914             // [ ][ ][ ] Counter data for each definition above
915 
916             long perfObjectOffset = perfData.HeaderLength;
917 
918             // Iterate object types. For Process should only be one here
919             for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
920                 PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
921                 // Identify where counter definitions start
922                 long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
923                 // If this isn't the Process object, ignore
924                 if (perfObject.ObjectNameTitleIndex == this.processIndex) {
925                     for (int counter = 0; counter < perfObject.NumCounters; counter++) {
926                         PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
927                                 pPerfData.share(perfCounterOffset));
928                         if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
929                             this.priorityBaseOffset = perfCounter.CounterOffset;
930                         } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
931                             this.elapsedTimeOffset = perfCounter.CounterOffset;
932                         } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
933                             this.creatingProcessIdOffset = perfCounter.CounterOffset;
934                         } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
935                             this.idProcessOffset = perfCounter.CounterOffset;
936                         } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
937                             this.ioReadOffset = perfCounter.CounterOffset;
938                         } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
939                             this.ioWriteOffset = perfCounter.CounterOffset;
940                         } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
941                             this.workingSetPrivateOffset = perfCounter.CounterOffset;
942                         }
943                         // Increment for next Counter
944                         perfCounterOffset += perfCounter.ByteLength;
945                     }
946                     // We're done, break the loop
947                     break;
948                 }
949                 // Increment for next object (should never need this)
950                 perfObjectOffset += perfObject.TotalByteLength;
951             }
952         }

954         private Map<Integer, OSProcess> buildProcessMapFromRegistry(OperatingSystem os, Collection<Integer> pids) {
955             Map<Integer, OSProcess> processMap = new HashMap<>();
956             // Grab the PERF_DATA_BLOCK from the registry.
957             // Sequentially increase the buffer until everything fits.
958             IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
959             Pointer pPerfData = new Memory(this.perfDataBufferSize);
960             int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
961                     pPerfData, lpcbData);
962             if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
963                 LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
964                 return processMap;
965             }
966             while (ret == WinError.ERROR_MORE_DATA) {
967                 this.perfDataBufferSize += 4096;
968                 lpcbData.setValue(this.perfDataBufferSize);
969                 pPerfData = new Memory(this.perfDataBufferSize);
970                 ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
971                         pPerfData, lpcbData);
972             }
973 
974             PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
975             long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
976             long now = System.currentTimeMillis(); // 1970 epoch
977 
978             // See format at
979             // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
980             // [ ] Object Type
981             // [ ][ ][ ] Multiple counter definitions
982             // Then multiple:
983             // [ ] Instance Definition
984             // [ ] Instance name
985             // [ ] Counter Block
986             // [ ][ ][ ] Counter data for each definition above
987             long perfObjectOffset = perfData.HeaderLength;
988 
989             // Iterate object types. For Process should only be one here
990             for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
991                 PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
992                 // If this isn't the Process object, ignore
993                 if (perfObject.ObjectNameTitleIndex == this.processIndex) {
994                     // Skip over counter definitions
995                     // There will be many of these, this points to the first one
996                     long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
997 
998                     // We need this for every process, initialize outside loop to
999                     // save overhead
1000                     PERF_COUNTER_BLOCK perfCounterBlock = null;
1001                     // Iterate instances.
1002                     // The last instance is _Total so subtract 1 from max
1003                     for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
1004                         PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
1005                                 pPerfData.share(perfInstanceOffset));
1006                         long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
1007 
1008                         int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
1009                         if (pids == null || pids.contains(pid)) {
1010                             OSProcessml#OSProcess">OSProcess proc = new OSProcess(os);
1011                             processMap.put(pid, proc);
1012 
1013                             proc.setProcessID(pid);
1014                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
1015                             long upTime = (perfTime100nSec
1016                                     - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset)) / 10_000L;
1017                             proc.setUpTime(upTime < 1L ? 1L : upTime);
1018                             proc.setStartTime(now - upTime);
1019                             proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
1020                             proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
1021                             proc.setResidentSetSize(
1022                                     pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
1023                             proc.setParentProcessID(
1024                                     pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
1025                             proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
1026                         }
1027 
1028                         // Increment to next instance
1029                         perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
1030                         perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
1031                     }
1032                     // We've found the process object and are done, no need to look at any other
1033                     // objects (shouldn't be any). Break the loop
1034                     break;
1035                 }
1036                 // Increment for next object (should never need this)
1037                 perfObjectOffset += perfObject.TotalByteLength;
1038             }

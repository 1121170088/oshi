32  /**

38  public class FreeBsdOperatingSystem extends AbstractOperatingSystem {
39  
40      private static final long serialVersionUID = 1L;
41  
42      public FreeBsdOperatingSystem() {
43          this.manufacturer = "Unix/BSD";
44          this.family = BsdSysctlUtil.sysctl("kern.ostype", "FreeBSD");
45          this.version = new FreeBsdOSVersionInfoEx();
46      }
47  
48      /**
49       * {@inheritDoc}
50       */
51      @Override
52      public FileSystem getFileSystem() {
53          return new FreeBsdFileSystem();
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public OSProcess[] getProcesses(int limit, ProcessSort sort) {
61          ArrayList<String> procList = ExecutingCommand
62                  .runNative("ps -awwxo state,pid,ppid,nlwp,pri,vsz,rss,etimes,systime,time,command");
63          if (procList.isEmpty() || procList.size() < 2) {
64              return new OSProcess[0];
65          }
66          // remove header row
67          procList.remove(0);
68          // Fill list
69          List<OSProcess> procs = new ArrayList<>();
70          for (String proc : procList) {
71              String[] split = proc.trim().split("\\s+");
72              // Elements should match ps command order. Args will make split
73              // bigger than 11 but we ignore thems
74              if (split.length < 11) {
75                  continue;
76              }
77              String path = split[10];
78              long now = System.currentTimeMillis();
79              procs.add(new FreeBsdProcess(path.substring(path.lastIndexOf('/') + 1), // name
80                      path, // path
81                      split[0].charAt(0), // state, one of DILRSTWZ
82                      ParseUtil.parseIntOrDefault(split[1], 0), // pid
83                      ParseUtil.parseIntOrDefault(split[2], 0), // ppid
84                      ParseUtil.parseIntOrDefault(split[3], 0), // thread count
85                      ParseUtil.parseIntOrDefault(split[4], 0), // priority
86                      ParseUtil.parseLongOrDefault(split[5], 0L), // VSZ in kb
87                      ParseUtil.parseLongOrDefault(split[6], 0L), // RSS in kb
88                      ParseUtil.parseLongOrDefault(split[7], 0L), // elapsed secs
89                      ParseUtil.parseDHMSOrDefault(split[8], 0L), // system ms
90                      ParseUtil.parseDHMSOrDefault(split[9], 0L), // usr+sys ms
91                      now //
92              ));
93          }
94          List<OSProcess> sorted = processSort(procs, limit, sort);
95          return sorted.toArray(new OSProcess[sorted.size()]);
96      }
97  
98      /**
99       * {@inheritDoc}
100      */
101     @Override
102     public OSProcess getProcess(int pid) {
103         ArrayList<String> procList = ExecutingCommand
104                 .runNative("ps -awwxo state,pid,ppid,nlwp,pri,vsz,rss,etimes,systime,time,command -p " + pid);
105         if (procList.isEmpty() || procList.size() < 2) {
106             return null;
107         }
108         // remove header row
109         String[] split = procList.get(1).trim().split("\\s+");
110         // Elements should match ps command order
111         if (split.length < 11) {
112             return null;
113         }
114         String path = split[10];
115         return new FreeBsdProcess(path.substring(path.lastIndexOf('/') + 1), // name
116                 path, // path
117                 split[0].charAt(0), // state, one of DILRSTWZ
118                 pid, // also split[1] but we already have
119                 ParseUtil.parseIntOrDefault(split[2], 0), // ppid
120                 ParseUtil.parseIntOrDefault(split[3], 0), // thread count
121                 ParseUtil.parseIntOrDefault(split[4], 0), // priority
122                 ParseUtil.parseLongOrDefault(split[5], 0L), // VSZ in kb
123                 ParseUtil.parseLongOrDefault(split[6], 0L), // RSS in kb
124                 ParseUtil.parseLongOrDefault(split[7], 0L), // elapsed secs
125                 ParseUtil.parseDHMSOrDefault(split[8], 0L), // system ms
126                 ParseUtil.parseDHMSOrDefault(split[9], 0L), // process ms
127                 System.currentTimeMillis() //
128         );
129     }
130 
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     public int getProcessId() {
136         return Libc.INSTANCE.getpid();
137     }
138 
139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public int getProcessCount() {
144         ArrayList<String> procList = ExecutingCommand.runNative("ps -axo pid");
145         if (!procList.isEmpty()) {
146             // Subtract 1 for header
147             return procList.size() - 1;
148         }
149         return 0;
150     }
151 
152     /**
153      * {@inheritDoc}
154      */
155     @Override
156     public int getThreadCount() {
157         ArrayList<String> threadList = ExecutingCommand.runNative("ps -axo nlwp");
158         int threads = 0;
159         for (String proc : threadList) {
160             threads += ParseUtil.parseIntOrDefault(proc.trim(), 0);
161         }
162         return threads;
163     }

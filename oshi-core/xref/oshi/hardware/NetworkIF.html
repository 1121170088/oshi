79      /**
80       * Sets the network interface and calculates other information derived from it
81       *
82       * @param networkInterface
83       *            The network interface to set
84       */
85      public void setNetworkInterface(NetworkInterface networkInterface) {
86          this.networkInterface = networkInterface;
87          try {
88              // Set MTU
89              this.mtu = networkInterface.getMTU();
90              // Set MAC
91              byte[] hwmac = networkInterface.getHardwareAddress();
92              if (hwmac != null) {
93                  List<String> octets = new ArrayList<>(6);
94                  for (byte b : hwmac) {
95                      octets.add(String.format("%02x", b));
96                  }
97                  this.mac = String.join(":", octets);
98              } else {
99                  this.mac = "Unknown";
100             }
101             // Set IP arrays
102             ArrayList<String> ipv4list = new ArrayList<>();
103             ArrayList<String> ipv6list = new ArrayList<>();
104             for (InetAddress address : Collections.list(networkInterface.getInetAddresses())) {
105                 if (address.getHostAddress().length() > 0) {
106                     if (address.getHostAddress().contains(":")) {
107                         ipv6list.add(address.getHostAddress().split("%")[0]);
108                     } else {
109                         ipv4list.add(address.getHostAddress());
110                     }
111                 }
112             }
113             this.ipv4 = ipv4list.toArray(new String[0]);
114             this.ipv6 = ipv6list.toArray(new String[0]);
115         } catch (SocketException e) {
116             LOG.error("Socket exception: {}", e);
117         }
118     }

404     /**
405      * Updates interface network statistics on this interface. Statistics include
406      * packets and bytes sent and received, and interface speed.
407      */
408     public void updateAttributes() {
409         switch (SystemInfo.getCurrentPlatformEnum()) {
410         case WINDOWS:
411             WindowsNetworks.updateNetworkStats(this);
412             break;
413         case LINUX:
414             LinuxNetworks.updateNetworkStats(this);
415             break;
416         case MACOSX:
417             MacNetworks.updateNetworkStats(this);
418             break;
419         case SOLARIS:
420             SolarisNetworks.updateNetworkStats(this);
421             break;
422         case FREEBSD:
423             FreeBsdNetworks.updateNetworkStats(this);
424             break;
425         default:
426             LOG.error("Unsupported platform. No update performed.");
427             break;
428         }
429     }

431     @Override
432     public String toString() {
433         StringBuilder sb = new StringBuilder();
434         sb.append("Name: ").append(getName()).append(" ").append("(").append(getDisplayName()).append(")").append("\n");
435         sb.append("  MAC Address: ").append(getMacaddr()).append("\n");
436         sb.append("  MTU: ").append(getMTU()).append(", ").append("Speed: ").append(getSpeed()).append("\n");
437         sb.append("  IPv4: ").append(Arrays.toString(getIPv4addr())).append("\n");
438         sb.append("  IPv6: ").append(Arrays.toString(getIPv6addr())).append("\n");
439         sb.append("  Traffic: received ").append(getPacketsRecv()).append(" packets/")
440                 .append(FormatUtil.formatBytes(getBytesRecv())).append(" (" + getInErrors() + " err);");
441         sb.append(" transmitted ").append(getPacketsSent()).append(" packets/")
442                 .append(FormatUtil.formatBytes(getBytesSent())).append(" (" + getOutErrors() + " err)");
443         return sb.toString();
444     }

128     public long getVendorFreq() {
129         if (this.cpuVendorFreq == 0) {
130             Pattern pattern = Pattern.compile("@ (.*)$");
131             Matcher matcher = pattern.matcher(getName());
132 
133             if (matcher.find()) {
134                 String unit = matcher.group(1);
135                 this.cpuVendorFreq = ParseUtil.parseHertz(unit);
136             } else {
137                 this.cpuVendorFreq = -1L;
138             }
139         }
140         return this.cpuVendorFreq;
141     }

154             sb.append(" Model ").append(getModel());
155             sb.append(" Stepping ").append(getStepping());
156             setIdentifier(sb.toString());
157         }
158         return this.cpuIdentifier;

314     /**
315      * Parses identifier string
316      *
317      * @param id
318      *            the id to retrieve
319      * @return the string following id
320      */
321     private String parseIdentifier(String id) {
322         String[] idSplit = ParseUtil.whitespaces.split(getIdentifier());
323         boolean found = false;
324         for (String s : idSplit) {
325             // If id string found, return next value
326             if (found) {
327                 return s;
328             }
329             found = s.equals(id);
330         }
331         // If id string not found, return empty string
332         return "";
333     }

335     /** {@inheritDoc} */
336     @Override
337     public synchronized double getSystemCpuLoadBetweenTicks(long[] oldTicks) {
338         if (oldTicks.length != TickType.values().length) {
339             throw new IllegalArgumentException(
340                     "Tick array " + oldTicks.length + " should have " + TickType.values().length + " elements");
341         }
342         long[] ticks = getSystemCpuLoadTicks();
343         // Calculate total
344         long total = 0;
345         for (int i = 0; i < ticks.length; i++) {
346             total += ticks[i] - oldTicks[i];
347         }
348         // Calculate idle from difference in idle and IOwait
349         long idle = ticks[TickType.IDLE.getIndex()] + ticks[TickType.IOWAIT.getIndex()]
350                 - oldTicks[TickType.IDLE.getIndex()] - oldTicks[TickType.IOWAIT.getIndex()];
351         LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
352 
353         return total > 0 && idle >= 0 ? (double) (total - idle) / total : 0d;
354     }

356     /** {@inheritDoc} */
357     @Override
358     public double[] getProcessorCpuLoadBetweenTicks(long[][] oldTicks) {
359         if (oldTicks.length != this.logicalProcessorCount || oldTicks[0].length != TickType.values().length) {
360             throw new IllegalArgumentException(
361                     "Tick array " + oldTicks.length + " should have " + this.logicalProcessorCount
362                             + " arrays, each of which has " + TickType.values().length + " elements");
363         }
364         long[][] ticks = getProcessorCpuLoadTicks();
365         double[] load = new double[this.logicalProcessorCount];
366         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
367             long total = 0;
368             for (int i = 0; i < ticks[cpu].length; i++) {
369                 total += ticks[cpu][i] - oldTicks[cpu][i];
370             }
371             // Calculate idle from difference in idle and IOwait
372             long idle = ticks[cpu][TickType.IDLE.getIndex()] + ticks[cpu][TickType.IOWAIT.getIndex()]
373                     - oldTicks[cpu][TickType.IDLE.getIndex()] - oldTicks[cpu][TickType.IOWAIT.getIndex()];
374             LOG.trace("CPU: {}  Total ticks: {}  Idle ticks: {}", cpu, total, idle);
375             // update
376             load[cpu] = total > 0 && idle >= 0 ? (double) (total - idle) / total : 0d;
377         }
378         return load;
379     }

405     /**
406      * Creates a Processor ID by encoding the stepping, model, family, and feature
407      * flags.
408      *
409      * @param stepping
410      *            The CPU stepping
411      * @param model
412      *            The CPU model
413      * @param family
414      *            The CPU family
415      * @param flags
416      *            A space-delimited list of CPU feature flags
417      * @return The Processor ID string
418      */
419     protected String createProcessorID(String stepping, String model, String family, String[] flags) {
420         long processorIdBytes = 0L;
421         long steppingL = ParseUtil.parseLongOrDefault(stepping, 0L);
422         long modelL = ParseUtil.parseLongOrDefault(model, 0L);
423         long familyL = ParseUtil.parseLongOrDefault(family, 0L);
424         // 3:0 – Stepping
425         processorIdBytes |= steppingL & 0xf;
426         // 19:16,7:4 – Model
427         processorIdBytes |= (modelL & 0x0f) << 4;
428         processorIdBytes |= (modelL & 0xf0) << 16;
429         // 27:20,11:8 – Family
430         processorIdBytes |= (familyL & 0x0f) << 8;
431         processorIdBytes |= (familyL & 0xf0) << 20;
432         // 13:12 – Processor Type, assume 0
433         for (String flag : flags) {
434             switch (flag) { // NOSONAR squid:S1479
435             case "fpu":
436                 processorIdBytes |= 1L << 32;
437                 break;
438             case "vme":
439                 processorIdBytes |= 1L << 33;
440                 break;
441             case "de":
442                 processorIdBytes |= 1L << 34;
443                 break;
444             case "pse":
445                 processorIdBytes |= 1L << 35;
446                 break;
447             case "tsc":
448                 processorIdBytes |= 1L << 36;
449                 break;
450             case "msr":
451                 processorIdBytes |= 1L << 37;
452                 break;
453             case "pae":
454                 processorIdBytes |= 1L << 38;
455                 break;
456             case "mce":
457                 processorIdBytes |= 1L << 39;
458                 break;
459             case "cx8":
460                 processorIdBytes |= 1L << 40;
461                 break;
462             case "apic":
463                 processorIdBytes |= 1L << 41;
464                 break;
465             case "sep":
466                 processorIdBytes |= 1L << 43;
467                 break;
468             case "mtrr":
469                 processorIdBytes |= 1L << 44;
470                 break;
471             case "pge":
472                 processorIdBytes |= 1L << 45;
473                 break;
474             case "mca":
475                 processorIdBytes |= 1L << 46;
476                 break;
477             case "cmov":
478                 processorIdBytes |= 1L << 47;
479                 break;
480             case "pat":
481                 processorIdBytes |= 1L << 48;
482                 break;
483             case "pse-36":
484                 processorIdBytes |= 1L << 49;
485                 break;
486             case "psn":
487                 processorIdBytes |= 1L << 50;
488                 break;
489             case "clfsh":
490                 processorIdBytes |= 1L << 51;
491                 break;
492             case "ds":
493                 processorIdBytes |= 1L << 53;
494                 break;
495             case "acpi":
496                 processorIdBytes |= 1L << 54;
497                 break;
498             case "mmx":
499                 processorIdBytes |= 1L << 55;
500                 break;
501             case "fxsr":
502                 processorIdBytes |= 1L << 56;
503                 break;
504             case "sse":
505                 processorIdBytes |= 1L << 57;
506                 break;
507             case "sse2":
508                 processorIdBytes |= 1L << 58;
509                 break;
510             case "ss":
511                 processorIdBytes |= 1L << 59;
512                 break;
513             case "htt":
514                 processorIdBytes |= 1L << 60;
515                 break;
516             case "tm":
517                 processorIdBytes |= 1L << 61;
518                 break;
519             case "ia64":
520                 processorIdBytes |= 1L << 62;
521                 break;
522             case "pbe":
523                 processorIdBytes |= 1L << 63;
524                 break;
525             default:
526                 break;
527             }
528         }
529         return String.format("%016X", processorIdBytes);
530     }

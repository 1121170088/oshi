19  package oshi.hardware;
20  
21  import java.io.Serializable;
22  
23  /**
24   * Memory refers to the state information of a computing system, as it is kept
25   * active in some physical structure. The term "memory" is used for the
26   * information in physical systems which are fast (ie. RAM), as a distinction
27   * from physical systems which are slow to access (ie. data storage). By design,
28   * the term "memory" refers to temporary state devices, whereas the term
29   * "storage" is reserved for permanent data.
30   *
31   * @author dblock[at]dblock[dot]org
32   */
33  public interface GlobalMemory extends Serializable {
34      /**
35       * The amount of actual physical memory, in bytes.
36       *
37       * @return Total number of bytes.
38       */
39      long getTotal();
40  
41      /**
42       * The amount of physical memory currently available, in bytes.
43       *
44       * @return Available number of bytes.
45       */
46      long getAvailable();
47  
48      /**
49       * The current size of the paging/swap file(s), in bytes. If the paging/swap
50       * file can be extended, this is a soft limit.
51       *
52       * @return Total swap in bytes.
53       */
54      long getSwapTotal();
55  
56      /**
57       * The current memory committed to the paging/swap file(s), in bytes
58       *
59       * @return Swap used in bytes
60       */
61      long getSwapUsed();
62  
63      /**
64       * Number of pages read from paging/swap file(s) to resolve hard page
65       * faults. (Hard page faults occur when a process requires code or data that
66       * is not in its working set or elsewhere in physical memory, and must be
67       * retrieved from disk.) This property was designed as a primary indicator
68       * of the kinds of faults that cause system-wide delays. It includes pages
69       * retrieved to satisfy faults in the file system cache (usually requested
70       * by applications) and in non-cached mapped memory files.
71       *
72       * @return Pages swapped in
73       */
74      long getSwapPagesIn();
75  
76      /**
77       * Number of pages written to paging/swap file(s) to free up space in
78       * physical memory. Pages are written back to disk only if they are changed
79       * in physical memory, so they are likely to hold data, not code. A high
80       * rate of pages output might indicate a memory shortage. The operating
81       * system writes more pages back to disk to free up space when physical
82       * memory is in short supply.
83       *
84       * @return Pages swapped out
85       */
86      long getSwapPagesOut();
87  
88      /**
89       * The number of bytes in a memory page
90       *
91       * @return Page size in bytes.
92       */
93      long getPageSize();
94  }

53      private static final Pattern VENDOR_PRODUCT_ID = Pattern
54              .compile(".*(?:VID|VEN)_(\\p{XDigit}{4})&(?:PID|DEV)_(\\p{XDigit}{4}).*");

61      private static Map<String, String> serialMap = new HashMap<>();
62      private static Map<String, List<String>> hubMap = new HashMap<>();

69      /**
70       * {@inheritDoc}
71       */
72      public static UsbDevice[] getUsbDevices(boolean tree) {
73          UsbDevice[] devices = getUsbDevices();
74          if (tree) {
75              return devices;
76          }
77          List<UsbDevice> deviceList = new ArrayList<>();
78          // Top level is controllers; they won't be added to the list, but all
79          // their connected devices will be
80          for (UsbDevice device : devices) {
81              addDevicesToList(deviceList, device.getConnectedDevices());
82          }
83          return deviceList.toArray(new UsbDevice[deviceList.size()]);
84      }
85  
86      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
87          for (UsbDevice device : connectedDevices) {
88              deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
89                      device.getProductId(), device.getSerialNumber(), new MacUsbDevice[0]));
90              addDevicesToList(deviceList, device.getConnectedDevices());
91          }
92      }

94      private static UsbDevice[] getUsbDevices() {
95          // Start by collecting information for all PNP devices. While in theory
96          // these could be individually queried with a WHERE clause, grabbing
97          // them all up front incurs minimal memory overhead in exchange for
98          // faster access later
99  
100         // Clear maps
101         nameMap.clear();
102         vendorMap.clear();
103         serialMap.clear();
104         hubMap.clear();
105 
106         // Query Win32_PnPEntity to populate the maps
107         Map<String, List<String>> usbMap = WmiUtil.selectStringsFrom(null, "Win32_PnPEntity",
108                 "Name,Manufacturer,PnPDeviceID", null);
109         for (int i = 0; i < usbMap.get("Name").size(); i++) {
110             String pnpDeviceID = usbMap.get("PnPDeviceID").get(i);
111             nameMap.put(pnpDeviceID, usbMap.get("Name").get(i));
112             if (usbMap.get("Manufacturer").get(i).length() > 0) {
113                 vendorMap.put(pnpDeviceID, usbMap.get("Manufacturer").get(i));
114             }
115         }
116 
117         // Get serial # for disk drives or other physical media
118         usbMap = WmiUtil.selectStringsFrom(null, "Win32_DiskDrive", "PNPDeviceID,SerialNumber", null);
119         for (int i = 0; i < usbMap.get("PNPDeviceID").size(); i++) {
120             serialMap.put(usbMap.get("PNPDeviceID").get(i),
121                     ParseUtil.hexStringToString(usbMap.get("PNPDeviceID").get(i)));
122         }
123         usbMap = WmiUtil.selectStringsFrom(null, "Win32_PhysicalMedia", "PNPDeviceID,SerialNumber", null);
124         for (int i = 0; i < usbMap.get("PNPDeviceID").size(); i++) {
125             serialMap.put(usbMap.get("PNPDeviceID").get(i),
126                     ParseUtil.hexStringToString(usbMap.get("PNPDeviceID").get(i)));
127         }
128 
129         // Build the device tree. Start with the USB Controllers
130         // and recurse downward to devices as needed
131         usbMap = WmiUtil.selectStringsFrom(null, "Win32_USBController", "PNPDeviceID", null);
132         List<UsbDevice> controllerDevices = new ArrayList<>();
133         for (String controllerDeviceId : usbMap.get("PNPDeviceID")) {
134             putChildrenInDeviceTree(controllerDeviceId, 0);
135             controllerDevices.add(getDeviceAndChildren(controllerDeviceId, "0000", "0000"));
136         }
137         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
138     }

140     /**
141      * Navigates the Device Tree to place all children PNPDeviceIDs into the map
142      * for the specified deviceID. Recursively adds children's children, etc.
143      *
144      * @param deviceId
145      *            The device to add respective children to the map
146      * @param deviceInstance
147      *            The device instance (devnode handle), if known. If set to 0,
148      *            the code will search for a match.
149      */
150     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance) {
151         int devInst = deviceInstance;
152         // If no devInst provided, find it by matching deviceId
153         if (devInst == 0) {
154             // Get a handle to the device with this deviceId
155             // Start with all classes
156             HANDLE hinfoSet = SetupApi.INSTANCE.SetupDiGetClassDevs(null, null, null, SetupApi.DIGCF_ALLCLASSES);
157             if (hinfoSet == WinNT.INVALID_HANDLE_VALUE) {
158                 LOG.error("Invalid handle value for {}. Error code: {}", deviceId, Native.getLastError());
159                 return;
160             }
161             // Iterate to find matching parent
162             SP_DEVINFO_DATA dinfo = new SP_DEVINFO_DATA();
163             dinfo.cbSize = dinfo.size();
164             int i = 0;
165             while (SetupApi.INSTANCE.SetupDiEnumDeviceInfo(hinfoSet, i++, dinfo)) {
166                 if (deviceId.equals(getDeviceId(dinfo.DevInst))) {
167                     devInst = dinfo.DevInst;
168                     break;
169                 }
170             }
171         }
172         if (devInst == 0) {
173             LOG.error("Unable to find a devnode handle for {}.", deviceId);
174             return;
175         }
176         // Now iterate the children. Call CM_Get_Child to get first child
177         IntByReference child = new IntByReference();
178         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
179             // Add first child to a list
180             List<String> childList = new ArrayList<>();
181             String childId = getDeviceId(child.getValue());
182             childList.add(childId);
183             hubMap.put(deviceId, childList);
184             putChildrenInDeviceTree(childId, child.getValue());
185             // Find any other children
186             IntByReference sibling = new IntByReference();
187             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
188                 // Add to the list
189                 String siblingId = getDeviceId(sibling.getValue());
190                 hubMap.get(deviceId).add(siblingId);
191                 putChildrenInDeviceTree(siblingId, sibling.getValue());
192                 // Make this sibling the new child to find other siblings
193                 child = sibling;
194             }
195         }
196     }

198     /**
199      * Gets the device id for a devnode
200      *
201      * @param devInst
202      *            the handle to the devnode
203      * @return The PNPDeviceID
204      */
205     private static String getDeviceId(int devInst) {
206         NativeLongByReference ulLen = new NativeLongByReference();
207         if (0 != Cfgmgr32.INSTANCE.CM_Get_Device_ID_Size(ulLen, devInst, 0)) {
208             LOG.error("Couldn't get device string for device instance {}", devInst);
209             return "";
210         }
211         // Add 1 for null terminator
212         int size = ulLen.getValue().intValue() + 1;
213         char[] buffer = new char[size];
214         if (0 != Cfgmgr32.INSTANCE.CM_Get_Device_ID(devInst, buffer, size, 0)) {
215             LOG.error("Couldn't get device string for device instance {} with size {}", devInst, size);
216             return "";
217         }
218         return new String(buffer).trim();
219     }

221     /**
222      * Recursively creates WindowsUsbDevices by fetching information from maps
223      * to populate fields
224      *
225      * @param hubDeviceId
226      *            The PNPdeviceID of this device.
227      * @param vid
228      *            The default (parent) vendor ID
229      * @param pid
230      *            The default (parent) product ID
231      * @return A WindowsUsbDevice corresponding to this deviceID
232      */
233     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid) {
234         String vendorId = vid;
235         String productId = pid;
236         Matcher m = VENDOR_PRODUCT_ID.matcher(hubDeviceId);
237         if (m.matches()) {
238             vendorId = m.group(1).toLowerCase();
239             productId = m.group(2).toLowerCase();
240         }
241         List<String> pnpDeviceIds = hubMap.getOrDefault(hubDeviceId, new ArrayList<String>());
242         List<WindowsUsbDevice> usbDevices = new ArrayList<>();
243         for (String pnpDeviceId : pnpDeviceIds) {
244             usbDevices.add(getDeviceAndChildren(pnpDeviceId, vendorId, productId));
245         }
246         Collections.sort(usbDevices);
247         return new WindowsUsbDevice(nameMap.getOrDefault(hubDeviceId, vendorId + ":" + productId),
248                 vendorMap.getOrDefault(hubDeviceId, ""), vendorId, productId, serialMap.getOrDefault(hubDeviceId, ""),
249                 usbDevices.toArray(new UsbDevice[usbDevices.size()]));
250     }

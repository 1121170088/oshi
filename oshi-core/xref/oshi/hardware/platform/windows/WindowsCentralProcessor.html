132     /**
133      * Initializes Class variables
134      */
135     private void initVars() {
136         final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
137         String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
138         if (processorIds.length > 0) {
139             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
140             setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
141                     "VendorIdentifier"));
142             setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
143                     "ProcessorNameString"));
144             setIdentifier(
145                     Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
146         }
147         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
148         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
149         if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
150                 || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
151             setCpu64(true);
152         } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
153             setCpu64(false);
154         }
155 
156         WmiQuery<ProcessorProperty> processorIdQuery = new WmiQuery<>("Win32_Processor", ProcessorProperty.class);
157         WmiResult<ProcessorProperty> processorId = WmiUtil.queryWMI(processorIdQuery);
158         if (processorId.getResultCount() > 0) {
159             setProcessorID(WmiUtil.getString(processorId, ProcessorProperty.PROCESSORID, 0));
160         }
161     }

163     /**
164      * Initializes PDH Tick Counters
165      */
166     private void initPdhCounters() {
167         this.contextSwitchesPerSecCounter = PerfDataUtil.createCounter("System", null, "Context Switches/sec");
168         if (!PerfDataUtil.addCounterToQuery(this.contextSwitchesPerSecCounter)) {
169             this.contextSwitchesPerSecCounter = null;
170             this.contextSwitchQuery = new WmiQuery<>("Win32_PerfRawData_PerfOS_System", ContextSwitchProperty.class);
171         }
172 
173         boolean enumeration = true;
174         try {
175             PdhEnumObjectItems objectItems = PdhUtil.PdhEnumObjectItems(null, null, PROCESSOR_LOCALIZED, 100);
176 
177             if (!objectItems.getInstances().isEmpty()) {
178                 // The Idle time counter is inconsistent across Windows versions
179                 // and vs. WMI, but the Processor Time matches. Subtract
180                 // User+Privileged from Processor to get Idle.
181                 this.dpcTickCounter = new PerfCounter[this.logicalProcessorCount];
182                 this.interruptTickCounter = new PerfCounter[this.logicalProcessorCount];
183                 this.privilegedTickCounter = new PerfCounter[this.logicalProcessorCount];
184                 this.processorTickCounter = new PerfCounter[this.logicalProcessorCount];
185                 this.userTickCounter = new PerfCounter[this.logicalProcessorCount];
186 
187                 List<String> instances = objectItems.getInstances();
188                 PerfCounter counter;
189                 for (int i = 0; i < instances.size() && i < this.logicalProcessorCount; i++) {
190                     String instance = instances.get(i);
191 
192                     counter = PerfDataUtil.createCounter(PROCESSOR, instance, "% DPC Time");
193                     this.dpcTickCounter[i] = counter;
194                     if (!PerfDataUtil.addCounterToQuery(counter)) {
195                         throw new PdhException(0);
196                     }
197 
198                     counter = PerfDataUtil.createCounter(PROCESSOR, instance, "% Interrupt Time");
199                     this.interruptTickCounter[i] = counter;
200                     if (!PerfDataUtil.addCounterToQuery(counter)) {
201                         throw new PdhException(0);
202                     }
203 
204                     counter = PerfDataUtil.createCounter(PROCESSOR, instance, "% Privileged Time");
205                     this.privilegedTickCounter[i] = counter;
206                     if (!PerfDataUtil.addCounterToQuery(counter)) {
207                         throw new PdhException(0);
208                     }
209 
210                     counter = PerfDataUtil.createCounter(PROCESSOR, instance, "% Processor Time");
211                     this.processorTickCounter[i] = counter;
212                     if (!PerfDataUtil.addCounterToQuery(counter)) {
213                         throw new PdhException(0);
214                     }
215 
216                     counter = PerfDataUtil.createCounter(PROCESSOR, instance, "% User Time");
217                     this.userTickCounter[i] = counter;
218                     if (!PerfDataUtil.addCounterToQuery(counter)) {
219                         throw new PdhException(0);
220                     }
221                 }
222 
223                 counter = PerfDataUtil.createCounter(PROCESSOR, TOTAL_INSTANCE, "% Interrupt Time");
224                 this.irqTickCounter = counter;
225                 if (!PerfDataUtil.addCounterToQuery(counter)) {
226                     throw new PdhException(0);
227                 }
228 
229                 counter = PerfDataUtil.createCounter(PROCESSOR, TOTAL_INSTANCE, "% DPC Time");
230                 this.softIrqTickCounter = counter;
231                 if (!PerfDataUtil.addCounterToQuery(counter)) {
232                     throw new PdhException(0);
233                 }
234 
235                 counter = PerfDataUtil.createCounter(PROCESSOR, TOTAL_INSTANCE, "Interrupts/sec");
236                 this.interruptsPerSecCounter = counter;
237                 if (!PerfDataUtil.addCounterToQuery(counter)) {
238                     throw new PdhException(0);
239                 }
240             }
241         } catch (PdhException e) {
242             LOG.warn("Unable to enumerate performance counter instances for {}.", PROCESSOR_LOCALIZED);
243             enumeration = false;
244         }
245         if (!enumeration) {
246             PerfDataUtil.removeAllCounters(PROCESSOR);
247             this.dpcTickCounter = null;
248             this.interruptTickCounter = null;
249             this.privilegedTickCounter = null;
250             this.processorTickCounter = null;
251             this.userTickCounter = null;
252             this.irqTickCounter = null;
253             this.softIrqTickCounter = null;
254             this.interruptsPerSecCounter = null;
255 
256             processorTickCountQuery = new WmiQuery<>("Win32_PerfRawData_PerfOS_Processor WHERE NOT Name=\"_Total\"",
257                     ProcessorTickCountProperty.class);
258             systemTickCountQuery = new WmiQuery<>("Win32_PerfRawData_PerfOS_Processor WHERE Name=\"_Total\"",
259                     SystemTickCountProperty.class);
260             interruptsQuery = new WmiQuery<>("Win32_PerfRawData_PerfOS_Processor WHERE Name=\"_Total\"",
261                     InterruptsProperty.class);
262         }
263         // REMOVE
264         processorTickCountQuery = new WmiQuery<>("Win32_PerfRawData_PerfOS_Processor WHERE NOT Name=\"_Total\"",
265                 ProcessorTickCountProperty.class);
266     }

268     /**
269      * Updates logical and physical processor counts
270      */
271     @Override
272     protected void calculateProcessorCounts() {
273         // Get number of logical processors
274         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
275         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
276         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
277 
278         // Get number of physical processors
279         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
280         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
281             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
282                 this.physicalPackageCount++;
283             }
284             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
285                 this.physicalProcessorCount++;
286             }
287         }
288     }

290     /**
291      * {@inheritDoc}
292      */
293     @Override
294     public long[] getSystemCpuLoadTicks() {
295         long[] ticks = new long[TickType.values().length];
296         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
297         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
298         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
299         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
300             LOG.error("Failed to update system idle/kernel/user times. Error code: {}", Native.getLastError());
301             return ticks;
302         }
303         // IOwait:
304         // Windows does not measure IOWait.
305 
306         // IRQ:
307         // Percent time raw value is cumulative 100NS-ticks
308         // Divide by 10000 to get milliseconds
309         if (this.systemTickCountQuery == null) {
310             refreshTickCounters();
311             ticks[TickType.IRQ.getIndex()] = PerfDataUtil.queryCounter(this.irqTickCounter) / 10000L;
312             ticks[TickType.SOFTIRQ.getIndex()] = PerfDataUtil.queryCounter(this.softIrqTickCounter) / 10000L;
313         } else {
314             WmiResult<SystemTickCountProperty> result = WmiUtil.queryWMI(this.systemTickCountQuery);
315             if (result.getResultCount() > 0) {
316                 ticks[TickType.IRQ.getIndex()] = WmiUtil.getUint64(result, SystemTickCountProperty.PERCENTINTERRUPTTIME,
317                         0) / 10000L;
318                 ticks[TickType.SOFTIRQ.getIndex()] = WmiUtil.getUint64(result, SystemTickCountProperty.PERCENTDPCTIME,
319                         0) / 10000L;
320             }
321         }
322 
323         ticks[TickType.IDLE.getIndex()] = lpIdleTime.toDWordLong().longValue() / 10000L;
324         ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.toDWordLong().longValue() / 10000L
325                 - ticks[TickType.IDLE.getIndex()];
326         ticks[TickType.USER.getIndex()] = lpUserTime.toDWordLong().longValue() / 10000L;
327         // Additional decrement to avoid double counting in the total array
328         ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()] + ticks[TickType.SOFTIRQ.getIndex()];
329         return ticks;
330     }

333      * {@inheritDoc}
334      */
335     @Override
336     public double[] getSystemLoadAverage(int nelem) {
337         if (nelem < 1 || nelem > 3) {
338             throw new IllegalArgumentException("Must include from one to three elements.");
339         }
340         double[] average = new double[nelem];
341         // Windows doesn't have load average
342         for (int i = 0; i < average.length; i++) {
343             average[i] = -1;
344         }
345         return average;
346     }

348     /**
349      * {@inheritDoc}
350      */
351     @Override
352     public long[][] getProcessorCpuLoadTicks() {
353         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
354         if (this.processorTickCountQuery == null) {
355             refreshTickCounters();
356             for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
357                 ticks[cpu][TickType.SYSTEM.getIndex()] = PerfDataUtil.queryCounter(this.privilegedTickCounter[cpu]);
358                 ticks[cpu][TickType.USER.getIndex()] = PerfDataUtil.queryCounter(this.userTickCounter[cpu]);
359                 ticks[cpu][TickType.IRQ.getIndex()] = PerfDataUtil.queryCounter(this.interruptTickCounter[cpu]);
360                 ticks[cpu][TickType.SOFTIRQ.getIndex()] = PerfDataUtil.queryCounter(this.dpcTickCounter[cpu]);
361                 // Fetch total processor ticks
362                 // Later decrement by system + user
363                 ticks[cpu][TickType.IDLE.getIndex()] = PerfDataUtil.queryCounter(this.processorTickCounter[cpu]);
364             }
365         } else {
366             ticks = new long[this.logicalProcessorCount][TickType.values().length];
367             WmiResult<ProcessorTickCountProperty> result = WmiUtil.queryWMI(this.processorTickCountQuery);
368             for (int cpu = 0; cpu < result.getResultCount() && cpu < this.logicalProcessorCount; cpu++) {
369                 ticks[cpu][TickType.SYSTEM.getIndex()] = WmiUtil.getUint64(result,
370                         ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME, cpu);
371                 ticks[cpu][TickType.USER.getIndex()] = WmiUtil.getUint64(result,
372                         ProcessorTickCountProperty.PERCENTUSERTIME, cpu);
373                 ticks[cpu][TickType.IRQ.getIndex()] = WmiUtil.getUint64(result,
374                         ProcessorTickCountProperty.PERCENTINTERRUPTTIME, cpu);
375                 ticks[cpu][TickType.SOFTIRQ.getIndex()] = WmiUtil.getUint64(result,
376                         ProcessorTickCountProperty.PERCENTDPCTIME, cpu);
377                 // Fetch total processor ticks
378                 // Later decrement by system + user
379                 ticks[cpu][TickType.IDLE.getIndex()] = WmiUtil.getUint64(result,
380                         ProcessorTickCountProperty.PERCENTPROCESSORTIME, cpu);
381             }
382         }
383         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
384             ticks[cpu][TickType.IDLE.getIndex()] -= ticks[cpu][TickType.SYSTEM.getIndex()]
385                     + ticks[cpu][TickType.USER.getIndex()];
386             // Additional decrement to avoid double counting in the
387             // total array
388             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
389                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
390 
391             // Raw value is cumulative 100NS-ticks
392             // Divide by 10000 to get milliseconds
393             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10000L;
394             ticks[cpu][TickType.USER.getIndex()] /= 10000L;
395             ticks[cpu][TickType.IRQ.getIndex()] /= 10000L;
396             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10000L;
397             ticks[cpu][TickType.IDLE.getIndex()] /= 10000L;
398         }
399         // Skipping nice and IOWait, they'll stay 0
400         return ticks;
401     }

403     /**
404      * {@inheritDoc}
405      */
406     @Override
407     public long getSystemUptime() {
408         // GetTickCount64 requires Vista (6.0) or later
409         if (MAJOR_VERSION >= 6) {
410             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
411         } else {
412             // 32 bit rolls over at ~ 49 days
413             return Kernel32.INSTANCE.GetTickCount() / 1000L;
414         }
415     }

426     /**
427      * {@inheritDoc}
428      */
429     @Override
430     public long getContextSwitches() {
431         if (this.contextSwitchQuery == null) {
432             PerfDataUtil.updateQuery(this.contextSwitchesPerSecCounter);
433             return PerfDataUtil.queryCounter(this.contextSwitchesPerSecCounter);
434         }
435         WmiResult<ContextSwitchProperty> result = WmiUtil.queryWMI(this.contextSwitchQuery);
436         if (result.getResultCount() > 0) {
437             return WmiUtil.getUint32(result, ContextSwitchProperty.CONTEXTSWITCHESPERSEC, 0);
438         }
439         return 0L;
440     }

442     /**
443      * {@inheritDoc}
444      */
445     @Override
446     public long getInterrupts() {
447         if (this.interruptsQuery == null) {
448             refreshTickCounters();
449             return PerfDataUtil.queryCounter(this.interruptsPerSecCounter);
450         }
451         WmiResult<InterruptsProperty> result = WmiUtil.queryWMI(this.interruptsQuery);
452         if (result.getResultCount() > 0) {
453             return WmiUtil.getUint32(result, InterruptsProperty.INTERRUPTSPERSEC, 0);
454         }
455         return 0L;
456     }

458     /**
459      * Refresh PDH counters no more often than 100ms
460      */
461     private static void refreshTickCounters() {
462         if (System.currentTimeMillis() - lastRefresh > 100L) {
463             lastRefresh = PerfDataUtil.updateQuery(PROCESSOR);
464         }

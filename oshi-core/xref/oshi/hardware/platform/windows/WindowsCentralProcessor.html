58       * Create a Processor

60      public WindowsCentralProcessor() {
61          super();

64          // Initialize tick arrays
65          initTicks();
66  
67          LOG.debug("Initialized Processor");
68      }

70      /**
71       * Initializes Class variables
72       */
73      private void initVars() {
74          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor";
75          String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
76          if (processorIds.length > 0) {
77              String cpuRegistryPath = cpuRegistryRoot + "\\" + processorIds[0];
78              setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
79                      "VendorIdentifier"));
80              setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
81                      "ProcessorNameString"));
82              setIdentifier(
83                      Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
84          }
85          SYSTEM_INFO sysinfo = new SYSTEM_INFO();
86          Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
87          if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
88                  || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
89              setCpu64(true);
90          } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
91              setCpu64(false);
92          }
93          setProcessorID(WmiUtil.selectStringFrom(null, "Win32_Processor", "ProcessorID", null));
94      }

96      /**
97       * Updates logical and physical processor counts
98       */
99      @Override
100     protected void calculateProcessorCounts() {
101         // Get number of logical processors
102         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
103         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
104         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
105 
106         // Get number of physical processors
107         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
108         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
109             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
110                 this.physicalPackageCount++;
111             }
112             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
113                 this.physicalProcessorCount++;
114             }
115         }
116     }

118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     public long[] getSystemCpuLoadTicks() {
123         long[] ticks = new long[TickType.values().length];
124         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
125         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
126         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
127         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
128             LOG.error("Failed to update system idle/kernel/user times. Error code: " + Native.getLastError());
129             return ticks;
130         }
131         // IOwait:
132         // Windows does not measure IOWait.
133 
134         // IRQ:
135         // Percent time raw value is cumulative 100NS-ticks
136         // Divide by 10000 to get milliseconds
137         Map<String, List<String>> irq = WmiUtil.selectStringsFrom(null,
138                 "Win32_PerfRawData_Counters_ProcessorInformation", "PercentInterruptTime,PercentDPCTime",
139                 "WHERE Name=\"_Total\"");
140         if (!irq.get("PercentInterruptTime").isEmpty()) {
141             ticks[TickType.IRQ.getIndex()] = ParseUtil.parseLongOrDefault(irq.get("PercentInterruptTime").get(0), 0L)
142                     / 10000L;
143             ticks[TickType.SOFTIRQ.getIndex()] = ParseUtil.parseLongOrDefault(irq.get("PercentDPCTime").get(0), 0L)
144                     / 10000L;
145         }
146 
147         // Units are in 100-ns, divide by 10000 for ms
148         ticks[TickType.IDLE.getIndex()] = lpIdleTime.toDWordLong().longValue() / 10000L;
149         ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.toDWordLong().longValue() / 10000L
150                 - ticks[TickType.IDLE.getIndex()];
151         ticks[TickType.USER.getIndex()] = lpUserTime.toDWordLong().longValue() / 10000L;
152         // Additional decrement to avoid double counting in the total array
153         ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()] + ticks[TickType.SOFTIRQ.getIndex()];
154         return ticks;
155     }

157     /**
158      * {@inheritDoc}
159      */
160     @Override
161     public double[] getSystemLoadAverage(int nelem) {
162         if (nelem < 1 || nelem > 3) {
163             throw new IllegalArgumentException("Must include from one to three elements.");
164         }
165         double[] average = new double[nelem];
166         // TODO: If Windows ever actually implements a laod average for 1/5/15,
167         // return it
168         for (int i = 0; i < average.length; i++) {
169             average[i] = -1;
170         }
171         return average;
172     }

174     /**
175      * {@inheritDoc}
176      */
177     @Override
178     public long[][] getProcessorCpuLoadTicks() {
179         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
180         // Percent time raw value is cumulative 100NS-ticks
181         // Divide by 10000 to get milliseconds
182         Map<String, List<String>> wmiTicks = WmiUtil.selectStringsFrom(null,
183                 "Win32_PerfRawData_Counters_ProcessorInformation",
184                 "Name,PercentIdleTime,PercentPrivilegedTime,PercentUserTime,PercentInterruptTime,PercentDPCTime",
185                 "WHERE NOT Name LIKE \"%_Total\"");
186         for (int index = 0; index < wmiTicks.get("Name").size(); index++) {
187             // It would be too easy if the WMI order matched logical processors
188             // but alas, it goes "0,3"; "0,2"; "0,1"; "0,0". So let's do it
189             // right and actually string match the name. The first 0 will be
190             // there unless we're dealing with NUMA nodes
191             for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
192                 String name = "0," + cpu;
193                 if (wmiTicks.get("Name").get(index).equals(name)) {
194                     // Skipping nice and IOWait, they'll stay 0
195                     ticks[cpu][TickType.USER.getIndex()] = ParseUtil
196                             .parseLongOrDefault(wmiTicks.get("PercentUserTime").get(index), 0L) / 10000L;
197                     ticks[cpu][TickType.SYSTEM.getIndex()] = ParseUtil
198                             .parseLongOrDefault(wmiTicks.get("PercentPrivilegedTime").get(index), 0L) / 10000L;
199                     ticks[cpu][TickType.IDLE.getIndex()] = ParseUtil
200                             .parseLongOrDefault(wmiTicks.get("PercentIdleTime").get(index), 0L) / 10000L;
201                     ticks[cpu][TickType.IRQ.getIndex()] = ParseUtil
202                             .parseLongOrDefault(wmiTicks.get("PercentInterruptTime").get(index), 0L) / 10000L;
203                     ticks[cpu][TickType.SOFTIRQ.getIndex()] = ParseUtil
204                             .parseLongOrDefault(wmiTicks.get("PercentDPCTime").get(index), 0L) / 10000L;
205                     // Additional decrement to avoid double counting in the
206                     // total array
207                     ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
208                             + ticks[cpu][TickType.SOFTIRQ.getIndex()];
209                     break;
210                 }
211             }
212         }
213         return ticks;
214     }

216     /**
217      * {@inheritDoc}
218      */
219     @Override
220     public long getSystemUptime() {
221         // GetTickCount64 requires Vista (6.0) or later
222         if (majorVersion >= 6) {
223             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
224         } else {
225             // 32 bit rolls over at ~ 49 days
226             return Kernel32.INSTANCE.GetTickCount() / 1000L;
227         }
228     }

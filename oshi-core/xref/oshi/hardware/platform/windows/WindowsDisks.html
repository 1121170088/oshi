113     public static boolean updateDiskStats(HWDiskStore diskStore) {
114         String index = null;
115         HWPartition[] partitions = diskStore.getPartitions();
116         if (partitions.length > 0) {
117             // If a partition exists on this drive, the major property
118             // corresponds to the disk index, so use it.
119             index = Integer.toString(partitions[0].getMajor());
120         } else if (diskStore.getName().startsWith(PHYSICALDRIVE_PREFIX)) {
121             // If no partition exists, Windows reliably uses a name to match the
122             // disk index. That said, the skeptical person might wonder why a
123             // disk has read/write statistics without a partition, and wonder
124             // why this branch is even relevant as an option. The author of this
125             // comment does not have an answer for this valid question.
126             index = diskStore.getName().substring(PHYSICALDRIVE_PREFIX.length(), diskStore.getName().length());
127         } else {
128             // The author of this comment cannot fathom a circumstance in which
129             // the code reaches this point, but just in case it does, here's the
130             // correct response. If you get this log warning, the circumstances
131             // would be of great interest to the project's maintainers.
132             LOG.warn("Couldn't match index for {}", diskStore.getName());
133             return false;
134         }
135         populateReadWriteMaps(index);
136         if (readMap.containsKey(index)) {
137             diskStore.setReads(MapUtil.getOrDefault(readMap, index, 0L));
138             diskStore.setReadBytes(MapUtil.getOrDefault(readByteMap, index, 0L));
139             diskStore.setWrites(MapUtil.getOrDefault(writeMap, index, 0L));
140             diskStore.setWriteBytes(MapUtil.getOrDefault(writeByteMap, index, 0L));
141             diskStore.setTransferTime(MapUtil.getOrDefault(xferTimeMap, index, 0L));
142             diskStore.setTimeStamp(MapUtil.getOrDefault(timeStampMap, index, 0L));
143             return true;
144         } else {
145             return false;
146         }

150     @Override
151     public HWDiskStore[] getDisks() {
152         List<HWDiskStore> result;
153         result = new ArrayList<>();
154         populateReadWriteMaps(null);
155         populatePartitionMaps();
156 
157         Map<String, List<Object>> vals = WmiUtil.selectObjectsFrom(null, DISK_DRIVE_CLASS, DRIVE_PROPERTIES, null,
158                 DRIVE_TYPES);
159         for (int i = 0; i < vals.get(NAME_PROPERTY).size(); i++) {
160             HWDiskStore ds = new HWDiskStore();
161             ds.setName((String) vals.get(NAME_PROPERTY).get(i));
162             ds.setModel(String.format("%s %s", vals.get(MODEL_PROPERTY).get(i), vals.get(MANUFACTURER_PROPERTY).get(i))
163                     .trim());
164             // Most vendors store serial # as a hex string; convert
165             ds.setSerial(ParseUtil.hexStringToString((String) vals.get(SERIALNUMBER_PROPERTY).get(i)));
166             String index = vals.get(INDEX_PROPERTY).get(i).toString();
167             ds.setReads(MapUtil.getOrDefault(readMap, index, 0L));
168             ds.setReadBytes(MapUtil.getOrDefault(readByteMap, index, 0L));
169             ds.setWrites(MapUtil.getOrDefault(writeMap, index, 0L));
170             ds.setWriteBytes(MapUtil.getOrDefault(writeByteMap, index, 0L));
171             ds.setTransferTime(MapUtil.getOrDefault(xferTimeMap, index, 0L));
172             ds.setTimeStamp(MapUtil.getOrDefault(timeStampMap, index, 0L));
173             // If successful this line is the desired value
174             ds.setSize(ParseUtil.parseLongOrDefault((String) vals.get(SIZE_PROPERTY).get(i), 0L));
175             // Get partitions
176             List<HWPartition> partitions = new ArrayList<>();
177             List<String> partList = driveToPartitionMap.get(ds.getName());
178             if (partList != null && !partList.isEmpty()) {
179                 for (String part : partList) {
180                     partitions.add(partitionMap.get(part));
181                 }
182             }
183             ds.setPartitions(partitions.toArray(new HWPartition[partitions.size()]));
184             // Add to list
185             result.add(ds);
186         }
187         return result.toArray(new HWDiskStore[result.size()]);
188     }

190     /**
191      * Populates the maps for the specified index. If the index is null,
192      * populates all the maps
193      * 
194      * @param index
195      *            The index to populate/update maps for
196      */
197     private static void populateReadWriteMaps(String index) {
198         // If index is null, start from scratch.
199         if (index == null) {
200             readMap.clear();
201             readByteMap.clear();
202             writeMap.clear();
203             writeByteMap.clear();
204             xferTimeMap.clear();
205             timeStampMap.clear();
206         }
207         // Fetch the instance names
208         // Call once to get string lengths
209         DWORDByReference pcchCounterListLength = new DWORDByReference(new DWORD(0));
210         DWORDByReference pcchInstanceListLength = new DWORDByReference(new DWORD(0));
211         Pdh.INSTANCE.PdhEnumObjectItems(null, null, "PhysicalDisk", null, pcchCounterListLength, null,
212                 pcchInstanceListLength, 100, 0);
213         // Allocate memory and call again to populate strings
214         char[] mszCounterList = new char[pcchCounterListLength.getValue().intValue()];
215         char[] mszInstanceList = new char[pcchInstanceListLength.getValue().intValue()];
216         Pdh.INSTANCE.PdhEnumObjectItems(null, null, "PhysicalDisk", mszCounterList, pcchCounterListLength,
217                 mszInstanceList, pcchInstanceListLength, 100, 0);
218         List<String> instances = Native.toStringList(mszInstanceList);
219         instances.remove("_Total");
220         // At this point we have a list of strings that PDH understands. Fetch
221         // the counters.
222         // Although the field names say "PerSec" this is the Raw Data/counters
223         // from which the associated fields are populated in the Formatted Data
224         for (String i : instances) {
225             String name = ParseUtil.whitespaces.split(i)[0];
226             String readString = String.format(PDH_DISK_READS_FORMAT, i);
227             if (!PdhUtil.isCounter(readString)) {
228                 PdhUtil.addCounter(readString);
229             }
230             String readBytesString = String.format(PDH_DISK_READ_BYTES_FORMAT, i);
231             if (!PdhUtil.isCounter(readBytesString)) {
232                 PdhUtil.addCounter(readBytesString);
233             }
234             String writeString = String.format(PDH_DISK_WRITES_FORMAT, i);
235             if (!PdhUtil.isCounter(writeString)) {
236                 PdhUtil.addCounter(writeString);
237             }
238             String writeBytesString = String.format(PDH_DISK_WRITE_BYTES_FORMAT, i);
239             if (!PdhUtil.isCounter(writeBytesString)) {
240                 PdhUtil.addCounter(writeBytesString);
241             }
242             String xferTimeString = String.format(PDH_DISK_TIME_FORMAT, i);
243             if (!PdhUtil.isCounter(xferTimeString)) {
244                 PdhUtil.addCounter(xferTimeString);
245             }
246             readMap.put(name, PdhUtil.queryCounter(readString));
247             readByteMap.put(name, PdhUtil.queryCounter(readBytesString));
248             writeMap.put(name, PdhUtil.queryCounter(writeString));
249             writeByteMap.put(name, PdhUtil.queryCounter(writeBytesString));
250             xferTimeMap.put(name, PdhUtil.queryCounter(xferTimeString) / 10000L);
251             timeStampMap.put(name, PdhUtil.queryCounterTimestamp(xferTimeString));
252         }
253     }

255     private void populatePartitionMaps() {
256         driveToPartitionMap.clear();
257         partitionToLogicalDriveMap.clear();
258         partitionMap.clear();
259         // For Regexp matching DeviceIDs
260         Matcher mAnt;
261         Matcher mDep;
262 
263         // Map drives to partitions
264         Map<String, List<String>> partitionQueryMap = WmiUtil.selectStringsFrom(null, "Win32_DiskDriveToDiskPartition",
265                 DRIVE_TO_PARTITION_PROPERTIES, null);
266         for (int i = 0; i < partitionQueryMap.get(ANTECEDENT_PROPERTY).size(); i++) {
267             mAnt = DEVICE_ID.matcher(partitionQueryMap.get(ANTECEDENT_PROPERTY).get(i));
268             mDep = DEVICE_ID.matcher(partitionQueryMap.get(DEPENDENT_PROPERTY).get(i));
269             if (mAnt.matches() && mDep.matches()) {
270                 MapUtil.createNewListIfAbsent(driveToPartitionMap, mAnt.group(1).replaceAll("\\\\\\\\", "\\\\"))
271                         .add(mDep.group(1));
272             }
273         }
274 
275         // Map partitions to logical disks
276         partitionQueryMap = WmiUtil.selectStringsFrom(null, "Win32_LogicalDiskToPartition",
277                 LOGICAL_DISK_TO_PARTITION_PROPERTIES, null);
278         for (int i = 0; i < partitionQueryMap.get(ANTECEDENT_PROPERTY).size(); i++) {
279             mAnt = DEVICE_ID.matcher(partitionQueryMap.get(ANTECEDENT_PROPERTY).get(i));
280             mDep = DEVICE_ID.matcher(partitionQueryMap.get(DEPENDENT_PROPERTY).get(i));
281             if (mAnt.matches() && mDep.matches()) {
282                 partitionToLogicalDriveMap.put(mAnt.group(1), mDep.group(1) + "\\");
283             }
284         }
285 
286         // Next, get all partitions and create objects
287         final Map<String, List<Object>> hwPartitionQueryMap = WmiUtil.selectObjectsFrom(null, "Win32_DiskPartition",
288                 PARTITION_PROPERTIES, null, PARTITION_TYPES);
289         for (int i = 0; i < hwPartitionQueryMap.get(NAME_PROPERTY).size(); i++) {
290             String deviceID = (String) hwPartitionQueryMap.get(DEVICE_ID_PROPERTY).get(i);
291             String logicalDrive = MapUtil.getOrDefault(partitionToLogicalDriveMap, deviceID, "");
292             String uuid = "";
293             if (!logicalDrive.isEmpty()) {
294                 // Get matching volume for UUID
295                 char[] volumeChr = new char[BUFSIZE];
296                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive, volumeChr, BUFSIZE);
297                 uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
298             }
299             partitionMap.put(deviceID,
300                     new HWPartition((String) hwPartitionQueryMap.get(NAME_PROPERTY).get(i),
301                             (String) hwPartitionQueryMap.get(TYPE_PROPERTY).get(i),
302                             (String) hwPartitionQueryMap.get(DESCRIPTION_PROPERTY).get(i), uuid,
303                             ParseUtil.parseLongOrDefault((String) hwPartitionQueryMap.get(SIZE_PROPERTY).get(i), 0L),
304                             ((Long) hwPartitionQueryMap.get(DISK_INDEX_PROPERTY).get(i)).intValue(),
305                             ((Long) hwPartitionQueryMap.get(INDEX_PROPERTY).get(i)).intValue(), logicalDrive));
306         }
307     }

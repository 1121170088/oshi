19  package oshi.hardware.platform.linux;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import oshi.hardware.HWDiskStore;
28  import oshi.hardware.common.AbstractDisks;
29  import oshi.jna.platform.linux.Udev;
30  import oshi.util.ParseUtil;
31  
32  /**
33   * Linux hard disk implementation.
34   *
35   * @author enrico[dot]bianchi[at]gmail[dot]com
36   */
37  public class LinuxDisks extends AbstractDisks {
38  
39      private static final long serialVersionUID = 1L;
40  
41      private static final Logger LOG = LoggerFactory.getLogger(LinuxDisks.class);
42  
43      private final int SECTORSIZE = 512;
44  
45      private void computeDiskStats(HWDiskStore store, Udev.UdevDevice disk) {
46          LinuxBlockDevStats stats;
47          stats = new LinuxBlockDevStats(store.getName(), disk);
48  
49          // Reads and writes are converted in bytes
50          store.setReads(stats.read_ops);
51          store.setReadBytes(stats.read_512bytes * this.SECTORSIZE);
52          store.setWrites(stats.write_ops);
53          store.setWriteBytes(stats.write_512bytes * this.SECTORSIZE);
54          store.setTransferTime(stats.active_ms);
55      }
56  
57      @Override
58      public HWDiskStore[] getDisks() {
59          HWDiskStore store;
60          List<HWDiskStore> result;
61  
62          Udev.UdevHandle handle = null;
63          Udev.UdevDevice device = null;
64          Udev.UdevEnumerate enumerate = null;
65          Udev.UdevListEntry entry;
66          Udev.UdevListEntry oldEntry;
67  
68          result = new ArrayList<>();
69  
70          handle = Udev.INSTANCE.udev_new();
71          enumerate = Udev.INSTANCE.udev_enumerate_new(handle);
72          Udev.INSTANCE.udev_enumerate_add_match_subsystem(enumerate, "block");
73          Udev.INSTANCE.udev_enumerate_scan_devices(enumerate);
74  
75          entry = Udev.INSTANCE.udev_enumerate_get_list_entry(enumerate);
76          while (true) {
77              store = new HWDiskStore();
78              try {
79                  oldEntry = entry;
80                  device = Udev.INSTANCE.udev_device_new_from_syspath(handle,
81                          Udev.INSTANCE.udev_list_entry_get_name(entry));
82                  if (Udev.INSTANCE.udev_device_get_devtype(device).equals("disk")
83                          && !Udev.INSTANCE.udev_device_get_devnode(device).startsWith("/dev/loop")
84                          && !Udev.INSTANCE.udev_device_get_devnode(device).startsWith("/dev/ram")) {
85                      store.setName(Udev.INSTANCE.udev_device_get_devnode(device));
86  
87                      // Avoid model and serial in virtual environments
88                      store.setModel((Udev.INSTANCE.udev_device_get_property_value(device, "ID_MODEL") == null)
89                              ? "Unknown" : Udev.INSTANCE.udev_device_get_property_value(device, "ID_MODEL"));
90                      store.setSerial((Udev.INSTANCE.udev_device_get_property_value(device, "ID_SERIAL_SHORT") == null)
91                              ? "Unknown"
92                              : Udev.INSTANCE.udev_device_get_property_value(device, "ID_SERIAL_SHORT"));
93  
94                      store.setSize(ParseUtil.parseLongOrDefault(
95                              Udev.INSTANCE.udev_device_get_sysattr_value(device, "size"), 0L) * SECTORSIZE);
96  
97                      this.computeDiskStats(store, device);
98                      result.add(store);
99                  }
100                 entry = Udev.INSTANCE.udev_list_entry_get_next(oldEntry);
101             } catch (Exception ex) {
102                 LOG.debug("Reached all disks. Exiting ");
103                 break;
104             } finally {
105                 if (device instanceof Udev.UdevDevice) {
106                     Udev.INSTANCE.udev_device_unref(device);
107                 }
108             }
109         }
110 
111         Udev.INSTANCE.udev_enumerate_unref(enumerate);
112         Udev.INSTANCE.udev_unref(handle);
113 
114         return result.toArray(new HWDiskStore[result.size()]);
115     }
116 }

19  package oshi.hardware.platform.linux;
20  
21  import oshi.hardware.NetworkIF;
22  import oshi.hardware.common.AbstractNetworks;
23  import oshi.util.FileUtil;
24  
25  /**
26   * @author enrico[dot]bianchi[at]gmail[dot]com
27   */
28  public class LinuxNetworks extends AbstractNetworks {
29  
30      private static final long serialVersionUID = 1L;
31  
32      /**
33       * Updates interface network statistics on the given interface. Statistics
34       * include packets and bytes sent and received, and interface speed.
35       *
36       * @param netIF
37       *            The interface on which to update statistics
38       */
39      public static void updateNetworkStats(NetworkIF netIF) {
40          String txBytesPath = String.format("/sys/class/net/%s/statistics/tx_bytes", netIF.getName());
41          String rxBytesPath = String.format("/sys/class/net/%s/statistics/rx_bytes", netIF.getName());
42          String txPacketsPath = String.format("/sys/class/net/%s/statistics/tx_packets", netIF.getName());
43          String rxPacketsPath = String.format("/sys/class/net/%s/statistics/rx_packets", netIF.getName());
44          String txErrorsPath = String.format("/sys/class/net/%s/statistics/tx_errors", netIF.getName());
45          String rxErrorsPath = String.format("/sys/class/net/%s/statistics/rx_errors", netIF.getName());
46          String speed = String.format("/sys/class/net/%s/speed", netIF.getName());
47  
48          netIF.setTimeStamp(System.currentTimeMillis());
49          netIF.setBytesSent(FileUtil.getUnsignedLongFromFile(txBytesPath));
50          netIF.setBytesRecv(FileUtil.getUnsignedLongFromFile(rxBytesPath));
51          netIF.setPacketsSent(FileUtil.getUnsignedLongFromFile(txPacketsPath));
52          netIF.setPacketsRecv(FileUtil.getUnsignedLongFromFile(rxPacketsPath));
53          netIF.setOutErrors(FileUtil.getUnsignedLongFromFile(txErrorsPath));
54          netIF.setInErrors(FileUtil.getUnsignedLongFromFile(rxErrorsPath));
55          long netSpeed = FileUtil.getUnsignedLongFromFile(speed) * 1024 * 1024;
56          netIF.setSpeed(netSpeed < 0 ? 0 : netSpeed);
57      }
58  }

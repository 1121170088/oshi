44  public class MacCentralProcessor extends AbstractCentralProcessor {

46      private static final long serialVersionUID = 1L;
47  
48      private static final Logger LOG = LoggerFactory.getLogger(MacCentralProcessor.class);
49  
50      private static final long BOOTTIME;
51      static {
52          Timeval tv = new Timeval();
53          if (!SysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
54              // Usually this works. If it doesn't, fall back to text parsing.
55              // Boot time will be the first consecutive string of digits.
56              BOOTTIME = ParseUtil.parseLongOrDefault(
57                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
58                      System.currentTimeMillis() / 1000);
59          } else {
60              // tv now points to a 64-bit timeval structure for boot time.
61              // First 4 bytes are seconds, second 4 bytes are microseconds
62              // (we ignore)
63              BOOTTIME = tv.tv_sec;
64          }
65      }
66  
67      /**
68       * Create a Processor
69       */
70      public MacCentralProcessor() {
71          super();
72          // Initialize class variables
73          initVars();
74          // Initialize tick arrays
75          initTicks();
76  
77          LOG.debug("Initialized Processor");
78      }
79  
80      private void initVars() {
81          setVendor(SysctlUtil.sysctl("machdep.cpu.vendor", ""));
82          setName(SysctlUtil.sysctl("machdep.cpu.brand_string", ""));
83          setCpu64(SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0);
84          int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
85          setStepping(i < 0 ? "" : Integer.toString(i));
86          i = SysctlUtil.sysctl("machdep.cpu.model", -1);
87          setModel(i < 0 ? "" : Integer.toString(i));
88          i = SysctlUtil.sysctl("machdep.cpu.family", -1);
89          setFamily(i < 0 ? "" : Integer.toString(i));
90          long processorID = 0L;
91          processorID |= SysctlUtil.sysctl("machdep.cpu.signature", 0);
92          processorID |= (SysctlUtil.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
93          setProcessorID(String.format("%016X", processorID));
94      }

96      /**
97       * Updates logical and physical processor counts from sysctl calls
98       */
99      @Override
100     protected void calculateProcessorCounts() {
101         this.logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
102         this.physicalProcessorCount = SysctlUtil.sysctl("hw.physicalcpu", 1);
103     }

105     /**
106      * {@inheritDoc}
107      */
108     @Override
109     public long[] getSystemCpuLoadTicks() {
110         long[] ticks = new long[TickType.values().length];
111         int machPort = SystemB.INSTANCE.mach_host_self();
112         HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
113         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
114                 new IntByReference(cpuLoadInfo.size()))) {
115             LOG.error("Failed to get System CPU ticks. Error code: " + Native.getLastError());
116             return ticks;
117         }
118 
119         ticks[TickType.USER.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
120         ticks[TickType.NICE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
121         ticks[TickType.SYSTEM.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
122         ticks[TickType.IDLE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
123         // Leave IOWait and IRQ values as 0
124         return ticks;
125     }

127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public double[] getSystemLoadAverage(int nelem) {
132         if (nelem < 1 || nelem > 3) {
133             throw new IllegalArgumentException("Must include from one to three elements.");
134         }
135         double[] average = new double[nelem];
136         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
137         if (retval < nelem) {
138             for (int i = Math.max(retval, 0); i < average.length; i++) {
139                 average[i] = -1d;
140             }
141         }
142         return average;
143     }
144 
145     /**
146      * {@inheritDoc}
147      */
148     @Override
149     public long[][] getProcessorCpuLoadTicks() {
150         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
151 
152         int machPort = SystemB.INSTANCE.mach_host_self();
153 
154         IntByReference procCount = new IntByReference();
155         PointerByReference procCpuLoadInfo = new PointerByReference();
156         IntByReference procInfoCount = new IntByReference();
157         if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
158                 procCpuLoadInfo, procInfoCount)) {
159             LOG.error("Failed to update CPU Load. Error code: " + Native.getLastError());
160             return ticks;
161         }
162 
163         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
164         for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
165             int offset = cpu * SystemB.CPU_STATE_MAX;
166             ticks[cpu][TickType.USER.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
167             ticks[cpu][TickType.NICE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
168             ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil
169                     .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
170             ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
171         }
172         return ticks;
173     }

175     /**
176      * {@inheritDoc}
177      */
178     @Override
179     public long getSystemUptime() {
180         return System.currentTimeMillis() / 1000 - BOOTTIME;
181     }

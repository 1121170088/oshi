19  package oshi.hardware.platform.mac;
20  
21  import java.util.regex.Pattern;
22  
23  import oshi.hardware.common.AbstractComputerSystem;
24  import oshi.util.ExecutingCommand;
25  
26  /**
27   * Hardware data obtained by system_profiler
28   * 
29   * @author SchiTho1 [at] Securiton AG
30   * @author widdis [at] gmail [dot] com
31   */
32  final class MacComputerSystem extends AbstractComputerSystem {
33  
34      private static final long serialVersionUID = 1L;
35  
36      private static final String APPLE = "Apple Inc.";
37  
38      MacComputerSystem() {
39          init();
40      }
41  
42      private void init() {
43  
44          setManufacturer(APPLE);
45  
46          // $ system_profiler SPHardwareDataType
47          // Hardware:
48          //
49          // Hardware Overview:
50          //
51          // Model Name: MacBook Pro
52          // Model Identifier: MacBookPro8,2
53          // Processor Name: Intel Core i7
54          // Processor Speed: 2.3 GHz
55          // Number of Processors: 1
56          // Total Number of Cores: 4
57          // L2 Cache (per Core): 256 KB
58          // L3 Cache: 8 MB
59          // Memory: 16 GB
60          // Boot ROM Version: MBP81.0047.B2C
61          // SMC Version (system): 1.69f4
62          // Serial Number (system): C02FH4XYCB71
63          // Hardware UUID: D92CE829-65AD-58FA-9C32-88968791B7BD
64          // Sudden Motion Sensor:
65          // State: Enabled
66  
67          String modelName = "";
68          final String modelNameMarker = "Model Name:";
69          String modelIdentifier = "";
70          final String modelIdMarker = "Model Identifier:";
71          String serialNumberSystem = "";
72          final String serialNumMarker = "Serial Number (system):";
73          String smcVersion = "";
74          final String smcMarker = "SMC Version (system):";
75          String bootRomVersion = "";
76          final String bootRomMarker = "Boot ROM Version:";
77  
78          final MacFirmware firmware = new MacFirmware();
79          firmware.setManufacturer(APPLE);
80          firmware.setName("EFI");
81  
82          final MacBaseboard baseboard = new MacBaseboard();
83          baseboard.setManufacturer(APPLE);
84          baseboard.setModel("SMC");
85  
86          // Populate name and ID
87          for (final String checkLine : ExecutingCommand.runNative("system_profiler SPHardwareDataType")) {
88              if (checkLine.contains(modelNameMarker)) {
89                  modelName = checkLine.split(modelNameMarker)[1].trim();
90              }
91              if (checkLine.contains(modelIdMarker)) {
92                  modelIdentifier = checkLine.split(modelIdMarker)[1].trim();
93              }
94              if (checkLine.contains(bootRomMarker)) {
95                  bootRomVersion = checkLine.split(bootRomMarker)[1].trim();
96              }
97              if (checkLine.contains(smcMarker)) {
98                  smcVersion = checkLine.split(Pattern.quote(smcMarker))[1].trim();
99              }
100             if (checkLine.contains(serialNumMarker)) {
101                 serialNumberSystem = checkLine.split(Pattern.quote(serialNumMarker))[1].trim();
102             }
103         }
104         // Use name (id) if both available; else either one
105         if (!modelName.isEmpty()) {
106             if (!modelIdentifier.isEmpty()) {
107                 setModel(modelName + " (" + modelIdentifier + ")");
108             } else {
109                 setModel(modelName);
110             }
111         } else {
112             if (!modelIdentifier.isEmpty()) {
113                 setModel(modelIdentifier);
114             }
115         }
116 
117         if (!serialNumberSystem.isEmpty()) {
118             setSerialNumber(serialNumberSystem);
119             baseboard.setSerialNumber(serialNumberSystem);
120         }
121         if (!smcVersion.isEmpty()) {
122             baseboard.setVersion(smcVersion);
123         }
124         if (bootRomVersion != null && !bootRomVersion.isEmpty()) {
125             firmware.setVersion(bootRomVersion);
126         }
127 
128         setFirmware(firmware);
129         setBaseboard(baseboard);
130     }
131 }

54  /**
55   * Mac hard disk implementation.
56   *
57   * @author enrico[dot]bianchi[at]gmail[dot]com
58   */
59  public class MacDisks extends AbstractDisks {
60  
61      private static final long serialVersionUID = 1L;
62  
63      private static final Logger LOG = LoggerFactory.getLogger(MacDisks.class);
64  
65      private static final Map<String, String> mountPointMap = new HashMap<>();
66  
67      @Override
68      public HWDiskStore[] getDisks() {
69          mountPointMap.clear();
70          List<HWDiskStore> result = new ArrayList<>();
71  
72          // Use statfs to find all drives
73          int numfs = SystemB.INSTANCE.getfsstat64(null, 0, 0);
74          // Create array to hold results
75          Statfs[] fs = new Statfs[numfs];
76          // Fill array with results
77          SystemB.INSTANCE.getfsstat64(fs, numfs * new Statfs().size(), SystemB.MNT_NOWAIT);
78  
79          // Open a DiskArbitration session
80          DASessionRef session = DiskArbitration.INSTANCE.DASessionCreate(CfUtil.ALLOCATOR);
81          if (session == null) {
82              LOG.error("Unable to open session to DiskArbitration framework.");
83              return new HWDiskStore[0];
84          }
85  
86          // Create a set in case we have multiple partitions mounted
87          Set<String> bsdNames = new HashSet<String>();
88          // Iterate all mounted file systems
89          for (Statfs f : fs) {
90              String mntFrom = new String(f.f_mntfromname).trim();
91              // OS X registry uses the BSD Name, e.g. disk0, disk1, etc.
92              // Strip off the partition # to get base disk name
93              String[] split = mntFrom.split("/dev/|s\\d+");
94              if (split.length > 1) {
95                  bsdNames.add(split[1]);
96              }
97              mountPointMap.put(mntFrom.replace("/dev/", ""), new String(f.f_mntonname).trim());
98          }
99          // Now iterate the bsdNames
100         for (String bsdName : bsdNames) {
101             String model = "";
102             String serial = "";
103             long size = 0L;
104             long xferTime = 0L;
105 
106             // Get a reference to the disk - only matching /dev/disk*
107             String path = "/dev/" + bsdName;
108 
109             // Get the DiskArbitration dictionary for this disk, which has model
110             // and size (capacity)
111             DADiskRef disk = DiskArbitration.INSTANCE.DADiskCreateFromBSDName(CfUtil.ALLOCATOR, session, path);
112             if (disk != null) {
113                 CFDictionaryRef diskInfo = DiskArbitration.INSTANCE.DADiskCopyDescription(disk);
114                 if (diskInfo != null) {
115                     // Parse out model and size from their respective keys
116                     Pointer modelPtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
117                             CfUtil.getCFString("DADeviceModel"));
118                     model = CfUtil.cfPointerToString(modelPtr);
119                     Pointer sizePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
120                             CfUtil.getCFString("DAMediaSize"));
121                     size = CfUtil.cfPointerToLong(sizePtr);
122                     CfUtil.release(diskInfo);
123 
124                     // Use the model as a key to get serial from IOKit
125                     if (!model.equals("Disk Image")) {
126                         CFStringRef modelNameRef = CFStringRef.toCFString(model);
127                         CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE
128                                 .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
129                         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, CfUtil.getCFString("Model"),
130                                 modelNameRef);
131                         CFMutableDictionaryRef matchingDict = CoreFoundation.INSTANCE
132                                 .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
133                         CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict,
134                                 CfUtil.getCFString("IOPropertyMatch"), propertyDict);
135 
136                         // search for all IOservices that match the model
137                         IntByReference serviceIterator = new IntByReference();
138                         IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
139                         // getMatchingServices releases matchingDict
140                         CfUtil.release(modelNameRef);
141                         CfUtil.release(propertyDict);
142                         int sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
143                         while (sdService != 0) {
144                             // look up the serial number
145                             serial = IOKitUtil.getIORegistryStringProperty(sdService, "Serial Number");
146                             IOKit.INSTANCE.IOObjectRelease(sdService);
147                             if (serial != null) {
148                                 break;
149                             }
150                             // iterate
151                             sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
152                         }
153                         if (serial == null) {
154                             serial = "";
155                         }
156                         IOKit.INSTANCE.IOObjectRelease(serviceIterator.getValue());
157                     }
158                 }
159                 CfUtil.release(disk);
160 
161                 // If empty, ignore
162                 if (size <= 0) {
163                     continue;
164                 }
165                 HWDiskStore diskStore = new HWDiskStore(bsdName, model.trim(), serial.trim(), size, 0L, 0L, 0L, 0L, 0L,
166                         new HWPartition[0]);
167 
168                 // Now look up the device using the BSD Name to get its
169                 // statistics
170                 CFMutableDictionaryRef matchingDict = IOKitUtil.getBSDNameMatchingDict(bsdName);
171                 if (matchingDict != null) {
172                     // search for all IOservices that match the bsd name
173                     IntByReference driveList = new IntByReference();
174                     IOKitUtil.getMatchingServices(matchingDict, driveList);
175                     // getMatchingServices releases matchingDict
176                     int drive = IOKit.INSTANCE.IOIteratorNext(driveList.getValue());
177                     // Should only match one drive
178                     if (drive != 0) {
179                         // Should be an IOMedia object with a parent
180                         // IOBlockStorageDriver object
181                         // Get the properties from the parent
182                         IntByReference parent = new IntByReference();
183                         if (IOKit.INSTANCE.IOObjectConformsTo(drive, "IOMedia")
184                                 && IOKit.INSTANCE.IORegistryEntryGetParentEntry(drive, "IOService", parent) == 0) {
185                             PointerByReference propsPtr = new PointerByReference();
186                             if (IOKit.INSTANCE.IOObjectConformsTo(parent.getValue(), "IOBlockStorageDriver")
187                                     && IOKit.INSTANCE.IORegistryEntryCreateCFProperties(parent.getValue(), propsPtr,
188                                             CfUtil.ALLOCATOR, 0) == 0) {
189                                 CFMutableDictionaryRef properties = new CFMutableDictionaryRef();
190                                 properties.setPointer(propsPtr.getValue());
191                                 // We now have a properties object with the
192                                 // statistics we need on it. Fetch them
193                                 Pointer statsPtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties,
194                                         CfUtil.getCFString("Statistics"));
195                                 CFDictionaryRef statistics = new CFDictionaryRef();
196                                 statistics.setPointer(statsPtr);
197 
198                                 // Now get the stats we want
199                                 Pointer stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
200                                         CfUtil.getCFString("Operations (Read)"));
201                                 diskStore.setReads(CfUtil.cfPointerToLong(stat));
202                                 stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
203                                         CfUtil.getCFString("Bytes (Read)"));
204                                 diskStore.setReadBytes(CfUtil.cfPointerToLong(stat));
205 
206                                 stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
207                                         CfUtil.getCFString("Operations (Write)"));
208                                 diskStore.setWrites(CfUtil.cfPointerToLong(stat));
209                                 stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
210                                         CfUtil.getCFString("Bytes (Write)"));
211                                 diskStore.setWriteBytes(CfUtil.cfPointerToLong(stat));
212 
213                                 // Total time is in nanoseconds. Add read+write
214                                 // and convert total to ms
215                                 stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
216                                         CfUtil.getCFString("Total Time (Read)"));
217                                 xferTime = CfUtil.cfPointerToLong(stat);
218                                 stat = CoreFoundation.INSTANCE.CFDictionaryGetValue(statistics,
219                                         CfUtil.getCFString("Total Time (Write)"));
220                                 xferTime += CfUtil.cfPointerToLong(stat);
221                                 diskStore.setTransferTime(xferTime / 10000L);
222 
223                                 CfUtil.release(properties);
224                             } else {
225                                 LOG.error("Unable to find block storage driver properties for {}", bsdName);
226                             }
227                             // Now get partitions for this disk.
228                             List<HWPartition> partitions = new ArrayList<>();
229                             if (IOKit.INSTANCE.IORegistryEntryCreateCFProperties(drive, propsPtr, CfUtil.ALLOCATOR,
230                                     0) == 0) {
231                                 CFMutableDictionaryRef properties = new CFMutableDictionaryRef();
232                                 properties.setPointer(propsPtr.getValue());
233                                 // Partitions will match BSD Unit property
234                                 Pointer p = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties,
235                                         CfUtil.getCFString("BSD Unit"));
236                                 CFNumberRef bsdUnit = new CFNumberRef();
237                                 bsdUnit.setPointer(p);
238                                 // Whole disk has 'true' for Whole and 'false'
239                                 // for leaf; store the boolean true
240                                 p = CoreFoundation.INSTANCE.CFDictionaryGetValue(properties,
241                                         CfUtil.getCFString("Whole"));
242                                 CFBooleanRef leaf = new CFBooleanRef();
243                                 leaf.setPointer(p);
244                                 // create a matching dict for BSD Unit
245                                 CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE
246                                         .CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0, null, null);
247                                 CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict,
248                                         CfUtil.getCFString("BSD Unit"), bsdUnit);
249                                 CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, CfUtil.getCFString("Leaf"),
250                                         leaf);
251                                 matchingDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0,
252                                         null, null);
253                                 CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict,
254                                         CfUtil.getCFString("IOPropertyMatch"), propertyDict);
255 
256                                 // search for all IOservices that match the
257                                 // BSD Unit with leaf=true; these are partitions
258                                 IntByReference serviceIterator = new IntByReference();
259                                 IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
260                                 // getMatchingServices releases matchingDict
261                                 CfUtil.release(properties);
262                                 CfUtil.release(propertyDict);
263                                 int sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
264                                 while (sdService != 0) {
265                                     // look up the BSD Name
266                                     String partBsdName = IOKitUtil.getIORegistryStringProperty(sdService, "BSD Name");
267                                     String name = partBsdName;
268                                     String type = "";
269                                     // Get the DiskArbitration dictionary for
270                                     // this partition
271                                     disk = DiskArbitration.INSTANCE.DADiskCreateFromBSDName(CfUtil.ALLOCATOR, session,
272                                             partBsdName);
273                                     if (disk != null) {
274                                         diskInfo = DiskArbitration.INSTANCE.DADiskCopyDescription(disk);
275                                         if (diskInfo != null) {
276                                             // get volume name from its key
277                                             Pointer volumePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
278                                                     CfUtil.getCFString("DAMediaName"));
279                                             type = CfUtil.cfPointerToString(volumePtr);
280                                             volumePtr = CoreFoundation.INSTANCE.CFDictionaryGetValue(diskInfo,
281                                                     CfUtil.getCFString("DAVolumeName"));
282                                             name = CfUtil.cfPointerToString(volumePtr);
283 
284                                             CfUtil.release(diskInfo);
285                                         }
286                                         CfUtil.release(disk);
287                                     }
288                                     partitions.add(new HWPartition(partBsdName, name, type,
289                                             IOKitUtil.getIORegistryStringProperty(sdService, "UUID"),
290                                             IOKitUtil.getIORegistryLongProperty(sdService, "Size"),
291                                             IOKitUtil.getIORegistryIntProperty(sdService, "BSD Major"),
292                                             IOKitUtil.getIORegistryIntProperty(sdService, "BSD Minor"),
293                                             mountPointMap.getOrDefault(partBsdName, "")));
294                                     IOKit.INSTANCE.IOObjectRelease(sdService);
295                                     // iterate
296                                     sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
297                                 }
298                                 IOKit.INSTANCE.IOObjectRelease(serviceIterator.getValue());
299 
300                             } else {
301                                 LOG.error("Unable to find properties for {}", bsdName);
302                             }
303                             Collections.sort(partitions);
304                             diskStore.setPartitions(partitions.toArray(new HWPartition[partitions.size()]));
305                             IOKit.INSTANCE.IOObjectRelease(parent.getValue());
306                         } else {
307                             LOG.error("Unable to find IOMedia device or parent for ", bsdName);
308                         }
309                         IOKit.INSTANCE.IOObjectRelease(drive);
310                     }
311                     IOKit.INSTANCE.IOObjectRelease(driveList.getValue());
312                 }
313                 result.add(diskStore);
314             }
315         }
316         // Close DA session
317         CfUtil.release(session);
318         Collections.sort(result);
319         return result.toArray(new HWDiskStore[result.size()]);
320     }

19  package oshi.hardware.platform.unix.freebsd;
20  
21  import oshi.hardware.common.AbstractGlobalMemory;
22  import oshi.util.ExecutingCommand;
23  import oshi.util.ParseUtil;
24  import oshi.util.platform.unix.freebsd.BsdSysctlUtil;
25  
26  /**
27   * Memory obtained by /proc/meminfo and sysinfo.totalram
28   *
29   * @author alessandro[at]perucchi[dot]org
30   * @author widdis[at]gmail[dot]com
31   */
32  public class FreeBsdGlobalMemory extends AbstractGlobalMemory {
33  
34      private static final long serialVersionUID = 1L;
35  
36      public FreeBsdGlobalMemory() {
37          this.pageSize = BsdSysctlUtil.sysctl("hw.pagesize", 4096L);
38          this.memTotal = BsdSysctlUtil.sysctl("hw.physmem", 0L);
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      @Override
45      protected void updateMeminfo() {
46          long inactive = BsdSysctlUtil.sysctl("vm.stats.vm.v_inactive_count", 0L);
47          long cache = BsdSysctlUtil.sysctl("vm.stats.vm.v_cache_count", 0L);
48          long free = BsdSysctlUtil.sysctl("vm.stats.vm.v_free_count", 0L);
49          this.memAvailable = (inactive + cache + free) * this.pageSize;
50          this.swapPagesIn = BsdSysctlUtil.sysctl("vm.stats.vm.v_swappgsin", 0L);
51          this.swapPagesOut = BsdSysctlUtil.sysctl("vm.stats.vm.v_swappgsout", 0L);
52      }
53  
54      /**
55       * {@inheritDoc}
56       */
57      @Override
58      protected void updateSwap() {
59          this.swapTotal = BsdSysctlUtil.sysctl("vm.swap_total", 0L);
60          String swapInfo = ExecutingCommand.getAnswerAt("swapinfo -k", 1);
61          String[] split = ParseUtil.whitespaces.split(swapInfo);
62          if (split.length < 5) {
63              return;
64          }
65          this.swapUsed = ParseUtil.parseLongOrDefault(split[2], 0L) << 10;
66      }
67  }

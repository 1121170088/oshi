60      private void initVars() {
61          // Get first result
62          Kstat ksp = KstatUtil.kstatLookup("cpu_info", -1, null);
63          // Set values
64          if (ksp != null && KstatUtil.kstatRead(ksp)) {
65              setVendor(KstatUtil.kstatDataLookupString(ksp, "vendor_id"));
66              setName(KstatUtil.kstatDataLookupString(ksp, "brand"));
67              setStepping(KstatUtil.kstatDataLookupString(ksp, "stepping"));
68              setModel(KstatUtil.kstatDataLookupString(ksp, "model"));
69              setFamily(KstatUtil.kstatDataLookupString(ksp, "family"));
70          }
71          setCpu64("64".equals(ExecutingCommand.getFirstAnswer("isainfo -b").trim()));
72          setProcessorID(getProcessorID(getStepping(), getModel(), getFamily()));
73  
74      }

76      /**
77       * Updates logical and physical processor counts from psrinfo
78       */
79      @Override
80      protected void calculateProcessorCounts() {
81          List<Kstat> kstats = KstatUtil.kstatLookupAll("cpu_info", -1, null);
82          Set<String> chipIDs = new HashSet<>();
83          Set<String> coreIDs = new HashSet<>();
84          this.logicalProcessorCount = 0;
85          for (Kstat ksp : kstats) {
86              if (ksp != null && KstatUtil.kstatRead(ksp)) {
87                  this.logicalProcessorCount++;
88                  chipIDs.add(KstatUtil.kstatDataLookupString(ksp, "chip_id"));
89                  coreIDs.add(KstatUtil.kstatDataLookupString(ksp, "core_id"));
90              }
91          }
92  
93          this.physicalPackageCount = chipIDs.size();
94          if (this.physicalPackageCount < 1) {
95              LOG.error("Couldn't find physical package count. Assuming 1.");
96              this.physicalPackageCount = 1;
97          }
98          this.physicalProcessorCount = coreIDs.size();
99          if (this.physicalProcessorCount < 1) {
100             LOG.error("Couldn't find physical processor count. Assuming 1.");
101             this.physicalProcessorCount = 1;
102         }
103         if (this.logicalProcessorCount < 1) {
104             LOG.error("Couldn't find logical processor count. Assuming 1.");
105             this.logicalProcessorCount = 1;
106         }
107     }

109     /**
110      * {@inheritDoc}
111      */
112     @Override
113     public synchronized long[] getSystemCpuLoadTicks() {
114         long[] ticks = new long[TickType.values().length];
115         // Average processor ticks
116         long[][] procTicks = getProcessorCpuLoadTicks();
117         for (int i = 0; i < ticks.length; i++) {
118             for (long[] procTick : procTicks) {
119                 ticks[i] += procTick[i];
120             }
121             ticks[i] /= procTicks.length;
122         }
123         return ticks;
124     }

126     /**
127      * {@inheritDoc}
128      */
129     @Override
130     public double[] getSystemLoadAverage(int nelem) {
131         if (nelem < 1 || nelem > 3) {
132             throw new IllegalArgumentException("Must include from one to three elements.");
133         }
134         double[] average = new double[nelem];
135         int retval = Libc.INSTANCE.getloadavg(average, nelem);
136         if (retval < nelem) {
137             for (int i = Math.max(retval, 0); i < average.length; i++) {
138                 average[i] = -1d;
139             }
140         }
141         return average;
142     }

144     /**
145      * {@inheritDoc}
146      */
147     @Override
148     public long[][] getProcessorCpuLoadTicks() {
149         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
150         int cpu = -1;
151         for (Kstat ksp : KstatUtil.kstatLookupAll("cpu", -1, "sys")) {
152             // This is a new CPU
153             if (++cpu >= ticks.length) {
154                 // Shouldn't happen
155                 break;
156             }
157             if (KstatUtil.kstatRead(ksp)) {
158                 ticks[cpu][TickType.IDLE.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_idle");
159                 ticks[cpu][TickType.SYSTEM.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_kernel");
160                 ticks[cpu][TickType.USER.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_user");
161             }
162         }
163         return ticks;
164     }

166     /**
167      * {@inheritDoc}
168      */
169     @Override
170     public long getSystemUptime() {
171         Kstat ksp = KstatUtil.kstatLookup("unix", 0, "system_misc");
172         if (ksp == null) {
173             return 0L;
174         }
175         // Snap Time is in nanoseconds; divide for seconds
176         return ksp.ks_snaptime / 1000000000L;
177     }

188     /**
189      * Fetches the ProcessorID by encoding the stepping, model, family, and
190      * feature flags.
191      *
192      * @param stepping
193      * @param model
194      * @param family
195      * @return The Processor ID string
196      */
197     private String getProcessorID(String stepping, String model, String family) {
198         List<String> isainfo = ExecutingCommand.runNative("isainfo -v");
199         StringBuilder flags = new StringBuilder();
200         for (String line : isainfo) {
201             if (line.startsWith("32-bit")) {
202                 break;
203             } else if (!line.startsWith("64-bit")) {
204                 flags.append(' ').append(line.trim());
205             }
206         }
207         return createProcessorID(stepping, model, family, ParseUtil.whitespaces.split(flags.toString().toLowerCase()));
208     }

210     /**
211      * {@inheritDoc}
212      */
213     @Override
214     public long getContextSwitches() {
215         long swtch = 0;
216         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/pswitch\\\\|inv_swtch/");
217         for (String s : kstat) {
218             swtch += ParseUtil.parseLastLong(s, 0L);
219         }
220         return swtch > 0 ? swtch : -1L;
221     }

226     @Override
227     public long getInterrupts() {
228         long intr = 0;
229         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/intr/");
230         for (String s : kstat) {
231             intr += ParseUtil.parseLastLong(s, 0L);
232         }
233         return intr > 0 ? intr : -1L;
234     }

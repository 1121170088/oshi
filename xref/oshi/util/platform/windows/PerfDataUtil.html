65      private static final DWORD_PTR PZERO = new DWORD_PTR(0);
66      private static final DWORDByReference PDH_FMT_RAW = new DWORDByReference(new DWORD(Pdh.PDH_FMT_RAW));

69      private static final String HEX_ERROR_FMT = "0x%08X";
70      private static final String LOG_COUNTER_NOT_EXISTS = "Counter does not exist: {}";
71      private static final String LOG_COUNTER_RECREATE = "Removing and re-adding counter: {}";
72      private static final String LOG_COUNTER_QUERY = "Querying counter: {}";

74      // PDH timestamps are 1601 epoch, local time
75      // Constants to convert to UTC millis
76      private static final long EPOCH_DIFF = 11644473600000L;
77      private static final int TZ_OFFSET = TimeZone.getDefault().getOffset(System.currentTimeMillis());

79      // Maps to hold pointers to the relevant counter information
80      private static final Map<PerfCounter, HANDLEByReference> counterMap = new HashMap<>();
81      private static final Map<String, HANDLEByReference> queryMap = new HashMap<>();
82      private static final Set<String> disabledQueries = new HashSet<>();

95              this.counter = counterName;

99      private PerfDataUtil() {
100         // Set up hook to close all queries on shutdown
101         Runtime.getRuntime().addShutdownHook(new Thread() {
102             @Override
103             public void run() {
104                 removeAllCounters();
105             }
106         });

125     /**
126      * Begin monitoring a Performance Data counter
127      *
128      * @param counter
129      *            A PerfCounter object
130      * @return True if the counter was successfully added.
131      */
132     public static boolean addCounterToQuery(PerfCounter counter) {
133         HANDLEByReference q = openQuery(counter.object);
134         if (q == null) {
135             LOG.error("Failed to open a query for PDH object: {}", counter.object);
136             return false;
137         }
138         HANDLEByReference p = new HANDLEByReference();
139         String path = counterPath(counter);
140         if (addCounter(q, path, p)) {
141             counterMap.put(counter, p);
142             return true;
143         }
144         return false;
145     }
146 
147     /**
148      * Stop monitoring a Performance Data counter
149      *
150      * @param counter
151      *            A PerfCounter object
152      * @return True if the counter was successfully removed.
153      */
154     public static boolean removeCounterFromQuery(PerfCounter counter) {
155         if (counterMap.containsKey(counter)) {
156             return WinError.ERROR_SUCCESS == PDH.PdhRemoveCounter(counterMap.get(counter).getValue());
157         }
158         return false;
159     }

161     /**
162      * Update a counter, and all other counters on that object
163      *
164      * @param counter
165      *            The counter whose object to update counters on
166      * @return The timestamp for the update of all the counters, in milliseconds
167      *         since the epoch, or 0 if the update failed
168      */
169     public static long updateQuery(PerfCounter counter) {
170         if (disabledQueries.contains(counter.object)) {
171             return 0L;
172         }
173         if (!queryMap.containsKey(counter.object) || !counterMap.containsKey(counter)) {
174             if (LOG.isErrorEnabled()) {
175                 LOG.error(LOG_COUNTER_NOT_EXISTS, counterPath(counter));
176             }
177             return 0L;
178         }
179         long timestamp = updateQueryTimestamp(queryMap.get(counter.object));
180         if (timestamp == 0L) {
181             LOG.error("Disabling future updates for {}.", counter.object);
182             disabledQueries.add(counter.object);
183             return 0L;
184         }
185         return timestamp;
186     }
187 
188     /**
189      * Update all counters on an object
190      *
191      * @param queryKey
192      *            The counter object to update counters on
193      * @return The timestamp for the update of all the counters, in milliseconds
194      *         since the epoch, or 0 if the update failed
195      */
196     public static long updateQuery(String queryKey) {
197         if (disabledQueries.contains(queryKey) || !queryMap.containsKey(queryKey)) {
198             return 0L;
199         }
200         return updateQueryTimestamp(queryMap.get(queryKey));
201     }
202 
203     /**
204      * Query the raw counter value of a Performance Data counter. Further
205      * mathematical manipulation/conversion is left to the caller.
206      *
207      * @param counter
208      *            The counter to query
209      * @return The raw value of the counter
210      */
211     public static long queryCounter(PerfCounter counter) {
212         if (!queryMap.containsKey(counter.object) || !counterMap.containsKey(counter)) {
213             if (LOG.isErrorEnabled()) {
214                 LOG.error(LOG_COUNTER_NOT_EXISTS, counterPath(counter));
215             }
216             return 0;
217         }
218         if (LOG.isTraceEnabled()) {
219             LOG.trace(LOG_COUNTER_QUERY, counterPath(counter));
220         }
221         long value = queryCounter(counterMap.get(counter));
222         if (value == PdhMsg.PDH_INVALID_HANDLE) {
223             if (LOG.isWarnEnabled()) {
224                 LOG.warn(LOG_COUNTER_RECREATE, counterPath(counter));
225             }
226             removeCounterFromQuery(counter);
227             addCounterToQuery(counter);
228         }
229         return value < 0 ? 0 : value;
230     }

232     /**
233      * Stop monitoring Performance Data counters for a particular queryKey and
234      * release their resources
235      *
236      * @param queryKey
237      *            The counter object to remove counters from
238      */
239     public static void removeAllCounters(String queryKey) {
240         // Remove counters from query
241         Iterator<Entry<PerfCounter, HANDLEByReference>> it = counterMap.entrySet().iterator();
242         while (it.hasNext()) {
243             Entry<PerfCounter, HANDLEByReference> entry = it.next();
244             if (entry.getKey().object.equals(queryKey)) {
245                 PDH.PdhRemoveCounter(entry.getValue().getValue());
246                 it.remove();
247             }
248         }
249         // Remove query
250         HANDLEByReference query = queryMap.get(queryKey);
251         if (query != null) {
252             PDH.PdhCloseQuery(query.getValue());
253         }
254         queryMap.remove(queryKey);
255         disabledQueries.remove(queryKey);
256     }

258     /**
259      * Open a query for the given string, or confirm a query is already open for
260      * that string. Multiple counters may be added to this string, but will all
261      * be queried at the same time.
262      *
263      * @param objectName
264      *            String to associate with the counter. Normally the English PDH
265      *            object name.
266      * @return A handle to the query, or null if an error occurred.
267      */
268     private static HANDLEByReference openQuery(String objectName) {
269         if (queryMap.containsKey(objectName)) {
270             return queryMap.get(objectName);
271         }
272         HANDLEByReference q = new HANDLEByReference();
273         if (openQuery(q)) {
274             queryMap.put(objectName, q);
275             return q;
276         }
277         return null;
278     }

280     /**
281      * Build a counter path
282      *
283      * counter A Counter object with the object, counter, and (optional)
284      * instance
285      *
286      * @return A string representing the complete counter
287      */
288     private static String counterPath(PerfCounter counter) {
289         StringBuilder sb = new StringBuilder();
290         sb.append('\\').append(counter.object);
291         if (counter.instance != null) {
292             sb.append('(').append(counter.instance).append(')');
293         }
294         sb.append('\\').append(counter.counter);
295         return sb.toString();
296     }

298     /**
299      * Stop monitoring all Performance Data counters and release their resources
300      */
301     public static void removeAllCounters() {
302         Set<String> queries = new HashSet<>(queryMap.keySet());
303         for (String query : queries) {
304             removeAllCounters(query);
305         }
306     }

308     /**
309      * Open a pdh query
310      *
311      * @param q
312      *            pointer to the query
313      * @return true if successful
314      */
315     private static boolean openQuery(HANDLEByReference q) {
316         int pdhOpenQueryError = PDH.PdhOpenQuery(null, PZERO, q);
317         if (pdhOpenQueryError != WinError.ERROR_SUCCESS && LOG.isErrorEnabled()) {
318             LOG.error("Failed to open PDH Query. Error code: {}", String.format(HEX_ERROR_FMT, pdhOpenQueryError));
319         }
320         return pdhOpenQueryError == WinError.ERROR_SUCCESS;
321     }

323     /**
324      * Adds a pdh counter to a query
325      *
326      * @param query
327      *            Pointer to the query to add the counter
328      * @param path
329      *            String name of the PerfMon counter
330      * @param p
331      *            Pointer to the counter
332      * @return
333      */
334     private static boolean addCounter(WinNT.HANDLEByReference query, String path, WinNT.HANDLEByReference p) {
335         int pdhAddCounterError;
336         if (IS_VISTA_OR_GREATER) {
337             // Localized version for Vista+
338             pdhAddCounterError = PDH.PdhAddEnglishCounter(query.getValue(), path, PZERO, p);
339         } else {
340             // Only will work for English counters
341             pdhAddCounterError = PDH.PdhAddCounter(query.getValue(), path, PZERO, p);
342         }
343         if (pdhAddCounterError != WinError.ERROR_SUCCESS && LOG.isWarnEnabled()) {
344             LOG.warn("Failed to add PDH Counter: {}, Error code: {}", path,
345                     String.format(HEX_ERROR_FMT, pdhAddCounterError));
346         }
347         return pdhAddCounterError == WinError.ERROR_SUCCESS;
348     }

350     /**
351      * Get value of pdh counter
352      *
353      * @param counter
354      *            The counter to get the value of
355      * @return long value of the counter, or negative value representing an
356      *         error code
357      */
358     private static long queryCounter(WinNT.HANDLEByReference counter) {
359         PDH_RAW_COUNTER counterValue = new PDH_RAW_COUNTER();
360         int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
361         if (ret != WinError.ERROR_SUCCESS) {
362             if (LOG.isWarnEnabled()) {
363                 LOG.warn("Failed to get counter. Error code: {}", String.format(HEX_ERROR_FMT, ret));
364             }
365             // Return error code. Will be a negative value
366             return ret;
367         }
368         return counterValue.FirstValue;
369     }

371     /**
372      * Update a query and get the timestamp
373      *
374      * @param query
375      *            The query to update all counters in
376      * @return The update timestamp of the first counter in the query
377      */
378     private static long updateQueryTimestamp(WinNT.HANDLEByReference query) {
379         LONGLONGByReference pllTimeStamp = new LONGLONGByReference();
380         int ret;
381         if (IS_VISTA_OR_GREATER) {
382             ret = PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp);
383         } else {
384             ret = PDH.PdhCollectQueryData(query.getValue());
385         }
386         // Due to race condition, initial update may fail with PDH_NO_DATA.
387         int retries = 0;
388         while (ret == PdhMsg.PDH_NO_DATA && retries++ < 3) {
389             // Exponential fallback.
390             Util.sleep(1 << retries);
391             if (IS_VISTA_OR_GREATER) {
392                 ret = PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp);
393             } else {
394                 ret = PDH.PdhCollectQueryData(query.getValue());
395             }
396         }
397         if (ret != WinError.ERROR_SUCCESS) {
398             if (LOG.isWarnEnabled()) {
399                 LOG.warn("Failed to update counter. Error code: {}", String.format(HEX_ERROR_FMT, ret));
400             }
401             return 0;
402         }
403         // Perf Counter timestamp is in local time
404         if (IS_VISTA_OR_GREATER) {
405             return filetimeToUtcMs(pllTimeStamp.getValue().longValue(), true);
406         } else {
407             // Approximate timestamp
408             return System.currentTimeMillis();
409         }

413     /**
414      * Convert a long representing filetime (100-ns since 1601 epoch) to ms
415      * since 1970 epoch
416      *
417      * @param filetime
418      *            A 64-bit value equivalent to FILETIME
419      * @param local
420      *            True if converting from a local filetime (PDH counter); false
421      *            if already UTC (WMI PerfRawData classes)
422      * @return Equivalent milliseconds since the epoch
423      */
424     public static long filetimeToUtcMs(long filetime, boolean local) {
425         return filetime / 10000L - EPOCH_DIFF - (local ? TZ_OFFSET : 0L);
426     }

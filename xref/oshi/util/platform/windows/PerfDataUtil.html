43  /**
44   * Helper class to centralize the boilerplate portions of PDH counter setup and
45   * allow applications to easily add, query, and remove counters.
46   * 
47   * @author widdis[at]gmail[dot]com
48   */
49  public class PerfDataUtil {
50      private static final Logger LOG = LoggerFactory.getLogger(PerfDataUtil.class);
51  
52      private static final DWORD_PTR PZERO = new DWORD_PTR(0);
53      private static final DWORDByReference PDH_FMT_RAW = new DWORDByReference(new DWORD(Pdh.PDH_FMT_RAW));
54      private static final PDH_RAW_COUNTER counterValue = new PDH_RAW_COUNTER();
55      private static final Pdh PDH = Pdh.INSTANCE;
56  
57      private static final String HEX_ERROR_FMT = "0x%08X";
58      private static final String LOG_COUNTER_NOT_EXISTS = "Counter does not exist: {}";
59  
60      // Maps to hold pointers to the relevant counter information
61      private static final Map<String, HANDLEByReference> counterMap = new HashMap<>();
62      private static final Map<String, HANDLEByReference> queryMap = new HashMap<>();
63      private static final Set<String> disabledCounters = new HashSet<>();
64  
65      // Regexp to match PDH counter string
66      // Format is \Path(Instance)\Counter or \Path\Counter
67      private static Pattern COUNTER_PATTERN = Pattern.compile("\\\\(.*?)(\\(.*\\))?\\\\(.*)");
68  
69  
70      private PerfDataUtil() {
71          // Set up hook to close all queries on shutdown
72          Runtime.getRuntime().addShutdownHook(new Thread() {
73              @Override
74              public void run() {
75                  for (HANDLEByReference query : queryMap.values()) {
76                      PDH.PdhCloseQuery(query.getValue());
77                  }
78              }
79          });
80      }

82      /**
83       * Translate an English counter path to its locale-specific string
84       * 
85       * @param englishPath
86       *            The english path of the counter
87       * @return The path of the counter in the machine's locale
88       */
89      private static String localizeCounterPath(String englishPath) {
90          Matcher match = COUNTER_PATTERN.matcher(englishPath);
91          if (match.matches()) {
92              StringBuilder sb = new StringBuilder();
93              sb.append('\\');
94              sb.append(PdhUtil.PdhLookupPerfNameByIndex(null, PdhUtil.PdhLookupPerfIndexByEnglishName(match.group(1))));
95              if (match.group(2) != null) {
96                  sb.append(match.group(2));
97              }
98              sb.append('\\');
99              sb.append(PdhUtil.PdhLookupPerfNameByIndex(null, PdhUtil.PdhLookupPerfIndexByEnglishName(match.group(3))));
100             return sb.toString();
101         }
102         return englishPath;
103     }

105     /**
106      * Report if a performance counter is being monitored
107      * 
108      * @param counterString
109      *            The counter to monitor
110      * @return True if the counter already exists
111      */
112     public static boolean isCounter(String counterString) {
113         return counterMap.containsKey(counterString);
114     }
115 
116     /**
117      * Begin monitoring a Performance Data counter
118      * 
119      * @param counterString
120      *            The counter to monitor
121      * @return True if the counter has been successfully added or already exists
122      */
123     public static boolean addCounter(String counterString) {
124         if (queryMap.containsKey(counterString)) {
125             LOG.warn("Counter already exists: {}", counterString);
126             return true;
127         }
128         HANDLEByReference q = new HANDLEByReference();
129         if (openQuery(q)) {
130             HANDLEByReference p = new HANDLEByReference();
131             addCounter(q, localizeCounterPath(counterString), p);
132             counterMap.put(counterString, p);
133             queryMap.put(counterString, q);
134             return true;
135         }
136         return false;
137     }

139     /**
140      * Begin monitoring a 2D array of Performance Data counters
141      * 
142      * @param name
143      *            A unique name that will always correspond to the same String
144      *            array
145      * @param counterStringArray
146      *            A 2D array of string counter names to monitor
147      * @return True if the counters have been successfully added or already
148      *         exist
149      */
150     public static boolean addCounter2DArray(String name, String[][] counterStringArray) {
151         if (queryMap.containsKey(name)) {
152             LOG.warn("Counters already exists: {}", name);
153             return true;
154         }
155         if (counterStringArray.length == 0 || counterStringArray[0].length == 0) {
156             LOG.error("This array has a zero dimension: {}", name);
157             return false;
158         }
159         HANDLEByReference q = new HANDLEByReference();
160         if (openQuery(q)) {
161             for (int i = 0; i < counterStringArray.length; i++) {
162                 for (int j = 0; j < counterStringArray[i].length; j++) {
163                     if (counterStringArray[i][j] != null) {
164                         HANDLEByReference p = new HANDLEByReference();
165                         addCounter(q, counterStringArray[i][j], p);
166                         counterMap.put(counterStringArray[i][j], p);
167                     }
168                 }
169             }
170             queryMap.put(name, q);
171             return true;
172         }
173         return false;
174     }

176     /**
177      * Stop monitoring a Performance Data counter
178      * 
179      * @param counterString
180      *            The counter to stop monitoring
181      */
182     public static void removeCounter(String counterString) {
183         if (queryMap.containsKey(counterString)) {
184             PDH.PdhCloseQuery(queryMap.get(counterString).getValue());
185             counterMap.remove(counterString);
186             queryMap.remove(counterString);
187             disabledCounters.remove(counterString);
188         } else {
189             LOG.warn(LOG_COUNTER_NOT_EXISTS, counterString);
190         }
191     }
192 
193     /**
194      * Query the raw counter value of a Performance Data counter. Further
195      * mathematical manipulation/conversion is left to the caller.
196      * 
197      * @param counterString
198      *            The counter to query
199      * @return The raw value of the counter
200      */
201     public static long queryCounter(String counterString) {
202         if (!queryMap.containsKey(counterString)) {
203             LOG.error(LOG_COUNTER_NOT_EXISTS, counterString);
204             return 0;
205         }
206         if (!disabledCounters.contains(counterString) && !updateCounters(queryMap.get(counterString))) {
207             LOG.error("Disabling future updates for {}. Call removeCounter and addCounter again to reset.",
208                     counterString);
209             disabledCounters.add(counterString);
210             return 0L;
211         }
212         return queryCounter(counterMap.get(counterString));
213     }

215     /**
216      * Get the timestamp of a raw counter value of a Performance Data counter.
217      * Does not update the counter, and should normally be called after querying
218      * the counter.
219      * 
220      * @param counterString
221      *            The counter to query
222      * @return The raw value of the counter
223      */
224     public static long queryCounterTimestamp(String counterString) {
225         if (!queryMap.containsKey(counterString)) {
226             LOG.error(LOG_COUNTER_NOT_EXISTS, counterString);
227             return 0;
228         }
229         return queryCounterTimestamp(counterMap.get(counterString)).toDWordLong().longValue() / 10000L;
230     }

232     /**
233      * Query the raw counter value of an array of Performance Data counters.
234      * Further mathematical manipulation/conversion is left to the caller.
235      * 
236      * @param name
237      *            A unique name that will always correspond to the same String
238      *            array
239      * @param counterStringArray
240      *            A 2D array of string counter names to monitor
241      * @return The raw values of the counters corresponding to the string
242      */
243     public static long[][] queryCounter2DArray(String name, String[][] counterStringArray) {
244         if (!queryMap.containsKey(name)) {
245             LOG.error(LOG_COUNTER_NOT_EXISTS, name);
246             return new long[0][0];
247         }
248         if (counterStringArray.length == 0 || counterStringArray[0].length == 0) {
249             LOG.error("This array has a zero dimension: {}", name);
250             return new long[0][0];
251         }
252         updateCounters(queryMap.get(name));
253         long[][] values = new long[counterStringArray.length][counterStringArray[0].length];
254         for (int i = 0; i < counterStringArray.length; i++) {
255             for (int j = 0; j < counterStringArray[i].length; j++) {
256                 if (counterStringArray[i][j] != null) {
257                     values[i][j] = queryCounter(counterMap.get(counterStringArray[i][j]));
258                 }
259             }
260         }
261         return values;
262     }

264     /**
265      * Open a pdh query
266      * 
267      * @param p
268      *            pointer to the query
269      * @return true if successful
270      */
271     private static boolean openQuery(HANDLEByReference p) {
272         int pdhOpenQueryError = PDH.PdhOpenQuery(null, PZERO, p);
273         if (pdhOpenQueryError != WinError.ERROR_SUCCESS && LOG.isErrorEnabled()) {
274             LOG.error("Failed to open PDH Query. Error code: {}", String.format(HEX_ERROR_FMT, pdhOpenQueryError));
275         }
276         return pdhOpenQueryError == WinError.ERROR_SUCCESS;
277     }

279     /**
280      * Adds a pdh counter to a query
281      * 
282      * @param query
283      *            Pointer to the query to add the counter
284      * @param path
285      *            String name of the PerfMon counter
286      * @param p
287      *            Pointer to the counter
288      */
289     private static void addCounter(WinNT.HANDLEByReference query, String path, WinNT.HANDLEByReference p) {
290         int pdhAddCounterError = PDH.PdhAddEnglishCounter(query.getValue(), path, PZERO, p);
291         if (pdhAddCounterError != WinError.ERROR_SUCCESS && LOG.isErrorEnabled()) {
292             LOG.error("Failed to add PDH Counter: {}, Error code: {}", path,
293                     String.format(HEX_ERROR_FMT, pdhAddCounterError));
294         }
295     }

297     /**
298      * Update counters to values since the last call
299      * 
300      * @param query
301      *            The query whose counters to update
302      * @return True if successful
303      */
304     private static boolean updateCounters(WinNT.HANDLEByReference query) {
305         int ret = PDH.PdhCollectQueryData(query.getValue());
306         if (ret != WinError.ERROR_SUCCESS) {
307             if (LOG.isErrorEnabled()) {
308                 LOG.error("Failed to update counters. Error code: {}", String.format(HEX_ERROR_FMT, ret));
309             }
310             return false;
311         }
312         return true;
313     }

315     /**
316      * Get value of pdh counter
317      * 
318      * @param counter
319      *            The counter to get the value of
320      * @return long value of the counter
321      */
322     private static long queryCounter(WinNT.HANDLEByReference counter) {
323         int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
324         if (ret != WinError.ERROR_SUCCESS) {
325             if (LOG.isWarnEnabled()) {
326                 LOG.warn("Failed to get counter. Error code: {}", String.format(HEX_ERROR_FMT, ret));
327             }
328             return 0L;
329         }
330         return counterValue.FirstValue;
331     }

333     /**
334      * Get timestamp of pdh counter
335      * 
336      * @param counter
337      *            The counter to get the value of
338      * @return FILETIME value of the counter. This is in 100-ns increments and
339      *         uses the 1601 Epoch.
340      */
341     private static FILETIME queryCounterTimestamp(WinNT.HANDLEByReference counter) {
342         int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
343         if (ret != WinError.ERROR_SUCCESS) {
344             if (LOG.isWarnEnabled()) {
345                 LOG.warn("Failed to get counter. Error code: {}", String.format(HEX_ERROR_FMT, ret));
346             }
347             return new FILETIME();
348         }
349         return counterValue.TimeStamp;
350     }

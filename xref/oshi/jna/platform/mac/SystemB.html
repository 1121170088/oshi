132      * path argument is the path name of any file or directory within the
133      * mounted file system. The buf argument is a pointer to a statfs structure.

136         public int f_bsize; /* fundamental file system block size */
137         public int f_iosize; /* optimal transfer block size */
138         public long f_blocks; /* total data blocks in file system */
139         public long f_bfree; /* free blocks in fs */
140         public long f_bavail; /* free blocks avail to non-superuser */
141         public long f_files; /* total file nodes in file system */
142         public long f_ffree; /* free file nodes in fs */
143         public int[] f_fsid = new int[2]; /* file system id */
144         public int f_owner; /* user that mounted the filesystem */
145         public int f_type; /* type of filesystem */
146         public int f_flags; /* copy of mount exported flags */
147         public int f_fssubtype; /* fs sub-type (flavor) */

150         /* directory on which mounted */

390     /**
391      * The getfsstat() function returns information about all mounted file
392      * systems. The buf argument is a pointer to an array of statfs structures.
393      *
394      * Fields that are undefined for a particular file system are set to -1. The
395      * buffer is filled with an array of statfs structures, one for each mounted
396      * file system up to the size specified by bufsize.
397      *
398      * @param buf
399      *            Array of statfs structures that will be filled with results.
400      *            If buf is given as NULL, getfsstat() returns just the number
401      *            of mounted file systems.
402      * @param bufsize
403      *            Size of the buffer to fill
404      * @param flags
405      *            If flags is set to MNT_NOWAIT, getfsstat() will directly
406      *            return the information retained in the kernel to avoid delays
407      *            caused by waiting for updated information from a file system
408      *            that is perhaps temporarily unable to respond. Some of the
409      *            information returned may be out of date, however; if flags is
410      *            set to MNT_WAIT or MNT_DWAIT instead, getfsstat() will request
411      *            updated information from each mounted filesystem before
412      *            returning.
413      * @return Upon successful completion, the number of statfs structures is
414      *         returned. Otherwise, -1 is returned and the global variable errno
415      *         is set to indicate the error.
416      */
417     int getfsstat64(Statfs[] buf, int bufsize, int flags);

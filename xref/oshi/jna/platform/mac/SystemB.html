75      /**
76       * The statfs() routine returns information about a mounted file system. The
77       * path argument is the path name of any file or directory within the
78       * mounted file system. The buf argument is a pointer to a statfs structure.
79       */
80      class Statfs extends Structure {
81          public int f_bsize; /* fundamental file system block size */
82          public int f_iosize; /* optimal transfer block size */
83          public long f_blocks; /* total data blocks in file system */
84          public long f_bfree; /* free blocks in fs */
85          public long f_bavail; /* free blocks avail to non-superuser */
86          public long f_files; /* total file nodes in file system */
87          public long f_ffree; /* free file nodes in fs */
88          public int[] f_fsid = new int[2]; /* file system id */
89          public int f_owner; /* user that mounted the filesystem */
90          public int f_type; /* type of filesystem */
91          public int f_flags; /* copy of mount exported flags */
92          public int f_fssubtype; /* fs sub-type (flavor) */
93          /* fs type name */
94          public byte[] f_fstypename = new byte[MFSTYPENAMELEN];
95          /* directory on which mounted */
96          public byte[] f_mntonname = new byte[MAXPATHLEN];
97          /* mounted filesystem */
98          public byte[] f_mntfromname = new byte[MAXPATHLEN];
99          public int[] f_reserved = new int[8]; /* For future use */
100

102         protected List<String> getFieldOrder() {
103             return Arrays.asList(new String[] { "f_bsize", "f_iosize", "f_blocks", "f_bfree", "f_bavail", "f_files",

325     /**
326      * The getfsstat() function returns information about all mounted file
327      * systems. The buf argument is a pointer to an array of statfs structures.
328      * 
329      * Fields that are undefined for a particular file system are set to -1. The
330      * buffer is filled with an array of statfs structures, one for each mounted
331      * file system up to the size specified by bufsize.
332      * 
333      * @param buf
334      *            Array of statfs structures that will be filled with results.
335      *            If buf is given as NULL, getfsstat() returns just the number
336      *            of mounted file systems.
337      * @param bufsize
338      *            Size of the buffer to fill
339      * @param flags
340      *            If flags is set to MNT_NOWAIT, getfsstat() will directly
341      *            return the information retained in the kernel to avoid delays
342      *            caused by waiting for updated information from a file system
343      *            that is perhaps temporarily unable to respond. Some of the
344      *            information returned may be out of date, however; if flags is
345      *            set to MNT_WAIT or MNT_DWAIT instead, getfsstat() will request
346      *            updated information from each mounted filesystem before
347      *            returning.
348      * @return Upon successful completion, the number of statfs structures is
349      *         returned. Otherwise, -1 is returned and the global variable errno
350      *         is set to indicate the error.
351      */
352     int getfsstat64(Statfs[] buf, int bufsize, int flags);

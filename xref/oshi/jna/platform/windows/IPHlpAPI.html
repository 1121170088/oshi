18  package oshi.jna.platform.windows;
19  
20  import java.util.Arrays;
21  import java.util.List;
22  
23  import com.sun.jna.Library;
24  import com.sun.jna.Native;
25  import com.sun.jna.Structure;
26  
27  /**
28   * Windows IP Helper API. This class should be considered non-API as it may be
29   * removed if/when its code is incorporated public into the JNA project.
30   * 
31   * @author widdis[at]gmail[dot]com
32   */
33  public interface IPHlpAPI extends Library {
34      IPHlpAPI INSTANCE = (IPHlpAPI) Native.loadLibrary("IPHlpAPI", IPHlpAPI.class);
35  
36      int MAX_INTERFACE_NAME_LEN = 256;
37      int MAXLEN_IFDESCR = 256;
38      int MAXLEN_PHYSADDR = 8;
39  
40      class MIB_IFROW extends Structure {
41          public char[] wszName = new char[MAX_INTERFACE_NAME_LEN];
42          public int dwIndex;
43          public int dwType;
44          public int dwMtu;
45          public int dwSpeed;
46          public int dwPhysAddrLen;
47          public byte[] bPhysAddr = new byte[MAXLEN_PHYSADDR];
48          public int dwAdminStatus;
49          public int dwOperStatus;
50          public int dwLastChange;
51          public int dwInOctets;
52          public int dwInUcastPkts;
53          public int dwInNUcastPkts;
54          public int dwInDiscards;
55          public int dwInErrors;
56          public int dwInUnknownProtos;
57          public int dwOutOctets;
58          public int dwOutUcastPkts;
59          public int dwOutNUcastPkts;
60          public int dwOutDiscards;
61          public int dwOutErrors;
62          public int dwOutQLen;
63          public int dwDescrLen;
64          public byte[] bDescr = new byte[MAXLEN_IFDESCR];
65  
66          @Override
67          protected List<String> getFieldOrder() {
68              return Arrays.asList(new String[] { "wszName", "dwIndex", "dwType", "dwMtu", "dwSpeed", "dwPhysAddrLen",
69                      "bPhysAddr", "dwAdminStatus", "dwOperStatus", "dwLastChange", "dwInOctets", "dwInUcastPkts",
70                      "dwInNUcastPkts", "dwInDiscards", "dwInErrors", "dwInUnknownProtos", "dwOutOctets",
71                      "dwOutUcastPkts", "dwOutNUcastPkts", "dwOutDiscards", "dwOutErrors", "dwOutQLen", "dwDescrLen",
72                      "bDescr" });
73          }
74      }
75  
76      /**
77       * The GetIfEntry function retrieves information for the specified interface
78       * on the local computer
79       * 
80       * @param pIfRow
81       *            A pointer to a MIB_IFROW structure that, on successful return,
82       *            receives information for an interface on the local computer.
83       *            On input, set the dwIndex member of MIB_IFROW to the index of
84       *            the interface for which to retrieve information.
85       * @return If the function succeeds, the return value is NO_ERROR.
86       */
87      int GetIfEntry(MIB_IFROW pIfRow);
88  
89  }

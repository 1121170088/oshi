19  package oshi.jna.platform.windows;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import com.sun.jna.Native;
25  import com.sun.jna.Structure;
26  import com.sun.jna.platform.win32.BaseTSD.SIZE_T;
27  import com.sun.jna.platform.win32.WinDef.DWORD;
28  
29  /**
30   * Windows PSAPI. This class should be considered non-API as it may be removed
31   * if/when its code is incorporated public into the JNA project.
32   *
33   * @author widdis[at]gmail[dot]com
34   */
35  public interface Psapi extends com.sun.jna.platform.win32.Psapi {
36      Psapi INSTANCE = Native.loadLibrary("Psapi", Psapi.class);
37  
38      // TODO: Submit this change to JNA Psapi class
39      class PERFORMANCE_INFORMATION extends Structure {
40          public DWORD cb;
41          public SIZE_T CommitTotal;
42          public SIZE_T CommitLimit;
43          public SIZE_T CommitPeak;
44          public SIZE_T PhysicalTotal;
45          public SIZE_T PhysicalAvailable;
46          public SIZE_T SystemCache;
47          public SIZE_T KernelTotal;
48          public SIZE_T KernelPaged;
49          public SIZE_T KernelNonpaged;
50          public SIZE_T PageSize;
51          public DWORD HandleCount;
52          public DWORD ProcessCount;
53          public DWORD ThreadCount;
54  
55          @Override
56          protected List<String> getFieldOrder() {
57              return Arrays.asList(new String[] { "cb", "CommitTotal", "CommitLimit", "CommitPeak", "PhysicalTotal",
58                      "PhysicalAvailable", "SystemCache", "KernelTotal", "KernelPaged", "KernelNonpaged", "PageSize",
59                      "HandleCount", "ProcessCount", "ThreadCount" });
60          }
61      }
62  
63      /**
64       * Retrieves the performance values contained in the
65       * {@link PERFORMANCE_INFORMATION} structure.
66       *
67       * @param pPerformanceInformation
68       *            A pointer to a {@link PERFORMANCE_INFORMATION} structure that
69       *            receives the performance information.
70       * @param cb
71       *            The size of the {@link PERFORMANCE_INFORMATION} structure, in
72       *            bytes.
73       * @return If the function succeeds, the return value is TRUE. If the
74       *         function fails, the return value is FALSE. To get extended error
75       *         information, call GetLastError.
76       */
77      boolean GetPerformanceInfo(PERFORMANCE_INFORMATION pPerformanceInformation, int cb);
78  }

19  package oshi.jna.platform.windows;
20  
21  import com.sun.jna.Memory;
22  import com.sun.jna.Native;
23  import com.sun.jna.ptr.IntByReference;
24  
25  /**
26   * Windows Cfgmgr32Util. This class should be considered non-API as it may be
27   * removed if/when its code is incorporated into the JNA project.
28   *
29   * @author widdis[at]gmail[dot]com
30   */
31  public abstract class Cfgmgr32Util {
32      @SuppressWarnings("serial")
33      public static class Cfgmgr32Exception extends RuntimeException {
34          private final int errorCode;
35  
36          public Cfgmgr32Exception(int errorCode) {
37              this.errorCode = errorCode;
38          }
39  
40          public int getErrorCode() {
41              return errorCode;
42          }
43      }
44  
45      /**
46       * Utility method to call Cfgmgr32's CM_Get_Device_ID that allocates the
47       * required memory for the Buffer parameter based on the type mapping used,
48       * calls to CM_Get_Device_ID, and returns the received string.
49       * 
50       * @param devInst
51       *            Caller-supplied device instance handle that is bound to the
52       *            local machine.
53       * @return The device instance ID string.
54       * @throws Cfgmgr32Exception
55       */
56      public static String CM_Get_Device_ID(int devInst) throws Cfgmgr32Exception {
57          int charToBytes = Boolean.getBoolean("w32.ascii") ? 1 : Native.WCHAR_SIZE;
58  
59          // Get Device ID character count
60          IntByReference pulLen = new IntByReference();
61          int ret = Cfgmgr32.INSTANCE.CM_Get_Device_ID_Size(pulLen, devInst, 0);
62          if (ret != Cfgmgr32.CR_SUCCESS) {
63              throw new Cfgmgr32Exception(ret);
64          }
65  
66          // Add one to length to allow null terminator
67          Memory buffer = new Memory((pulLen.getValue() + 1) * charToBytes);
68          // Zero the buffer (including the extra character)
69          buffer.clear();
70          // Fetch the buffer specifying only the current length
71          ret = Cfgmgr32.INSTANCE.CM_Get_Device_ID(devInst, buffer, pulLen.getValue(), 0);
72          // In the unlikely event the device id changes this might not be big
73          // enough, try again. This happens rarely enough one retry should be
74          // sufficient.
75          if (ret == Cfgmgr32.CR_BUFFER_SMALL) {
76              ret = Cfgmgr32.INSTANCE.CM_Get_Device_ID_Size(pulLen, devInst, 0);
77              if (ret != Cfgmgr32.CR_SUCCESS) {
78                  throw new Cfgmgr32Exception(ret);
79              }
80              buffer = new Memory((pulLen.getValue() + 1) * charToBytes);
81              buffer.clear();
82              ret = Cfgmgr32.INSTANCE.CM_Get_Device_ID(devInst, buffer, pulLen.getValue(), 0);
83          }
84          // If we still aren't successful throw an exception
85          if (ret != Cfgmgr32.CR_SUCCESS) {
86              throw new Cfgmgr32Exception(ret);
87          }
88          // Convert buffer to Java String
89          if (charToBytes == 1) {
90              return buffer.getString(0);
91          } else {
92              return buffer.getWideString(0);
93          }
94      }
95  }

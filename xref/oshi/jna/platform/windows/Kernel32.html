24  package oshi.jna.platform.windows;
25  
26  import com.sun.jna.Native;
27  
28  /**
29   * Kernel32 and WinNT info submitted to JNA as #1050. This class should be
30   * considered non-API as it may be removed if/when its code is incorporated into
31   * the JNA project.
32   */
33  public interface Kernel32 extends com.sun.jna.platform.win32.Kernel32 {
34      Kernel32 INSTANCE = Native.load("Kernel32", Kernel32.class);
35  
36      /*
37       * WIN32_WINNT version constants
38       */
39      short WIN32_WINNT_NT4 = 0x0400; // Windows NT 4.0
40      short WIN32_WINNT_WIN2K = 0x0500; // Windows 2000
41      short WIN32_WINNT_WINXP = 0x0501; // Windows XP
42      short WIN32_WINNT_WS03 = 0x0502; // Windows Server 2003
43      short WIN32_WINNT_WIN6 = 0x0600; // Windows Vista
44      short WIN32_WINNT_VISTA = 0x0600; // Windows Vista
45      short WIN32_WINNT_WS08 = 0x0600; // Windows Server 2008
46      short WIN32_WINNT_LONGHORN = 0x0600; // Windows Vista
47      short WIN32_WINNT_WIN7 = 0x0601; // Windows 7
48      short WIN32_WINNT_WIN8 = 0x0602; // Windows 8
49      short WIN32_WINNT_WINBLUE = 0x0603; // Windows 8.1
50      short WIN32_WINNT_WINTHRESHOLD = 0x0A00; // Windows 10
51      short WIN32_WINNT_WIN10 = 0x0A00; // Windows 10
52  
53      /**
54       * Compares a set of operating system version requirements to the
55       * corresponding values for the currently running version of the system.This
56       * function is subject to manifest-based behavior.
57       * 
58       * @param lpVersionInformation
59       *            A pointer to an OSVERSIONINFOEX structure containing the
60       *            operating system version requirements to compare. The
61       *            dwTypeMask parameter indicates the members of this structure
62       *            that contain information to compare.
63       * 
64       *            You must set the dwOSVersionInfoSize member of this structure
65       *            to sizeof(OSVERSIONINFOEX). You must also specify valid data
66       *            for the members indicated by dwTypeMask. The function ignores
67       *            structure members for which the corresponding dwTypeMask bit
68       *            is not set.
69       * @param dwTypeMask
70       *            A mask that indicates the members of the OSVERSIONINFOEX
71       *            structure to be tested.
72       * @param dwlConditionMask
73       *            The type of comparison to be used for each lpVersionInfo
74       *            member being compared. To build this value, call the
75       *            VerSetConditionMask function once for each OSVERSIONINFOEX
76       *            member being compared.
77       * @return If the currently running operating system satisfies the specified
78       *         requirements, the return value is a nonzero value.
79       * 
80       *         If the current system does not satisfy the requirements, the
81       *         return value is zero and GetLastError returns
82       *         ERROR_OLD_WIN_VERSION.
83       * 
84       *         If the function fails, the return value is zero and GetLastError
85       *         returns an error code other than ERROR_OLD_WIN_VERSION.
86       */
87      boolean VerifyVersionInfoW(OSVERSIONINFOEX lpVersionInformation, int dwTypeMask, long dwlConditionMask);
88  
89      /**
90       * Sets the bits of a 64-bit value to indicate the comparison operator to
91       * use for a specified operating system version attribute. This function is
92       * used to build the dwlConditionMask parameter of the VerifyVersionInfo
93       * function.
94       * 
95       * @param conditionMask
96       *            A value to be passed as the dwlConditionMask parameter of the
97       *            VerifyVersionInfo function. The function stores the comparison
98       *            information in the bits of this variable.
99       * 
100      *            Before the first call to VerSetCondition, initialize this
101      *            variable to zero. For subsequent calls, pass in the variable
102      *            used in the previous call.
103      * @param typeMask
104      *            A mask that indicates the member of the OSVERSIONINFOEX
105      *            structure whose comparison operator is being set. This value
106      *            corresponds to one of the bits specified in the dwTypeMask
107      *            parameter for the VerifyVersionInfo function.
108      * @param condition
109      *            The operator to be used for the comparison. The
110      *            VerifyVersionInfo function uses this operator to compare a
111      *            specified attribute value to the corresponding value for the
112      *            currently running system.
113      * @return The function returns the condition mask value.
114      */
115     long VerSetConditionMask(long conditionMask, int typeMask, byte condition);
116 
117 }

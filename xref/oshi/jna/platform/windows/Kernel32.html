19  package oshi.jna.platform.windows;
20  
21  import java.util.Arrays;
22  import java.util.List;
23  
24  import com.sun.jna.Native;
25  import com.sun.jna.Structure;
26  
27  /**
28   * Windows Kernel32
29   *
30   * @author widdis[at]gmail[dot]com
31   */
32  public interface Kernel32 extends com.sun.jna.platform.win32.Kernel32 {
33      Kernel32 INSTANCE = Native.loadLibrary("Kernel32", Kernel32.class);
34  
35      class IO_COUNTERS extends Structure {
36          public long ReadOperationCount;
37          public long WriteOperationCount;
38          public long OtherOperationCount;
39          public long ReadTransferCount;
40          public long WriteTransferCount;
41          public long OtherTransferCount;
42  
43          @Override
44          protected List<String> getFieldOrder() {
45              return Arrays.asList(new String[] { "ReadOperationCount", "WriteOperationCount", "OtherOperationCount",
46                      "ReadTransferCount", "WriteTransferCount", "OtherTransferCount" });
47          }
48      }
49  
50      /**
51       * Retrieves timing information for the specified process.
52       *
53       * @param hProcess
54       *            A handle to the process whose timing information is sought.
55       *            The handle must have the PROCESS_QUERY_INFORMATION or
56       *            PROCESS_QUERY_LIMITED_INFORMATION access right.
57       * @param lpCreationTime
58       *            A pointer to a FILETIME structure that receives the creation
59       *            time of the process.
60       * @param lpExitTime
61       *            A pointer to a FILETIME structure that receives the exit time
62       *            of the process. If the process has not exited, the content of
63       *            this structure is undefined.
64       * @param lpKernelTime
65       *            A pointer to a FILETIME structure that receives the amount of
66       *            time that the process has executed in kernel mode. The time
67       *            that each of the threads of the process has executed in kernel
68       *            mode is determined, and then all of those times are summed
69       *            together to obtain this value.
70       * @param lpUserTime
71       *            A pointer to a FILETIME structure that receives the amount of
72       *            time that the process has executed in user mode. The time that
73       *            each of the threads of the process has executed in user mode
74       *            is determined, and then all of those times are summed together
75       *            to obtain this value. Note that this value can exceed the
76       *            amount of real time elapsed (between lpCreationTime and
77       *            lpExitTime) if the process executes across multiple CPU cores.
78       * @return If the function succeeds, the return value is nonzero. If the
79       *         function fails, the return value is zero. To get extended error
80       *         information, call GetLastError.
81       */
82      boolean GetProcessTimes(HANDLE hProcess, FILETIME lpCreationTime, FILETIME lpExitTime, FILETIME lpKernelTime,
83              FILETIME lpUserTime);
84  
85      /**
86       * Retrieves accounting information for all I/O operations performed by the
87       * specified process.
88       *
89       * @param hProcess
90       *            A handle to the process. The handle must have the
91       *            PROCESS_QUERY_INFORMATION or PROCESS_QUERY_LIMITED_INFORMATION
92       *            access right.
93       * @param lpIoCounters
94       *            A pointer to an IO_COUNTERS structure that receives the I/O
95       *            accounting information for the process.
96       * @return If the function succeeds, the return value is nonzero. If the
97       *         function fails, the return value is zero. To get extended error
98       *         information, call GetLastError.
99       */
100     boolean GetProcessIoCounters(HANDLE hProcess, IO_COUNTERS lpIoCounters);
101 }

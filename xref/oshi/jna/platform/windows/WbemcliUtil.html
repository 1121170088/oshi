143         /**
144          * Query WMI for values, with no timeout.
145          *
146          * @return a WmiResult object containing the query results, wrapping an
147          *         EnumMap
148          */
149         public WmiResult<T> execute() {
150             try {
151                 return execute(Wbemcli.WBEM_INFINITE);
152             } catch (TimeoutException e) {
153                 throw new COMException("Got a WMI timeout when infinite wait was specified. This should never happen.");
154             }
155         }
156 
157         /**
158          * Query WMI for values, with a specified timeout.
159          *
160          * @param timeout
161          *            Number of milliseconds to wait for results before timing
162          *            out. If {@link IEnumWbemClassObject#WBEM_INFINITE} (-1),
163          *            will always wait for results. If a timeout occurs, throws
164          *            a {@link TimeoutException}.
165          *
166          * @return a WmiResult object containing the query results, wrapping an
167          *         EnumMap
168          *
169          * @throws TimeoutException
170          *             if the query times out before completion
171          */
172         public WmiResult<T> execute(int timeout) throws TimeoutException {
173             // Idiot check
174             if (getPropertyEnum().getEnumConstants().length < 1) {
175                 throw new IllegalArgumentException("The query's property enum has no values.");
176             }
177 
178             // Connect to the server
179             IWbemServices svc = connectServer(getNameSpace());
180 
181             // Send query
182             try {
183                 IEnumWbemClassObject enumerator = selectProperties(svc, this);
184 
185                 try {
186                     return enumerateProperties(enumerator, getPropertyEnum(), timeout);
187                 } finally {
188                     // Cleanup
189                     enumerator.Release();
190                 }
191             } finally {
192                 // Cleanup
193                 svc.Release();
194             }
195 
196         }

198         /**
199          * Selects properties from WMI. Returns immediately (asynchronously),
200          * even while results are being retrieved; results may begun to be
201          * enumerated in the forward direction only.
202          *
203          * @param svc
204          *            A WbemServices object to make the calls
205          * @param query
206          *            A WmiQuery object encapsulating the details of the query
207          *
208          * @return An enumerator to receive the results of the query
209          */
210         private static <T extends Enum<T>> IEnumWbemClassObject selectProperties(IWbemServices svc, WmiQuery<T> query) {
211             // Step 6: --------------------------------------------------
212             // Use the IWbemServices pointer to make requests of WMI ----
213             T[] props = query.getPropertyEnum().getEnumConstants();
214             StringBuilder sb = new StringBuilder("SELECT ");
215             // We earlier checked for at least one enum constant
216             sb.append(props[0].name());
217             for (int i = 1; i < props.length; i++) {
218                 sb.append(',').append(props[i].name());
219             }
220             sb.append(" FROM ").append(query.getWmiClassName());
221             // Send the query. The flags allow us to return immediately and
222             // begin
223             // enumerating in the forward direction as results come in.
224             return svc.ExecQuery("WQL", sb.toString().replaceAll("\\\\", "\\\\\\\\"),
225                     Wbemcli.WBEM_FLAG_FORWARD_ONLY | Wbemcli.WBEM_FLAG_RETURN_IMMEDIATELY, null);
226         }

228         /*-
229          * The following table maps WMI return types (CIM type) to the VT type of
230          * the returned VARIANT.
231          *
232          * CIM type  |  VT type
233          * ----------|----------
234          * BOOLEAN   |  VT_BOOL
235          * ----------|----------
236          * UINT8     |  VT_UI1
237          * ----------|----------
238          * SINT8     |  VT_I2
239          * SINT16    |  VT_I2
240          * CHAR16    |  VT_I2
241          * ----------|----------
242          * UINT16    |  VT_I4
243          * SINT32    |  VT_I4
244          * UINT32    |  VT_I4
245          * ----------|----------
246          * SINT64    |  VT_BSTR
247          * UINT64    |  VT_BSTR
248          * DATETIME  |  VT_BSTR
249          * REFERENCE |  VT_BSTR
250          * STRING    |  VT_BSTR
251          * ----------|----------
252          * REAL32    |  VT_R4
253          * ----------|----------
254          * REAL64    |  VT_R8
255          * ----------|----------
256          * OBJECT    |  VT_UNKNOWN (not implemented)
257          */
258         /**
259          * Enumerate the results of a WMI query. This method is called while
260          * results are still being retrieved and may iterate in the forward
261          * direction only.
262          *
263          * @param enumerator
264          *            The enumerator with the results
265          * @param propertyEnum
266          *            The enum containing the properties to enumerate, which are
267          *            the keys to the WmiResult map
268          * @param timeout
269          *            Number of milliseconds to wait for results before timing
270          *            out. If {@link IEnumWbemClassObject#WBEM_INFINITE} (-1),
271          *            will always wait for results.
272          *
273          * @return A WmiResult object encapsulating an EnumMap which will hold
274          *         the results.
275          *
276          * @throws TimeoutException
277          *             if the query times out before completion
278          */
279         private static <T extends Enum<T>> WmiResult<T> enumerateProperties(IEnumWbemClassObject enumerator,
280                 Class<T> propertyEnum, int timeout) throws TimeoutException {
281             WmiResult<T> values = INSTANCE.new WmiResult<>(propertyEnum);
282             // Step 7: -------------------------------------------------
283             // Get the data from the query in step 6 -------------------
284             Pointer[] pclsObj = new Pointer[1];
285             IntByReference uReturn = new IntByReference(0);
286             Map<T, WString> wstrMap = new HashMap<>();
287             HRESULT hres = null;
288             for (T property : propertyEnum.getEnumConstants()) {
289                 wstrMap.put(property, new WString(property.name()));
290             }
291             while (enumerator.getPointer() != Pointer.NULL) {
292                 // Enumerator will be released by calling method so no need to
293                 // release it here.
294                 hres = enumerator.Next(timeout, pclsObj.length, pclsObj, uReturn);
295                 // Enumeration complete or no more data; we're done, exit the
296                 // loop
297                 if (hres.intValue() == Wbemcli.WBEM_S_FALSE || hres.intValue() == Wbemcli.WBEM_S_NO_MORE_DATA) {
298                     break;
299                 }
300                 // Throw exception to notify user of timeout
301                 if (hres.intValue() == Wbemcli.WBEM_S_TIMEDOUT) {
302                     throw new TimeoutException("No results after " + timeout + " ms.");
303                 }
304                 // Other exceptions here.
305                 if (COMUtils.FAILED(hres)) {
306                     throw new COMException("Failed to enumerate results: " + hres.intValue());
307                 }
308 
309                 VARIANT.ByReference pVal = new VARIANT.ByReference();
310                 IntByReference pType = new IntByReference();
311 
312                 // Get the value of the properties
313                 IWbemClassObject clsObj = new IWbemClassObject(pclsObj[0]);
314                 for (T property : propertyEnum.getEnumConstants()) {
315                     clsObj.Get(wstrMap.get(property), 0, pVal, pType, null);
316                     int vtType = (pVal.getValue() == null ? Variant.VT_NULL : pVal.getVarType()).intValue();
317                     int cimType = pType.getValue();
318                     switch (vtType) {
319                     case Variant.VT_BSTR:
320                         values.add(vtType, cimType, property, pVal.stringValue());
321                         break;
322                     case Variant.VT_I4:
323                         values.add(vtType, cimType, property, pVal.intValue());
324                         break;
325                     case Variant.VT_UI1:
326                         values.add(vtType, cimType, property, pVal.byteValue());
327                         break;
328                     case Variant.VT_I2:
329                         values.add(vtType, cimType, property, pVal.shortValue());
330                         break;
331                     case Variant.VT_BOOL:
332                         values.add(vtType, cimType, property, pVal.booleanValue());
333                         break;
334                     case Variant.VT_R4:
335                         values.add(vtType, cimType, property, pVal.floatValue());
336                         break;
337                     case Variant.VT_R8:
338                         values.add(vtType, cimType, property, pVal.doubleValue());
339                         break;
340                     case Variant.VT_NULL:
341                         values.add(vtType, cimType, property, null);
342                         break;
343                     // Unimplemented type. User must cast
344                     default:
345                         values.add(vtType, cimType, property, pVal.getValue());
346                     }
347                     OleAuto.INSTANCE.VariantClear(pVal);
348                 }
349                 clsObj.Release();
350 
351                 values.incrementResultCount();
352             }
353             return values;
354         }

357     /**
358      * Helper class wrapping an EnumMap containing the results of a query.
359      */
360     public class WmiResult<T extends Enum<T>> {
361         private Map<T, List<Object>> propertyMap;
362         private Map<T, Integer> vtTypeMap;
363         private Map<T, Integer> cimTypeMap;
364         private int resultCount = 0;
365 
366         /**
367          * @param propertyEnum
368          *            The enum associated with this map
369          */
370         public WmiResult(Class<T> propertyEnum) {
371             this.propertyMap = new EnumMap<>(propertyEnum);
372             this.vtTypeMap = new EnumMap<>(propertyEnum);
373             this.cimTypeMap = new EnumMap<>(propertyEnum);
374             for (T prop : propertyEnum.getEnumConstants()) {
375                 this.propertyMap.put(prop, new ArrayList<>());
376                 this.vtTypeMap.put(prop, Variant.VT_NULL);
377                 this.cimTypeMap.put(prop, Wbemcli.CIM_EMPTY);
378             }
379         }
380 
381         /**
382          * Gets a value from the WmiResult, which may be null. User must check
383          * for null and cast the result. Types correlate to the CIM Type of the
384          * enumerated WMI property and will be consistent for a given property,
385          * and may be validated by the user using {@link #getVtType} or the
386          * Class of the returned Object.
387          *
388          * @param property
389          *            The property (column) to fetch
390          * @param index
391          *            The index (row) to fetch
392          * @return The Object containing the specified value, which may be null
393          */
394         public Object getValue(T property, int index) {
395             return this.propertyMap.get(property).get(index);
396         }
397 
398         /**
399          * Gets the Variant type from the WmiResult. The integer value is
400          * defined as a VT_* constant in the
401          * {@link com.sun.jna.platform.win32.Variant} interface.
402          *
403          * @param property
404          *            The property (column) whose type to fetch
405          * @return An integer representing the Variant type
406          */
407         public int getVtType(T property) {
408             return this.vtTypeMap.get(property);
409         }
410 
411         /**
412          * Gets the CIM type from the WmiResult. The integer value is defined as
413          * a CIM_* constant in the {@link Wbemcli} interface.
414          *
415          * @param property
416          *            The property (column) whose type to fetch
417          * @return An integer representing the CIM type
418          */
419         public int getCIMType(T property) {
420             return this.cimTypeMap.get(property);
421         }
422 
423         /**
424          * Adds a value to the WmiResult at the next index for that property
425          *
426          * @param vtType
427          *            The Variant type of this object
428          * @param cimType
429          *            The CIM type of this property
430          * @param property
431          *            The property (column) to store
432          * @param o
433          *            The object to store
434          */
435         private void add(int vtType, int cimType, T property, Object o) {
436             this.propertyMap.get(property).add(o);
437             if (vtType != Variant.VT_NULL && this.vtTypeMap.get(property).equals(Variant.VT_NULL)) {
438                 this.vtTypeMap.put(property, vtType);
439             }
440             if (this.cimTypeMap.get(property).equals(Wbemcli.CIM_EMPTY)) {
441                 this.cimTypeMap.put(property, cimType);
442             }
443         }
444 
445         /**
446          * @return The number of results in each mapped list
447          */
448         public int getResultCount() {
449             return this.resultCount;
450         }
451 
452         /**
453          * Increment the result count by one.
454          */
455         private void incrementResultCount() {
456             this.resultCount++;
457         }
458     }

460     /**
461      * Determine if WMI has the requested namespace. Some namespaces only exist
462      * on newer versions of Windows.
463      *
464      * @param namespace
465      *            The namespace to test
466      * @return true if the namespace exists, false otherwise
467      */
468     public static boolean hasNamespace(String namespace) {
469         // Strip off leading ROOT\ for valid match
470         String ns = namespace;
471         if (namespace.toUpperCase().startsWith("ROOT\\")) {
472             ns = namespace.substring(5);
473         }
474         // Test
475         WmiQuery<NamespaceProperty> namespaceQuery = new WmiQuery<>("ROOT", "__NAMESPACE", NamespaceProperty.class);
476         WmiResult<NamespaceProperty> namespaces = namespaceQuery.execute();
477         for (int i = 0; i < namespaces.getResultCount(); i++) {
478             if (ns.equalsIgnoreCase((String) namespaces.getValue(NamespaceProperty.NAME, i))) {
479                 return true;
480             }
481         }
482         return false;
483     }

496     /**
497      * Obtains a locator to the WMI server and connects to the specified
498      * namespace
499      *
500      * @param namespace
501      *            The namespace to connect to
502      * @return A service representing the connected namespace, which can be
503      *         queried. This service may be re-used for multiple queries and
504      *         should be released by the user
505      */
506     public static IWbemServices connectServer(String namespace) {
507         // Step 3: ---------------------------------------------------
508         // Obtain the initial locator to WMI -------------------------
509         IWbemLocator loc = IWbemLocator.create();
510         if (loc == null) {
511             throw new COMException("Failed to create WbemLocator object.");
512         }
513 
514         // Step 4: -----------------------------------------------------
515         // Connect to WMI through the IWbemLocator::ConnectServer method
516         // Connect to the namespace with the current user and obtain pointer
517         // pSvc to make IWbemServices calls.
518         IWbemServices services = loc.ConnectServer(namespace, null, null, null, 0, null, null);
519         // Release the locator. If successful, pSvc contains connection
520         // information
521         loc.Release();
522 
523         // Step 5: --------------------------------------------------
524         // Set security levels on the proxy -------------------------
525         HRESULT hres = Ole32.INSTANCE.CoSetProxyBlanket(services, Ole32.RPC_C_AUTHN_WINNT, Ole32.RPC_C_AUTHZ_NONE, null,
526                 Ole32.RPC_C_AUTHN_LEVEL_CALL, Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE);
527         if (COMUtils.FAILED(hres)) {
528             services.Release();
529             throw new COMException("Could not set proxy blanket: " + hres.intValue());
530         }
531         return services;
532     }

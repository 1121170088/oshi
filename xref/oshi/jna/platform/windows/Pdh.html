21  import com.sun.jna.Native; // NOSONAR
22  import com.sun.jna.Pointer;

26  /**
27   * Windows Performance Data Helper (a.k.a. PDH). This class should be considered
28   * non-API as it may be removed if/when its code is incorporated into the JNA
29   * project.
30   *
31   * @author widdis[at]gmail[dot]com
32   */
33  public interface Pdh extends com.sun.jna.platform.win32.Pdh {
34      Pdh INSTANCE = Native.loadLibrary("Pdh", Pdh.class, W32APIOptions.DEFAULT_OPTIONS);
35      int PDH_MORE_DATA = 0x800007D2;
36      int PDH_INVALID_ARGUMENT = 0xC0000BBD;
37      int PDH_MEMORY_ALLOCATION_FAILURE = 0xC0000BBB;
38      int PDH_CSTATUS_NO_MACHINE = 0x800007D0;
39      int PDH_CSTATUS_NO_OBJECT = 0xC0000BB8;
40  
41      /**
42       * Returns the specified object's counter and instance names that exist on
43       * the specified computer or in the specified log file.
44       *
45       * @param szDataSource
46       *            String that specifies the name of the log file used to
47       *            enumerate the counter and instance names. If NULL, the
48       *            function uses the computer specified in the szMachineName
49       *            parameter to enumerate the names.
50       * @param szMachineName
51       *            String that specifies the name of the computer that contains
52       *            the counter and instance names that you want to enumerate.
53       *            Include the leading slashes in the computer name, for example,
54       *            \\computername. If the szDataSource parameter is NULL, you can
55       *            set szMachineName to NULL to specify the local computer.
56       * @param szObjectName
57       *            String that specifies the name of the object whose counter and
58       *            instance names you want to enumerate.
59       * @param mszCounterList
60       *            Caller-allocated buffer that receives a list of
61       *            null-terminated counter names provided by the specified
62       *            object. The list contains unique counter names. The list is
63       *            terminated by two NULL characters. Set to NULL if the
64       *            pcchCounterListLengthparameter is zero.
65       * @param pcchCounterListLength
66       *            Size of the mszCounterList buffer, in TCHARs. If zero on input
67       *            and the object exists, the function returns PDH_MORE_DATA and
68       *            sets this parameter to the required buffer size. If the buffer
69       *            is larger than the required size, the function sets this
70       *            parameter to the actual size of the buffer that was used. If
71       *            the specified size on input is greater than zero but less than
72       *            the required size, you should not rely on the returned size to
73       *            reallocate the buffer.
74       * @param mszInstanceList
75       *            Caller-allocated buffer that receives a list of
76       *            null-terminated instance names provided by the specified
77       *            object. The list contains unique instance names. The list is
78       *            terminated by two NULL characters. Set to NULL if
79       *            pcchInstanceListLength is zero.
80       * @param pcchInstanceListLength
81       *            Size of the mszInstanceList buffer, in TCHARs. If zero on
82       *            input and the object exists, the function returns
83       *            PDH_MORE_DATA and sets this parameter to the required buffer
84       *            size. If the buffer is larger than the required size, the
85       *            function sets this parameter to the actual size of the buffer
86       *            that was used. If the specified size on input is greater than
87       *            zero but less than the required size, you should not rely on
88       *            the returned size to reallocate the buffer. If the specified
89       *            object does not support variable instances, then the returned
90       *            value will be zero. If the specified object does support
91       *            variable instances, but does not currently have any instances,
92       *            then the value returned is 2, which is the size of an empty
93       *            MULTI_SZ list string.
94       * @param dwDetailLevel
95       *            Detail level of the performance items to return. All items
96       *            that are of the specified detail level or less will be
97       *            returned.
98       * @param dwFlags
99       *            This parameter must be zero.
100      * @return If the function succeeds, it returns ERROR_SUCCESS. If the
101      *         function fails, the return value is a system error code or a PDH
102      *         error code.
103      * @see <A HREF=
104      *      "https://msdn.microsoft.com/en-us/library/windows/desktop/aa372677(v=vs.85).aspx">
105      *      PdhEnumObjectItems</A>
106      */
107     int PdhEnumObjectItems(String szDataSource, String szMachineName, String szObjectName, Pointer mszCounterList,
108             DWORDByReference pcchCounterListLength, Pointer mszInstanceList, DWORDByReference pcchInstanceListLength,
109             int dwDetailLevel, int dwFlags);
110 
111     /**
112      * Returns the counter index corresponding to the specified counter name.
113      *
114      * @param szMachineName
115      *            Null-terminated string that specifies the name of the computer
116      *            where the specified counter is located. The computer name can
117      *            be specified by the DNS name or the IP address. If NULL, the
118      *            function uses the local computer.
119      * @param szNameBuffer
120      *            Null-terminated string that contains the counter name.
121      * @param pdwIndex
122      *            Index of the counter.
123      * @return If the function succeeds, it returns ERROR_SUCCESS. If the
124      *         function fails, the return value is a system error code or a PDH
125      *         error code.
126      * @see <A HREF=
127      *      "https://msdn.microsoft.com/en-us/library/windows/desktop/aa372647(v=vs.85).aspx">
128      *      PdhLookupPerfIndexByName</A>
129      */
130     int PdhLookupPerfIndexByName(String szMachineName, String szNameBuffer, DWORDByReference pdwIndex);
131 
132     /**
133      * Returns the performance object name or counter name corresponding to the
134      * specified index.
135      *
136      * @param szMachineName
137      *            Null-terminated string that specifies the name of the computer
138      *            where the specified performance object or counter is located.
139      *            The computer name can be specified by the DNS name or the IP
140      *            address. If NULL, the function uses the local computer.
141      * @param dwNameIndex
142      *            Index of the performance object or counter.
143      * @param szNameBuffer
144      *            Caller-allocated buffer that receives the null-terminated name
145      *            of the performance object or counter. Set to NULL if
146      *            pcchNameBufferSize is zero.
147      * @param pcchNameBufferSize
148      *            Size of the szNameBuffer buffer, in TCHARs. If zero on input,
149      *            the function returns PDH_MORE_DATA and sets this parameter to
150      *            the required buffer size. If the buffer is larger than the
151      *            required size, the function sets this parameter to the actual
152      *            size of the buffer that was used. If the specified size on
153      *            input is greater than zero but less than the required size,
154      *            you should not rely on the returned size to reallocate the
155      *            buffer.
156      * @return If the function succeeds, it returns ERROR_SUCCESS. If the
157      *         function fails, the return value is a system error code or a PDH
158      *         error code.
159      * @see <A HREF=
160      *      "https://msdn.microsoft.com/en-us/library/windows/desktop/aa372648(v=vs.85).aspx">
161      *      PdhLookupPerfNameByIndex</A>
162      */
163     int PdhLookupPerfNameByIndex(String szMachineName, int dwNameIndex, Pointer szNameBuffer,
164             DWORDByReference pcchNameBufferSize);
165 }

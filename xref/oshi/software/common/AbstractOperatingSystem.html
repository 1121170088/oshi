21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.Collections;

26  import java.util.List;

28  import oshi.software.os.OSProcess;
29  import oshi.software.os.OperatingSystem;

32  public abstract class AbstractOperatingSystem implements OperatingSystem {
33  
34      private static final long serialVersionUID = 1L;
35  
36      protected String manufacturer;
37      protected String family;
38      protected OperatingSystemVersion version;
39  
40      /*
41       * Comparators for use in processSort()
42       */
43      private static final Comparator<OSProcess> CPU_DESC_SORT = new Comparator<OSProcess>() {
44          @Override
45          public int compare(OSProcess p1, OSProcess p2) {
46              return Double.compare((p2.getKernelTime() + p2.getUserTime()) / (double) p2.getUpTime(),
47                      (p1.getKernelTime() + p1.getUserTime()) / (double) p1.getUpTime());
48          }
49      };
50      private static final Comparator<OSProcess> RSS_DESC_SORT = new Comparator<OSProcess>() {
51          @Override
52          public int compare(OSProcess p1, OSProcess p2) {
53              return Long.compare(p2.getResidentSetSize(), p1.getResidentSetSize());
54          }
55      };
56      private static final Comparator<OSProcess> UPTIME_DESC_SORT = new Comparator<OSProcess>() {
57          @Override
58          public int compare(OSProcess p1, OSProcess p2) {
59              return Long.compare(p2.getUpTime(), p1.getUpTime());
60          }
61      };
62      private static final Comparator<OSProcess> UPTIME_ASC_SORT = new Comparator<OSProcess>() {
63          @Override
64          public int compare(OSProcess p1, OSProcess p2) {
65              return Long.compare(p1.getUpTime(), p2.getUpTime());
66          }
67      };
68      private static final Comparator<OSProcess> PID_ASC_SORT = new Comparator<OSProcess>() {
69          @Override
70          public int compare(OSProcess p1, OSProcess p2) {
71              return Integer.compare(p1.getProcessID(), p2.getProcessID());
72          }
73      };
74      private static final Comparator<OSProcess> PARENTPID_ASC_SORT = new Comparator<OSProcess>() {
75          @Override
76          public int compare(OSProcess p1, OSProcess p2) {
77              return Integer.compare(p1.getParentProcessID(), p2.getParentProcessID());
78          }
79      };
80      private static final Comparator<OSProcess> NAME_ASC_SORT = new Comparator<OSProcess>() {
81          @Override
82          public int compare(OSProcess p1, OSProcess p2) {
83              return p1.getName().toLowerCase().compareTo(p2.getName().toLowerCase());
84          }
85      };

88       * {@inheritDoc}

90      @Override
91      public OperatingSystemVersion getVersion() {
92          return this.version;
93      }

98      @Override

100         return this.family;
101     }

103     /**
104      * {@inheritDoc}
105      */
106     @Override
107     public String getManufacturer() {
108         return this.manufacturer;
109     }

111     /**
112      * Sorts an array of processes using the specified sorting, returning an
113      * array with the top limit results if positive.
114      *
115      * @param processes
116      *            The array to sort
117      * @param limit
118      *            The number of results to return if positive; if zero returns
119      *            all results
120      * @param sort
121      *            The sorting to use, or null
122      * @return An array of size limit (if positive) or of all processes, sorted
123      *         as specified
124      */
125     protected List<OSProcess> processSort(List<OSProcess> processes, int limit, ProcessSort sort) {
126         if (sort != null) {
127             switch (sort) {
128             case CPU:
129                 Collections.sort(processes, CPU_DESC_SORT);
130                 break;
131             case MEMORY:
132                 Collections.sort(processes, RSS_DESC_SORT);
133                 break;
134             case OLDEST:
135                 Collections.sort(processes, UPTIME_DESC_SORT);
136                 break;
137             case NEWEST:
138                 Collections.sort(processes, UPTIME_ASC_SORT);
139                 break;
140             case PID:
141                 Collections.sort(processes, PID_ASC_SORT);
142                 break;
143             case PARENTPID:
144                 Collections.sort(processes, PARENTPID_ASC_SORT);
145                 break;
146             case NAME:
147                 Collections.sort(processes, NAME_ASC_SORT);
148                 break;
149             default:
150                 // Should never get here! If you get this exception you've
151                 // added something to the enum without adding it here. Tsk.
152                 throw new IllegalArgumentException("Unimplemented enum type: " + sort.toString());
153             }
154         }
155         // Return max of limit or process size
156         // Nonpositive limit means return all
157         int maxProcs = processes.size();
158         if (limit > 0 && maxProcs > limit) {
159             maxProcs = limit;
160         } else {
161             return processes;
162         }
163         List<OSProcess> procs = new ArrayList<>();
164         for (int i = 0; i < maxProcs; i++) {
165             procs.add(processes.get(i));
166         }
167         return procs;
168     }
169 
170     @Override
171     public String toString() {
172         StringBuilder sb = new StringBuilder();
173         sb.append(getManufacturer()).append(' ').append(getFamily()).append(' ').append(getVersion().toString());
174         return sb.toString();
175     }
176 
177     /**
178      * {@inheritDoc}
179      */
180     @Override
181     public List<OSProcess> getProcesses(Collection<Integer> pids) {
182         List<OSProcess> returnValue = new LinkedList<>();
183         for (Integer pid : pids) {
184             OSProcess process = getProcess(pid);
185             if (process != null)
186                 returnValue.add(process);
187         }
188         return returnValue;
189     }
190 }

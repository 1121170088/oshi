142     private void initRegistry() {
143         // Get the title indices
144         DWORDByReference index = new DWORDByReference();
145         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Process", index);
146         processIndex = index.getValue().intValue();
147         processIndexStr = Integer.toString(processIndex);
148         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Priority Base", index);
149         int priorityBaseIndex = index.getValue().intValue();
150         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Elapsed Time", index);
151         int elapsedTimeIndex = index.getValue().intValue();
152         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "ID Process", index);
153         int idProcessIndex = index.getValue().intValue();
154         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Creating Process ID", index);
155         int creatingProcessIdIndex = index.getValue().intValue();
156         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Read Bytes/sec", index);
157         int ioReadIndex = index.getValue().intValue();
158         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Write Bytes/sec", index);
159         int ioWriteIndex = index.getValue().intValue();
160         Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Working Set - Private", index);
161         int workingSetPrivateIndex = index.getValue().intValue();
162 
163         // now load the Process registry to match up the offsets
164         // Sequentially increase the buffer until everything fits.
165         // Save this buffer size for later use
166         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
167         Pointer pPerfData = new Memory(this.perfDataBufferSize);
168         while (WinError.ERROR_MORE_DATA == Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA,
169                 processIndexStr, 0, null, pPerfData, lpcbData)) {
170             this.perfDataBufferSize += 4096;
171             lpcbData.setValue(this.perfDataBufferSize);
172             pPerfData = new Memory(this.perfDataBufferSize);
173         }
174 
175         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
176 
177         // See format at
178         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
179         // [ ] Object Type
180         // [ ][ ][ ] Multiple counter definitions
181         // Then multiple:
182         // [ ] Instance Definition
183         // [ ] Instance name
184         // [ ] Counter Block
185         // [ ][ ][ ] Counter data for each definition above
186 
187         long perfObjectOffset = perfData.HeaderLength;
188 
189         // Iterate object types. For Process should only be one here
190         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
191             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
192             // Identify where counter definitions start
193             long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
194             // If this isn't the Process object, ignore
195             if (perfObject.ObjectNameTitleIndex == processIndex) {
196                 for (int counter = 0; counter < perfObject.NumCounters; counter++) {
197                     PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
198                             pPerfData.share(perfCounterOffset));
199                     if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
200                         this.priorityBaseOffset = perfCounter.CounterOffset;
201                     } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
202                         this.elapsedTimeOffset = perfCounter.CounterOffset;
203                     } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
204                         this.creatingProcessIdOffset = perfCounter.CounterOffset;
205                     } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
206                         this.idProcessOffset = perfCounter.CounterOffset;
207                     } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
208                         this.ioReadOffset = perfCounter.CounterOffset;
209                     } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
210                         this.ioWriteOffset = perfCounter.CounterOffset;
211                     } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
212                         this.workingSetPrivateOffset = perfCounter.CounterOffset;
213                     }
214                     // Increment for next Counter
215                     perfCounterOffset += perfCounter.ByteLength;
216                 }
217                 // We're done, break the loop
218                 break;
219             }
220             // Increment for next object (should never need this)
221             perfObjectOffset += perfObject.TotalByteLength;
222         }
223     }

225     private void initBitness() {
226         // If bitness is 64 we are 64 bit.
227         // If 32 test if we are on 64-bit OS
228         if (bitness < 64) {
229             // Try the easy way
230             if (System.getenv("ProgramFiles(x86)") != null) {
231                 this.bitness = 64;
232             } else {
233                 // Fetch from WMI
234                 this.bitness = (WmiUtil.selectUint32From(null, "Win32_Processor", "AddressWidth", null)).intValue();
235             }
236         }

260     @Override
261     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
262         Set<Integer> childPids = new HashSet<>();
263         // Get processes from ToolHelp API for parent PID
264         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
265         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
266         try {
267             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
268                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
269                     childPids.add(processEntry.th32ProcessID.intValue());
270                 }
271             }
272         } finally {
273             Kernel32.INSTANCE.CloseHandle(snapshot);
274         }
275         List<OSProcess> procList = getProcesses(childPids);
276         List<OSProcess> sorted = processSort(procList, limit, sort);
277         return sorted.toArray(new OSProcess[sorted.size()]);
278     }

299     /**
300      * Private method to do the heavy lifting for all the getProcess functions.
301      *
302      * @param pids
303      *         A collection of pids to query. If null, the entire process
304      *         list will be queried.
305      * @return A corresponding list of processes
306      */
307     private List<OSProcess> processMapToList(Collection<Integer> pids) {
308         // Get data from the registry to update cache
309         updateProcessMapFromRegistry(pids);
310 
311         // define here to avoid object repeated creation overhead later
312         List<String> groupList = new LinkedList<>();
313         List<String> groupIDList = new LinkedList<>();
314         int myPid = getProcessId();
315 
316         // Get processes from WTS
317         final PointerByReference ppProcessInfo = new PointerByReference();
318         IntByReference pCount = new IntByReference(0);
319         if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
320                 Wtsapi32.WTSTypeProcessInfoLevel1, Wtsapi32.WTS_ANY_SESSION, ppProcessInfo, pCount)) {
321             LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
322             return new LinkedList<OSProcess>();
323         }
324         // extract the pointed-to pointer and create array
325         final Pointer pProcessInfo = ppProcessInfo.getValue();
326         final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
327         final WTS_PROCESS_INFO_EX[] processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
328 
329         // Store a subset of processes in a list to later return.
330         List<OSProcess> tempProcessList = new LinkedList<>();
331 
332         for (WTS_PROCESS_INFO_EX procInfo : processInfo) {
333             // Skip if only updating a subset of pids, or if not in cache.
334             // (Cache should have just been updated from registry so this will
335             // only occur in a race condition for a just-started process.)
336             int pid = procInfo.ProcessId.intValue();
337             if (!this.processMap.containsKey(pid) || (pids != null && !pids.contains(pid))) {
338                 continue;
339             }
340             OSProcess proc = this.processMap.get(pid);
341             // For my own process, set CWD
342             if (pid == myPid) {
343                 String cwd = new File(".").getAbsolutePath();
344                 // trim off trailing "."
345                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
346             }
347 
348             proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
349             proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
350 
351             // Get a handle to the process for various extended info. Only gets
352             // current user unless running as administrator
353             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
354                     proc.getProcessID());
355             if (pHandle != null) {
356                 // Full path
357                 proc.setPath(Kernel32Util.QueryFullProcessImageName(pHandle, 0));
358 
359                 final HANDLEByReference phToken = new HANDLEByReference();
360                 if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY, phToken)) {
361                     Account account = Advapi32Util.getTokenAccount(phToken.getValue());
362                     proc.setUser(account.name);
363                     proc.setUserID(account.sidString);
364                     // Fetching group information incurs ~10ms latency per
365                     // process. Skip for full process list
366                     if (pids != null) {
367                         Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
368                         // get groups
369                         groupList.clear();
370                         groupIDList.clear();
371                         for (Account a : accounts) {
372                             groupList.add(a.name);
373                             groupIDList.add(a.sidString);
374                         }
375                         proc.setGroup(FormatUtil.join(",", groupList));
376                         proc.setGroupID(FormatUtil.join(",", groupIDList));
377                     }
378                 } else {
379                     int error = Kernel32.INSTANCE.GetLastError();
380                     // Access denied errors are common. Fail silently.
381                     if (error != WinError.ERROR_ACCESS_DENIED) {
382                         LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
383                                 Kernel32.INSTANCE.GetLastError());
384                     }
385                 }
386             }
387             Kernel32.INSTANCE.CloseHandle(pHandle);
388 
389             // There is no easy way to get ExecutuionState for a process.
390             // The WMI value is null. It's possible to get thread Execution
391             // State and possibly roll up.
392             proc.setState(OSProcess.State.RUNNING);
393 
394             proc.setThreadCount(procInfo.NumberOfThreads.intValue());
395             proc.setVirtualSize(procInfo.PagefileUsage.longValue());
396 
397             proc.setOpenFiles(procInfo.HandleCount.intValue());
398             tempProcessList.add(proc);
399         }
400         // Clean up memory allocated in C
401         if (!Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTSTypeProcessInfoLevel1.getValue(), pProcessInfo,
402                 pCount.getValue())) {
403             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
404             return new LinkedList<OSProcess>();
405         }
406 
407         // Command Line only accessible via WMI.
408         // Utilize cache to only update new processes
409         Set<Integer> emptyCommandLines = new HashSet<>();
410         for (OSProcess cachedProcess : tempProcessList) {
411             // If the process in the cache has an empty command line.
412             if (cachedProcess.getCommandLine().isEmpty()) {
413                 emptyCommandLines.add(cachedProcess.getProcessID());
414             }
415         }
416         if (!emptyCommandLines.isEmpty()) {
417             StringBuilder query = new StringBuilder("WHERE ");
418             for (Integer pid : emptyCommandLines) {
419                 query.append(String.format("ProcessID=%d OR ", pid));
420             }
421             query.setLength(query.length() - 3);
422             Map<String, List<Object>> commandLineProcs = WmiUtil.selectObjectsFrom(null, "Win32_Process",
423                     PROCESS_STRINGS, query.toString(), PROCESS_TYPES);
424             for (int p = 0; p < commandLineProcs.get(WmiProperty.PROCESSID.name()).size(); p++) {
425                 int pid = ((Long) commandLineProcs.get(WmiProperty.PROCESSID.name()).get(p)).intValue();
426                 // This should always be true because emptyCommandLines was
427                 // built from a subset of the cache, but just in case, protect
428                 // against dereferencing null
429                 if (this.processMap.containsKey(pid)) {
430                     OSProcess proc = this.processMap.get(pid);
431                     proc.setCommandLine((String) commandLineProcs.get(WmiProperty.COMMANDLINE.name()).get(p));
432                 }
433             }
434         }
435 
436         // Deep copy for return
437         List<OSProcess> procList = new ArrayList<>(tempProcessList.size());
438         for (OSProcess proc : tempProcessList) {
439             try {
440                 procList.add((OSProcess) proc.clone());
441             } catch (CloneNotSupportedException e) {
442                 LOG.error("Unable to clone process ID {}", proc.getProcessID());
443             }
444         }
445         return procList;
446     }

448     private void updateProcessMapFromRegistry(Collection<Integer> pids) {
449         List<Integer> pidsToKeep = new LinkedList<>();
450 
451         // Grab the PERF_DATA_BLOCK from the registry.
452         // Sequentially increase the buffer until everything fits.
453         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
454         Pointer pPerfData = new Memory(this.perfDataBufferSize);
455         while (WinError.ERROR_MORE_DATA == Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA,
456                 processIndexStr, 0, null, pPerfData, lpcbData)) {
457             this.perfDataBufferSize += 4096;
458             lpcbData.setValue(this.perfDataBufferSize);
459             pPerfData = new Memory(this.perfDataBufferSize);
460         }
461 
462         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
463         long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
464         long now = System.currentTimeMillis(); // 1970 epoch
465 
466         // See format at
467         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
468         // [ ] Object Type
469         // [ ][ ][ ] Multiple counter definitions
470         // Then multiple:
471         // [ ] Instance Definition
472         // [ ] Instance name
473         // [ ] Counter Block
474         // [ ][ ][ ] Counter data for each definition above
475 
476         long perfObjectOffset = perfData.HeaderLength;
477 
478         // Iterate object types. For Process should only be one here
479         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
480             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
481             // If this isn't the Process object, ignore
482             if (perfObject.ObjectNameTitleIndex == processIndex) {
483                 // Skip over counter definitions
484                 // There will be many of these, this points to the first one
485                 long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
486 
487                 // We need this for every process, initialize outside loop to
488                 // save overhead
489                 PERF_COUNTER_BLOCK perfCounterBlock = null;
490                 // Iterate instances.
491                 // The last instance is _Total so subtract 1 from max
492                 for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
493                     PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
494                             pPerfData.share(perfInstanceOffset));
495                     long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
496 
497                     int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
498                     if (pids == null || pids.contains(pid)) {
499                         pidsToKeep.add(pid);
500 
501                         // If process exists fetch from cache
502                         OSProcess proc = null;
503                         if (this.processMap.containsKey(pid)) {
504                             proc = this.processMap.get(pid);
505                         }
506                         // If not in cache or if start time differs too much,
507                         // create new process to add/replace cache value
508                         long upTime = (perfTime100nSec
509                                 - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset)) / 10_000L;
510                         long startTime = now - upTime;
511                         if (proc == null || Math.abs(startTime - proc.getStartTime()) > 200) {
512                             proc = new OSProcess();
513                             proc.setProcessID(pid);
514                             proc.setUpTime(upTime);
515                             proc.setStartTime(startTime);
516                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
517                             // Adds or replaces previous
518                             this.processMap.put(pid, proc);
519                         }
520                         
521                         // Update stats
522                         proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
523                         proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
524                         proc.setResidentSetSize(
525                                 pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
526                         proc.setParentProcessID(
527                                 pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
528                         proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
529                     }
530 
531                     // Increment to next instance
532                     perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
533                     perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
534                 }
535                 // We've found the process object and are done, no need to look
536                 // at any other objects (shouldn't be any). Break the loop
537                 break;
538             }
539             // Increment for next object (should never need this)
540             perfObjectOffset += perfObject.TotalByteLength;
541         }
542         // If this was a full update, delete any pid we didn't find from the
543         // cache.
544         if (pids == null) {
545             for (Integer pid : new HashSet<>(this.processMap.keySet())) {
546                 if (!pidsToKeep.contains(pid)) {
547                     this.processMap.remove(pid);
548                 }
549             }
550         }
551     }

595     /**
596      * Enables debug privileges for this process, required for OpenProcess() to
597      * get processes other than the current user
598      */
599     private static void enableDebugPrivilege() {
600         HANDLEByReference hToken = new HANDLEByReference();
601         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
602                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
603         if (!success) {
604             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
605             return;
606         }
607         WinNT.LUID luid = new WinNT.LUID();
608         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
609         if (!success) {
610             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
611             return;
612         }
613         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
614         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
615         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
616         if (!success) {
617             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
618         }
619         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
620     }

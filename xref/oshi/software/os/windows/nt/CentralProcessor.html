104   // Return values from PDH methods, if nonzero signifies error
105   private static int pdhOpenTickQueryError = 0;
106 
107   private static int pdhAddTickCounterError = 0;
108 
109   private static int pdhOpenUptimeQueryError = 0;
110 
111   private static int pdhAddUptimeCounterError = 0;
112 
113   static {
114     // Get number of processors
115     SYSTEM_INFO sysinfo = new SYSTEM_INFO();
116     Kernel32.INSTANCE.GetSystemInfo(sysinfo);
117     numCPU = sysinfo.dwNumberOfProcessors.intValue();
118 
119     // Open tick query for this processor
120     pdhOpenTickQueryError = Pdh.INSTANCE.PdhOpenQuery(null, zero, phQuery);
121     if (pdhOpenTickQueryError != 0) {
122       LOG.error("Failed to open PDH Tick Query. Error code: {}",
123           String.format("0x%08X", pdhOpenTickQueryError));
124     }
125 
126     // Open uptime query for this processor
127     pdhOpenUptimeQueryError = Pdh.INSTANCE.PdhOpenQuery(null, one, uptimeQuery);
128     if (pdhOpenTickQueryError != 0) {
129       LOG.error("Failed to open PDH Uptime Query. Error code: {}",
130           String.format("0x%08X", pdhOpenUptimeQueryError));
131     }
132 
133     // Set up hook to close the query on shutdown
134     Runtime.getRuntime().addShutdownHook(new Thread() {
135 
136       @Override
137       public void run() {
138         Pdh.INSTANCE.PdhCloseQuery(phQuery.getValue());
139         Pdh.INSTANCE.PdhCloseQuery(uptimeQuery.getValue());
140       }
141     });
142   }

144   // Set up a counter for each processor
145   private static PointerByReference[] phUserCounters = new PointerByReference[numCPU];
146 
147   private static PointerByReference[] phIdleCounters = new PointerByReference[numCPU];
148 
149   static {
150     if (pdhOpenTickQueryError == 0) {
151       for (int p = 0; p < numCPU; p++) {
152         // Options are (only need 2 to calculate all)
153         // "\Processor(0)\% processor time"
154         // "\Processor(0)\% idle time" (1 - processor)
155         // "\Processor(0)\% privileged time" (subset of processor)
156         // "\Processor(0)\% user time" (other subset of processor)
157         // Note need to make \ = \\ for Java Strings and %% for format
158         String counterPath = String.format("\\Processor(%d)\\%% user time", p);
159         phUserCounters[p] = new PointerByReference();
160         // Open tick query for this processor
161         pdhAddTickCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(
162             phQuery.getValue(), counterPath, zero, phUserCounters[p]);
163         if (pdhAddTickCounterError != 0) {
164           LOG.error(
165               "Failed to add PDH User Tick Counter for processor {}. Error code: {}",
166               p, String.format("0x%08X", pdhAddTickCounterError));
167           break;
168         }
169         counterPath = String.format("\\Processor(%d)\\%% idle time", p);
170         phIdleCounters[p] = new PointerByReference();
171         pdhAddTickCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(
172             phQuery.getValue(), counterPath, zero, phIdleCounters[p]);
173         if (pdhAddTickCounterError != 0) {
174           LOG.error(
175               "Failed to add PDH Idle Tick Counter for processor {}. Error code: {}",
176               p, String.format("0x%08X", pdhAddTickCounterError));
177           break;
178         }
179       }
180 
181       LOG.debug("Tick counter queries added.  Initializing with first query.");
182       // Initialize by collecting data the first time
183       int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
184       if (ret != 0) {
185         LOG.warn("Failed to update Tick Counters. Error code: {}",
186             String.format("0x%08X", ret));
187       }
188     }
189   }

225   /**
226    * Create a Processor with the given number
227    * 
228    * @param procNo
229    *        The processor number
230    */
231   public CentralProcessor(int procNo) {
232     if (procNo >= numCPU) {
233       throw new IllegalArgumentException("Processor number (" + procNo
234           + ") must be less than the number of CPUs: " + numCPU);
235     }
236     this.processorNumber = procNo;
237     updateProcessorTicks();
238     System.arraycopy(allProcessorTicks[this.processorNumber], 0,
239         this.curProcTicks, 0, this.curProcTicks.length);
240     LOG.debug("Initialized Processor {}", procNo);
241   }

293   /**
294    * Vendor frequency (in Hz), eg. for processor named Intel(R) Core(TM)2 Duo
295    * CPU T7300 @ 2.00GHz the vendor frequency is 2000000000.
296    * 
297    * @return Processor frequency or -1 if unknown.
298    */
299   @Override
300   public long getVendorFreq() {
301     if (this.cpuVendorFreq == null) {
302       Pattern pattern = Pattern.compile("@ (.*)$");
303       Matcher matcher = pattern.matcher(getName());
304 
305       if (matcher.find()) {
306         String unit = matcher.group(1);
307         this.cpuVendorFreq = Long.valueOf(ParseUtil.parseHertz(unit));
308       } else {
309         this.cpuVendorFreq = Long.valueOf(-1L);
310       }
311     }
312 
313     return this.cpuVendorFreq.longValue();
314   }

422   /**
423    * {@inheritDoc}
424    */
425   @Override
426   public synchronized double getSystemCpuLoadBetweenTicks() {
427     // Check if > ~ 0.95 seconds since last tick count.
428     long now = System.currentTimeMillis();
429     LOG.trace("Current time: {}  Last tick time: {}", now, tickTime);
430     boolean update = (now - tickTime > 950);
431     if (update) {
432       // Enough time has elapsed.
433       // Update latest
434       updateSystemTicks();
435       tickTime = now;
436       LOG.debug("System Ticks Updated");
437     }
438     // Calculate total
439     long total = 0;
440     for (int i = 0; i < curTicks.length; i++) {
441       total += (curTicks[i] - prevTicks[i]);
442     }
443     // Calculate idle from last field [3]
444     long idle = curTicks[3] - prevTicks[3];
445     LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
446 
447     // Copy latest ticks to earlier position for next call
448     if (update) {
449       System.arraycopy(curTicks, 0, prevTicks, 0, curTicks.length);
450     }
451 
452     // return
453     if (total > 0 && idle >= 0) {
454       return (double) (total - idle) / total;
455     }
456     return 0d;
457   }

471   /**
472    * Updates system tick information from native call to GetSystemTimes(). Array
473    * with four elements representing clock ticks or milliseconds (platform
474    * dependent) spent in User (0), Nice (1), System (2), and Idle (3) states. By
475    * measuring the difference between ticks across a time interval, CPU load
476    * over that interval may be calculated.
477    */
478   private static void updateSystemTicks() {
479     LOG.trace("Updating System Ticks");
480     WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
481     WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
482     WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
483     if (0 == Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime,
484         lpUserTime)) {
485       LOG.error("Failed to update system idle/kernel/user times. Error code: "
486           + Native.getLastError());
487       return;
488     }
489     // Array order is user,nice,kernel,idle
490     curTicks[0] = lpUserTime.toLong() + Kernel32.WIN32_TIME_OFFSET;
491     curTicks[1] = 0L; // Windows is not 'nice'
492     curTicks[2] = lpKernelTime.toLong() - lpIdleTime.toLong();
493     curTicks[3] = lpIdleTime.toLong() + Kernel32.WIN32_TIME_OFFSET;
494   }

517   /**
518    * {@inheritDoc}
519    */
520   @Override
521   public double getProcessorCpuLoadBetweenTicks() {
522     // Check if > ~ 0.95 seconds since last tick count.
523     long now = System.currentTimeMillis();
524     LOG.trace("Current time: {}  Last processor tick time: {}", now,
525         this.procTickTime);
526     if (now - this.procTickTime > 950) {
527       // Enough time has elapsed. Update array in place
528       updateProcessorTicks();
529       // Copy arrays in place
530       System.arraycopy(this.curProcTicks, 0, this.prevProcTicks, 0,
531           this.curProcTicks.length);
532       System.arraycopy(allProcessorTicks[this.processorNumber], 0,
533           this.curProcTicks, 0, this.curProcTicks.length);
534       this.procTickTime = now;
535     }
536     long total = 0;
537     for (int i = 0; i < this.curProcTicks.length; i++) {
538       total += (this.curProcTicks[i] - this.prevProcTicks[i]);
539     }
540     // Calculate idle from last field [3]
541     long idle = this.curProcTicks[3] - this.prevProcTicks[3];
542     LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
543     // update
544     return (total > 0 && idle >= 0) ? (double) (total - idle) / total : 0d;
545   }

556   /**
557    * Updates the tick array for all processors if more than 100ms has elapsed
558    * since the last update. This permits using the allProcessorTicks as a cache
559    * when iterating over processors since pdh query updates all counters at once
560    * so we can't separate individual processors
561    */
562   private void updateProcessorTicks() {
563     // Do nothing if we have PDH errors
564     if (pdhOpenTickQueryError != 0 || pdhAddTickCounterError != 0) {
565       LOG.warn("PDH Tick Counters not initialized. Processor ticks not updated.");
566       return;
567     }
568     // Update no more frequently than 100ms so this is only triggered once
569     // during iteration over Processors
570     long now = System.currentTimeMillis();
571     LOG.trace("Current time: {}  Last all processor tick time: {}", now,
572         allProcTickTime);
573     if (now - allProcTickTime < 100)
574       return;
575 
576     // This call updates all process counters to % used since last call
577     int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
578     if (ret != 0) {
579       LOG.warn("Failed to update Tick Counters. Error code: {}",
580           String.format("0x%08X", ret));
581       return;
582     }
583     // Multiply % usage times elapsed MS to recreate ticks, then increment
584     long elapsed = now - allProcTickTime;
585     for (int cpu = 0; cpu < numCPU; cpu++) {
586       PdhFmtCounterValue phUserCounterValue = new PdhFmtCounterValue();
587       ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(
588           phUserCounters[cpu].getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000,
589           null, phUserCounterValue);
590       if (ret != 0) {
591         LOG.warn("Failed to get Uer Tick Counters. Error code: {}",
592             String.format("0x%08X", ret));
593         return;
594       }
595 
596       PdhFmtCounterValue phIdleCounterValue = new PdhFmtCounterValue();
597       ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(
598           phIdleCounters[cpu].getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000,
599           null, phIdleCounterValue);
600       if (ret != 0) {
601         LOG.warn("Failed to get idle Tick Counters. Error code: {}",
602             String.format("0x%08X", ret));
603         return;
604       }
605 
606       // Returns results in 1000's of percent, e.g. 5% is 5000
607       // Multiply by elapsed to get total ms and Divide by 100 * 1000
608       // Putting division at end avoids need to cast division to double
609       long user = elapsed * phUserCounterValue.value.largeValue / 100000;
610       long idle = elapsed * phIdleCounterValue.value.largeValue / 100000;
611       // Elasped is only since last read, so increment previous value
612       allProcessorTicks[cpu][0] += user;
613       // allProcessorTicks[cpu][1] is ignored, Windows is not nice
614       allProcessorTicks[cpu][2] += (elapsed - user - idle); // u+i+sys=100%
615       allProcessorTicks[cpu][3] += idle;
616     }
617     allProcTickTime = now;
618   }

620   /**
621    * {@inheritDoc}
622    */
623   @Override
624   public long getSystemUptime() {
625     // Return 0 if we have PDH errors
626     if (pdhOpenUptimeQueryError != 0 || pdhAddUptimeCounterError != 0) {
627       LOG.warn("Uptime Counters not initialized. Returning 0.");
628       return 0L;
629     }
630 
631     int ret = Pdh.INSTANCE.PdhCollectQueryData(uptimeQuery.getValue());
632     if (ret != 0) {
633       LOG.error("Failed to update Uptime Counters. Error code: {}",
634           String.format("0x%08X", ret));
635       return 0L;
636     }
637 
638     PdhFmtCounterValue uptimeCounterValue = new PdhFmtCounterValue();
639     ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pUptime.getValue(),
640         Pdh.PDH_FMT_LARGE, null, uptimeCounterValue);
641     if (ret != 0) {
642       LOG.error("Failed to get Uptime Counters. Error code: {}",
643           String.format("0x%08X", ret));
644       return 0L;
645     }
646 
647     return uptimeCounterValue.value.largeValue;
648   }

650   /**
651    * {@inheritDoc}
652    */
653   @Override
654   public String getSystemSerialNumber() {
655     String sn = null;
656     // This should always work
657     ArrayList<String> hwInfo = ExecutingCommand
658         .runNative("wmic bios get serialnumber");
659     for (String checkLine : hwInfo) {
660       if (checkLine.length() == 0
661           || checkLine.toLowerCase().contains("serialnumber")) {
662         continue;
663       } else {
664         sn = checkLine.trim();
665         break;
666       }
667     }
668     // Just in case the above doesn't
669     if (sn == null || sn.length() == 0) {
670       hwInfo = ExecutingCommand
671           .runNative("wmic csproduct get identifyingnumber");
672       for (String checkLine : hwInfo) {
673         if (checkLine.length() == 0
674             || checkLine.toLowerCase().contains("identifyingnumber")) {
675           continue;
676         } else {
677           sn = checkLine.trim();
678           break;
679         }
680       }
681     }
682     return (sn == null || sn.length() == 0) ? "unknown" : sn;
683   }

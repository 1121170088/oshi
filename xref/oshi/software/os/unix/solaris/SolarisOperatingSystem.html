45      public SolarisOperatingSystem() {
46          this.manufacturer = "Oracle";
47          this.family = "SunOS";
48          this.version = new SolarisOSVersionInfoEx();
49      }

51      /**
52       * {@inheritDoc}
53       */
54      @Override
55      public FileSystem getFileSystem() {
56          return new SolarisFileSystem();
57      }

59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public OSProcess[] getProcesses(int limit, ProcessSort sort) {
64          List<OSProcess> procs = getProcessListFromPS(
65                  "ps -eo s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args", -1);
66          List<OSProcess> sorted = processSort(procs, limit, sort);
67          return sorted.toArray(new OSProcess[sorted.size()]);
68      }
69  
70      /**
71       * {@inheritDoc}
72       */
73      @Override
74      public OSProcess getProcess(int pid) {
75          List<OSProcess> procs = getProcessListFromPS(
76                  "ps -o s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args -p ", pid);
77          if (procs.isEmpty()) {
78              return null;
79          }
80          return procs.get(0);
81      }
82  
83      private List<OSProcess> getProcessListFromPS(String psCommand, int pid) {
84          Map<Integer, String> cwdMap = LsofUtil.getCwdMap(pid);
85          List<OSProcess> procs = new ArrayList<>();
86          List<String> procList = ExecutingCommand.runNative(psCommand + (pid < 0 ? "" : pid));
87          if (procList.isEmpty() || procList.size() < 2) {
88              return procs;
89          }
90          // remove header row
91          procList.remove(0);
92          // Fill list
93          for (String proc : procList) {
94              String[] split = ParseUtil.whitespaces.split(proc.trim(), 15);
95              // Elements should match ps command order
96              if (split.length < 15) {
97                  continue;
98              }
99              long now = System.currentTimeMillis();
100             OSProcess sproc = new OSProcess();
101             switch (split[0].charAt(0)) {
102             case 'O':
103                 sproc.setState(OSProcess.State.RUNNING);
104                 break;
105             case 'S':
106                 sproc.setState(OSProcess.State.SLEEPING);
107                 break;
108             case 'R':
109             case 'W':
110                 sproc.setState(OSProcess.State.WAITING);
111                 break;
112             case 'Z':
113                 sproc.setState(OSProcess.State.ZOMBIE);
114                 break;
115             case 'T':
116                 sproc.setState(OSProcess.State.STOPPED);
117                 break;
118             default:
119                 sproc.setState(OSProcess.State.OTHER);
120                 break;
121             }
122             sproc.setProcessID(ParseUtil.parseIntOrDefault(split[1], 0));
123             sproc.setParentProcessID(ParseUtil.parseIntOrDefault(split[2], 0));
124             sproc.setUser(split[3]);
125             sproc.setUserID(split[4]);
126             sproc.setGroup(split[5]);
127             sproc.setGroupID(split[6]);
128             sproc.setThreadCount(ParseUtil.parseIntOrDefault(split[7], 0));
129             sproc.setPriority(ParseUtil.parseIntOrDefault(split[8], 0));
130             // These are in KB, multiply
131             sproc.setVirtualSize(ParseUtil.parseLongOrDefault(split[9], 0) * 1024);
132             sproc.setResidentSetSize(ParseUtil.parseLongOrDefault(split[10], 0) * 1024);
133             // Avoid divide by zero for processes up less than a second
134             long elapsedTime = ParseUtil.parseDHMSOrDefault(split[11], 0L);
135             sproc.setUpTime(elapsedTime < 1L ? 1L : elapsedTime);
136             sproc.setStartTime(now - sproc.getUpTime());
137             sproc.setUserTime(ParseUtil.parseDHMSOrDefault(split[12], 0L));
138             sproc.setPath(split[13]);
139             sproc.setName(sproc.getPath().substring(sproc.getPath().lastIndexOf('/') + 1));
140             sproc.setCommandLine(split[14]);
141             sproc.setCurrentWorkingDirectory(MapUtil.getOrDefault(cwdMap, sproc.getProcessID(), ""));
142             // bytes read/written not easily available
143             
144             //gets the open files count
145             String openFilesString = ExecutingCommand.getFirstAnswer(String.format("lsof -p %d | wc -l", pid));
146             sproc.setOpenFiles(ParseUtil.parseLongOrDefault(openFilesString, -1));
147                 
148             procs.add(sproc);
149         }
150         return procs;
151     }

153     /**
154      * {@inheritDoc}
155      */
156     @Override
157     public int getProcessId() {
158         return Libc.INSTANCE.getpid();
159     }
160 
161     /**
162      * {@inheritDoc}
163      */
164     @Override
165     public int getProcessCount() {
166         return ProcUtil.getPidFiles().length;
167     }
168 
169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public int getThreadCount() {
174         List<String> threadList = ExecutingCommand.runNative("ps -eLo pid");
175         if (!threadList.isEmpty()) {
176             // Subtract 1 for header
177             return threadList.size() - 1;
178         }
179         return getProcessCount();
180     }

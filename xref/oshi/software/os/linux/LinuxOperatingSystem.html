22  import java.io.IOException;
23  import java.util.List;

31  /**
32   * Linux is a family of free operating systems most commonly used on personal
33   * computers.
34   *
35   * @author alessandro[at]perucchi[dot]org
36   */
37  public class LinuxOperatingSystem extends AbstractOperatingSystem {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(LinuxOperatingSystem.class);
40  
41      private List<String> osRelease;
42  
43      public LinuxOperatingSystem() {
44          this.manufacturer = "GNU/Linux";
45          this.family = parseFamily(); // populates osRelease
46          this.version = (osRelease != null) ? new LinuxOSVersionInfoEx(osRelease) : new LinuxOSVersionInfoEx();
47      }
48  
49      private String parseFamily() {
50          if (this.family == null) {
51              String etcOsRelease = getReleaseFilename();
52              try {
53                  this.osRelease = FileUtil.readFile(etcOsRelease);
54                  for (String line : this.osRelease) {
55                      String[] splittedLine = line.split("=");
56                      if ((splittedLine[0].equals("NAME") || splittedLine[0].equals("DISTRIB_ID"))
57                              && splittedLine.length > 1) {
58                          // remove beginning and ending '"' characters, etc from
59                          // NAME="Ubuntu"
60                          this.family = splittedLine[1].replaceAll("^\"|\"$", "");
61                          break;
62                      }
63                  }
64                  // If we couldn't parse the os-release or lsb-release formats,
65                  // see if we can parse first line of /etc/*-release
66                  if (this.family == null && this.osRelease.size() > 0) {
67                      // Get everything before " release" or " VERSION"
68                      String[] split = this.osRelease.get(0).split("release");
69                      if (split.length > 1) {
70                          this.family = split[0].trim();
71                      } else {
72                          split = this.osRelease.get(0).split("VERSION");
73                          if (split.length > 1) {
74                              this.family = split[0].trim();
75                          }
76                      }
77                  }
78                  // If we've gotten to the end without matching, use the filename
79                  if (this.family == null) {
80                      this.family = filenameToFamily(etcOsRelease.replace("/etc/", "").replace("release", "")
81                              .replace("version", "").replace("-", "").replace("_", ""));
82                  }
83              } catch (IOException e) {
84                  LOG.trace("", e);
85                  return "";
86              }
87          }
88          return this.family;
89  
90      }
91  
92      protected static String getReleaseFilename() {
93          // Check for existence of primary sources of info:
94          if (new File("/etc/os-release").exists()) {
95              return "/etc/os-release";
96          }
97          if (new File("/etc/lsb-release").exists()) {
98              return "/etc/lsb-release";
99          }
100         // Look for any /etc/*-release, *-version, and variants
101         File etc = new File("/etc");
102         File[] files = etc.listFiles(new FilenameFilter() {
103             @Override
104             public boolean accept(File dir, String name) {
105                 return name.endsWith("-release") || name.endsWith("-version") || name.endsWith("_release")
106                         || name.endsWith("_version");
107             }
108         });
109         if (files != null && files.length > 0) {
110             return files[0].getPath();
111         }
112         if (new File("/etc/release").exists()) {
113             return "/etc/release";
114         }
115         // If all else fails, try this
116         return "/etc/issue";
117     }

119     /**
120      * Converts a portion of a filename (e.g. the 'redhat' in
121      * /etc/redhat-release) to a mixed case string representing the family
122      * (e.g., Red Hat)
123      * 
124      * @param name
125      *            Stripped version of filename after removing /etc and -release
126      * @return Mixed case family
127      */
128     private static String filenameToFamily(String name) {
129         switch (name) {
130         // Handle known special cases
131         case "":
132             return "Solaris";
133         case "blackcat":
134             return "Black Cat";
135         case "e-smith":
136             return "SME Server";
137         case "eos":
138             return "FreeEOS";
139         case "hlfs":
140             return "HLFS";
141         case "linuxppc":
142             return "Linux-PPC";
143         case "mklinux":
144             return "MkLinux";
145         case "nld":
146             return "Novell Linux Desktop";
147         case "pld":
148             return "PLD";
149         case "redhat":
150             return "Red Hat";
151         case "novell":
152         case "sles":
153             return "SuSE";
154         case "synoinfo":
155             return "Synology";
156         case "tinysofa":
157             return "Tiny Sofa";
158         case "turbolinux":
159             return "TurboLinux";
160         case "ultrapenguin":
161             return "UltraPenguin";
162         case "va":
163             return "VA-Linux";
164         case "yellowdog":
165             return "Yellow Dog";
166         // /etc/issue will end up here:
167         case "issue":
168             return "Unknown";
169         // If not a special case just capitalize first letter
170         default:
171             return name.substring(0, 1).toUpperCase() + name.substring(1);
172         }
173     }

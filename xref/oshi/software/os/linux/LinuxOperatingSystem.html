36  /**
37   * Linux is a family of free operating systems most commonly used on personal
38   * computers.
39   *
40   * @author alessandro[at]perucchi[dot]org
41   */
42  public class LinuxOperatingSystem implements OperatingSystem {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(LinuxOperatingSystem.class);
45  
46      private OperatingSystemVersion _version;
47  
48      private String _family;
49  
50      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
51  
52      private List<String> osRelease;
53  
54      protected static String getReleaseFilename() {
55          // Check for existence of primary sources of info:
56          if ((new File("/etc/os-release")).exists()) {
57              return "/etc/os-release";
58          }
59          if ((new File("/etc/lsb-release")).exists()) {
60              return "/etc/lsb-release";
61          }
62          // Look for any /etc/*-release, *-version, and variants
63          File etc = new File("/etc");
64          File[] files = etc.listFiles(new FilenameFilter() {
65              @Override
66              public boolean accept(File dir, String name) {
67                  return (name.endsWith("-release") || name.endsWith("-version") || name.endsWith("_release")
68                          || name.endsWith("_version"));
69              }
70          });
71          if (files.length > 0) {
72              return files[0].getPath();
73          }
74          if ((new File("/etc/release")).exists()) {
75              return "/etc/release";
76          }
77          // If all else fails, try this
78          return "/etc/issue";
79      }
80  
81      @Override
82      public String getFamily() {
83          if (this._family == null) {
84              String etcOsRelease = getReleaseFilename();
85              try {
86                  this.osRelease = FileUtil.readFile(etcOsRelease);
87                  for (String line : this.osRelease) {
88                      String[] splittedLine = line.split("=");
89                      if ((splittedLine[0].equals("NAME") || splittedLine[0].equals("DISTRIB_ID"))
90                              && splittedLine.length > 1) {
91                          // remove beginning and ending '"' characters, etc from
92                          // NAME="Ubuntu"
93                          this._family = splittedLine[1].replaceAll("^\"|\"$", "");
94                          break;
95                      }
96                  }
97                  // If we couldn't parse the os-release or lsb-release formats,
98                  // see if we can parse first line of /etc/*-release
99                  if (this._family == null && this.osRelease.size() > 0) {
100                     // Get everything before " release" or " VERSION"
101                     String[] split = this.osRelease.get(0).split("release");
102                     if (split.length > 1) {
103                         this._family = split[0].trim();
104                     } else {
105                         split = this.osRelease.get(0).split("VERSION");
106                         if (split.length > 1) {
107                             this._family = split[0].trim();
108                         }
109                     }
110                 }
111                 // If we've gotten to the end without matching, use the filename
112                 if (this._family == null) {
113                     this._family = filenameToFamily(etcOsRelease.replace("/etc/", "").replace("release", "")
114                             .replace("version", "").replace("-", "").replace("_", ""));
115                 }
116             } catch (IOException e) {
117                 LOG.trace("", e);
118                 return "";
119             }
120         }
121         return this._family;
122 
123     }
124 
125     /**
126      * Converts a portion of a filename (e.g. the 'redhat' in
127      * /etc/redhat-release) to a mixed case string representing the family
128      * (e.g., Red Hat)
129      * 
130      * @param name
131      *            Stripped version of filename after removing /etc and -release
132      * @return Mixed case family
133      */
134     private String filenameToFamily(String name) {
135         switch (name) {
136         // Handle known special cases
137         case "":
138             return "Solaris";
139         case "blackcat":
140             return "Black Cat";
141         case "e-smith":
142             return "SME Server";
143         case "eos":
144             return "FreeEOS";
145         case "hlfs":
146             return "HLFS";
147         case "linuxppc":
148             return "Linux-PPC";
149         case "mklinux":
150             return "MkLinux";
151         case "nld":
152             return "Novell Linux Desktop";
153         case "pld":
154             return "PLD";
155         case "redhat":
156             return "Red Hat";
157         case "novell":
158         case "sles":
159             return "SuSE";
160         case "synoinfo":
161             return "Synology";
162         case "tinysofa":
163             return "Tiny Sofa";
164         case "turbolinux":
165             return "TurboLinux";
166         case "ultrapenguin":
167             return "UltraPenguin";
168         case "va":
169             return "VA-Linux";
170         case "yellowdog":
171             return "Yellow Dog";
172         // /etc/issue will end up here:
173         case "issue":
174             return "Unknown";
175         // If not a special case just capitalize first letter
176         default:
177             return name.substring(0, 1).toUpperCase() + name.substring(1);
178         }
179     }
180 
181     @Override
182     public String getManufacturer() {
183         return "GNU/Linux";
184     }
185 
186     @Override
187     public OperatingSystemVersion getVersion() {
188         if (this._version == null) {
189             this._version = (osRelease != null) ? new LinuxOSVersionInfoEx(osRelease) : new LinuxOSVersionInfoEx();
190         }
191         return this._version;
192     }
193 
194     @Override
195     public JsonObject toJSON() {
196         return NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder()).add("manufacturer", getManufacturer())
197                 .add("family", getFamily()).add("version", getVersion().toJSON()).build();
198     }
199 
200     @Override
201     public String toString() {
202         StringBuilder sb = new StringBuilder();
203         sb.append(getManufacturer());
204         sb.append(" ");
205         sb.append(getFamily());
206         sb.append(" ");
207         sb.append(getVersion().toString());
208         return sb.toString();
209     }
210 }

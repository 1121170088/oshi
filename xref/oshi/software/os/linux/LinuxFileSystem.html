20  import java.io.IOException;
21  import java.nio.file.FileStore;
22  import java.nio.file.FileSystems;
23  import java.util.ArrayList;
24  import java.util.HashMap;
25  import java.util.List;
26  import java.util.Map;
27  
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import oshi.software.common.AbstractFileSystem;
32  import oshi.software.os.OSFileStore;
33  import oshi.util.FileUtil;
34  
35  /**
36   * The Mac File System contains {@link OSFileStore}s which are a storage pool,
37   * device, partition, volume, concrete file system or other implementation
38   * specific means of file storage. In Linux, these are found in the /proc/mount
39   * filesystem, excluding temporary and kernel mounts.
40   * 
41   * @author widdis[at]gmail[dot]com
42   */
43  public class LinuxFileSystem extends AbstractFileSystem {
44  
45      private static final Logger LOG = LoggerFactory.getLogger(LinuxFileSystem.class);
46  
47      /**
48       * Gets File System Information.
49       * 
50       * @return An array of {@link FileStore} objects representing mounted
51       *         volumes. May return disconnected volumes with
52       *         {@link OSFileStore#getTotalSpace()} = 0.
53       */
54      public OSFileStore[] getFileStores() {
55          // Parse /proc/self/mounts to map filesystem paths to types
56          Map<String, String> fstype = new HashMap<>();
57          try {
58              List<String> mounts = FileUtil.readFile("/proc/self/mounts");
59              for (String mount : mounts) {
60                  String[] split = mount.split(" ");
61                  // 2nd field is path with spaces escaped as \040
62                  // 3rd field is fs type
63                  if (split.length < 6) {
64                      continue;
65                  }
66                  fstype.put(split[1].replaceAll("\\\\040", " "), split[2]);
67              }
68          } catch (IOException e) {
69              LOG.error("Error reading /proc/self/mounts. Can't detect filetypes.");
70          }
71          // Format
72          // Now list file systems
73          List<OSFileStore> fsList = new ArrayList<>();
74          for (FileStore store : FileSystems.getDefault().getFileStores()) {
75              // FileStore toString starts with path, then a space, then name in
76              // parentheses e.g., "/ (/dev/sda1)" and "/proc (proc)"
77              String path = store.toString().replace(" (" + store.name() + ")", "");
78              // Exclude special directories
79              if (path.startsWith("/proc") || path.startsWith("/sys") || path.startsWith("/run") || path.equals("/dev")
80                      || path.equals("/dev/pts"))
81                  continue;
82              String name = store.name();
83              if (path.equals("/"))
84                  name = "/";
85              String description = "Mount Point";
86              if (store.name().startsWith("/dev"))
87                  description = "Local Disk";
88              String type = "unknown";
89              if (fstype.containsKey(path)) {
90                  type = fstype.get(path);
91              }
92              try {
93                  fsList.add(new OSFileStore(name, description, type, store.getUsableSpace(), store.getTotalSpace()));
94              } catch (IOException e) {
95                  // get*Space() may fail for ejected CD-ROM, etc.
96                  LOG.trace("", e);
97                  continue;
98              }
99          }
100         return fsList.toArray(new OSFileStore[fsList.size()]);
101     }

21  import java.io.Serializable;

23  /**
24   * The Central Processing Unit (CPU) or the processor is the portion of a
25   * computer system that carries out the instructions of a computer program, and
26   * is the primary element carrying out the computer's functions.
27   *
28   * @author dblock[at]dblock[dot]org
29   */
30  public interface CentralProcessor extends Serializable {
31  
32      /**
33       * Index of CPU tick counters in the {@link #getSystemCpuLoadTicks()} and
34       * {@link #getProcessorCpuLoadTicks()} arrays.
35       */
36      enum TickType {
37          /**
38           * CPU utilization that occurred while executing at the user level
39           * (application).
40           */
41          USER(0),
42          /**
43           * CPU utilization that occurred while executing at the user level with
44           * nice priority.
45           */
46          NICE(1),
47          /**
48           * CPU utilization that occurred while executing at the system level
49           * (kernel). Also includes CPU time which the hypervisor dedicated for
50           * other guests in the system (steal).
51           */
52          SYSTEM(2),
53          /**
54           * Time that the CPU or CPUs were idle and the system did not have an
55           * outstanding disk I/O request.
56           */
57          IDLE(3),
58          /**
59           * Time that the CPU or CPUs were idle during which the system had an
60           * outstanding disk I/O request.
61           */
62          IOWAIT(4),
63          /**
64           * Time that the CPU used to service hardware IRQs
65           */
66          IRQ(5),
67          /**
68           * Time that the CPU used to service soft IRQs
69           */
70          SOFTIRQ(6);
71  
72          private int index;
73  
74          TickType(int value) {
75              this.index = value;
76          }
77  
78          /**
79           * @return The integer index of this ENUM in the processor tick arrays,
80           *         which matches the output of Linux /proc/cpuinfo
81           */
82          public int getIndex() {
83              return index;
84          }
85      }

196     /**
197      * Returns the "recent cpu usage" for the whole system by counting ticks
198      * from {@link #getSystemCpuLoadTicks()} between successive calls of this
199      * method, with a minimum interval slightly less than 1 second. If less than
200      * one second has elapsed since the last call of this method, it will return
201      * a calculation based on the tick counts and times of the previous two
202      * calls. If at least a second has elapsed, it will return the average CPU
203      * load for the interval and update the "last called" times. This method is
204      * intended to be used for periodic polling at intervals of 1 second or
205      * longer.
206      *
207      * @return CPU load between 0 and 1 (100%)
208      */
209     double getSystemCpuLoadBetweenTicks();
210 
211     /**
212      * Get System-wide CPU Load tick counters. Returns an array with seven
213      * elements representing either clock ticks or milliseconds (platform
214      * dependent) spent in User (0), Nice (1), System (2), Idle (3), IOwait (4),
215      * Hardware interrupts (IRQ) (5), or Software interrupts/DPC (SoftIRQ) (6)
216      * states. Use {@link TickType#getIndex()} to retrieve the appropriate
217      * index. By measuring the difference between ticks across a time interval,
218      * CPU load over that interval may be calculated.
219      *
220      * Nice and IOWait information is not available on Windows, and IOwait and
221      * IRQ information is not available on macOS, so these ticks will always be
222      * zero.
223      *
224      * To calculate overall Idle time using this method, include both Idle and
225      * IOWait ticks. Similarly, IRQ and SoftIRQ ticks should be added to the
226      * System value to get the total. System ticks also include time executing
227      * other virtual hosts (steal).
228      *
229      * @return An array of 7 long values representing time spent in User, Nice,
230      *         System, Idle, IOwait, IRQ, and SoftIRQ states.
231      */
232     long[] getSystemCpuLoadTicks();
233 
234     /**
235      * Returns the "recent cpu usage" for the whole system from
236      * {@link com.sun.management.OperatingSystemMXBean#getSystemCpuLoad()} if a
237      * user is running the Oracle JVM. This value is a double in the [0.0,1.0]
238      * interval. A value of 0.0 means that all CPUs were idle during the recent
239      * period of time observed, while a value of 1.0 means that all CPUs were
240      * actively running 100% of the time during the recent period being
241      * observed. All values between 0.0 and 1.0 are possible depending of the
242      * activities going on in the system. If the system recent cpu usage is not
243      * available, the method returns a negative value. Calling this method
244      * immediately upon instantiating the {@link CentralProcessor} may give
245      * unreliable results. If a user is not running the Oracle JVM, this method
246      * will default to the behavior and return value of
247      * {@link #getSystemCpuLoadBetweenTicks()}.
248      *
249      * @return the "recent cpu usage" for the whole system; a negative value if
250      *         not available.
251      */
252     @SuppressWarnings("restriction")
253     double getSystemCpuLoad();
254 
255     /**
256      * Returns the system load average for the last minute. This is equivalent
257      * to calling {@link CentralProcessor#getSystemLoadAverage(int)} with an
258      * argument of 1 and returning the first value, and is retained for
259      * compatibility.
260      *
261      * @return the system load average; or a negative value if not available.
262      */
263     double getSystemLoadAverage();
264 
265     /**
266      * Returns the system load average for the number of elements specified, up
267      * to 3, representing 1, 5, and 15 minutes. The system load average is the
268      * sum of the number of runnable entities queued to the available processors
269      * and the number of runnable entities running on the available processors
270      * averaged over a period of time. The way in which the load average is
271      * calculated is operating system specific but is typically a damped
272      * time-dependent average. If the load average is not available, a negative
273      * value is returned. This method is designed to provide a hint about the
274      * system load and may be queried frequently. The load average may be
275      * unavailable on some platforms (e.g., Windows) where it is expensive to
276      * implement this method.
277      *
278      * @param nelem
279      *            Number of elements to return.
280      * @return an array of the system load averages for 1, 5, and 15 minutes
281      *         with the size of the array specified by nelem; or negative values
282      *         if not available.
283      */
284     double[] getSystemLoadAverage(int nelem);
285 
286     /**
287      * Returns the "recent cpu usage" for all logical processors by counting
288      * ticks for the processors from {@link #getProcessorCpuLoadTicks()} between
289      * successive calls of this method, with a minimum interval slightly less
290      * than 1 second. If less than one second has elapsed since the last call of
291      * this method, it will return a calculation based on the tick counts and
292      * times of the previous two calls. If at least a second has elapsed, it
293      * will return the average CPU load for the interval and update the
294      * "last called" times. This method is intended to be used for periodic
295      * polling (iterating over all processors) at intervals of 1 second or
296      * longer.
297      *
298      * @return array of CPU load between 0 and 1 (100%) for each logical
299      *         processor
300      */
301     double[] getProcessorCpuLoadBetweenTicks();
302 
303     /**
304      * Get Processor CPU Load tick counters. Returns a two dimensional array,
305      * with {@link #getLogicalProcessorCount()} arrays, each containing seven
306      * elements representing either clock ticks or milliseconds (platform
307      * dependent) spent in User (0), Nice (1), System (2), Idle (3), IOwait (4),
308      * Hardware interrupts (IRQ) (5), or Software interrupts/DPC (SoftIRQ) (6)
309      * states. Use {@link TickType#getIndex()} to retrieve the appropriate
310      * index. By measuring the difference between ticks across a time interval,
311      * CPU load over that interval may be calculated.
312      *
313      * Nice and IOwait per processor information is not available on Windows,
314      * and IOwait and IRQ information is not available on macOS, so these ticks
315      * will always be zero.
316      *
317      * To calculate overall Idle time using this method, include both Idle and
318      * IOWait ticks. Similarly, IRQ and SoftIRQ ticks should be added to the
319      * System value to get the total. System ticks also include time executing
320      * other virtual hosts (steal).
321      *
322      * @return A 2D array of logicalProcessorCount x 7 long values representing
323      *         time spent in User, Nice, System, Idle, IOwait, IRQ, and SoftIRQ
324      *         states.
325      */
326     long[][] getProcessorCpuLoadTicks();
327 
328     /**
329      * Get the System uptime (time since boot).
330      *
331      * @return Number of seconds since boot.
332      */
333     long getSystemUptime();

335     /**
336      * Get the System/CPU Serial Number, if available.
337      *
338      * On Linux and FreeBSD, this requires either root permissions, installation
339      * of the (deprecated) HAL library (lshal command). Linux also attempts to
340      * read the dmi/id serial number files in sysfs, which are read-only root by
341      * default but may have permissions altered by the user.
342      *
343      * @return the System/CPU Serial Number, if available, otherwise returns
344      *         "unknown"
345      */
346     String getSystemSerialNumber();
347

349      * Get the number of logical CPUs available for processing.
350      *
351      * @return The number of logical CPUs available.
352      */
353     int getLogicalProcessorCount();
354 
355     /**
356      * Get the number of physical CPUs/cores available for processing.
357      *
358      * @return The number of physical CPUs available.
359      */
360     int getPhysicalProcessorCount();

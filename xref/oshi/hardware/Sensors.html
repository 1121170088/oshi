17  package oshi.hardware;
18  
19  import oshi.json.OshiJsonObject;
20  
21  /**
22   * Sensors include hardwore sensors to monitor temperature, fan speed, and other
23   * information. Drivers may or may not exist to collect this data depending on
24   * Operating System and CPU. In addition, software-hardware communication may
25   * suffer intermittent errors when attempting to access this information, so it
26   * should be considered for information only. Users should test for zero values
27   * and empty arrays which will result if the OS is unable to provide the
28   * information.
29   * 
30   * Windows information is retrieved via Windows Management Instrumentation
31   * (WMI). Unfortunately, most hardware providers do not publish values to WMI.
32   * If a value is not available through the Microsoft API, Oshi will attempt to
33   * retrieve values as published by the Open Hardware Monitor
34   * (http://openhardwaremonitor.org/) if it is running.
35   * 
36   * @author widdis[at]gmail[dot]com
37   */
38  public interface Sensors extends OshiJsonObject {
39      /**
40       * CPU Temperature
41       * 
42       * @return CPU Temperature in degrees Celsius if available, 0 otherwise.
43       */
44      double getCpuTemperature();
45  
46      /**
47       * Fan speeds
48       * 
49       * @return Speed in rpm for all fans. May return empty array if no fans
50       *         detected or 0 fan speed if unable to measure fan speed.
51       */
52      int[] getFanSpeeds();
53  
54      /**
55       * CPU Voltage
56       * 
57       * @return CPU Voltage in Volts if available, 0 otherwise.
58       */
59      double getCpuVoltage();
60  }

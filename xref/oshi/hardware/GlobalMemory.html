19  package oshi.hardware;
20  
21  import java.io.Serializable;
22  
23  /**
24   * Memory refers to the state information of a computing system, as it is kept
25   * active in some physical structure. The term "memory" is used for the
26   * information in physical systems which are fast (ie. RAM), as a distinction
27   * from physical systems which are slow to access (ie. data storage). By design,
28   * the term "memory" refers to temporary state devices, whereas the term
29   * "storage" is reserved for permanent data.
30   *
31   * @author dblock[at]dblock[dot]org
32   */
33  public interface GlobalMemory extends Serializable {
34      /**
35       * The amount of actual physical memory, in bytes.
36       *
37       * @return Total number of bytes.
38       */
39      long getTotal();
40  
41      /**
42       * The amount of physical memory currently available, in bytes.
43       *
44       * @return Available number of bytes.
45       */
46      long getAvailable();
47  
48      /**
49       * The current size of the paging/swap file(s), in bytes. If the paging/swap
50       * file can be extended, this is a soft limit.
51       *
52       * @return Total swap in bytes.
53       */
54      long getSwapTotal();
55  
56      /**
57       * The current memory committed to the paging/swap file(s), in bytes
58       *
59       * @return Swap used in bytes
60       */
61      long getSwapUsed();
62  }

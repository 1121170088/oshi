36  /**
37   * Memory info.
38   */
39  public abstract class AbstractGlobalMemory implements GlobalMemory {
40  
41      @Override
42      public PhysicalMemory[] getPhysicalMemory() {
43          // dmidecode requires sudo permission but is the only option on Linux
44          // and Unix
45          List<PhysicalMemory> pmList = new ArrayList<>();
46          List<String> dmi = ExecutingCommand.runNative("dmidecode --type 17");
47          int bank = 0;
48          String bankLabel = Constants.UNKNOWN;
49          String locator = "";
50          long capacity = 0L;
51          long speed = 0L;
52          String manufacturer = Constants.UNKNOWN;
53          String memoryType = Constants.UNKNOWN;
54          for (String line : dmi) {
55              if (line.trim().contains("DMI type 17")) {
56                  // Save previous bank
57                  if (bank++ > 0) {
58                      if (capacity > 0) {
59                          pmList.add(new PhysicalMemory(bankLabel + locator, capacity, speed, manufacturer, memoryType));
60                      }
61                      bankLabel = Constants.UNKNOWN;
62                      locator = "";
63                      capacity = 0L;
64                      speed = 0L;
65                  }
66              } else if (bank > 0) {
67                  String[] split = line.trim().split(":");
68                  if (split.length == 2) {
69                      switch (split[0]) {
70                      case "Bank Locator":
71                          bankLabel = split[1].trim();
72                          break;
73                      case "Locator":
74                          locator = "/" + split[1].trim();
75                          break;
76                      case "Size":
77                          capacity = parsePhysicalMemorySize(split[1].trim());
78                          break;
79                      case "Type":
80                          memoryType = split[1].trim();
81                          break;
82                      case "Speed":
83                          speed = ParseUtil.parseHertz(split[1]);
84                          break;
85                      case "Manufacturer":
86                          manufacturer = split[1].trim();
87                          break;
88                      default:
89                          break;
90                      }
91                  }
92              }
93          }
94          if (capacity > 0) {
95              pmList.add(new PhysicalMemory(bankLabel + locator, capacity, speed, manufacturer, memoryType));
96          }
97          return pmList.toArray(new PhysicalMemory[0]);
98      }

100     /**
101      * Parses a string such as "4096 MB" to its long. Used to parse macOS and *nix
102      * memory chip sizes. Although the units given are decimal they must parse to
103      * binary units.
104      *
105      * @param size
106      *            A string of memory sizes like "4096 MB"
107      * @return the size parsed to a long
108      */
109     protected long parsePhysicalMemorySize(String size) {
110         String[] mem = ParseUtil.whitespaces.split(size);
111         long capacity = ParseUtil.parseLongOrDefault(mem[0], 0L);
112         if (mem.length == 2 && mem[1].length() > 1) {
113             switch (mem[1].charAt(0)) {
114             case 'T':
115                 capacity <<= 40;
116                 break;
117             case 'G':
118                 capacity <<= 30;
119                 break;
120             case 'M':
121                 capacity <<= 20;
122                 break;
123             case 'K':
124             case 'k':
125                 capacity <<= 10;
126                 break;
127             default:
128                 break;
129             }
130         }
131         return capacity;
132     }

134     /**
135      * Convert SMBIOS type number to a human readable string
136      *
137      * @param type
138      *            The SMBIOS type
139      * @return A string describing the type
140      */
141     protected static String smBiosMemoryType(int type) {
142         // https://www.dmtf.org/sites/default/files/standards/documents/DSP0134_3.2.0.pdf
143         // table 76
144         switch (type) {
145         case 0x01:
146             return "Other";
147         case 0x03:
148             return "DRAM";
149         case 0x04:
150             return "EDRAM";
151         case 0x05:
152             return "VRAM";
153         case 0x06:
154             return "SRAM";
155         case 0x07:
156             return "RAM";
157         case 0x08:
158             return "ROM";
159         case 0x09:
160             return "FLASH";
161         case 0x0A:
162             return "EEPROM";
163         case 0x0B:
164             return "FEPROM";
165         case 0x0C:
166             return "EPROM";
167         case 0x0D:
168             return "CDRAM";
169         case 0x0E:
170             return "3DRAM";
171         case 0x0F:
172             return "SDRAM";
173         case 0x10:
174             return "SGRAM";
175         case 0x11:
176             return "RDRAM";
177         case 0x12:
178             return "DDR";
179         case 0x13:
180             return "DDR2";
181         case 0x14:
182             return "DDR2 FB-DIMM";
183         case 0x18:
184             return "DDR3";
185         case 0x19:
186             return "FBD2";
187         case 0x1A:
188             return "DDR4";
189         case 0x1B:
190             return "LPDDR";
191         case 0x1C:
192             return "LPDDR2";
193         case 0x1D:
194             return "LPDDR3";
195         case 0x1E:
196             return "LPDDR4";
197         case 0x1F:
198             return "Logical non-volatile device";
199         case 0x02:
200         default:
201             return "Unknown";
202         }
203     }
204 
205     @Override
206     public String toString() {
207         StringBuilder sb = new StringBuilder();
208         sb.append("Available: ");
209         sb.append(FormatUtil.formatBytes(getAvailable()));
210         sb.append("/");
211         sb.append(FormatUtil.formatBytes(getTotal()));
212         return sb.toString();
213     }

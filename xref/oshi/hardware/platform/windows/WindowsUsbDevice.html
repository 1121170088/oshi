50      public WindowsUsbDevice(String name, String vendor, String serialNumber, UsbDevice[] connectedDevices) {
51          super(name, vendor, serialNumber, connectedDevices);
52      }

54      /*
55       * Maps to store information using PNPDeviceID as the key
56       */
57      private static Map<String, String> nameMap = new HashMap<>();
58      private static Map<String, String> vendorMap = new HashMap<>();
59      private static Map<String, String> serialMap = new HashMap<>();
60      private static Map<String, List<String>> hubMap = new HashMap<>();
61  
62      /**
63       * {@inheritDoc}
64       */
65      public static UsbDevice[] getUsbDevices() {
66          // Start by collecting information for all PNP devices. While in theory
67          // these could be individually queried with a WHERE clause, grabbing
68          // them all up front incurs minimal memory overhead in exchange for
69          // faster access later
70  
71          // Clear maps
72          nameMap.clear();
73          vendorMap.clear();
74          serialMap.clear();
75  
76          // Query Win32_PnPEntity to populate the maps
77          Map<String, List<String>> usbMap = WmiUtil.selectStringsFrom(null, "Win32_PnPEntity",
78                  "Name,Manufacturer,PnPDeviceID", null);
79          for (int i = 0; i < usbMap.get("Name").size(); i++) {
80              String pnpDeviceID = usbMap.get("PnPDeviceID").get(i);
81              nameMap.put(pnpDeviceID, usbMap.get("Name").get(i));
82              if (usbMap.get("Manufacturer").get(i).length() > 0) {
83                  vendorMap.put(pnpDeviceID, usbMap.get("Manufacturer").get(i));
84              }
85          }
86  
87          // Get serial # for disk drives or other physical media
88          usbMap = WmiUtil.selectStringsFrom(null, "Win32_DiskDrive", "PNPDeviceID,SerialNumber", null);
89          for (int i = 0; i < usbMap.get("PNPDeviceID").size(); i++) {
90              serialMap.put(usbMap.get("PNPDeviceID").get(i),
91                      ParseUtil.hexStringToString(usbMap.get("PNPDeviceID").get(i)));
92          }
93          usbMap = WmiUtil.selectStringsFrom(null, "Win32_PhysicalMedia", "PNPDeviceID,SerialNumber", null);
94          for (int i = 0; i < usbMap.get("PNPDeviceID").size(); i++) {
95              serialMap.put(usbMap.get("PNPDeviceID").get(i),
96                      ParseUtil.hexStringToString(usbMap.get("PNPDeviceID").get(i)));
97          }
98  
99          // Build the device tree. Start with the USB Controllers
100         // and recurse downward to devices as needed
101         usbMap = WmiUtil.selectStringsFrom(null, "Win32_USBController", "PNPDeviceID", null);
102         List<UsbDevice> controllerDevices = new ArrayList<UsbDevice>();
103         for (String controllerDeviceId : usbMap.get("PNPDeviceID")) {
104             putChildrenInDeviceTree(controllerDeviceId, 0);
105             controllerDevices.add(getDeviceAndChildren(controllerDeviceId));
106         }
107         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
108     }
109 
110     /**
111      * Navigates the Device Tree to place all children PNPDeviceIDs into the map
112      * for the specified deviceID. Recursively adds children's children, etc.
113      * 
114      * @param deviceId
115      *            The device to add respective children to the map
116      * @param devInst
117      *            The device instance (devnode handle), if known. If set to 0,
118      *            the code will search for a match.
119      */
120     private static void putChildrenInDeviceTree(String deviceId, int devInst) {
121         // If no devInst provided, find it by matching deviceId
122         if (devInst == 0) {
123             // Get a handle to the device with this deviceId
124             // Start with all classes
125             HANDLE hinfoSet = SetupApi.INSTANCE.SetupDiGetClassDevs(null, null, null, SetupApi.DIGCF_ALLCLASSES);
126             if (hinfoSet == WinNT.INVALID_HANDLE_VALUE) {
127                 LOG.error("Invalid handle value for {}. Error code: {}", deviceId, Native.getLastError());
128                 return;
129             }
130             // Iterate to find matching parent
131             SP_DEVINFO_DATA dinfo = new SP_DEVINFO_DATA();
132             dinfo.cbSize = dinfo.size();
133             int i = 0;
134             while (SetupApi.INSTANCE.SetupDiEnumDeviceInfo(hinfoSet, i++, dinfo)) {
135                 if (deviceId.equals(getDeviceId(dinfo.DevInst))) {
136                     devInst = dinfo.DevInst;
137                     break;
138                 }
139             }
140         }
141         if (devInst == 0) {
142             LOG.error("Unable to find a devnode handle for {}.", deviceId);
143             return;
144         }
145         // Now iterate the children. Call CM_Get_Child to get first child
146         IntByReference child = new IntByReference();
147         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
148             // Add first child to a list
149             List<String> childList = new ArrayList<>();
150             String childId = getDeviceId(child.getValue());
151             childList.add(childId);
152             hubMap.put(deviceId, childList);
153             putChildrenInDeviceTree(childId, child.getValue());
154             // Find any other children
155             IntByReference sibling = new IntByReference();
156             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
157                 // Add to the list
158                 String siblingId = getDeviceId(sibling.getValue());
159                 hubMap.get(deviceId).add(siblingId);
160                 putChildrenInDeviceTree(siblingId, sibling.getValue());
161                 // Make this sibling the new child to find other siblings
162                 child = sibling;
163             }
164         }
165     }

167     /**
168      * Gets the device id for a devnode
169      * 
170      * @param devInst
171      *            the handle to the devnode
172      * @return The PNPDeviceID
173      */
174     private static String getDeviceId(int devInst) {
175         NativeLongByReference ulLen = new NativeLongByReference();
176         if (0 != Cfgmgr32.INSTANCE.CM_Get_Device_ID_Size(ulLen, devInst, 0)) {
177             LOG.error("Couldn't get device string for device instance {}", devInst);
178             return "";
179         }
180         // Add 1 for null terminator
181         int size = ulLen.getValue().intValue() + 1;
182         char[] buffer = new char[size];
183         if (0 != Cfgmgr32.INSTANCE.CM_Get_Device_ID(devInst, buffer, size, 0)) {
184             LOG.error("Couldn't get device string for device instance {} with size {}", devInst, size);
185             return "";
186         }
187         return new String(buffer).trim();
188     }

190     /**
191      * Recursively creates WindowsUsbDevices by fetching information from maps
192      * to populate fields
193      * 
194      * @param hubDeviceID
195      *            The PNPdeviceID of this device.
196      * @return A WindowsUsbDevice corresponding to this deviceID
197      */
198     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceID) {
199         List<String> pnpDeviceIDs = hubMap.containsKey(hubDeviceID) ? hubMap.get(hubDeviceID) : new ArrayList<String>();
200         List<WindowsUsbDevice> usbDevices = new ArrayList<>();
201         for (String pnpDeviceID : pnpDeviceIDs) {
202             usbDevices.add(getDeviceAndChildren(pnpDeviceID));
203         }
204         Collections.sort(usbDevices);
205         return new WindowsUsbDevice(nameMap.containsKey(hubDeviceID) ? nameMap.get(hubDeviceID) : "",
206                 vendorMap.containsKey(hubDeviceID) ? vendorMap.get(hubDeviceID) : "",
207                 serialMap.containsKey(hubDeviceID) ? serialMap.get(hubDeviceID) : "",
208                 usbDevices.toArray(new UsbDevice[usbDevices.size()]));
209     }

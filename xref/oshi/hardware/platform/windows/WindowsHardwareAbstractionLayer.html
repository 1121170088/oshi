17  package oshi.hardware.platform.windows;
18  
19  import javax.json.Json;
20  import javax.json.JsonArrayBuilder;
21  import javax.json.JsonBuilderFactory;
22  import javax.json.JsonObject;
23  
24  import com.sun.jna.platform.win32.Advapi32Util;
25  import com.sun.jna.platform.win32.WinReg;
26  
27  import oshi.hardware.CentralProcessor;
28  import oshi.hardware.Display;
29  import oshi.hardware.GlobalMemory;
30  import oshi.hardware.HardwareAbstractionLayer;
31  import oshi.hardware.PowerSource;
32  import oshi.software.os.OSFileStore;
33  import oshi.software.os.windows.WindowsFileSystem;
34  
35  public class WindowsHardwareAbstractionLayer implements HardwareAbstractionLayer {
36  
37      private CentralProcessor processor;
38  
39      private GlobalMemory _memory;
40  
41      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
42  
43      @Override
44      public GlobalMemory getMemory() {
45          if (this._memory == null) {
46              this._memory = new WindowsGlobalMemory();
47          }
48          return this._memory;
49      }
50  
51      @Override
52      public CentralProcessor getProcessor() {
53          if (this.processor == null) {
54              processor = new WindowsCentralProcessor();
55              final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor";
56              String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
57              if (processorIds.length > 0) {
58                  String cpuRegistryPath = cpuRegistryRoot + "\\" + processorIds[0];
59                  processor.setIdentifier(
60                          Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
61                  processor.setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
62                          "ProcessorNameString"));
63                  processor.setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
64                          "VendorIdentifier"));
65              }
66          }
67          return this.processor;
68      }
69  
70      @Override
71      public PowerSource[] getPowerSources() {
72          return WindowsPowerSource.getPowerSources();
73      }
74  
75      @Override
76      public OSFileStore[] getFileStores() {
77          return WindowsFileSystem.getFileStores();
78      }
79  
80      @Override
81      public Display[] getDisplays() {
82          return WindowsDisplay.getDisplays();
83      }
84  
85      @Override
86      public JsonObject toJSON() {
87          JsonArrayBuilder powerSourceArrayBuilder = jsonFactory.createArrayBuilder();
88          for (PowerSource powerSource : getPowerSources()) {
89              powerSourceArrayBuilder.add(powerSource.toJSON());
90          }
91          JsonArrayBuilder fileStoreArrayBuilder = jsonFactory.createArrayBuilder();
92          for (OSFileStore fileStore : getFileStores()) {
93              fileStoreArrayBuilder.add(fileStore.toJSON());
94          }
95          JsonArrayBuilder displayArrayBuilder = jsonFactory.createArrayBuilder();
96          for (Display display : getDisplays()) {
97              displayArrayBuilder.add(display.toJSON());
98          }
99          return jsonFactory.createObjectBuilder().add("processor", getProcessor().toJSON())
100                 .add("memory", getMemory().toJSON()).add("powerSources", powerSourceArrayBuilder.build())
101                 .add("fileStores", fileStoreArrayBuilder.build()).add("displays", displayArrayBuilder.build()).build();
102     }
103 }

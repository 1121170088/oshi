168     /**
169      * Create a Processor

171     public WindowsCentralProcessor() {
172         // Processor counts
173         calculateProcessorCounts();
174 
175         // PDH counter setup
176         initPdhCounters();
177 
178         // System ticks
179         this.prevTicks = new long[4];
180         this.curTicks = new long[4];
181         updateSystemTicks();
182 
183         // Per-processor ticks
184         this.allProcTicks = new long[logicalProcessorCount][4];
185         this.prevProcTicks = new long[logicalProcessorCount][4];
186         this.curProcTicks = new long[logicalProcessorCount][4];
187         updateProcessorTicks();
188 
189         // Get and discard initial read to reset timers
190         getSystemIOWaitTicks();
191         getSystemIrqTicks();
192 
193         LOG.debug("Initialized Processor");
194     }

196     /**
197      * Updates logical and physical processor counts from /proc/cpuinfo
198      */
199     private void calculateProcessorCounts() {
200         // Get number of logical processors
201         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
202         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
203         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
204 
205         // Get number of physical processors
206         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
207         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
208             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
209                 this.physicalProcessorCount++;
210             }
211         }
212     }

214     /**
215      * Initializes PDH Tick and Uptime Counters
216      */
217     private void initPdhCounters() {
218         // Open tick query
219         int pdhOpenTickQueryError = Pdh.INSTANCE.PdhOpenQuery(null, pZero, phQuery);
220         if (pdhOpenTickQueryError != 0) {
221             LOG.error("Failed to open PDH Tick Query. Error code: {}", String.format("0x%08X", pdhOpenTickQueryError));
222         }
223 
224         // Set up counters
225         phUserCounters = new PointerByReference[logicalProcessorCount];
226         phIdleCounters = new PointerByReference[logicalProcessorCount];
227 
228         if (pdhOpenTickQueryError == 0) {
229             for (int p = 0; p < logicalProcessorCount; p++) {
230                 // Options are (only need 2 to calculate all)
231                 // "\Processor(0)\% processor time"
232                 // "\Processor(0)\% idle time" (1 - processor)
233                 // "\Processor(0)\% privileged time" (subset of processor)
234                 // "\Processor(0)\% user time" (other subset of processor)
235                 // Note need to make \ = \\ for Java Strings and %% for format
236                 String counterPath = String.format("\\Processor(%d)\\%% user time", p);
237                 phUserCounters[p] = new PointerByReference();
238                 // Open tick query for this processor
239                 int pdhAddTickCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(phQuery.getValue(), counterPath, pZero,
240                         phUserCounters[p]);
241                 if (pdhAddTickCounterError != 0) {
242                     LOG.error("Failed to add PDH User Tick Counter for processor {}. Error code: {}", p,
243                             String.format("0x%08X", pdhAddTickCounterError));
244                     break;
245                 }
246                 counterPath = String.format("\\Processor(%d)\\%% idle time", p);
247                 phIdleCounters[p] = new PointerByReference();
248                 pdhAddTickCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(phQuery.getValue(), counterPath, pZero,
249                         phIdleCounters[p]);
250                 if (pdhAddTickCounterError != 0) {
251                     LOG.error("Failed to add PDH Idle Tick Counter for processor {}. Error code: {}", p,
252                             String.format("0x%08X", pdhAddTickCounterError));
253                     break;
254                 }
255             }
256 
257             LOG.debug("Tick counter queries added.  Initializing with first query.");
258             // Initialize by collecting data the first time
259             int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
260             if (ret != 0) {
261                 LOG.warn("Failed to update Tick Counters. Error code: {}", String.format("0x%08X", ret));
262             }
263         }
264 
265         // Open uptime query
266         int pdhOpenUptimeQueryError = Pdh.INSTANCE.PdhOpenQuery(null, pOne, uptimeQuery);
267         if (pdhOpenUptimeQueryError != 0) {
268             LOG.error("Failed to open PDH Uptime Query. Error code: {}",
269                     String.format("0x%08X", pdhOpenUptimeQueryError));
270         }
271         if (pdhOpenUptimeQueryError == 0) {
272             // \System\System Up Time
273             String uptimePath = "\\System\\System Up Time";
274             pUptime = new PointerByReference();
275             int pdhAddUptimeCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(uptimeQuery.getValue(), uptimePath, pOne,
276                     pUptime);
277             if (pdhAddUptimeCounterError != 0) {
278                 LOG.error("Failed to add PDH Uptime Counter. Error code: {}",
279                         String.format("0x%08X", pdhAddUptimeCounterError));
280             }
281         }
282 
283         // Open iowait query
284         int pdhOpenIOwaitQueryError = Pdh.INSTANCE.PdhOpenQuery(null, pOne, iowaitQuery);
285         if (pdhOpenIOwaitQueryError != 0) {
286             LOG.error("Failed to open PDH iowait Query. Error code: {}",
287                     String.format("0x%08X", pdhOpenIOwaitQueryError));
288         }
289         if (pdhOpenIOwaitQueryError == 0) {
290             // \LogicalDisk(_Total)\Avg. Disk sec/Transfer
291             String latencyPath = "\\LogicalDisk(_Total)\\Avg. Disk sec/Transfer";
292             pLatency = new PointerByReference();
293             int pdhAddLatencyCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(iowaitQuery.getValue(), latencyPath,
294                     pTwo, pLatency);
295             if (pdhAddLatencyCounterError != 0) {
296                 LOG.error("Failed to add PDH latency Counter. Error code: {}",
297                         String.format("0x%08X", pdhAddLatencyCounterError));
298             }
299             // \LogicalDisk(_Total)\Disk Transfers/sec
300             String transfersPath = "\\LogicalDisk(_Total)\\Disk Transfers/sec";
301             pTransfers = new PointerByReference();
302             int pdhAddTransfersCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(iowaitQuery.getValue(), transfersPath,
303                     pTwo, pTransfers);
304             if (pdhAddTransfersCounterError != 0) {
305                 LOG.error("Failed to add PDH latency Counter. Error code: {}",
306                         String.format("0x%08X", pdhAddTransfersCounterError));
307             }
308         }
309 
310         // Open irq query
311         int pdhOpenIrqQueryError = Pdh.INSTANCE.PdhOpenQuery(null, pOne, irqQuery);
312         if (pdhOpenIrqQueryError != 0) {
313             LOG.error("Failed to open PDH Irq Query. Error code: {}", String.format("0x%08X", pdhOpenIrqQueryError));
314         }
315         if (pdhOpenIrqQueryError == 0) {
316             // \Processor(_Total)\% Interrupt Time
317             String irqPath = "\\Processor(_Total)\\% Interrupt Time";
318             pIrq = new PointerByReference();
319             int pdhAddIrqCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(irqQuery.getValue(), irqPath, pThree, pIrq);
320             if (pdhAddIrqCounterError != 0) {
321                 LOG.error("Failed to add PDH Irq Counter. Error code: {}",
322                         String.format("0x%08X", pdhAddIrqCounterError));
323             }
324             // \Processor(_Total)\% DPC Time
325             String dpcPath = "\\Processor(_Total)\\% DPC Time";
326             pDpc = new PointerByReference();
327             int pdhAddDpcCounterError = Pdh.INSTANCE.PdhAddEnglishCounterA(irqQuery.getValue(), dpcPath, pThree, pDpc);
328             if (pdhAddDpcCounterError != 0) {
329                 LOG.error("Failed to add PDH DPC Counter. Error code: {}",
330                         String.format("0x%08X", pdhAddDpcCounterError));
331             }
332         }
333 
334         // Set up hook to close the queries on shutdown
335         Runtime.getRuntime().addShutdownHook(new Thread() {
336             @Override
337             public void run() {
338                 Pdh.INSTANCE.PdhCloseQuery(phQuery.getValue());
339                 Pdh.INSTANCE.PdhCloseQuery(uptimeQuery.getValue());
340                 Pdh.INSTANCE.PdhCloseQuery(iowaitQuery.getValue());
341                 Pdh.INSTANCE.PdhCloseQuery(irqQuery.getValue());
342             }
343         });
344     }

388     /**
389      * Vendor frequency (in Hz), eg. for processor named Intel(R) Core(TM)2 Duo
390      * CPU T7300 @ 2.00GHz the vendor frequency is 2000000000.
391      * 
392      * @return Processor frequency or -1 if unknown.
393      */
394     @Override
395     public long getVendorFreq() {
396         if (this.cpuVendorFreq == null) {
397             Pattern pattern = Pattern.compile("@ (.*)$");
398             Matcher matcher = pattern.matcher(getName());
399 
400             if (matcher.find()) {
401                 String unit = matcher.group(1);
402                 this.cpuVendorFreq = Long.valueOf(ParseUtil.parseHertz(unit));
403             } else {
404                 this.cpuVendorFreq = Long.valueOf(-1L);
405             }
406         }
407 
408         return this.cpuVendorFreq.longValue();
409     }

438     @Override
439     public void setIdentifier(String identifier) {
440         this.cpuIdentifier = identifier;
441         if (this.cpuIdentifier != null) {
442             String[] id = this.cpuIdentifier.split(" ");
443             for (int i = 0; i < id.length; i++) {
444                 if (id[i].equals("Intel64")) {
445                     setCpu64(true);
446                 } else if (id[i].equals("Intel32")) {
447                     setCpu64(false);
448                 } else if (id[i].equals("Family")) {
449                     setFamily((i < id.length - 1) ? id[i + 1] : "");
450                 } else if (id[i].equals("Model")) {
451                     setModel((i < id.length - 1) ? id[i + 1] : "");
452                 } else if (id[i].equals("Stepping")) {
453                     setStepping((i < id.length - 1) ? id[i + 1] : "");
454                 }
455             }
456         }
457     }

459     /**
460      * {@inheritDoc}
461      */
462     @Override
463     public boolean isCpu64bit() {
464         if (this.cpu64 == null) {
465             SYSTEM_INFO sysinfo = new SYSTEM_INFO();
466             Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
467             if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
468                     || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
469                 this.cpu64 = true;
470             } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
471                 this.cpu64 = false;
472             }
473         }
474         return this.cpu64 != null ? this.cpu64 : false;
475     }

533     /**
534      * {@inheritDoc}
535      */
536     @Override
537     public synchronized double getSystemCpuLoadBetweenTicks() {
538         // Check if > ~ 0.95 seconds since last tick count.
539         long now = System.currentTimeMillis();
540         LOG.trace("Current time: {}  Last tick time: {}", now, tickTime);
541         if (now - tickTime > 950) {
542             // Enough time has elapsed.
543             updateSystemTicks();
544         }
545         // Calculate total
546         long total = 0;
547         for (int i = 0; i < curTicks.length; i++) {
548             total += (curTicks[i] - prevTicks[i]);
549         }
550         // Calculate idle from last field [3]
551         long idle = curTicks[3] - prevTicks[3];
552         LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
553 
554         return (total > 0 && idle >= 0) ? (double) (total - idle) / total : 0d;
555     }

557     /**
558      * {@inheritDoc}
559      */
560     @Override
561     public long[] getSystemCpuLoadTicks() {
562         long[] ticks = new long[curTicks.length];
563         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
564         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
565         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
566         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
567             LOG.error("Failed to update system idle/kernel/user times. Error code: " + Native.getLastError());
568             return ticks;
569         }
570         // Array order is user,nice,kernel,idle
571         // TODO: Change to lp*Time.toDWordLong.longValue() with JNA 4.2.2
572         ticks[3] = WinBase.FILETIME.dateToFileTime(lpIdleTime.toDate());
573         ticks[2] = WinBase.FILETIME.dateToFileTime(lpKernelTime.toDate()) - ticks[3];
574         ticks[1] = 0L; // Windows is not 'nice'
575         ticks[0] = WinBase.FILETIME.dateToFileTime(lpUserTime.toDate());
576         return ticks;
577     }

579     /**
580      * {@inheritDoc}
581      */
582     @Override
583     public long getSystemIOWaitTicks() {
584         // Avg. Disk sec/Transfer x Avg. Disk Transfers/sec * seconds between
585         // reads
586 
587         // This call updates all process counters to % used since last call
588         int ret = Pdh.INSTANCE.PdhCollectQueryData(iowaitQuery.getValue());
589         if (ret != 0) {
590             LOG.error("Failed to update Uptime Counters. Error code: {}", String.format("0x%08X", ret));
591             return 0;
592         }
593 
594         long now = System.currentTimeMillis();
595 
596         // We'll manufacture our own ticks by multiplying the % used (from the
597         // counter) by time elapsed since the last call to get a tick increment
598         long elapsed = now - iowaitTime;
599 
600         PdhFmtCounterValue phLatencyCounterValue = new PdhFmtCounterValue();
601         ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pLatency.getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000, null,
602                 phLatencyCounterValue);
603         if (ret != 0) {
604             LOG.warn("Failed to get Latency Tick Counters. Error code: {}", String.format("0x%08X", ret));
605             return 0;
606         }
607 
608         PdhFmtCounterValue phTransfersCounterValue = new PdhFmtCounterValue();
609         ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pTransfers.getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000,
610                 null, phTransfersCounterValue);
611         if (ret != 0) {
612             LOG.warn("Failed to get Transfers Tick Counters. Error code: {}", String.format("0x%08X", ret));
613             return 0;
614         }
615 
616         // Returns results in 1000's of percent, e.g. 5% is 5000
617         // Multiply by elapsed to get total ms and Divide by 100 * 1000
618         // Putting division at end avoids need to cast division to double
619         // Elasped is only since last read, so increment previous value
620         iowaitTicks += elapsed * phLatencyCounterValue.value.largeValue / 100000
621                 * phTransfersCounterValue.value.largeValue / 100000;
622 
623         iowaitTime = now;
624 
625         return iowaitTicks;
626     }

632     public long[] getSystemIrqTicks() {
633         long[] ticks = new long[2];
634         // Time * seconds between reads
635         // This call updates all process counters to % used since last call
636         int ret = Pdh.INSTANCE.PdhCollectQueryData(irqQuery.getValue());
637         if (ret != 0) {
638             LOG.error("Failed to update Uptime Counters. Error code: {}", String.format("0x%08X", ret));
639             return ticks;
640         }
641 
642         long now = System.currentTimeMillis();
643 
644         // We'll manufacture our own ticks by multiplying the % used (from the
645         // counter) by time elapsed since the last call to get a tick increment
646         long elapsed = now - irqTime;
647 
648         PdhFmtCounterValue phIrqCounterValue = new PdhFmtCounterValue();
649         ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pIrq.getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000, null,
650                 phIrqCounterValue);
651         if (ret != 0) {
652             LOG.warn("Failed to get IRQ Tick Counters. Error code: {}", String.format("0x%08X", ret));
653             return ticks;
654         }
655 
656         PdhFmtCounterValue phDpcCounterValue = new PdhFmtCounterValue();
657         ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pDpc.getValue(), Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000, null,
658                 phDpcCounterValue);
659         if (ret != 0) {
660             LOG.warn("Failed to get DPC Tick Counters. Error code: {}", String.format("0x%08X", ret));
661             return ticks;
662         }
663 
664         // Returns results in 1000's of percent, e.g. 5% is 5000
665         // Multiply by elapsed to get total ms and Divide by 100 * 1000
666         // Putting division at end avoids need to cast division to double
667         // Elasped is only since last read, so increment previous value
668         irqTicks[0] += elapsed * phIrqCounterValue.value.largeValue / 100000;
669         irqTicks[1] += elapsed * phDpcCounterValue.value.largeValue / 100000;
670 
671         irqTime = now;
672 
673         // Make a copy of the array to return
674         System.arraycopy(irqTicks, 0, ticks, 0, irqTicks.length);
675         return ticks;
676     };

679      * Updates system tick information from native call to GetSystemTimes().
680      * Array with four elements representing clock ticks or milliseconds
681      * (platform dependent) spent in User (0), Nice (1), System (2), and Idle
682      * (3) states. By measuring the difference between ticks across a time
683      * interval, CPU load over that interval may be calculated.
684      */
685     private void updateSystemTicks() {
686         LOG.trace("Updating System Ticks");

689         this.tickTime = System.currentTimeMillis();
690         long[] ticks = getSystemCpuLoadTicks();
691         System.arraycopy(ticks, 0, curTicks, 0, ticks.length);

736     /**
737      * {@inheritDoc}
738      */
739     @Override
740     public double[] getProcessorCpuLoadBetweenTicks() {
741         // Check if > ~ 0.95 seconds since last tick count.
742         long now = System.currentTimeMillis();
743         LOG.trace("Current time: {}  Last tick time: {}", now, procTickTime);
744         if (now - procTickTime > 950) {
745             // Enough time has elapsed.
746             // Update latest
747             updateProcessorTicks();
748         }
749         double[] load = new double[logicalProcessorCount];
750         for (int cpu = 0; cpu < logicalProcessorCount; cpu++) {
751             long total = 0;
752             for (int i = 0; i < this.curProcTicks[cpu].length; i++) {
753                 total += (this.curProcTicks[cpu][i] - this.prevProcTicks[cpu][i]);
754             }
755             // Calculate idle from last field [3]
756             long idle = this.curProcTicks[cpu][3] - this.prevProcTicks[cpu][3];
757             LOG.trace("CPU: {}  Total ticks: {}  Idle ticks: {}", cpu, total, idle);
758             // update
759             load[cpu] = (total > 0 && idle >= 0) ? (double) (total - idle) / total : 0d;
760         }
761         return load;
762     }

764     /**
765      * {@inheritDoc}
766      */
767     @Override
768     public long[][] getProcessorCpuLoadTicks() {
769         long[][] ticks = new long[logicalProcessorCount][4];
770 
771         // This call updates all process counters to % used since last call
772         int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
773         if (ret != 0) {
774             LOG.warn("Failed to update Tick Counters. Error code: {}", String.format("0x%08X", ret));
775             return ticks;
776         }
777         long now = System.currentTimeMillis();
778 
779         // We'll manufacture our own ticks by multiplying the % used (from the
780         // counter) by time elapsed since the last call to get a tick increment
781         long elapsed = now - allProcTickTime;
782         for (int cpu = 0; cpu < logicalProcessorCount; cpu++) {
783             PdhFmtCounterValue phUserCounterValue = new PdhFmtCounterValue();
784             ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(phUserCounters[cpu].getValue(),
785                     Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000, null, phUserCounterValue);
786             if (ret != 0) {
787                 LOG.warn("Failed to get Uer Tick Counters. Error code: {}", String.format("0x%08X", ret));
788                 return ticks;
789             }
790 
791             PdhFmtCounterValue phIdleCounterValue = new PdhFmtCounterValue();
792             ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(phIdleCounters[cpu].getValue(),
793                     Pdh.PDH_FMT_LARGE | Pdh.PDH_FMT_1000, null, phIdleCounterValue);
794             if (ret != 0) {
795                 LOG.warn("Failed to get idle Tick Counters. Error code: {}", String.format("0x%08X", ret));
796                 return ticks;
797             }
798 
799             // Returns results in 1000's of percent, e.g. 5% is 5000
800             // Multiply by elapsed to get total ms and Divide by 100 * 1000
801             // Putting division at end avoids need to cast division to double
802             long user = elapsed * phUserCounterValue.value.largeValue / 100000;
803             long idle = elapsed * phIdleCounterValue.value.largeValue / 100000;
804             // Elasped is only since last read, so increment previous value
805             allProcTicks[cpu][0] += user;
806             // allProcTicks[cpu][1] is ignored, Windows is not nice
807             allProcTicks[cpu][2] += Math.max(0, elapsed - user - idle); // u+i+sys=100%
808             allProcTicks[cpu][3] += idle;
809         }
810         allProcTickTime = now;
811 
812         // Make a copy of the array to return
813         for (int cpu = 0; cpu < logicalProcessorCount; cpu++) {
814             System.arraycopy(allProcTicks[cpu], 0, ticks[cpu], 0, allProcTicks[cpu].length);
815         }
816         return ticks;
817     }

819     /**
820      * Updates the tick array for all processors by querying PDH counter. Stores
821      * in 2D array; an array for each logical processor with four elements
822      * representing clock ticks or milliseconds (platform dependent) spent in
823      * User (0), Nice (1), System (2), and Idle (3) states. By measuring the
824      * difference between ticks across a time interval, CPU load over that
825      * interval may be calculated.
826      */
827     private void updateProcessorTicks() {
828         LOG.trace("Updating Processor Ticks");
829         // Copy to previous
830         for (int cpu = 0; cpu < logicalProcessorCount; cpu++) {
831             System.arraycopy(curProcTicks[cpu], 0, prevProcTicks[cpu], 0, curProcTicks[cpu].length);
832         }
833         this.procTickTime = System.currentTimeMillis();
834         long[][] ticks = getProcessorCpuLoadTicks();
835         for (int cpu = 0; cpu < logicalProcessorCount; cpu++) {
836             System.arraycopy(ticks[cpu], 0, curProcTicks[cpu], 0, ticks[cpu].length);
837         }
838     }

862      * {@inheritDoc}
863      */
864     @Override
865     public String getSystemSerialNumber() {
866         if (this.cpuSerialNumber == null) {
867             // This should always work
868             ArrayList<String> hwInfo = ExecutingCommand.runNative("wmic bios get serialnumber");
869             for (String checkLine : hwInfo) {
870                 if (checkLine.length() == 0 || checkLine.toLowerCase().contains("serialnumber")) {
871                     continue;
872                 } else {
873                     this.cpuSerialNumber = checkLine.trim();
874                     break;
875                 }
876             }
877             // Just in case the above doesn't
878             if (this.cpuSerialNumber == null || this.cpuSerialNumber.length() == 0) {
879                 hwInfo = ExecutingCommand.runNative("wmic csproduct get identifyingnumber");
880                 for (String checkLine : hwInfo) {
881                     if (checkLine.length() == 0 || checkLine.toLowerCase().contains("identifyingnumber")) {
882                         continue;
883                     } else {
884                         this.cpuSerialNumber = checkLine.trim();
885                         break;
886                     }
887                 }
888             }
889             if (this.cpuSerialNumber == null) {
890                 this.cpuSerialNumber = "unknown";
891             }
892         }
893         return this.cpuSerialNumber;
894     }

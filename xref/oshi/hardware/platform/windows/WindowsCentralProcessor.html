78      /**
79       * Create a Processor
80       */
81      public WindowsCentralProcessor() {
82          // Initialize class variables
83          initVars();
84          // Initialize PDH
85          initPdhCounters();
86          // Initialize tick arrays
87          this.allProcTicks = new long[this.logicalProcessorCount][4];
88          initTicks();
89  
90          LOG.debug("Initialized Processor");
91      }

93      /**
94       * Initializes Class variables
95       */
96      private void initVars() {
97          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor";
98          String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
99          if (processorIds.length > 0) {
100             String cpuRegistryPath = cpuRegistryRoot + "\\" + processorIds[0];
101             this.setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
102                     "VendorIdentifier"));
103             this.setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
104                     "ProcessorNameString"));
105             this.setIdentifier(
106                     Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
107         }
108         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
109         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
110         if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
111                 || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
112             this.setCpu64(true);
113         } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
114             this.setCpu64(false);
115         }
116     }

118     /**
119      * Initializes PDH Tick Counters
120      */
121     private void initPdhCounters() {
122         // Set up counters
123         this.phUserCounters = new PointerByReference[this.logicalProcessorCount];
124         this.phIdleCounters = new PointerByReference[this.logicalProcessorCount];
125 
126         // Open tick query
127         if (PdhUtil.openQuery(phQuery)) {
128             for (int p = 0; p < this.logicalProcessorCount; p++) {
129                 // Options are (only need 2 to calculate all)
130                 // "\Processor(0)\% processor time"
131                 // "\Processor(0)\% idle time" (1 - processor)
132                 // "\Processor(0)\% privileged time" (subset of processor)
133                 // "\Processor(0)\% user time" (other subset of processor)
134                 // Note need to make \ = \\ for Java Strings and %% for format
135                 String counterPath = String.format("\\Processor(%d)\\%% user time", p);
136                 this.phUserCounters[p] = new PointerByReference();
137                 // Add tick query for this processor
138                 PdhUtil.addCounter(phQuery, counterPath, this.phUserCounters[p]);
139 
140                 counterPath = String.format("\\Processor(%d)\\%% idle time", p);
141                 this.phIdleCounters[p] = new PointerByReference();
142                 PdhUtil.addCounter(phQuery, counterPath, this.phIdleCounters[p]);
143             }
144             LOG.debug("Tick counter queries added.  Initializing with first query.");
145 
146             // Initialize by collecting data the first time
147             Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
148         }
149 
150         // Open iowait query
151         if (PdhUtil.openQuery(iowaitQuery)) {
152             // \LogicalDisk(_Total)\Avg. Disk sec/Transfer
153             PdhUtil.addCounter(iowaitQuery, "\\LogicalDisk(_Total)\\Avg. Disk sec/Transfer", pLatency);
154             // \LogicalDisk(_Total)\Disk Transfers/sec
155             PdhUtil.addCounter(iowaitQuery, "\\LogicalDisk(_Total)\\Disk Transfers/sec", pTransfers);
156             // Initialize by collecting data the first time
157             Pdh.INSTANCE.PdhCollectQueryData(iowaitQuery.getValue());
158         }
159 
160         // Open irq query
161         if (PdhUtil.openQuery(irqQuery)) {
162             // \Processor(_Total)\% Interrupt Time
163             PdhUtil.addCounter(irqQuery, "\\Processor(_Total)\\% Interrupt Time", pIrq);
164             // \Processor(_Total)\% DPC Time
165             PdhUtil.addCounter(irqQuery, "\\Processor(_Total)\\% DPC Time", pDpc);
166             // Initialize by collecting data the first time
167             Pdh.INSTANCE.PdhCollectQueryData(irqQuery.getValue());
168         }
169 
170         // Set up hook to close the queries on shutdown
171         Runtime.getRuntime().addShutdownHook(new Thread() {
172             @Override
173             public void run() {
174                 Pdh.INSTANCE.PdhCloseQuery(phQuery.getValue());
175                 Pdh.INSTANCE.PdhCloseQuery(iowaitQuery.getValue());
176                 Pdh.INSTANCE.PdhCloseQuery(irqQuery.getValue());
177             }
178         });
179     }

181     /**
182      * Updates logical and physical processor counts from /proc/cpuinfo
183      */
184     protected void calculateProcessorCounts() {
185         // Get number of logical processors
186         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
187         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
188         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
189 
190         // Get number of physical processors
191         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
192         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
193             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
194                 this.physicalProcessorCount++;
195             }
196         }
197     }

199     /**
200      * {@inheritDoc}
201      */
202     @Override
203     public long[] getSystemCpuLoadTicks() {
204         long[] ticks = new long[curTicks.length];
205         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
206         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
207         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
208         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
209             LOG.error("Failed to update system idle/kernel/user times. Error code: " + Native.getLastError());
210             return ticks;
211         }
212         // Array order is user,nice,kernel,idle
213         // TODO: Change to lp*Time.toDWordLong.longValue() with JNA 4.3
214         ticks[3] = WinBase.FILETIME.dateToFileTime(lpIdleTime.toDate());
215         ticks[2] = WinBase.FILETIME.dateToFileTime(lpKernelTime.toDate()) - ticks[3];
216         ticks[1] = 0L; // Windows is not 'nice'
217         ticks[0] = WinBase.FILETIME.dateToFileTime(lpUserTime.toDate());
218         return ticks;
219     }

221     /**
222      * {@inheritDoc}
223      */
224     @Override
225     public long getSystemIOWaitTicks() {
226         // Avg. Disk sec/Transfer x Avg. Disk Transfers/sec * seconds between
227         // reads
228         if (!PdhUtil.updateCounters(iowaitQuery)) {
229             return 0;
230         }
231 
232         // We'll manufacture our own ticks by multiplying the latency (sec per
233         // read) by transfers (reads per sec) by time elapsed since the last
234         // call to get a tick increment
235         long now = System.currentTimeMillis();
236         long elapsed = now - iowaitTime;
237 
238         long latency = PdhUtil.queryCounter(pLatency);
239         long transfers = PdhUtil.queryCounter(pTransfers);
240 
241         // Since PDH_FMT_1000 is used, results must be divided by 1000 to get
242         // actual. Units are sec (*1000) per read * reads (*1000) per sec time *
243         // ms time. Reads cancel so result is in sec (*1000*1000) per sec, which
244         // is a unitless percentage (times a million) multiplied by elapsed time
245         // Multiply by elapsed to get total ms and Divide by 1000 * 1000.
246         // Putting division at end avoids need to cast division to double
247         // Elasped is only since last read, so increment previous value
248         iowaitTicks += elapsed * latency * transfers / 1000000;
249 
250         iowaitTime = now;
251 
252         return iowaitTicks;
253     }

255     /**
256      * {@inheritDoc}
257      */
258     @Override
259     public long[] getSystemIrqTicks() {
260         long[] ticks = new long[2];
261         // Time * seconds between reads
262         if (!PdhUtil.updateCounters(irqQuery)) {
263             return ticks;
264         }
265 
266         // We'll manufacture our own ticks by multiplying the % used (from the
267         // counter) by time elapsed since the last call to get a tick increment
268         long now = System.currentTimeMillis();
269         long elapsed = now - irqTime;
270 
271         long irq = PdhUtil.queryCounter(pIrq);
272         long dpc = PdhUtil.queryCounter(pDpc);
273 
274         // Returns results in 1000's of percent, e.g. 5% is 5000
275         // Multiply by elapsed to get total ms and Divide by 100 * 1000
276         // Putting division at end avoids need to cast division to double
277         // Elasped is only since last read, so increment previous value
278         irqTicks[0] += elapsed * irq / 100000;
279         irqTicks[1] += elapsed * dpc / 100000;
280 
281         irqTime = now;
282 
283         // Make a copy of the array to return
284         System.arraycopy(irqTicks, 0, ticks, 0, irqTicks.length);
285         return ticks;
286     };

291     @Override
292     public double[] getSystemLoadAverage(int nelem) {
293         if (nelem < 1) {
294             throw new IllegalArgumentException("Must include at least one element.");
295         }
296         if (nelem > 3) {
297             LOG.warn("Max elements of SystemLoadAverage is 3. " + nelem + " specified. Ignoring extra.");
298             nelem = 3;
299         }
300         double[] average = new double[nelem];
301         // TODO: If Windows ever actually implements a laod average for 1/5/15,
302         // return it
303         for (int i = 0; i < average.length; i++) {
304             average[i] = -1;
305         }
306         return average;
307     }

309     /**
310      * {@inheritDoc}
311      */
312     @Override
313     public long[][] getProcessorCpuLoadTicks() {
314         long[][] ticks = new long[this.logicalProcessorCount][4];
315 
316         if (!PdhUtil.updateCounters(phQuery)) {
317             return ticks;
318         }
319 
320         // We'll manufacture our own ticks by multiplying the % used (from the
321         // counter) by time elapsed since the last call to get a tick increment
322         long now = System.currentTimeMillis();
323         long elapsed = now - allProcTickTime;
324 
325         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
326             long userPct = PdhUtil.queryCounter(this.phUserCounters[cpu]);
327             long idlePct = PdhUtil.queryCounter(this.phIdleCounters[cpu]);
328 
329             // Returns results in 1000's of percent, e.g. 5% is 5000
330             // Multiply by elapsed to get total ms and Divide by 100 * 1000
331             // Putting division at end avoids need to cast division to double
332             long user = elapsed * userPct / 100000;
333             long idle = elapsed * idlePct / 100000;
334             // Elasped is only since last read, so increment previous value
335             allProcTicks[cpu][0] += user;
336             // allProcTicks[cpu][1] is ignored, Windows is not nice
337             allProcTicks[cpu][2] += Math.max(0, elapsed - user - idle); // u+i+sys=100%
338             allProcTicks[cpu][3] += idle;
339         }
340         allProcTickTime = now;
341 
342         // Make a copy of the array to return
343         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
344             System.arraycopy(allProcTicks[cpu], 0, ticks[cpu], 0, allProcTicks[cpu].length);
345         }
346         return ticks;
347     }

357     /**
358      * {@inheritDoc}
359      */
360     @Override
361     public String getSystemSerialNumber() {
362         if (this.cpuSerialNumber == null) {
363             // This should always work
364             ArrayList<String> hwInfo = ExecutingCommand.runNative("wmic bios get serialnumber");
365             for (String checkLine : hwInfo) {
366                 if (checkLine.length() == 0 || checkLine.toLowerCase().contains("serialnumber")) {
367                     continue;
368                 } else {
369                     this.cpuSerialNumber = checkLine.trim();
370                     break;
371                 }
372             }
373             // Just in case the above doesn't
374             if (this.cpuSerialNumber == null || this.cpuSerialNumber.length() == 0) {
375                 hwInfo = ExecutingCommand.runNative("wmic csproduct get identifyingnumber");
376                 for (String checkLine : hwInfo) {
377                     if (checkLine.length() == 0 || checkLine.toLowerCase().contains("identifyingnumber")) {
378                         continue;
379                     } else {
380                         this.cpuSerialNumber = checkLine.trim();
381                         break;
382                     }
383                 }
384             }
385             if (this.cpuSerialNumber == null) {
386                 this.cpuSerialNumber = "unknown";
387             }
388         }
389         return this.cpuSerialNumber;
390     }

392     @Override
393     public int getProcessCount() {
394         PERFORMANCE_INFORMATION perfInfo = new PERFORMANCE_INFORMATION();
395         if (!Psapi.INSTANCE.GetPerformanceInfo(perfInfo, perfInfo.size())) {
396             LOG.error("Failed to get Performance Info. Error code: {}", Kernel32.INSTANCE.GetLastError());
397             return 0;
398         }
399         return perfInfo.ProcessCount.intValue();
400     }

38  /**
39   * Windows hard disk implementation.
40   *
41   * @author enrico[dot]bianchi[at]gmail[dot]com
42   */
43  public class WindowsDisks implements Disks {
44  
45      private static final long serialVersionUID = 1L;
46  
47      /**
48       * Maps to store read/write bytes per drive index
49       */
50      private static Map<String, Long> readMap = new HashMap<>();
51      private static Map<String, Long> readByteMap = new HashMap<>();
52      private static Map<String, Long> writeMap = new HashMap<>();
53      private static Map<String, Long> writeByteMap = new HashMap<>();
54      private static Map<String, Long> xferTimeMap = new HashMap<>();
55      private static Map<String, Long> timeStampMap = new HashMap<>();
56      private static Map<String, List<String>> driveToPartitionMap = new HashMap<>();
57      private static Map<String, String> partitionToLogicalDriveMap = new HashMap<>();
58      private static Map<String, HWPartition> partitionMap = new HashMap<>();
59  
60      private static final ValueType[] DRIVE_TYPES = { ValueType.STRING, ValueType.STRING, ValueType.STRING,
61              ValueType.STRING, ValueType.STRING, ValueType.UINT32 };
62  
63      private static final ValueType[] READ_WRITE_TYPES = { ValueType.STRING, ValueType.UINT32, ValueType.STRING,
64              ValueType.UINT32, ValueType.STRING, ValueType.STRING, ValueType.STRING };
65  
66      private static final ValueType[] PARTITION_TYPES = { ValueType.STRING, ValueType.STRING, ValueType.STRING,
67              ValueType.STRING, ValueType.STRING, ValueType.UINT32, ValueType.UINT32 };
68  
69      private static final Pattern DEVICE_ID = Pattern.compile(".*\\.DeviceID=\"(.*)\"");
70  
71      private static final int BUFSIZE = 255;
72  
73      @Override
74      public HWDiskStore[] getDisks() {
75          List<HWDiskStore> result;
76          result = new ArrayList<>();
77          populateReadWriteMaps();
78          populatePartitionMaps();
79  
80          Map<String, List<Object>> vals = WmiUtil.selectObjectsFrom(null, "Win32_DiskDrive",
81                  "Name,Manufacturer,Model,SerialNumber,Size,Index", null, DRIVE_TYPES);
82          for (int i = 0; i < vals.get("Name").size(); i++) {
83              HWDiskStore ds = new HWDiskStore();
84              ds.setName((String) vals.get("Name").get(i));
85              ds.setModel(String.format("%s %s", vals.get("Model").get(i), vals.get("Manufacturer").get(i)).trim());
86              // Most vendors store serial # as a hex string; convert
87              ds.setSerial(ParseUtil.hexStringToString((String) vals.get("SerialNumber").get(i)));
88              String index = vals.get("Index").get(i).toString();
89              ds.setReads(MapUtil.getOrDefault(readMap, index, 0L));
90              ds.setReadBytes(MapUtil.getOrDefault(readByteMap, index, 0L));
91              ds.setWrites(MapUtil.getOrDefault(writeMap, index, 0L));
92              ds.setWriteBytes(MapUtil.getOrDefault(writeByteMap, index, 0L));
93              ds.setTransferTime(MapUtil.getOrDefault(xferTimeMap, index, 0L));
94              ds.setTimeStamp(MapUtil.getOrDefault(timeStampMap, index, 0L));
95              // If successful this line is the desired value
96              ds.setSize(ParseUtil.parseLongOrDefault((String) vals.get("Size").get(i), 0L));
97              // Get partitions
98              List<HWPartition> partitions = new ArrayList<>();
99              List<String> partList = driveToPartitionMap.get(ds.getName());
100             if (partList != null && !partList.isEmpty()) {
101                 for (String part : partList) {
102                     partitions.add(partitionMap.get(part));
103                 }
104             }
105             ds.setPartitions(partitions.toArray(new HWPartition[partitions.size()]));
106             // Add to list
107             result.add(ds);
108         }
109         return result.toArray(new HWDiskStore[result.size()]);
110     }
111 
112     private void populateReadWriteMaps() {
113         readMap.clear();
114         readByteMap.clear();
115         writeMap.clear();
116         writeByteMap.clear();
117         xferTimeMap.clear();
118         timeStampMap.clear();
119         // Although the field names say "PerSec" this is the Raw Data from which
120         // the associated fields are populated in the Formatted Data class, so
121         // in fact this is the data we want
122         Map<String, List<Object>> vals = WmiUtil.selectObjectsFrom(null, "Win32_PerfRawData_PerfDisk_PhysicalDisk",
123                 "Name,DiskReadsPerSec,DiskReadBytesPerSec,DiskWritesPerSec,DiskWriteBytesPerSec,PercentDiskTime,Timestamp_Sys100NS",
124                 null, READ_WRITE_TYPES);
125         for (int i = 0; i < vals.get("Name").size(); i++) {
126             String name = ((String) vals.get("Name").get(i)).split("\\s+")[0];
127             readMap.put(name, (long) vals.get("DiskReadsPerSec").get(i));
128             readByteMap.put(name, ParseUtil.parseLongOrDefault((String) vals.get("DiskReadBytesPerSec").get(i), 0L));
129             writeMap.put(name, (long) vals.get("DiskWritesPerSec").get(i));
130             writeByteMap.put(name, ParseUtil.parseLongOrDefault((String) vals.get("DiskWriteBytesPerSec").get(i), 0L));
131             // Units are 100-ns, divide to get ms
132             xferTimeMap.put(name,
133                     ParseUtil.parseLongOrDefault((String) vals.get("PercentDiskTime").get(i), 0L) / 10000L);
134             timeStampMap.put(name,
135                     ParseUtil.parseLongOrDefault((String) vals.get("Timestamp_Sys100NS").get(i), 0L) / 10000L);
136         }
137     }
138 
139     private void populatePartitionMaps() {
140         driveToPartitionMap.clear();
141         partitionToLogicalDriveMap.clear();
142         partitionMap.clear();
143         // For Regexp matching DeviceIDs
144         Matcher mAnt;
145         Matcher mDep;
146 
147         // Map drives to partitions
148         Map<String, List<String>> partitionQueryMap = WmiUtil.selectStringsFrom(null, "Win32_DiskDriveToDiskPartition",
149                 "Antecedent,Dependent", null);
150         for (int i = 0; i < partitionQueryMap.get("Antecedent").size(); i++) {
151             mAnt = DEVICE_ID.matcher(partitionQueryMap.get("Antecedent").get(i));
152             mDep = DEVICE_ID.matcher(partitionQueryMap.get("Dependent").get(i));
153             if (mAnt.matches() && mDep.matches()) {
154                 MapUtil.createNewListIfAbsent(driveToPartitionMap, mAnt.group(1).replaceAll("\\\\\\\\", "\\\\"))
155                         .add(mDep.group(1));
156             }
157         }
158 
159         // Map partitions to logical disks
160         partitionQueryMap = WmiUtil.selectStringsFrom(null, "Win32_LogicalDiskToPartition", "Antecedent,Dependent",
161                 null);
162         for (int i = 0; i < partitionQueryMap.get("Antecedent").size(); i++) {
163             mAnt = DEVICE_ID.matcher(partitionQueryMap.get("Antecedent").get(i));
164             mDep = DEVICE_ID.matcher(partitionQueryMap.get("Dependent").get(i));
165             if (mAnt.matches() && mDep.matches()) {
166                 partitionToLogicalDriveMap.put(mAnt.group(1), mDep.group(1) + "\\");
167             }
168         }
169 
170         // Next, get all partitions and create objects
171         final Map<String, List<Object>> hwPartitionQueryMap = WmiUtil.selectObjectsFrom(null, "Win32_DiskPartition",
172                 "Name,Type,Description,DeviceID,Size,DiskIndex,Index", null, PARTITION_TYPES);
173         for (int i = 0; i < hwPartitionQueryMap.get("Name").size(); i++) {
174             String deviceID = (String) hwPartitionQueryMap.get("DeviceID").get(i);
175             String logicalDrive = MapUtil.getOrDefault(partitionToLogicalDriveMap, deviceID, "");
176             String uuid = "";
177             if (!logicalDrive.isEmpty()) {
178                 // Get matching volume for UUID
179                 char[] volumeChr = new char[BUFSIZE];
180                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive, volumeChr, BUFSIZE);
181                 uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
182             }
183             partitionMap.put(deviceID,
184                     new HWPartition((String) hwPartitionQueryMap.get("Name").get(i),
185                             (String) hwPartitionQueryMap.get("Type").get(i),
186                             (String) hwPartitionQueryMap.get("Description").get(i), uuid,
187                             ParseUtil.parseLongOrDefault((String) hwPartitionQueryMap.get("Size").get(i), 0L),
188                             ((Long) hwPartitionQueryMap.get("DiskIndex").get(i)).intValue(),
189                             ((Long) hwPartitionQueryMap.get("Index").get(i)).intValue(), logicalDrive));
190         }
191     }

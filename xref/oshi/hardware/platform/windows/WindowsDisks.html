19  package oshi.hardware.platform.windows;
20  
21  import java.util.ArrayList;
22  import java.util.HashMap;
23  import java.util.List;
24  import java.util.Map;
25  
26  import oshi.hardware.HWDiskStore;
27  import oshi.hardware.common.AbstractDisks;
28  import oshi.util.ParseUtil;
29  import oshi.util.platform.windows.WmiUtil;
30  import oshi.util.platform.windows.WmiUtil.ValueType;
31  
32  /**
33   * Windows hard disk implementation.
34   *
35   * @author enrico[dot]bianchi[at]gmail[dot]com
36   */
37  public class WindowsDisks extends AbstractDisks {
38  
39      private static final long serialVersionUID = 1L;
40  
41      /**
42       * Maps to store read/write bytes per drive index
43       */
44      private static Map<String, Long> readMap = new HashMap<>();
45      private static Map<String, Long> writeMap = new HashMap<>();
46  
47      private static final ValueType[] DRIVE_TYPES = { ValueType.STRING, ValueType.STRING, ValueType.STRING,
48              ValueType.STRING, ValueType.STRING, ValueType.LONG };
49  
50      @Override
51      public HWDiskStore[] getDisks() {
52          List<HWDiskStore> result;
53          result = new ArrayList<>();
54          readMap.clear();
55          writeMap.clear();
56          populateReadWriteMaps();
57  
58          Map<String, List<Object>> vals = WmiUtil.selectObjectsFrom(null, "Win32_DiskDrive",
59                  "Name,Manufacturer,Model,SerialNumber,Size,Index", null, DRIVE_TYPES);
60          for (int i = 0; i < vals.get("Name").size(); i++) {
61              HWDiskStore ds = new HWDiskStore();
62              ds.setName((String) vals.get("Name").get(i));
63              ds.setModel(String.format("%s %s", vals.get("Model").get(i), vals.get("Manufacturer").get(i)).trim());
64              // Most vendors store serial # as a hex string; convert
65              ds.setSerial(ParseUtil.hexStringToString((String) vals.get("SerialNumber").get(i)));
66              String index = vals.get("Index").get(i).toString();
67              if (readMap.containsKey(index)) {
68                  ds.setReads(readMap.get(index));
69              }
70              if (writeMap.containsKey(index)) {
71                  ds.setWrites(writeMap.get(index));
72              }
73              // If successful this line is the desired value
74              try {
75                  ds.setSize(Long.parseLong((String) vals.get("Size").get(i)));
76              } catch (NumberFormatException e) {
77                  // If we failed to parse, give up
78                  // This is expected for an empty string on some drives
79                  ds.setSize(0L);
80              }
81              result.add(ds);
82          }
83          return result.toArray(new HWDiskStore[result.size()]);
84      }
85  
86      private void populateReadWriteMaps() {
87          // Although the field names say "PerSec" this is the Raw Data from which
88          // the associated fields are populated in the Formatted Data class, so
89          // in fact this is the data we want
90          Map<String, List<String>> vals = WmiUtil.selectStringsFrom(null, "Win32_PerfRawData_PerfDisk_PhysicalDisk",
91                  "Name,DiskReadBytesPerSec,DiskWriteBytesPerSec", null);
92          for (int i = 0; i < vals.get("Name").size(); i++) {
93              String index = vals.get("Name").get(i).split("\\s+")[0];
94              try {
95                  readMap.put(index, Long.parseLong(vals.get("DiskReadBytesPerSec").get(i)));
96                  writeMap.put(index, Long.parseLong(vals.get("DiskWriteBytesPerSec").get(i)));
97              } catch (NumberFormatException e) {
98                  // If we can't parse, just ignore, we won't populate the map
99                  continue;
100             }
101         }
102     }
103 }

142     private static final String PDH_DISK_READS_FORMAT = "\\PhysicalDisk(%s)\\Disk Reads/sec";
143     private static final String PDH_DISK_READ_BYTES_FORMAT = "\\PhysicalDisk(%s)\\Disk Read Bytes/sec";
144     private static final String PDH_DISK_WRITES_FORMAT = "\\PhysicalDisk(%s)\\Disk Writes/sec";
145     private static final String PDH_DISK_WRITE_BYTES_FORMAT = "\\PhysicalDisk(%s)\\Disk Write Bytes/sec";
146     private static final String PDH_DISK_TIME_FORMAT = "\\PhysicalDisk(%s)\\%% Disk Time";
147 
148     private static final String PHYSICALDRIVE_PREFIX = "\\\\.\\PHYSICALDRIVE";

150     public static boolean updateDiskStats(HWDiskStore diskStore) {
151         String index = null;
152         HWPartition[] partitions = diskStore.getPartitions();
153         if (partitions.length > 0) {
154             // If a partition exists on this drive, the major property
155             // corresponds to the disk index, so use it.
156             index = Integer.toString(partitions[0].getMajor());
157         } else if (diskStore.getName().startsWith(PHYSICALDRIVE_PREFIX)) {
158             // If no partition exists, Windows reliably uses a name to match the
159             // disk index. That said, the skeptical person might wonder why a
160             // disk has read/write statistics without a partition, and wonder
161             // why this branch is even relevant as an option. The author of this
162             // comment does not have an answer for this valid question.
163             index = diskStore.getName().substring(PHYSICALDRIVE_PREFIX.length(), diskStore.getName().length());
164         } else {
165             // The author of this comment cannot fathom a circumstance in which
166             // the code reaches this point, but just in case it does, here's the
167             // correct response. If you get this log warning, the circumstances
168             // would be of great interest to the project's maintainers.
169             LOG.warn("Couldn't match index for {}", diskStore.getName());
170             return false;
171         }
172         populateReadWriteMaps(index);
173         if (readMap.containsKey(index)) {
174             diskStore.setReads(MapUtil.getOrDefault(readMap, index, 0L));
175             diskStore.setReadBytes(MapUtil.getOrDefault(readByteMap, index, 0L));
176             diskStore.setWrites(MapUtil.getOrDefault(writeMap, index, 0L));
177             diskStore.setWriteBytes(MapUtil.getOrDefault(writeByteMap, index, 0L));
178             diskStore.setTransferTime(MapUtil.getOrDefault(xferTimeMap, index, 0L));
179             diskStore.setTimeStamp(MapUtil.getOrDefault(timeStampMap, index, 0L));
180             return true;
181         } else {
182             return false;
183         }

187     @Override
188     public HWDiskStore[] getDisks() {
189         List<HWDiskStore> result;
190         result = new ArrayList<>();
191         populateReadWriteMaps(null);
192         populatePartitionMaps();
193 
194         WmiResult<DiskDriveProperty> vals = WmiUtil.queryWMI(DISK_DRIVE_QUERY);
195 
196         for (int i = 0; i < vals.getResultCount(); i++) {
197             HWDiskStore ds = new HWDiskStore();
198             ds.setName((String) vals.get(DiskDriveProperty.NAME).get(i));
199             ds.setModel(String.format("%s %s", vals.get(DiskDriveProperty.MODEL).get(i),
200                     vals.get(DiskDriveProperty.MANUFACTURER).get(i)).trim());
201             // Most vendors store serial # as a hex string; convert
202             ds.setSerial(ParseUtil.hexStringToString((String) vals.get(DiskDriveProperty.SERIALNUMBER).get(i)));
203             String index = vals.get(DiskDriveProperty.INDEX).get(i).toString();
204             ds.setReads(MapUtil.getOrDefault(readMap, index, 0L));
205             ds.setReadBytes(MapUtil.getOrDefault(readByteMap, index, 0L));
206             ds.setWrites(MapUtil.getOrDefault(writeMap, index, 0L));
207             ds.setWriteBytes(MapUtil.getOrDefault(writeByteMap, index, 0L));
208             ds.setTransferTime(MapUtil.getOrDefault(xferTimeMap, index, 0L));
209             ds.setTimeStamp(MapUtil.getOrDefault(timeStampMap, index, 0L));
210             ds.setSize((Long) vals.get(DiskDriveProperty.SIZE).get(i));
211             // Get partitions
212             List<HWPartition> partitions = new ArrayList<>();
213             List<String> partList = driveToPartitionMap.get(ds.getName());
214             if (partList != null && !partList.isEmpty()) {
215                 for (String part : partList) {
216                     if (partitionMap.containsKey(part)) {
217                         partitions.add(partitionMap.get(part));
218                     }
219                 }
220             }
221             ds.setPartitions(partitions.toArray(new HWPartition[partitions.size()]));
222             // Add to list
223             result.add(ds);
224         }
225         return result.toArray(new HWDiskStore[result.size()]);
226     }

228     /**
229      * Populates the maps for the specified index. If the index is null,
230      * populates all the maps
231      * 
232      * @param index
233      *            The index to populate/update maps for
234      */
235     private static void populateReadWriteMaps(String index) {
236         // If index is null, start from scratch.
237         if (index == null) {
238             readMap.clear();
239             readByteMap.clear();
240             writeMap.clear();
241             writeByteMap.clear();
242             xferTimeMap.clear();
243             timeStampMap.clear();
244         }
245         // Fetch the instance names
246         List<String> instances = PdhUtil.PdhEnumObjectItemInstances(null, null, "PhysicalDisk", 100);
247         instances.remove("_Total");
248         // At this point we have a list of strings that PDH understands. Fetch
249         // the counters.
250         // Although the field names say "PerSec" this is the Raw Data/counters
251         // from which the associated fields are populated in the Formatted Data
252         for (String i : instances) {
253             String name = ParseUtil.whitespaces.split(i)[0];
254             String readString = String.format(PDH_DISK_READS_FORMAT, i);
255             if (!PerfDataUtil.isCounter(readString)) {
256                 PerfDataUtil.addCounter(readString);
257             }
258             String readBytesString = String.format(PDH_DISK_READ_BYTES_FORMAT, i);
259             if (!PerfDataUtil.isCounter(readBytesString)) {
260                 PerfDataUtil.addCounter(readBytesString);
261             }
262             String writeString = String.format(PDH_DISK_WRITES_FORMAT, i);
263             if (!PerfDataUtil.isCounter(writeString)) {
264                 PerfDataUtil.addCounter(writeString);
265             }
266             String writeBytesString = String.format(PDH_DISK_WRITE_BYTES_FORMAT, i);
267             if (!PerfDataUtil.isCounter(writeBytesString)) {
268                 PerfDataUtil.addCounter(writeBytesString);
269             }
270             String xferTimeString = String.format(PDH_DISK_TIME_FORMAT, i);
271             if (!PerfDataUtil.isCounter(xferTimeString)) {
272                 PerfDataUtil.addCounter(xferTimeString);
273             }
274             readMap.put(name, PerfDataUtil.queryCounter(readString));
275             readByteMap.put(name, PerfDataUtil.queryCounter(readBytesString));
276             writeMap.put(name, PerfDataUtil.queryCounter(writeString));
277             writeByteMap.put(name, PerfDataUtil.queryCounter(writeBytesString));
278             xferTimeMap.put(name, PerfDataUtil.queryCounter(xferTimeString) / 10000L);
279             timeStampMap.put(name, PerfDataUtil.queryCounterTimestamp(xferTimeString));
280         }
281     }

283     private void populatePartitionMaps() {
284         driveToPartitionMap.clear();
285         partitionToLogicalDriveMap.clear();
286         partitionMap.clear();
287         // For Regexp matching DeviceIDs
288         Matcher mAnt;
289         Matcher mDep;
290 
291         // Map drives to partitions
292         WmiResult<DriveToPartitionProperty> drivePartitionMap = WmiUtil.queryWMI(DRIVE_TO_PARTITION_QUERY);
293         for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
294             mAnt = DEVICE_ID.matcher((String) drivePartitionMap.get(DriveToPartitionProperty.ANTECEDENT).get(i));
295             mDep = DEVICE_ID.matcher((String) drivePartitionMap.get(DriveToPartitionProperty.DEPENDENT).get(i));
296             if (mAnt.matches() && mDep.matches()) {
297                 MapUtil.createNewListIfAbsent(driveToPartitionMap, mAnt.group(1).replaceAll("\\\\\\\\", "\\\\"))
298                         .add(mDep.group(1));
299             }
300         }
301 
302         // Map partitions to logical disks
303         WmiResult<DriveToPartitionProperty> diskPartitionMap = WmiUtil.queryWMI(DISK_TO_PARTITION_QUERY);
304         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
305             mAnt = DEVICE_ID.matcher((String) diskPartitionMap.get(DriveToPartitionProperty.ANTECEDENT).get(i));
306             mDep = DEVICE_ID.matcher((String) diskPartitionMap.get(DriveToPartitionProperty.DEPENDENT).get(i));
307             if (mAnt.matches() && mDep.matches()) {
308                 partitionToLogicalDriveMap.put(mAnt.group(1), mDep.group(1) + "\\");
309             }
310         }
311 
312         // Next, get all partitions and create objects
313         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = WmiUtil.queryWMI(PARTITION_QUERY);
314         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
315             String deviceID = (String) hwPartitionQueryMap.get(DiskPartitionProperty.DEVICEID).get(i);
316             String logicalDrive = MapUtil.getOrDefault(partitionToLogicalDriveMap, deviceID, "");
317             String uuid = "";
318             if (!logicalDrive.isEmpty()) {
319                 // Get matching volume for UUID
320                 char[] volumeChr = new char[BUFSIZE];
321                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive, volumeChr, BUFSIZE);
322                 uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
323             }
324             partitionMap.put(deviceID,
325                     new HWPartition((String) hwPartitionQueryMap.get(DiskPartitionProperty.NAME).get(i),
326                             (String) hwPartitionQueryMap.get(DiskPartitionProperty.TYPE).get(i),
327                             (String) hwPartitionQueryMap.get(DiskPartitionProperty.DESCRIPTION).get(i), uuid,
328                             (Long) hwPartitionQueryMap.get(DiskPartitionProperty.SIZE).get(i),
329                             ((Long) hwPartitionQueryMap.get(DiskPartitionProperty.DISKINDEX).get(i)).intValue(),
330                             ((Long) hwPartitionQueryMap.get(DiskPartitionProperty.INDEX).get(i)).intValue(),
331                             logicalDrive));
332         }
333     }

19  package oshi.hardware.platform.windows;
20  
21  import java.util.List;
22  import java.util.Map;
23  
24  import oshi.hardware.common.AbstractComputerSystem;
25  import oshi.util.platform.windows.WmiUtil;
26  
27  /**
28   * Hardware data obtained from WMI
29   * 
30   * @author SchiTho1 [at] Securiton AG
31   * @author widdis [at] gmail [dot] com
32   */
33  final class WindowsComputerSystem extends AbstractComputerSystem {
34  
35      private static final long serialVersionUID = 1L;
36  
37      WindowsComputerSystem() {
38          init();
39      }
40  
41      private void init() {
42  
43          final Map<String, List<String>> win32ComputerSystem = WmiUtil.selectStringsFrom(null, "Win32_ComputerSystem",
44                  "Manufacturer,Model", null);
45  
46          final List<String> manufacturers = win32ComputerSystem.get("Manufacturer");
47          if (manufacturers != null && !manufacturers.isEmpty()) {
48              setManufacturer(manufacturers.get(0));
49          }
50  
51          final List<String> models = win32ComputerSystem.get("Model");
52          if (models != null && !models.isEmpty()) {
53              setModel(models.get(0));
54          }
55  
56          setSerialNumber(getSystemSerialNumber());
57  
58          setFirmware(new WindowsFirmware());
59  
60          setBaseboard(new WindowsBaseboard());
61      }
62  
63      private String getSystemSerialNumber() {
64          // This should always work
65          String serialNumber = WmiUtil.selectStringFrom(null, "Win32_BIOS", "SerialNumber", "where PrimaryBIOS=true");
66          // If the above doesn't work, this might
67          if ("".equals(serialNumber)) {
68              serialNumber = WmiUtil.selectStringFrom(null, "Win32_Csproduct", "IdentifyingNumber", null);
69          }
70          if ("".equals(serialNumber)) {
71              serialNumber = "unknown";
72          }
73          return serialNumber;
74      }
75  }

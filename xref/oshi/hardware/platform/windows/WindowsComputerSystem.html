19  package oshi.hardware.platform.windows;
20  
21  import oshi.hardware.common.AbstractComputerSystem;
22  import oshi.jna.platform.windows.WbemcliUtil;
23  import oshi.jna.platform.windows.WbemcliUtil.WmiQuery;
24  import oshi.jna.platform.windows.WbemcliUtil.WmiResult;
25  import oshi.util.platform.windows.WmiUtil;
26  
27  /**
28   * Hardware data obtained from WMI
29   *
30   * @author SchiTho1 [at] Securiton AG
31   * @author widdis [at] gmail [dot] com
32   */
33  final class WindowsComputerSystem extends AbstractComputerSystem {
34  
35      private static final long serialVersionUID = 1L;
36  
37      enum ComputerSystemProperty {
38          MANUFACTURER, MODEL;
39      }
40  
41      enum BiosProperty {
42          SERIALNUMBER;
43      }
44  
45      enum ComputerSystemProductProperty {
46          IDENTIFYINGNUMBER;
47      }
48  
49      private static final WmiQuery<ComputerSystemProductProperty> IDENTIFYINGNUMBER_QUERY = WbemcliUtil
50              .createQuery("Win32_ComputerSystemProduct", ComputerSystemProductProperty.class);
51  
52      private String systemSerialNumber = "";
53  
54      WindowsComputerSystem() {
55          init();
56      }
57  
58      private void init() {
59          WmiQuery<ComputerSystemProperty> computerSystemQuery = WbemcliUtil.createQuery("Win32_ComputerSystem",
60                  ComputerSystemProperty.class);
61          WmiResult<ComputerSystemProperty> win32ComputerSystem = WmiUtil.queryWMI(computerSystemQuery);
62          if (win32ComputerSystem.getResultCount() > 0) {
63              setManufacturer(win32ComputerSystem.getString(ComputerSystemProperty.MANUFACTURER, 0));
64              setModel(win32ComputerSystem.getString(ComputerSystemProperty.MODEL, 0));
65          }
66  
67          setSerialNumber(getSystemSerialNumber());
68          setFirmware(new WindowsFirmware());
69          setBaseboard(new WindowsBaseboard());
70      }
71  
72      private String getSystemSerialNumber() {
73          if (!"".equals(this.systemSerialNumber)) {
74              return this.systemSerialNumber;
75          }
76          // This should always work
77          WmiQuery<BiosProperty> serialNumberQuery = WbemcliUtil.createQuery("Win32_BIOS where PrimaryBIOS=true",
78                  BiosProperty.class);
79          WmiResult<BiosProperty> serialNumber = WmiUtil.queryWMI(serialNumberQuery);
80          if (serialNumber.getResultCount() > 0) {
81              this.systemSerialNumber = serialNumber.getString(BiosProperty.SERIALNUMBER, 0);
82          }
83          // If the above doesn't work, this might
84          if (!"".equals(this.systemSerialNumber)) {
85              WmiResult<ComputerSystemProductProperty> identifyingNumber = WmiUtil.queryWMI(IDENTIFYINGNUMBER_QUERY);
86              if (identifyingNumber.getResultCount() > 0) {
87                  this.systemSerialNumber = identifyingNumber.getString(ComputerSystemProductProperty.IDENTIFYINGNUMBER, 0);
88              }
89          }
90          // Nothing worked. Default.
91          if (!"".equals(this.systemSerialNumber)) {
92              this.systemSerialNumber = "unknown";
93          }
94          return this.systemSerialNumber;
95      }
96  }

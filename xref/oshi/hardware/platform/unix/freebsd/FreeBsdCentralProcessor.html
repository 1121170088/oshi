53      private static final Pattern CPUMASK = Pattern.compile(".*<cpu\\s.*mask=\"(?:0x)?(\\p{XDigit}+)\".*>.*</cpu>.*");
54  
55      private static final Pattern CPUINFO = Pattern
56              .compile("Origin=\"([^\"]*)\".*Id=(\\S+).*Family=(\\S+).*Model=(\\S+).*Stepping=(\\S+).*");
57      private static final Pattern CPUINFO2 = Pattern.compile("Features=(\\S+)<.*");
58  
59      private static final long BOOTTIME;
60      static {
61          Timeval tv = new Timeval();
62          if (!BsdSysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
63              // Usually this works. If it doesn't, fall back to text parsing.
64              // Boot time will be the first consecutive string of digits.
65              BOOTTIME = ParseUtil.parseLongOrDefault(
66                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
67                      System.currentTimeMillis() / 1000);
68          } else {
69              // tv now points to a 128-bit timeval structure for boot time.
70              // First 8 bytes are seconds, second 8 bytes are microseconds
71              // (we ignore)
72              BOOTTIME = tv.tv_sec;
73          }
74      }

89      private void initVars() {
90          setName(BsdSysctlUtil.sysctl("hw.model", ""));
91          // This is apparently the only reliable source for this stuff on
92          // FreeBSD...
93          long processorID = 0L;
94          List<String> cpuInfo = FileUtil.readFile("/var/run/dmesg.boot");
95          for (String line : cpuInfo) {
96              line = line.trim();
97              // Prefer hw.model to this one
98              if (line.startsWith("CPU:") && getName().isEmpty()) {
99                  setName(line.replace("CPU:", "").trim());
100             } else if (line.startsWith("Origin=")) {
101                 Matcher m = CPUINFO.matcher(line);
102                 if (m.matches()) {
103                     setVendor(m.group(1));
104                     processorID |= Long.decode(m.group(2));
105                     setFamily(Integer.decode(m.group(3)).toString());
106                     setModel(Integer.decode(m.group(4)).toString());
107                     setStepping(Integer.decode(m.group(5)).toString());
108                 }
109             } else if (line.startsWith("Features=")) {
110                 Matcher m = CPUINFO2.matcher(line);
111                 if (m.matches()) {
112                     processorID |= Long.decode(m.group(1)) << 32;
113                 }
114                 // No further interest in this file
115                 break;
116             }
117         }
118         setCpu64(ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64"));
119         setProcessorID(getProcessorID(processorID));
120     }

122     /**
123      * Updates logical and physical processor/package counts
124      */
125     @Override
126     protected void calculateProcessorCounts() {
127         String[] topology = BsdSysctlUtil.sysctl("kern.sched.topology_spec", "").split("\\n|\\r");
128         int physMask = 0;
129         int virtMask = 0;
130         int lastMask = 0;
131         int physPackage = 0;
132         for (String topo : topology) {
133             if (topo.contains("<cpu")) {
134                 // Find <cpu> tag and extract bits
135                 Matcher m = CPUMASK.matcher(topo);
136                 if (m.matches()) {
137                     // Add this processor mask to cpus. Regex guarantees parsing
138                     lastMask = Integer.parseInt(m.group(1), 16);
139                     physMask |= lastMask;
140                     virtMask |= lastMask;
141                 }
142             } else if (topo.contains("<flags>")
143                     && (topo.contains("HTT") || topo.contains("SMT") || topo.contains("THREAD"))) {
144                 // These are virtual cpus, remove processor mask from physical
145                 physMask &= ~lastMask;
146             } else if (topo.contains("<group level=\"2\"")) {
147                 // This is a physical package
148                 physPackage++;
149             }
150         }
151 
152         this.logicalProcessorCount = Integer.bitCount(virtMask);
153         if (this.logicalProcessorCount < 1) {
154             LOG.error("Couldn't find logical processor count. Assuming 1.");
155             this.logicalProcessorCount = 1;
156         }
157         this.physicalProcessorCount = Integer.bitCount(physMask);
158         if (this.physicalProcessorCount < 1) {
159             LOG.error("Couldn't find physical processor count. Assuming 1.");
160             this.physicalProcessorCount = 1;
161         }
162         this.physicalPackageCount = physPackage;
163         if (this.physicalPackageCount < 1) {
164             LOG.error("Couldn't find physical package count. Assuming 1.");
165             this.physicalPackageCount = 1;
166         }
167     }

169     /**
170      * {@inheritDoc}
171      */
172     @Override
173     public synchronized long[] getSystemCpuLoadTicks() {
174         long[] ticks = new long[TickType.values().length];
175         CpTime cpTime = new CpTime();
176         BsdSysctlUtil.sysctl("kern.cp_time", cpTime);
177         ticks[TickType.USER.getIndex()] = cpTime.cpu_ticks[Libc.CP_USER];
178         ticks[TickType.NICE.getIndex()] = cpTime.cpu_ticks[Libc.CP_NICE];
179         ticks[TickType.SYSTEM.getIndex()] = cpTime.cpu_ticks[Libc.CP_SYS];
180         ticks[TickType.IRQ.getIndex()] = cpTime.cpu_ticks[Libc.CP_INTR];
181         ticks[TickType.IDLE.getIndex()] = cpTime.cpu_ticks[Libc.CP_IDLE];
182         return ticks;
183     }

185     /**
186      * {@inheritDoc}
187      */
188     @Override
189     public double[] getSystemLoadAverage(int nelem) {
190         if (nelem < 1 || nelem > 3) {
191             throw new IllegalArgumentException("Must include from one to three elements.");
192         }
193         double[] average = new double[nelem];
194         int retval = Libc.INSTANCE.getloadavg(average, nelem);
195         if (retval < nelem) {
196             for (int i = Math.max(retval, 0); i < average.length; i++) {
197                 average[i] = -1d;
198             }
199         }
200         return average;
201     }

203     /**
204      * {@inheritDoc}
205      */
206     @Override
207     public long[][] getProcessorCpuLoadTicks() {
208         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
209 
210         // Allocate memory for array of CPTime
211         int offset = new CpTime().size();
212         int size = offset * this.logicalProcessorCount;
213         Pointer p = new Memory(size);
214         String name = "kern.cp_times";
215         // Fetch
216         if (0 != Libc.INSTANCE.sysctlbyname(name, p, new IntByReference(size), null, 0)) {
217             LOG.error("Failed syctl call: {}, Error code: {}", name, Native.getLastError());
218             return ticks;
219         }
220         // p now points to the data; need to copy each element
221         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
222             ticks[cpu][TickType.USER.getIndex()] = p.getLong((long) (offset * cpu + Libc.CP_USER * Libc.UINT64_SIZE)); // lgtm[java/evaluation-to-constant]
223             ticks[cpu][TickType.NICE.getIndex()] = p.getLong((long) (offset * cpu + Libc.CP_NICE * Libc.UINT64_SIZE));
224             ticks[cpu][TickType.SYSTEM.getIndex()] = p.getLong((long) (offset * cpu + Libc.CP_SYS * Libc.UINT64_SIZE));
225             ticks[cpu][TickType.IRQ.getIndex()] = p.getLong((long) (offset * cpu + Libc.CP_INTR * Libc.UINT64_SIZE));
226             ticks[cpu][TickType.IDLE.getIndex()] = p.getLong((long) (offset * cpu + Libc.CP_IDLE * Libc.UINT64_SIZE));
227         }
228         return ticks;
229     }

248     /**
249      * Fetches the ProcessorID from dmidecode (if possible with root
250      * permissions), otherwise uses the values from /var/run/dmesg.boot
251      *
252      * @param processorID
253      * @return The ProcessorID string
254      */
255     private String getProcessorID(long processorID) {
256         boolean procInfo = false;
257         String marker = "Processor Information";
258         for (String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
259             if (!procInfo && checkLine.contains(marker)) {
260                 marker = "ID:";
261                 procInfo = true;
262             } else if (procInfo && checkLine.contains(marker)) {
263                 return checkLine.split(marker)[1].trim();
264             }
265         }
266         // If we've gotten this far, dmidecode failed. Used the passed-in values
267         return String.format("%016X", processorID);
268     }

271      * {@inheritDoc}
272      */
273     @Override
274     public long getContextSwitches() {
275         String name = "vm.stats.sys.v_swtch";
276         IntByReference size = new IntByReference(Libc.INT_SIZE);
277         Pointer p = new Memory(size.getValue());
278         if (0 != Libc.INSTANCE.sysctlbyname(name, p, size, null, 0)) {
279             return -1;
280         }
281         return ParseUtil.unsignedIntToLong(p.getInt(0));
282     }

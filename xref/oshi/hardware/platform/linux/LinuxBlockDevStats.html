19  package oshi.hardware.platform.linux;
20  
21  import java.io.Serializable;
22  
23  import oshi.jna.platform.linux.Udev;
24  import oshi.util.ParseUtil;
25  
26  /**
27   * POJO for mapping Linux block device stats info
28   *
29   * @author enrico[dot]bianchi[at]gmail[dot]com
30   */
31  public class LinuxBlockDevStats implements Serializable {
32  
33      private static final long serialVersionUID = 1L;
34  
35      // Device name
36      public final String device;
37      // Number of read I/Os processed
38      public final long read_ops;
39      // Number of read I/Os merged with in-queue I/O
40      public final long read_merged;
41      // Number of sectors read
42      public final long read_512bytes;
43      // Total wait time for read requests, milliseconds
44      public final long read_waits_ms;
45      // Number of write I/Os processed
46      public final long write_ops;
47      // Number of write I/Os merged with in-queue I/O
48      public final long write_merged;
49      // Number of sectors written
50      public final long write_512bytes;
51      // Total wait time for write requests, milliseconds */
52      public final long write_waits_ms;
53      // Number of I/Os currently in flight
54      public final long in_flight;
55      // Total active time, milliseconds
56      public final long active_ms;
57      // Total wait time, milliseconds
58      public final long waits_ms;
59  
60      public LinuxBlockDevStats(String device, Udev.UdevDevice disk) {
61          String devstat;
62          String[] splitstats;
63  
64          devstat = Udev.INSTANCE.udev_device_get_sysattr_value(disk, "stat");
65          splitstats = ParseUtil.whitespaces.split(devstat);
66  
67          if (splitstats.length < 11) {
68              throw new IllegalStateException("Unexpected length of array: " + splitstats.length);
69          }
70  
71          // read last 11 elements
72          int startIndex = splitstats.length - 11;
73  
74          this.device = device;
75          this.read_ops = ParseUtil.parseLongOrDefault(splitstats[startIndex + 0], 0L);
76          this.read_merged = ParseUtil.parseLongOrDefault(splitstats[startIndex + 1], 0L);
77          this.read_512bytes = ParseUtil.parseLongOrDefault(splitstats[startIndex + 2], 0L);
78          this.read_waits_ms = ParseUtil.parseLongOrDefault(splitstats[startIndex + 3], 0L);
79          this.write_ops = ParseUtil.parseLongOrDefault(splitstats[startIndex + 4], 0L);
80          this.write_merged = ParseUtil.parseLongOrDefault(splitstats[startIndex + 5], 0L);
81          this.write_512bytes = ParseUtil.parseLongOrDefault(splitstats[startIndex + 6], 0L);
82          this.write_waits_ms = ParseUtil.parseLongOrDefault(splitstats[startIndex + 7], 0L);
83          this.in_flight = ParseUtil.parseLongOrDefault(splitstats[startIndex + 8], 0L);
84          this.active_ms = ParseUtil.parseLongOrDefault(splitstats[startIndex + 9], 0L);
85          this.waits_ms = ParseUtil.parseLongOrDefault(splitstats[startIndex + 10], 0L);
86      }
87  
88  }

17  package oshi.hardware.platform.linux;
18  
19  import java.io.IOException;
20  import java.util.List;
21  
22  import javax.json.Json;
23  import javax.json.JsonBuilderFactory;
24  import javax.json.JsonObject;
25  
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import com.sun.jna.Native;
30  
31  import oshi.hardware.GlobalMemory;
32  import oshi.jna.platform.linux.Libc;
33  import oshi.jna.platform.linux.Libc.Sysinfo;
34  import oshi.json.NullAwareJsonObjectBuilder;
35  import oshi.util.FileUtil;
36  
37  /**
38   * Memory obtained by /proc/meminfo and sysinfo.totalram
39   * 
40   * @author alessandro[at]perucchi[dot]org
41   * @author widdis[at]gmail[dot]com
42   */
43  public class LinuxGlobalMemory implements GlobalMemory {
44      private static final Logger LOG = LoggerFactory.getLogger(LinuxGlobalMemory.class);
45  
46      private long totalMemory = 0;
47  
48      private JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
49  
50      @Override
51      public long getAvailable() {
52          long availableMemory = 0;
53          List<String> memInfo = null;
54          try {
55              memInfo = FileUtil.readFile("/proc/meminfo");
56          } catch (IOException e) {
57              LOG.error("Problem with /proc/meminfo: {}", e.getMessage());
58              return availableMemory;
59          }
60          for (String checkLine : memInfo) {
61              // If we have MemAvailable, it trumps all. See code in
62              // https://git.kernel.org/cgit/linux/kernel/git/torvalds/
63              // linux.git/commit/?id=34e431b0ae398fc54ea69ff85ec700722c9da773
64              if (checkLine.startsWith("MemAvailable:")) {
65                  String[] memorySplit = checkLine.split("\\s+");
66                  availableMemory = parseMeminfo(memorySplit);
67                  break;
68              } else
69              // Otherwise we combine MemFree + Active(file), Inactive(file), and
70              // Reclaimable. Free+cached is no longer appropriate. MemAvailable
71              // reduces these values using watermarks to estimate when swapping
72              // is prevented, omitted here for simplicity (assuming 0 swap).
73              if (checkLine.startsWith("MemFree:")) {
74                  String[] memorySplit = checkLine.split("\\s+");
75                  availableMemory += parseMeminfo(memorySplit);
76              } else if (checkLine.startsWith("Active(file):")) {
77                  String[] memorySplit = checkLine.split("\\s+");
78                  availableMemory += parseMeminfo(memorySplit);
79              } else if (checkLine.startsWith("Inactive(file):")) {
80                  String[] memorySplit = checkLine.split("\\s+");
81                  availableMemory += parseMeminfo(memorySplit);
82              } else if (checkLine.startsWith("SReclaimable:")) {
83                  String[] memorySplit = checkLine.split("\\s+");
84                  availableMemory += parseMeminfo(memorySplit);
85              }
86          }
87          return availableMemory;
88      }
89  
90      @Override
91      public long getTotal() {
92          if (this.totalMemory == 0) {
93              // Try to get it from the libc sysinfo call
94              try {
95                  Sysinfo info = new Sysinfo();
96                  if (0 != Libc.INSTANCE.sysinfo(info)) {
97                      LOG.error("Failed to get total memory. Error code: " + Native.getLastError());
98                  } else {
99                      this.totalMemory = info.totalram.longValue() * info.mem_unit;
100                     return this.totalMemory;
101                 }
102             } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
103                 LOG.warn("Failed to get total memory from sysinfo. Falling back /proc/meminfo MemTotal. {}",
104                         e.getMessage());
105             }
106             // If still no success, populate from /proc/meminfo
107             List<String> memInfo = null;
108             try {
109                 memInfo = FileUtil.readFile("/proc/meminfo");
110             } catch (IOException e) {
111                 LOG.error("Problem with /proc/meminfo: {}", e.getMessage());
112                 return 0;
113             }
114             for (String checkLine : memInfo) {
115                 if (checkLine.startsWith("MemTotal:")) {
116                     String[] memorySplit = checkLine.split("\\s+");
117                     this.totalMemory = parseMeminfo(memorySplit);
118                     break;
119                 }
120             }
121         }
122         return this.totalMemory;
123     }
124 
125     private long parseMeminfo(String[] memorySplit) {
126         if (memorySplit.length < 2) {
127             return 0l;
128         }
129         long memory = Long.valueOf(memorySplit[1]);
130         if (memorySplit.length > 2 && memorySplit[2].equals("kB")) {
131             memory *= 1024;
132         }
133         return memory;
134     }
135 
136     @Override
137     public JsonObject toJSON() {
138         return NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder()).add("available", getAvailable())
139                 .add("total", getTotal()).build();
140     }
141 }

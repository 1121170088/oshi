69      public MacSensors() {
70          smcOpen();
71          // Do an initial read of temperature and fan speeds. This caches initial
72          // dataInfo and improves success of future queries
73          this.lastTemp = getCpuTemperature();
74          this.lastFanSpeeds = getFanSpeeds();
75  
76          Runtime.getRuntime().addShutdownHook(new Thread() {
77              @Override
78              public void run() {
79                  smcClose();
80              }
81          });
82      }

84      /*
85       * Open connection to SMC
86       */
87      private int smcOpen() {
88          int service = 0;
89          MachPort masterPort = new MachPort();
90  
91          int result = IOKit.INSTANCE.IOMasterPort(0, masterPort);
92          if (result != 0) {
93              LOG.error(String.format("Error: IOMasterPort() = %08x", result));
94              return result;
95          }
96  
97          service = IOKit.INSTANCE.IOServiceGetMatchingService(masterPort.getValue(),
98                  IOKit.INSTANCE.IOServiceMatching("AppleSMC"));
99          if (service == 0) {
100             LOG.error("Error: no SMC found\n");
101             return result;
102         }
103 
104         result = IOKit.INSTANCE.IOServiceOpen(service, SystemB.INSTANCE.mach_task_self(), 0, this.conn);
105         IOKit.INSTANCE.IOObjectRelease(service);
106         if (result != 0) {
107             LOG.error(String.format("Error: IOServiceOpen() = 0x%08x", result));
108             return result;
109         }
110         // Delay to improve success of next query
111         Util.sleep(5);
112         return 0;
113     }
114 
115     private int smcClose() {
116         return IOKit.INSTANCE.IOServiceClose(this.conn.getValue());
117     }

119     /*
120      * Get Temperature from SMC
121      */
122     private double smcGetTemperature(String key, int retries) {
123         SMCVal val = new SMCVal();
124         int result = smcReadKey(key, val, retries);
125         if (result == 0) {
126             int temp = 0;
127             if (val.dataSize > 0 && Arrays.equals(val.dataType, IOKit.DATATYPE_SP78)) {
128                 temp = (val.bytes[0] * 256 + val.bytes[1]) >> 2;
129             }
130             return temp / 64.0;
131         }
132         // Read failed
133         return 0d;
134     }

136     /*
137      * Get number of fans
138      */
139     private int smcGetFanNumber(int retries) {
140         SMCVal val = new SMCVal();
141         int result = smcReadKey(IOKit.SMC_KEY_FAN_NUM, val, retries);
142         if (result == 0) {
143             return strtoul10(val.bytes, val.dataSize);
144         }
145         // Read failed
146         return 0;
147     }

149     /*
150      * Get fan speed
151      */
152     private float smcGetFanSpeed(int fanNum, int retries) {
153         SMCVal val = new SMCVal();
154         String key = String.format(IOKit.SMC_KEY_FAN_SPEED, fanNum);
155         int result = smcReadKey(key, val, retries);
156         if (result == 0) {
157             return strtof(val.bytes, val.dataSize, 2);
158         }
159         // Read failed
160         return 0f;
161     }

163     /*
164      * Get voltage
165      */
166     private float smcGetVoltage(int retries) {
167         SMCVal val = new SMCVal();
168         int result = smcReadKey(IOKit.SMC_KEY_CPU_VOLTAGE, val, retries);
169         if (result == 0) {
170             return strtof(val.bytes, val.dataSize, 2) / 1000f;
171         }
172         // Read failed
173         return 0f;
174     }

176     /*
177      * Get cached keyInfo if it exists, or generate new
178      */
179     private int smcGetKeyInfo(SMCKeyData inputStructure, SMCKeyData outputStructure) {
180         if (keyInfoCache.containsKey(inputStructure.key)) {
181             SMCKeyDataKeyInfo keyInfo = keyInfoCache.get(inputStructure.key);
182             outputStructure.keyInfo.dataSize = keyInfo.dataSize;
183             outputStructure.keyInfo.dataType = keyInfo.dataType;
184             outputStructure.keyInfo.dataAttributes = keyInfo.dataAttributes;
185         } else {
186             int result = 0;
187             inputStructure.data8 = IOKit.SMC_CMD_READ_KEYINFO;
188             Util.sleep(4);
189             result = smcCall(IOKit.KERNEL_INDEX_SMC, inputStructure, outputStructure);
190             if (result != 0) {
191                 return result;
192             }
193             SMCKeyDataKeyInfo keyInfo = new SMCKeyDataKeyInfo();
194             keyInfo.dataSize = outputStructure.keyInfo.dataSize;
195             keyInfo.dataType = outputStructure.keyInfo.dataType;
196             keyInfo.dataAttributes = outputStructure.keyInfo.dataAttributes;
197             keyInfoCache.put(inputStructure.key, keyInfo);
198         }
199         return 0;
200     }

202     /*
203      * Read a key from SMC
204      */
205     private int smcReadKey(String key, SMCVal val, int retries) {
206         SMCKeyData inputStructure = new SMCKeyData();
207         SMCKeyData outputStructure = new SMCKeyData();
208 
209         inputStructure.key = strtoul16(key, 4);
210         int result;
211         do {
212             result = smcGetKeyInfo(inputStructure, outputStructure);
213             if (result != 0) {
214                 continue;
215             }
216 
217             val.dataSize = outputStructure.keyInfo.dataSize;
218             val.dataType = new byte[5];
219             for (int i = 3; i >= 0; i--) {
220                 val.dataType[i] = (byte) (outputStructure.keyInfo.dataType >> (8 * (3 - i)));
221             }
222             inputStructure.keyInfo.dataSize = val.dataSize;
223             inputStructure.data8 = IOKit.SMC_CMD_READ_BYTES;
224 
225             Util.sleep(4);
226             result = smcCall(IOKit.KERNEL_INDEX_SMC, inputStructure, outputStructure);
227             if (result != 0) {
228                 continue;
229             }
230             break;
231         } while (--retries > 0);
232         // If we errored out return code
233         if (result != 0) {
234             return result;
235         }
236 
237         System.arraycopy(outputStructure.bytes, 0, val.bytes, 0, val.bytes.length);
238         // Success
239         return 0;
240     }

242     /*
243      * Call SMC
244      */
245     private int smcCall(int index, SMCKeyData inputStructure, SMCKeyData outputStructure) {
246         int structureInputSize = inputStructure.size();
247         IntByReference structureOutputSizePtr = new IntByReference(outputStructure.size());
248 
249         int result = IOKit.INSTANCE.IOConnectCallStructMethod(this.conn.getValue(), index, inputStructure,
250                 structureInputSize, outputStructure, structureOutputSizePtr);
251         if (result != 0) {
252             // This seems to be a common error, suppressing output
253             // LOG.error(String.format("Error: IOConnectCallStructMethod() =
254             // 0x%08x", result));
255             return result;
256         }
257         // Success
258         return result;
259     }

261     private int strtoul16(String str, int size) {
262         int total = 0;
263         int i;
264 
265         for (i = 0; i < size; i++) {
266             total += str.charAt(i) << (size - 1 - i) * 8;
267         }
268         return total;
269     }
270 
271     private int strtoul10(byte[] bytes, int size) {
272         int total = 0;
273         int i;
274 
275         for (i = 0; i < size; i++) {
276             total += (byte) (bytes[i] << (size - 1 - i) * 8);
277         }
278         return total;
279     }
280 
281     float strtof(byte[] bytes, int size, int e) {
282         float total = 0;
283         for (int i = 0; i < size; i++) {
284             if (i == (size - 1))
285                 total += (bytes[i] & 0xff) >> e;
286             else
287                 total += bytes[i] << (size - 1 - i) * (8 - e);
288         }
289         total += (bytes[size - 1] & 0x03) * 0.25;
290         return total;
291     }

293     @Override
294     public double getCpuTemperature() {
295         // Only update every second
296         if (System.currentTimeMillis() - this.lastTempTime > 900) {
297             double temp = smcGetTemperature(IOKit.SMC_KEY_CPU_TEMP, 50);
298             if (temp > 0d) {
299                 this.lastTemp = temp;
300                 lastTempTime = System.currentTimeMillis();
301             }
302         }
303         return this.lastTemp;
304     }
305 
306     @Override
307     public int[] getFanSpeeds() {
308         // Only update every second
309         if (System.currentTimeMillis() - this.lastFanSpeedsTime > 900) {
310             // If we don't have fan # try to get it
311             if (this.numFans == 0) {
312                 this.numFans = smcGetFanNumber(50);
313                 this.lastFanSpeeds = new int[this.numFans];
314             }
315             for (int i = 0; i < this.numFans; i++) {
316                 int speed = (int) smcGetFanSpeed(i, 50);
317                 if (speed > 0) {
318                     this.lastFanSpeeds[i] = speed;
319                     lastFanSpeedsTime = System.currentTimeMillis();
320                 }
321             }
322         }
323         return this.lastFanSpeeds;
324     }
325 
326     @Override
327     public double getCpuVoltage() {
328         // Only update every second
329         if (System.currentTimeMillis() - this.lastVoltsTime > 900) {
330             double volts = smcGetVoltage(50);
331             if (volts > 0d) {
332                 this.lastVolts = volts;
333                 lastVoltsTime = System.currentTimeMillis();
334             }
335         }
336         return this.lastVolts;
337     }

339     @Override
340     public JsonObject toJSON() {
341         JsonArrayBuilder fanSpeedsArrayBuilder = jsonFactory.createArrayBuilder();
342         for (int speed : getFanSpeeds()) {
343             fanSpeedsArrayBuilder.add(speed);
344         }
345         return NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder()).add("cpuTemperature", getCpuTemperature())
346                 .add("fanSpeeds", fanSpeedsArrayBuilder.build()).add("cpuVoltage", getCpuVoltage()).build();
347     }

18  package oshi.hardware.platform.mac;
19  
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import oshi.hardware.HWDiskStore;
24  import oshi.hardware.common.AbstractDisks;
25  import oshi.util.ExecutingCommand;
26  
27  /**
28   * Mac hard disk implementation.
29   *
30   * @author enrico[dot]bianchi[at]gmail[dot]com
31   */
32  public class MacDisks extends AbstractDisks {
33  
34      @Override
35      public HWDiskStore[] getDisks() {
36          List<HWDiskStore> result;
37          result = new ArrayList<>();
38  
39          String model = "";
40          String serial = "";
41          String capacity = "";
42          long size = 0L;
43          int indent = 0;
44  
45          ArrayList<String> hwInfo = ExecutingCommand.runNative("system_profiler SPSerialATADataType");
46          // Add a non-indented line to the end of the string to prompt List add
47          hwInfo.add("END");
48          // Iterate the list and store info as it is found.
49          for (String checkLine : hwInfo) {
50              // Ignore blank lines
51              if (checkLine.length() == 0) {
52                  continue;
53              }
54              // If indentation backs up by more than 2 spaces, end of this entry
55              String s = checkLine.trim();
56  
57              if (checkLine.indexOf(s) < indent - 2) {
58                  HWDiskStore ds = new HWDiskStore();
59                  ds.setName(String.format("disk%d", result.size()));
60                  ds.setModel(model);
61                  ds.setSerial(serial);
62                  ds.setSize(size);
63                  result.add(ds);
64  
65                  // Clear to prep for next
66                  model = "";
67                  serial = "";
68                  capacity = "";
69                  size = 0L;
70              }
71              indent = checkLine.indexOf(s);
72  
73              String[] split = s.split(":");
74              if (split.length < 2) {
75                  continue;
76              }
77              switch (split[0]) {
78              case "Model":
79                  model = split[1].trim();
80                  break;
81              case "Serial Number":
82                  serial = split[1].trim();
83                  break;
84              case "Capacity":
85                  // Only use the first Capacity we find
86                  if (capacity.length() == 0) {
87                      // Capacity: 209.7 MB (209,715,200 bytes)
88                      capacity = split[1].trim();
89                      capacity = capacity.substring(capacity.indexOf("(") + 1, capacity.indexOf(" bytes)"))
90                              .replaceAll(",", "");
91                      // If successful this line is the desired value
92                      try {
93                          size = Long.parseLong(capacity);
94                      } catch (NumberFormatException e) {
95                          // If we failed to parse, give up
96                      }
97                  }
98                  break;
99              default:
100                 // do nothing
101             }
102         }
103 
104         return result.toArray(new HWDiskStore[result.size()]);
105     }
106 }

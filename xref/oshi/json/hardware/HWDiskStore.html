33  /**
34   * A storage mechanism where data are recorded by various electronic, magnetic,
35   * optical, or mechanical changes to a surface layer of one or more rotating
36   * disks or flash storage such as a removable or solid state drive. In constrast
37   * to a File System, defining the way an Operating system uses the storage, the
38   * Disk Store represents the hardware which a FileSystem uses for its File
39   * Stores.
40   *
41   * @author enrico[dot]bianchi[at]gmail[dot]com
42   */
43  public class HWDiskStore extends AbstractOshiJsonObject {
44  
45      private static final long serialVersionUID = 1L;
46  
47      private transient JsonBuilderFactory jsonFactory = Json.createBuilderFactory(null);
48  
49      private oshi.hardware.HWDiskStore hwDiskStore;
50  
51      /**
52       * Create an object with all values
53       * 
54       * @param name
55       *            Name of the disk (e.g., /dev/disk1)
56       * @param model
57       *            Model of the disk
58       * @param serial
59       *            Disk serial number, if available
60       * @param size
61       *            Disk capacity in bytes
62       * @param reads
63       *            Number of reads from the disk
64       * @param readBytes
65       *            Number of bytes read from the disk
66       * @param writes
67       *            Number of writes to the disk
68       * @param writeBytes
69       *            Number of bytes written to the disk
70       * @param transferTime
71       *            milliseconds spent reading or writing to the disk
72       * @param partitions
73       *            Partitions on this disk
74       */
75      public HWDiskStore(String name, String model, String serial, long size, long reads, long readBytes, long writes,
76              long writeBytes, long transferTime, HWPartition[] partitions) {
77          oshi.hardware.HWPartition[] parts = new oshi.hardware.HWPartition[partitions.length];
78          for (int i = 0; i < partitions.length; i++) {
79              parts[i] = new oshi.hardware.HWPartition(partitions[i].getIdentification(), partitions[i].getName(),
80                      partitions[i].getType(), partitions[i].getUuid(), partitions[i].getSize(), partitions[i].getMajor(),
81                      partitions[i].getMinor(), partitions[i].getMountPoint());
82          }
83          this.hwDiskStore = new oshi.hardware.HWDiskStore(name, model, serial, size, reads, readBytes, writes,
84                  writeBytes, transferTime, parts);
85      }

150     /**
151      * @return the partitions of this disk
152      */
153     public HWPartition[] getPartitions() {
154         HWPartition[] partitions = new HWPartition[this.hwDiskStore.getPartitions().length];
155         for (int i = 0; i < partitions.length; i++) {
156             partitions[i] = new HWPartition(this.hwDiskStore.getPartitions()[i].getIdentification(),
157                     this.hwDiskStore.getPartitions()[i].getName(), this.hwDiskStore.getPartitions()[i].getType(),
158                     this.hwDiskStore.getPartitions()[i].getUuid(), this.hwDiskStore.getPartitions()[i].getSize(),
159                     this.hwDiskStore.getPartitions()[i].getMajor(), this.hwDiskStore.getPartitions()[i].getMinor(),
160                     this.hwDiskStore.getPartitions()[i].getMountPoint());
161         }
162         return partitions;
163     }

251     /**
252      * {@inheritDoc}
253      */
254     @Override
255     public JsonObject toJSON(Properties properties) {
256         JsonObjectBuilder json = NullAwareJsonObjectBuilder.wrap(jsonFactory.createObjectBuilder());
257         if (PropertiesUtil.getBoolean(properties, "hardware.disks.name")) {
258             json.add("name", this.hwDiskStore.getName());
259         }
260         if (PropertiesUtil.getBoolean(properties, "hardware.disks.model")) {
261             json.add("model", this.hwDiskStore.getModel());
262         }
263         if (PropertiesUtil.getBoolean(properties, "hardware.disks.serial")) {
264             json.add("serial", this.hwDiskStore.getSerial());
265         }
266         if (PropertiesUtil.getBoolean(properties, "hardware.disks.size")) {
267             json.add("size", this.hwDiskStore.getSize());
268         }
269         if (PropertiesUtil.getBoolean(properties, "hardware.disks.reads")) {
270             json.add("reads", this.hwDiskStore.getReads());
271         }
272         if (PropertiesUtil.getBoolean(properties, "hardware.disks.readBytes")) {
273             json.add("readBytes", this.hwDiskStore.getReadBytes());
274         }
275         if (PropertiesUtil.getBoolean(properties, "hardware.disks.writes")) {
276             json.add("writes", this.hwDiskStore.getWrites());
277         }
278         if (PropertiesUtil.getBoolean(properties, "hardware.disks.writeBytes")) {
279             json.add("writeBytes", this.hwDiskStore.getWriteBytes());
280         }
281         if (PropertiesUtil.getBoolean(properties, "hardware.disks.writeBytes")) {
282             json.add("transferTime", this.hwDiskStore.getTransferTime());
283         }
284         if (PropertiesUtil.getBoolean(properties, "hardware.disks.partitions")) {
285             JsonArrayBuilder partitionArrayBuilder = jsonFactory.createArrayBuilder();
286             for (HWPartition partition : getPartitions()) {
287                 partitionArrayBuilder.add(partition.toJSON(properties));
288             }
289             json.add("partitions", partitionArrayBuilder.build());
290         }
291         return json.build();
292     }

19  package oshi.json.software.os;
20  
21  import oshi.json.json.OshiJsonObject;
22  import oshi.software.os.OperatingSystem.ProcessSort;
23  
24  /**
25   * An operating system (OS) is the software on a computer that manages the way
26   * different programs use its hardware, and regulates the ways that a user
27   * controls the computer.
28   *
29   * @author dblock[at]dblock[dot]org
30   */
31  public interface OperatingSystem extends OshiJsonObject {
32  
33      /**
34       * Operating system family.
35       *
36       * @return String.
37       */
38      String getFamily();
39  
40      /**
41       * Manufacturer.
42       *
43       * @return String.
44       */
45      String getManufacturer();
46  
47      /**
48       * Operating system version.
49       *
50       * @return Version.
51       */
52      OperatingSystemVersion getVersion();
53  
54      /**
55       * Instantiates a {@link FileSystem} object.
56       *
57       * @return A {@link FileSystem} object.
58       */
59      FileSystem getFileSystem();
60  
61      /**
62       * Gets currently running processes. If a positive limit is specified,
63       * returns only that number of processes; zero will return all processes.
64       * The order may be specified by the sort parameter, for example, to return
65       * the top cpu or memory consuming processes; if null, no order is
66       * guaranteed.
67       *
68       * In the JSON output, these parameters may be specified by
69       * "operatingSystem.processes.limit" and "operatingSystem.processes.sort"
70       * property values.
71       *
72       * @param limit
73       *            Max number of results to return, or 0 to return all results
74       * @param sort
75       *            If not null, determines sorting of results
76       * @return An array of {@link oshi.software.os.OSProcess} objects for the
77       *         specified number (or all) of currently running processes, sorted
78       *         as specified. The array may contain null elements if a process
79       *         terminates during iteration.
80       */
81      OSProcess[] getProcesses(int limit, ProcessSort sort);
82  
83      /**
84       * Gets information on a currently running process
85       *
86       * @param pid
87       *            A process ID
88       * @return An {@link oshi.software.os.OSProcess} object for the specified
89       *         process id if it is running; null otherwise
90       */
91      OSProcess getProcess(int pid);
92  
93      /**
94       * Gets the current process ID
95       *
96       * @return the Process ID of the current process
97       */
98      int getProcessId();
99  
100     /**
101      * Get the number of processes currently running
102      *
103      * @return The number of processes running
104      */
105     int getProcessCount();
106 
107     /**
108      * Get the number of threads currently running
109      *
110      * @return The number of threads running
111      */
112     int getThreadCount();
113 
114     /**
115      * Instantiates a {@link NetworkParams} object.
116      *
117      * @return A {@link NetworkParams} object.
118      */
119     NetworkParams getNetworkParams();
120 }
